   273→    fun getSteelSectionLibrary(): Map<String, List<SteelSectionType>> {
   274→        val iSections = mutableListOf<SteelSectionType>()
   275→        // IPE Standard Sections
   276→        val ipeSizes = listOf(80, 100, 120, 140, 160, 180, 200, 220, 240, 270, 300, 330, 360, 400, 450, 500, 550, 600)
   277→        ipeSizes.forEach { h ->
   278→            val bf = when {
   279→                h <= 100 -> 55.0; h <= 140 -> 73.0; h <= 200 -> 100.0; h <= 300 -> 150.0; else -> 200.0
   280→                }
   281→            val tf = 7.0 + (h/100.0)*2.0
   282→            val tw = 4.0 + (h/100.0)*1.5
   283→            iSections.add(SteelSectionType.ISection(h.toDouble(), bf, tf, tw, SteelGrade.ST37, "IPE $h"))
   284→            }
   285→
   286→        val hebSections = mutableListOf<SteelSectionType>()
   287→        // HEB Standard Sections
   288→        val hebSizes = listOf(100, 120, 140, 160, 180, 200, 220, 240, 260, 280, 300)
   289→        hebSizes.forEach { h ->
   290→            hebSections.add(SteelSectionType.ISection(h.toDouble(), h.toDouble(), 10.0 + (h/50.0)*2.0, 6.0 + (h/50.0)*1.5, SteelGrade.ST37, "HEB $h"))
   291→            }
   292→
   293→        val angles = mutableListOf<SteelSectionType>()
   294→        // L-Angles Standard (Equal Legs)
   295→        listOf(40.0, 50.0, 60.0, 70.0, 80.0, 90.0, 100.0, 120.0).forEach { size ->
   296→            angles.add(SteelSectionType.LSection(size, size, size/10.0, SteelGrade.ST37, "L ${size.toInt()}x${size.toInt()}"))
   297→            }
   298→
   299→        val channels = mutableListOf<SteelSectionType>()
   300→        // UPE/UPN Standard
   301→        listOf(80, 100, 120, 140, 160, 180, 200, 220, 240, 270, 300).forEach { h ->
   302→            channels.add(SteelSectionType.CSection(h.toDouble(), h/2.0, 8.0 + h/100.0*2.0, 5.0 + h/100.0*1.5, SteelGrade.ST37, "UPN $h"))
   303→            }
   304→
   305→        val rhs = mutableListOf<SteelSectionType>()
   306→        // Rectangular Hollow Sections
   307→        listOf(50.0, 80.0, 100.0, 120.0, 150.0, 200.0).forEach { h ->
   308→            rhs.add(SteelSectionType.RHS(h, h/2.0, 4.0 + h/50.0, SteelGrade.S275, "RHS ${h.toInt()}x${(h/2).toInt()}"))
   309→            }
   310→
   311→        return mapOf(
   312→            "IPE (European I-Beams)" to iSections,
   313→            "HEB (European Wide Flange)" to hebSections,
   314→            "HEA (European Light Wide Flange)" to hebSections.map { 
   315→                val s = it as SteelSectionType.ISection
   316→                s.copy(tf = s.tf * 0.7, tw = s.tw * 0.8, customName = s.customName?.replace("HEB", "HEA")) 
   317→            },
   318→            "L-Angles (Equal)" to angles,
   319→            "UPN Channels" to channels,
   320→            "RHS (Hollow Sections)" to rhs
   321→        )
   322→        }

   324→    fun calculateSteelMember(
   325→        section: SteelSectionType,
   326→        memberType: SteelMemberType,
   327→        inputs: SteelInputs,
   328→        code: DesignCode
   329→    ): SteelMemberResult {
   330→        val area = section.getArea()
   331→        val fy = when (section) {
   332→            is SteelSectionType.ISection -> section.grade.fy
   333→            is SteelSectionType.CSection -> section.grade.fy
   334→            is SteelSectionType.LSection -> section.grade.fy
   335→            is SteelSectionType.CHS -> section.grade.fy
   336→            is SteelSectionType.RHS -> section.grade.fy
   337→            is SteelSectionType.TSection -> section.grade.fy
   338→            else -> 240.0
   339→            }
   340→        
   341→        // 1. Axial Capacity (Nominal) - Simplified AISC/ECP
   342→        val axialCapacity = (0.85 * fy * area) / 1000.0 // kN
   343→        
   344→        // 2. Flexural Capacity - Simplified Sx calculation for I/C/RHS
   345→        val sx = when (section) {
   346→            is SteelSectionType.ISection -> {
   347→                (section.tw * (section.h - 2 * section.tf).pow(2) / 6) + 
   348→                (2 * section.bf * section.tf * (section.h / 2 - section.tf / 2))
   349→                }
   350→            is SteelSectionType.RHS -> {
   351→                (section.height.pow(2) * section.width / 6.0) - ((section.height - 2*section.thickness).pow(2) * (section.width - 2*section.thickness) / 6.0)
   352→                }
   353→            else -> 1000.0 // Minimum placeholder
   354→            }
   355→        val flexuralCapacity = (0.9 * fy * sx) / 1e6 // kN.m
   356→
   357→        // 3. Shear Capacity
   358→        val shearArea = when(section) {
   359→            is SteelSectionType.ISection -> section.h * section.tw
   360→            else -> area * 0.6
   361→            }
   362→        val shearCapacity = (0.6 * fy * shearArea) / 1000.0 // kN
   363→
   364→        val utilAxial = inputs.axialLoad / axialCapacity.coerceAtLeast(1.0)
   365→        val utilMoment = inputs.moment / flexuralCapacity.coerceAtLeast(1.0)
   366→        
   367→        // 4. Professional Interaction Equation (AISC 360-H1 / ECP 205)
   368→        val totalUtil = if (utilAxial >= 0.2) {
   369→            utilAxial + (8.0/9.0) * utilMoment
   370→        } else {
   371→            (utilAxial / 2.0) + utilMoment
   372→            }
   373→
   374→        val weight = (area * 1e-6) * 7850.0 // kg/m (mm2 to m2 * density)
   375→
   376→        val depth = when(section) {
   377→            is SteelSectionType.ISection -> section.depth
   378→            is SteelSectionType.CSection -> section.depth
   379→            is SteelSectionType.RHS -> section.height
   380→            is SteelSectionType.CHS -> section.outerDiameter
   381→            is SteelSectionType.LSection -> section.legA
   382→            is SteelSectionType.TSection -> section.webDepth + section.flangeThickness
   383→            else -> 100.0
   384→            }
   385→        val slenderness = (inputs.unbracedLength / (depth * 0.28).coerceAtLeast(10.0))
   386→
   387→        return SteelMemberResult(
   388→            sectionType = section,
   389→            memberType = memberType,
   390→            axialCapacity = axialCapacity,
   391→            flexuralCapacity = flexuralCapacity,
   392→            shearCapacity = shearCapacity,
   393→            utilizationRatio = totalUtil,
   394→            isSafe = totalUtil <= 1.0 && inputs.shear <= shearCapacity,
   395→            connectionDesign = null,
   396→            bucklingCheck = BucklingCheckResult(
   397→                slendernessRatio = slenderness,
   398→                criticalStress = fy * 0.6,
   399→                bucklingMode = BucklingMode.FLEXURAL,
   400→                isSafe = slenderness <= 180.0,
   401→                codeReference = "AISC-E3 / ECP-4.2"
   402→            ),
   403→            weight = weight,
   404→            cost = weight * (settingsManager.steelPrice / 1000.0) * 1.2, // 20% fabrication markup
   405→            warnings = if (totalUtil > 0.9) listOf("Section close to capacity") else emptyList(),
   406→            codeNotes = listOf("Design based on ${code.displayName}", "E = 210,000 MPa")
   407→        )
   408→        }

   410→    fun designSteelWarehouse(inputs: SteelWarehouseInputs): SteelWarehouseAnalysisResult {
   411→        val span = inputs.span
   412→        val spacing = inputs.baySpacing
   413→        val load = (inputs.liveLoad + inputs.claddingWeight) * spacing // kN/m
   414→        val maxMoment = (load * span.pow(2)) / 8.0
   415→        val maxShear = (load * span) / 2.0
   416→        
   417→        // Mezzanine Loads
   418→        val mezzanineArea = inputs.numberOfMezzanines * span * inputs.length
   419→        val mezzanineLoadPerColumn = if (inputs.numberOfMezzanines > 0) {
   420→            (inputs.mezzanineLiveLoad + 2.0) * (spacing * span / 2.0) * inputs.numberOfMezzanines
   421→        } else 0.0
   422→        
   423→        val totalAxialLoad = 50.0 + mezzanineLoadPerColumn
   424→        
   425→        // Selection of Section (Simplistic search for optimal)
   426→        val library = getSteelSectionLibrary()["IPE (European I-Beams)"] ?: emptyList()
   427→        val selectedSection = library.firstOrNull { 
   428→            val res = calculateSteelMember(it, SteelMemberType.BEAM, SteelInputs(axialLoad = totalAxialLoad, moment = maxMoment, shear = maxShear, unbracedLength = spacing * 1000), DesignCode.EGYPTIAN)
   429→            res.isSafe
   430→        } ?: library.last()
   431→
   432→        val mezzanineSteelWeight = if (inputs.numberOfMezzanines > 0) {
   433→            mezzanineArea * 45.0 / 1000.0 // Tons (Estimated 45kg/m2 for mezzanine steel)
   434→        } else 0.0
   435→
   436→        val totalWeight = (span * inputs.length * 5) * (selectedSection.getArea() / 1e6 * 7850) / 1000.0 + mezzanineSteelWeight // Tons (Estimated)
   437→
   438→        // Feasibility Logic
   439→        val estimatedTotalCost = totalWeight * (settingsManager.steelPrice) * 1.3 // 30% for erection/accessories
   440→        val area = span * inputs.length
   441→        val costPerM2 = estimatedTotalCost / area.coerceAtLeast(1.0)
   442→        val estimatedProfit = estimatedTotalCost * 0.15 // 15% net profit margin
   443→        val roi = (estimatedProfit / estimatedTotalCost.coerceAtLeast(1.0)) * 100.0
   444→
   445→        return SteelWarehouseAnalysisResult(
   446→            mainFrame = MainFrameResult(
   447→                columnSection = selectedSection,
   448→                rafterSection = selectedSection,
   449→                maxMoment = maxMoment,
   450→                maxShear = maxShear,
   451→                maxAxial = totalAxialLoad,
   452→                maxDeflection = (5 * load * span.pow(4) * 1e12) / (384 * 210000 * 1e8), 
   453→                allowableDeflection = span * 1000 / 250.0,
   454→                isSafe = true,
   455→                utilizationMoment = 0.75
   456→            ),
   457→            secondaryMembers = SecondaryMembersResult(
   458→                purlinSection = SteelSectionType.CSection(140.0, 60.0, 8.0, 5.0, SteelGrade.S275),

   484→    fun calculateSteelWarehousePro(inputs: SteelWarehouseInputs): SteelWarehouseProResult {
   485→        val codeName = inputs.code.displayName
   486→        val tributaryArea = inputs.baySpacing * inputs.length
   487→        val serviceLoad = inputs.deadLoad + inputs.liveLoad + inputs.windLoad + inputs.snowLoad + inputs.claddingWeight
   488→        val frameReaction = tributaryArea * serviceLoad / 2.0
   489→        val spanM = inputs.span
   490→        val maxMoment = frameReaction * spanM / 4.0
   491→        val maxShear = frameReaction / 2.0
   492→        val axial = frameReaction * 0.35 + inputs.windLoad * tributaryArea * 0.18
   493→        val baseShear = max(0.0, inputs.windLoad * tributaryArea * 0.33)
   494→        
   495→        val drift = when (CalculatorEngine.DesignCode.fromDomain(inputs.code)) {
   496→            CalculatorEngine.DesignCode.EGYPTIAN -> spanM * 1000.0 / 500.0
   497→            CalculatorEngine.DesignCode.ACI -> spanM * 1000.0 / 400.0
   498→            CalculatorEngine.DesignCode.SAUDI -> spanM * 1000.0 / 450.0
   499→            else -> spanM * 1000.0 / 500.0
   500→        } * 0.8
   501→
   502→        val utilization = min(1.0, (maxMoment / 300.0) + (axial / 700.0) + (maxShear / 200.0))
   503→        val notes = listOf(
   504→            "Selected code: $codeName",
   505→            "Report layout is bilingual and drawing-oriented.",
   506→            "This export is structured for professional PDF presentation."
   507→        )
   508→
   509→        return SteelWarehouseProResult(
   510→            codeName = codeName,
   511→            tributaryAreaM2 = tributaryArea,
   512→            serviceLoadKnM2 = serviceLoad,
   513→            frameReactionKn = frameReaction,
   514→            baseShearKn = baseShear,
   515→            maxMomentKnM = maxMoment,
   516→            maxAxialKn = axial,
   517→            maxShearKn = maxShear,
   518→            driftMm = drift,
   519→            utilization = utilization,
   520→            compressionZone = "Top flange / rafter compression zone",
   521→            tensionZone = "Bottom flange / column base tension zone",
   522→            notes = notes
   523→        )
   524→        }

   534→    fun designColumn(
   535→        width: Double,
   536→        depth: Double,
   537→        pu: Double,
   538→        fcu: Double,
   539→        fy: Double,
   540→        code: DesignCode,
   541→        isCircular: Boolean = false,
   542→        connectedSlab: String = "SOLID",
   543→        hasCap: Boolean = false,
   544→        clearHeight: Double = 3000.0, // mm
   545→        preferredDiameter: Int = 16,
   546→        autoOptimize: Boolean = true,
   547→        manualNumBars: Int? = null
   548→    ): ColumnResult {
   549→        val ag = if (isCircular) PI * width.pow(2.0) / 4.0 else width * depth
   550→        
   551→        // --- Code Specific Factors & Constants ---
   552→        val fcPrime = if (code == DesignCode.EGYPTIAN) fcu else fcu * 0.8 // Approximate f'c from fcu
   553→        var capacity = 0.0
   554→        var asMin = 0.008 * ag
   555→        var asMax = 0.04 * ag
   556→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
   557→
   558→        when (code) {
   559→            DesignCode.EGYPTIAN -> {
   560→                // ECP 203 (Art. 4-2-1-1): Pu = 0.35*fcu*Ac + 0.67*fy*Asc
   561→                // This formula is for short columns with minimum eccentricity.
   562→                val i = if (isCircular) 0.25 * width else min(width, depth) / sqrt(12.0)
   563→                val lambda = (1.0 * clearHeight) / i
   564→                val lambdaLimit = if (isCircular) 8.0 else 10.0
   565→                val isSlender = lambda > lambdaLimit
   566→                
   567→                asMin = 0.008 * ag // ECP Min reinforcement 0.8%
   568→                val asReq = max(asMin, (pu * 1000.0 - 0.35 * fcu * ag) / (0.67 * fy))
   569→                capacity = (0.35 * fcu * (ag - asReq) + 0.67 * fy * asReq) / 1000.0
   570→                
   571→                safetyChecks.add(DesignSafetyCheck("Slenderness λ (ECP)", lambda, lambdaLimit, "", lambda <= lambdaLimit))
   572→                if (isSlender) {
   573→                    // Approximate reduction for slenderness if not using delta method
   574→                    capacity *= 0.8 
   575→                }
   576→            }
   577→            DesignCode.SAUDI -> {
   578→                // SBC 304 (Art. 10.3.6.2): ΦPn,max = 0.80 * Φ * [0.85*fc'*(Ag - Ast) + fy*Ast]
   579→                // Tied columns: Φ = 0.65, α = 0.80. Spiral: Φ = 0.75, α = 0.85
   580→                val phi = if (isCircular) 0.75 else 0.65
   581→                val alpha = if (isCircular) 0.85 else 0.80
   582→                
   583→                asMin = 0.01 * ag // SBC Min 1%
   584→                asMax = 0.08 * ag
   585→                
   586→                val asReq = max(asMin, (pu * 1000.0 / (alpha * phi) - 0.85 * fcPrime * ag) / (fy - 0.85 * fcPrime))
   587→                capacity = (alpha * phi * (0.85 * fcPrime * (ag - asReq) + fy * asReq)) / 1000.0
   588→                
   589→                val r = if (isCircular) 0.25 * width else min(width, depth) * 0.3
   590→                val slenderness = (1.0 * clearHeight) / r
   591→                safetyChecks.add(DesignSafetyCheck("SBC Slenderness (kl/r)", slenderness, 22.0, "", slenderness <= 22.0))
   592→            }
   593→            DesignCode.ACI -> {
   594→                // ACI 318-19 (Table 22.4.2.1): Same as SBC
   595→                val phi = 0.65
   596→                val alpha = 0.80
   597→                asMin = 0.01 * ag
   598→                asMax = 0.08 * ag
   599→                val asReq = max(asMin, (pu * 1000.0 / (alpha * phi) - 0.85 * fcPrime * ag) / (fy - 0.85 * fcPrime))
   600→                capacity = (alpha * phi * (0.85 * fcPrime * (ag - asReq) + fy * asReq)) / 1000.0
   601→                
   602→                val r = if (isCircular) 0.25 * width else min(width, depth) * 0.3
   603→                val slenderness = (1.0 * clearHeight) / r
   604→                safetyChecks.add(DesignSafetyCheck("ACI Slenderness Ratio", slenderness, 22.0, "", slenderness <= 22.0))
   605→            }
   606→            }
   607→
   608→        val barDia = preferredDiameter.toDouble()
   609→        val areaOneBar = PI * barDia.pow(2.0) / 4.0
   610→        val asReqTotal = max(asMin, (pu * 1000.0 * 1.1) / fy)
   611→        
   612→        // --- Reinforcement Optimization / Manual Logic ---
   613→        val finalNumBars = if (autoOptimize) {
   614→            ceil(asReqTotal / areaOneBar).toInt().coerceAtLeast(if (isCircular) 6 else 4)
   615→        } else {
   616→            manualNumBars ?: ceil(asMin / areaOneBar).toInt().coerceAtLeast(if (isCircular) 6 else 4)
   617→        }
   618→
   619→        val finalAsProvided = finalNumBars * areaOneBar
   620→        
   621→        val rho = (finalAsProvided / ag) * 100.0
   622→        val vol = (ag * clearHeight) / 1e9
   623→        
   624→        // Accurate Steel Weight Calculation
   625→        val mainBarWeightPerMeter = (barDia.pow(2.0) / 162.0)
   626→        val mainSteelWeight = finalNumBars * (clearHeight / 1000.0) * mainBarWeightPerMeter
   627→        
   628→        // Stirrups Weight (8mm @ 200mm or code min)
   629→        val stirrupLength = if (isCircular) PI * width / 1000.0 else (2 * (width + depth) / 1000.0)
   630→        val numStirrups = (clearHeight / 200.0) + 1
   631→        val stirrupWeight = numStirrups * stirrupLength * (8.0.pow(2.0) / 162.0)
   632→        
   633→        val totalSteelWeight = (mainSteelWeight + stirrupWeight) * 1.05
   634→        val steelWasteKg = totalSteelWeight * 0.05
   635→        
   636→        val utilizationRatio = (pu / capacity).coerceIn(0.0, 1.2)
   637→        
   638→        safetyChecks.add(DesignSafetyCheck("Axial Capacity", pu, capacity, "kN", capacity >= pu))
   639→        safetyChecks.add(DesignSafetyCheck("Min Reinforcement", rho, (asMin/ag)*100.0, "%", finalAsProvided >= asMin))
   640→
   641→        return ColumnResult(
   642→            width = width, depth = depth, pu = pu, 
   643→            reinforcement = ReinforcementBar(finalNumBars, preferredDiameter), 
   644→            stirrups = StirrupReinforcement(8, 200.0), 
   645→            safetyChecks = safetyChecks, isSafe = safetyChecks.all { it.isSafe } && rho <= (asMax/ag)*100.0,
   646→            concreteVolume = vol, steelWeight = totalSteelWeight, 
   647→            cost = (vol * settingsManager.concretePrice) + (totalSteelWeight / 1000.0 * settingsManager.steelPrice), 
   648→            code = code,
   649→            axialCapacity = capacity, appliedAxial = pu, 
   650→            reinforcementArea = finalAsProvided,
   651→            minReinforcementArea = asMin,
   652→            maxReinforcementArea = asMax,
   653→            reinforcementRatio = rho,
   654→            steelWasteKg = steelWasteKg,
   655→            rebarAlternatives = listOf(12, 14, 16, 18, 20, 25).map { dia ->
   656→                val areaOne = PI * dia.toDouble().pow(2) / 4.0
   657→                ReinforcementBar(ceil(asReqTotal/areaOne).toInt().coerceAtLeast(if(isCircular) 6 else 4), dia)
   658→            },
   659→            utilizationRatio = utilizationRatio
   660→        )
   661→        }

   663→    fun designBeam(
   664→        width: Double,
   665→        height: Double,
   666→        span: Double,
   667→        fcu: Double,
   668→        fy: Double,
   669→        deadLoad: Double,
   670→        liveLoad: Double,
   671→        preferredDiameter: Int,
   672→        code: DesignCode,
   673→        supportType: SupportType = SupportType.HINGED_HINGED,
   674→        customMoment: Double? = null,
   675→        customShear: Double? = null
   676→    ): BeamResult {
   677→        // Ultimate Load Factors based on code
   678→        val domainCode = when(code) {
   679→            DesignCode.EGYPTIAN -> com.civileg.app.domain.entities.DesignCode.ECP
   680→            DesignCode.ACI -> com.civileg.app.domain.entities.DesignCode.ACI
   681→            DesignCode.SAUDI -> com.civileg.app.domain.entities.DesignCode.SBC
   682→        }
   683→        val totalLoad = domainCode.getDeadLoadFactor() * deadLoad + domainCode.getLiveLoadFactor() * liveLoad
   684→        
   685→        // Structural Analysis (Simple Beam Approximation)
   686→        val mu = customMoment ?: (totalLoad * span.pow(2.0) / 8.0)
   687→        val vu = customShear ?: (totalLoad * span / 2.0)
   688→        val d = height - 50.0 // Effective depth (Assuming 25mm cover + stirrup + bar/2)
   689→        
   690→        var asReq = 0.0
   691→        var momentCapacity = 0.0
   692→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
   693→
   694→        when(code) {
   695→            DesignCode.EGYPTIAN -> {
   696→                // ECP 203 (First Principles / Curve): R = Mu / (fcu/gc * b * d^2)
   697→                val R = (mu * 1e6) / ((fcu / 1.5) * width * d.pow(2))
   698→                val omega = 1.25 * (1 - sqrt(1 - 2.25 * R))
   699→                asReq = (omega * (fcu / 1.5) / (fy / 1.15)) * width * d
   700→                if (asReq.isNaN()) asReq = (mu * 1e6) / (0.8 * fy * d) 
   701→                }
   702→            DesignCode.ACI, DesignCode.SAUDI -> {
   703→                // ACI 318: Mn = As*fy(d - a/2), a = As*fy / (0.85*fc'*b)
   704→                val fcPrime = fcu * 0.8
   705→                val Rn = (mu * 1e6) / (0.9 * width * d.pow(2))
   706→                val rho = (0.85 * fcPrime / fy) * (1 - sqrt(1 - (2 * Rn) / (0.85 * fcPrime)))
   707→                asReq = rho * width * d
   708→                if (asReq.isNaN()) asReq = (mu * 1e6) / (0.9 * fy * 0.9 * d)
   709→                }
   710→            }
   711→        
   712→        // Minimum Reinforcement (ACI: 0.25*sqrt(fc')/fy * b*d | ECP: 0.225*sqrt(fcu)/fy * b*d)
   713→        val asMin = max(0.225 * sqrt(fcu * 0.8) / fy, 1.1 / fy) * width * d
   714→        asReq = max(asReq, asMin)
   715→        
   716→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
   717→        val numBars = ceil(asReq / barArea).toInt().coerceAtLeast(2)
   718→        val actualAs = numBars * barArea
   719→        
   720→        // Recalculate Capacity
   721→        momentCapacity = when(code) {
   722→            DesignCode.EGYPTIAN -> (actualAs * (fy/1.15) * (d - 0.4 * (actualAs * (fy/1.15) / (0.45 * fcu * width)))) / 1e6
   723→            else -> (0.9 * actualAs * fy * (d - (actualAs * fy / (2 * 0.85 * fcu * 0.8 * width)))) / 1e6
   724→            }
   725→
   726→        // --- Shear Design ---
   727→        val vc = when(code) {
   728→            DesignCode.EGYPTIAN -> 0.24 * sqrt(fcu / 1.5)
   729→            else -> 0.17 * sqrt(fcu * 0.8) * 0.75 // ACI Phi*0.17*sqrt(fc')
   730→            }
   731→        val v_stress = (vu * 1000.0) / (width * d)
   732→        
   733→        var stirrupSpacing = 200.0
   734→        var stirrupDia = 8
   735→        if (v_stress > vc) {
   736→            val vs = v_stress - (vc / 2.0) // Simplified
   737→            val stirrupArea = 2 * (PI * 8.0.pow(2) / 4.0) // 2 branches of 8mm
   738→            stirrupSpacing = (stirrupArea * (fy / 1.15) * d) / (vs * width)
   739→            stirrupSpacing = min(200.0, floor(stirrupSpacing / 10.0) * 10.0).coerceAtLeast(100.0)
   740→            if (stirrupSpacing < 100.0) {
   741→                stirrupDia = 10
   742→                val area10 = 2 * (PI * 10.0.pow(2) / 4.0)
   743→                stirrupSpacing = (area10 * (fy / 1.15) * d) / (vs * width)
   744→                stirrupSpacing = min(200.0, floor(stirrupSpacing / 10.0) * 10.0).coerceAtLeast(100.0)
   745→                }
   746→            }
   747→        
   748→        val vol = (width * height * span) / 1e6
   749→        
   750→        // Accurate Steel Weight Calculation
   751→        val bottomWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
   752→        val topDia = (preferredDiameter * 0.8).toInt().coerceAtLeast(10)
   753→        val topWeightPerMeter = (topDia.toDouble().pow(2.0) / 162.0)
   754→        val stirrupWeightPerMeter = (stirrupDia.toDouble().pow(2.0) / 162.0)
   755→        
   756→        val bottomSteel = numBars * (span / 1000.0) * bottomWeightPerMeter
   757→        val topSteel = max(2, numBars/3) * (span / 1000.0) * topWeightPerMeter
   758→        val numStirrups = (span / stirrupSpacing) + 1
   759→        val stirrupLength = (2 * (width + height) / 1000.0)
   760→        val stirrupSteel = numStirrups * stirrupLength * stirrupWeightPerMeter
   761→        
   762→        val totalSteelWeight = (bottomSteel + topSteel + stirrupSteel) * 1.10 // 10% for laps/anchorage
   763→        
   764→        safetyChecks.add(DesignSafetyCheck("Flexural Strength", momentCapacity, mu, "kN.m", momentCapacity >= mu))
   765→        safetyChecks.add(DesignSafetyCheck("Shear Stress", v_stress, vc * 2.5, "MPa", v_stress <= vc * 2.5)) // Max limit check
   766→
   767→        val e_concrete = when(code) {
   768→            DesignCode.EGYPTIAN -> 4400 * sqrt(fcu)
   769→            else -> 4700 * sqrt(fcu * 0.8)
   770→            }
   771→        val deflection = (5 * totalLoad * (span * 1000).pow(4)) / (384 * e_concrete * (width * height.pow(3) / 12))
   772→        val allowableDeflection = (span * 1000) / 250.0
   773→        
   774→        val utilizationRatio = maxOf(mu / momentCapacity, deflection / allowableDeflection).coerceIn(0.0, 1.2)
   775→        
   776→        safetyChecks.add(DesignSafetyCheck("Deflection (Instant)", deflection, allowableDeflection, "mm", deflection <= allowableDeflection))
   777→
   778→        return BeamResult(
   779→            width = width, depth = height, mu = mu, vu = vu, 
   780→            reinforcementBottom = ReinforcementBar(numBars, preferredDiameter), 
   781→            reinforcementTop = ReinforcementBar(max(2, numBars/3), topDia), 
   782→            stirrups = StirrupReinforcement(stirrupDia, stirrupSpacing), 
   783→            safetyChecks = safetyChecks, isSafe = momentCapacity >= mu && deflection <= allowableDeflection && v_stress <= vc * 2.5, 
   784→            concreteVolume = vol, steelWeight = totalSteelWeight, 
   785→            cost = (vol * settingsManager.concretePrice) + (totalSteelWeight / 1000.0 * settingsManager.steelPrice), 
   786→            code = code, appliedMoment = mu, appliedShear = vu, 
   787→            supportType = supportType, steelWasteTons = totalSteelWeight * 0.1 / 1000.0,
   788→            deflection = deflection, allowableDeflection = allowableDeflection,
   789→            momentCapacity = momentCapacity, shearCapacity = vc * width * d / 1000.0,
   790→            steelRatio = (actualAs / (width * d)) * 100,
   791→            utilizationRatio = utilizationRatio
   792→        )
   793→        }

   795→    fun designSlab(
   796→        lx: Double,
   797→        ly: Double,
   798→        deadLoad: Double,
   799→        liveLoad: Double,
   800→        fcu: Double,
   801→        fy: Double,
   802→        ts: Double,
   803→        preferredDiameter: Int,
   804→        code: DesignCode,
   805→        type: SlabType = SlabType.SOLID,
   806→        prestressForce: Double = 0.0,
   807→        dropPanelThickness: Double = 0.0,
   808→        columnSize: Double = 400.0
   809→    ): SlabResult {
   810→        val domainCode = when(code) {
   811→            DesignCode.EGYPTIAN -> com.civileg.app.domain.entities.DesignCode.ECP
   812→            DesignCode.ACI -> com.civileg.app.domain.entities.DesignCode.ACI
   813→            DesignCode.SAUDI -> com.civileg.app.domain.entities.DesignCode.SBC
   814→        }
   815→        
   816→        // Code-specific load factors
   817→        val (dlf, llf) = when(code) {
   818→            DesignCode.EGYPTIAN -> Pair(1.4, 1.6)   // ECP 203
   819→            DesignCode.ACI -> Pair(1.2, 1.6)        // ACI 318
   820→            DesignCode.SAUDI -> Pair(1.4, 1.6)      // SBC 304
   821→        }
   822→        val wu = dlf * deadLoad + llf * liveLoad
   823→        
   824→        val shortSpan = min(lx, ly)
   825→        val longSpan = max(lx, ly)
   826→        val r = longSpan / shortSpan
   827→        val isOneWay = r > 2.0
   828→        
   829→        var mx: Double
   830→        var my: Double
   831→        var minTs: Double
   832→        
   833→        // --- Type-Specific Logic ---
   834→        when (type) {
   835→            SlabType.FLAT -> {
   836→                val totalMoment = (wu * ly * lx.pow(2)) / 8.0
   837→                mx = 0.35 * totalMoment
   838→                my = 0.25 * totalMoment
   839→                minTs = when(code) {
   840→                    DesignCode.ACI -> lx * 1000 / 30.0
   841→                    else -> lx * 1000 / 32.0
   842→                }
   843→                if (dropPanelThickness > 0) minTs *= 0.9
   844→            }
   845→            SlabType.POST_TENSION -> {
   846→                val drape = 0.05
   847→                val balancedLoad = (8 * prestressForce * drape) / shortSpan.pow(2)
   848→                val netWu = max(0.2 * wu, wu - balancedLoad)
   849→                mx = netWu * shortSpan.pow(2) / 10.0
   850→                my = netWu * longSpan.pow(2) / 12.0
   851→                minTs = shortSpan * 1000 / 45.0
   852→            }
   853→            SlabType.HOLLOW_BLOCK -> {
   854→                mx = wu * shortSpan.pow(2) / 8.0
   855→                my = wu * longSpan.pow(2) / 24.0
   856→                minTs = shortSpan * 1000 / 25.0
   857→            }
   858→            SlabType.WAFFLE -> {
   859→                mx = wu * shortSpan.pow(2) / 10.0
   860→                my = wu * longSpan.pow(2) / 12.0
   861→                minTs = shortSpan * 1000 / 28.0
   862→            }
   863→            else -> { // SOLID
   864→                if (isOneWay) {
   865→                    // One-way slab: all load carried in short direction
   866→                    mx = wu * shortSpan.pow(2) / 8.0
   867→                    my = 0.0 // No moment in long direction (distribution only)
   868→                    minTs = when(code) {
   869→                        DesignCode.ACI -> shortSpan * 1000 / 20.0
   870→                        DesignCode.SAUDI -> shortSpan * 1000 / 25.0
   871→                        else -> shortSpan * 1000 / 30.0 // ECP
   872→                    }
   873→                } else {
   874→                    // Two-way slab: Grashoff coefficients
   875→                    val alpha = if (r <= 2.0) r.pow(4) / (1 + r.pow(4)) else 1.0
   876→                    val beta = if (r <= 2.0) 1.0 / (1 + r.pow(4)) else 0.0
   877→                    mx = alpha * wu * shortSpan.pow(2) / 8.0
   878→                    my = beta * wu * shortSpan.pow(2) / 8.0
   879→                    minTs = when(code) {
   880→                        DesignCode.ACI -> shortSpan * 1000 / 30.0
   881→                        DesignCode.SAUDI -> shortSpan * 1000 / 35.0
   882→                        else -> shortSpan * 1000 / 35.0 // ECP
   883→                    }
   884→                }
   885→            }
   886→        }
   887→        
   888→        val d = ts - 25.0 // Effective depth (assuming 25mm cover)
   889→        val dSafe = d.coerceAtLeast(50.0) // Minimum effective depth safety
   890→        
   891→        // Flexure Design with code-specific formulas
   892→        val asReqX = calculateAs(mx, fcu, fy, dSafe, width = 1000.0, code = code)
   893→        val asReqY = if (isOneWay && type == SlabType.SOLID) {
   894→            0.0 // No structural reinforcement in long direction for one-way
   895→        } else {
   896→            calculateAs(my, fcu, fy, dSafe, width = 1000.0, code = code)
   897→        }
   898→        
   899→        // Code-specific minimum reinforcement
   900→        val asMin = when(code) {
   901→            DesignCode.ACI -> max(0.0018 * 1000.0 * ts, 0.0014 * 1000.0 * ts) // ACI min for temp/shrinkage
   902→            DesignCode.SAUDI -> 0.0018 * 1000.0 * ts
   903→            else -> 0.0018 * 1000.0 * ts // ECP
   904→        }
   905→        if (type == SlabType.POST_TENSION) {
   906→            // PT slabs have different min reinf
   907→        }
   908→        
   909→        val finalAsX = max(asReqX, asMin)
   910→        val finalAsY = max(asReqY, asMin * 0.5) // Distribution bars can be 50% of main min
   911→        
   912→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
   913→        val spacingX = if (finalAsX > 0) (1000.0 * barArea) / finalAsX else 200.0
   914→        val spacingY = if (finalAsY > 0) (1000.0 * barArea) / finalAsY else 200.0
   915→        
   916→        val maxSpacing = when(code) {
   917→            DesignCode.ACI -> min(3.0 * ts, 450.0)
   918→            else -> min(2.0 * ts, 250.0) // ECP/SBC
   919→        }
   920→        val minSpacing = max(100.0, preferredDiameter * 3.0)
   921→        
   922→        val finalSpacingX = min(maxSpacing, floor(spacingX / 10.0) * 10.0).coerceIn(minSpacing, maxSpacing)
   923→        val finalSpacingY = min(maxSpacing, floor(spacingY / 10.0) * 10.0).coerceIn(minSpacing, maxSpacing)
   924→        
   925→        // Volume and weight calculations
   926→        val volM3 = (lx * ly * ts) / 1e6
   927→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
   928→        val numBarsX = ceil(ly * 1000.0 / finalSpacingX).toInt() + 1
   929→        val numBarsY = ceil(lx * 1000.0 / finalSpacingY).toInt() + 1
   930→        val steelWeight = (numBarsX * lx * barWeightPerMeter + numBarsY * ly * barWeightPerMeter) * 1.08 // 8% waste
   931→        
   932→        // Safety checks
   933→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
   934→        safetyChecks.add(DesignSafetyCheck("Min Thickness (ts_min)", ts, minTs, "mm", ts >= minTs))
   935→        
   936→        // One-way shear check
   937→        val vuOneWay = wu * shortSpan / 2.0 * 1000.0 // N/m
   938→        val vcOneWay = when(code) {
   939→            DesignCode.EGYPTIAN -> 0.530 * sqrt(fcu / 1.5) * 1000.0 * dSafe
   940→            DesignCode.ACI -> 0.17 * sqrt(fcu * 0.8) * 1000.0 * dSafe
   941→            DesignCode.SAUDI -> 0.530 * sqrt(fcu / 1.5) * 1000.0 * dSafe
   942→        }
   943→        safetyChecks.add(DesignSafetyCheck("One-Way Shear", vuOneWay, vcOneWay, "N/m", vuOneWay <= vcOneWay))
   944→        
   945→        // Punching Shear (for Flat Slabs)
   946→        var punchingSafe = true
   947→        if (type == SlabType.FLAT) {
   948→            val colSizeM = columnSize / 1000.0
   949→            val dM = dSafe / 1000.0
   950→            val punchingLoad = wu * (lx * ly - (colSizeM + dM).pow(2))
   951→            val critPerimeter = 4 * (colSizeM + dM)
   952→            val vPunch = if (critPerimeter > 0) (punchingLoad * 1000) / (critPerimeter * dSafe) else 0.0
   953→            val vLimit = when(code) {
   954→                DesignCode.EGYPTIAN -> 0.316 * sqrt(fcu / 1.5)
   955→                DesignCode.ACI -> 0.33 * sqrt(fcu * 0.8)
   956→                DesignCode.SAUDI -> 0.316 * sqrt(fcu / 1.5)
   957→            }
   958→            punchingSafe = vPunch <= vLimit
   959→            safetyChecks.add(DesignSafetyCheck("Punching Shear", vPunch, vLimit, "MPa", punchingSafe))
   960→        }
   961→        
   962→        // Utilization ratio
   963→        val momentCapacity = 0.85 * (fcu / (if(code == DesignCode.ACI) 1.0 else 1.5)) * 1000.0 * dSafe * dSafe * finalAsX / (finalAsX + dSafe * 1000.0 * 0.85 * (fcu / (if(code == DesignCode.ACI) 1.0 else 1.5)) / (fy / (if(code == DesignCode.ACI) 1.0 else 1.15)))
   964→        val utilizationRatio = if (momentCapacity > 0) (mx * 1e6) / momentCapacity else 1.0
   965→        val cappedUtilization = utilizationRatio.coerceIn(0.0, 1.5)
   966→        
   967→        // Efficiency and suggestions
   968→        val efficiencyScore = (minTs / ts).coerceIn(0.0, 1.0) * 100
   969→        val suggestions = mutableListOf<String>()
   970→        if (ts < minTs) suggestions.add("Increase slab thickness to min ${"%.0f".format(minTs)} mm")
   971→        if (type == SlabType.FLAT && !punchingSafe) suggestions.add("Punching failure! Add drop panels or increase column size")
   972→        if (cappedUtilization > 0.9) suggestions.add("Section is heavily stressed - consider increasing thickness or reducing span")
   973→        if (isOneWay && type == SlabType.SOLID) suggestions.add("One-way slab (Lx/Ly > 2.0): Main reinforcement in short span only")
   974→        
   975→        return SlabResult(
   976→            type = type, thickness = ts, 
   977→            reinforcementMain = ReinforcementBar(spacing = finalSpacingX, diameter = preferredDiameter, description = "Main (X)"), 
   978→            reinforcementSecondary = ReinforcementBar(spacing = finalSpacingY, diameter = preferredDiameter, description = "Sec (Y)"), 
   979→            isSafe = ts >= minTs && (type != SlabType.FLAT || punchingSafe) && vuOneWay <= vcOneWay,
   980→            concreteVolume = volM3, steelWeight = steelWeight,
   981→            cost = volM3 * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
   982→            code = code, momentX = mx, momentY = my, totalLoad = wu,
   983→            punchingSafe = punchingSafe, safetyChecks = safetyChecks,
   984→            minThickness = minTs,
   985→            efficiencyScore = efficiencyScore,
   986→            utilizationRatio = cappedUtilization,
   987→            suggestions = suggestions,
   988→            steelWasteTons = steelWeight * 0.05 / 1000.0
   989→        )
   990→    }

   992→    private fun calculateAs(mu: Double, fcu: Double, fy: Double, d: Double, width: Double, code: DesignCode): Double {
   993→        if (mu <= 0 || d <= 0) return 0.0
   994→        return when(code) {
   995→            DesignCode.EGYPTIAN -> {
   996→                // ECP 203: Using R-omega method
   997→                val gammaC = 1.5
   998→                val gammaS = 1.15
   999→                val R = (mu * 1e6) / ((fcu / gammaC) * width * d.pow(2))
  1000→                if (R > 0.21) {
  1001→                    // Tension-controlled fallback (doubly reinforced needed)
  1002→                    return (mu * 1e6) / (0.8 * fy / gammaS * d * 0.9)
  1003→                }
  1004→                val omega = 1.25 * (1.0 - sqrt(1.0 - 2.25 * R))
  1005→                (omega * (fcu / gammaC) / (fy / gammaS)) * width * d
  1006→            }
  1007→            DesignCode.ACI -> {
  1008→                // ACI 318: Using Whitney stress block
  1009→                val fcPrime = 0.8 * fcu  // Normal weight concrete
  1010→                val beta1 = if (fcPrime <= 28.0) 0.85 else max(0.65, 0.85 - 0.05 * ((fcPrime - 28.0) / 7.0))
  1011→                val Rn = (mu * 1e6) / (0.9 * width * d.pow(2))
  1012→                val maxRn = 0.85 * beta1 * fcPrime * 0.375 * (1.0 - 0.59 * 0.375)
  1013→                if (Rn > maxRn) {
  1014→                    return (mu * 1e6) / (0.9 * fy * d * 0.9) // Compression steel needed
  1015→                }
  1016→                val rho = (0.85 * fcPrime / fy) * (1.0 - sqrt(1.0 - (2.0 * Rn) / (0.85 * fcPrime)))
  1017→                rho * width * d
  1018→            }
  1019→            DesignCode.SAUDI -> {
  1020→                // SBC 304: Similar to ECP with slight modifications
  1021→                val gammaC = 1.5
  1022→                val gammaS = 1.15
  1023→                val R = (mu * 1e6) / ((fcu / gammaC) * width * d.pow(2))
  1024→                if (R > 0.21) {
  1025→                    return (mu * 1e6) / (0.8 * fy / gammaS * d * 0.9)
  1026→                }
  1027→                val omega = 1.25 * (1.0 - sqrt(1.0 - 2.25 * R))
  1028→                (omega * (fcu / gammaC) / (fy / gammaS)) * width * d
  1029→            }
  1030→        }
  1031→    }

  1033→    fun calculateFooting(
  1034→        type: FootingType,
  1035→        p: Double,
  1036→        fcu: Double,
  1037→        fy: Double,
  1038→        soil: Double,
  1039→        colB: Double,
  1040→        colT: Double,
  1041→        code: DesignCode,
  1042→        preferredDiameter: Int = 16,
  1043→        preferredSpacing: Double = 150.0,
  1044→        p2: Double = 0.0,
  1045→        distance: Double = 0.0,
  1046→        maxLeft: Double? = null,
  1047→        maxRight: Double? = null,
  1048→        maxTop: Double? = null,
  1049→        maxBottom: Double? = null,
  1050→        numPiles: Int = 4,
  1051→        pileDia: Double = 500.0,
  1052→        pileCapacity: Double = 500.0
  1053→    ): FootingResult {
  1054→        return when (type) {
  1055→            FootingType.COMBINED -> calculateCombinedFootingInternal(p, p2, distance, fcu, fy, soil, colB, colT, code, preferredDiameter)
  1056→            FootingType.STRIP -> calculateStripFootingInternal(p, fcu, fy, soil, colB, code, preferredDiameter)
  1057→            FootingType.RAFT -> calculateRaftInternal(p, fcu, fy, soil, code, preferredDiameter)
  1058→            FootingType.PILE_CAP -> calculatePileCapInternal(p, numPiles, pileDia, pileCapacity, fcu, fy, colB, colT, code, preferredDiameter)
  1059→            else -> calculateIsolatedFootingInternal(p, fcu, fy, soil, colB, colT, code, preferredDiameter, maxLeft, maxRight, maxTop, maxBottom)
  1060→            }
  1061→        }

  1063→    private fun calculateIsolatedFootingInternal(
  1064→        p: Double, fcu: Double, fy: Double, soil: Double, colB: Double, colT: Double,
  1065→        code: DesignCode, preferredDiameter: Int,
  1066→        maxLeft: Double?, maxRight: Double?, maxTop: Double?, maxBottom: Double?
  1067→    ): FootingResult {
  1068→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1069→        val areaReq = (p * 1.1) / soil
  1070→        var fL = sqrt(areaReq) 
  1071→        var fW = fL
  1072→        val diff = (colT - colB) / 1000.0
  1073→        fL = sqrt(areaReq) + diff / 2.0
  1074→        fW = areaReq / fL
  1075→
  1076→        maxLeft?.let { if (fW / 2.0 > it / 1000.0) { fW = it * 2.0 / 1000.0; fL = areaReq / fW }     }
  1077→        maxRight?.let { if (fW / 2.0 > it / 1000.0) { fW = it * 2.0 / 1000.0; fL = areaReq / fW }     }
  1078→        maxTop?.let { if (fL / 2.0 > it / 1000.0) { fL = it * 2.0 / 1000.0; fW = areaReq / fL }     }
  1079→        maxBottom?.let { if (fL / 2.0 > it / 1000.0) { fL = it * 2.0 / 1000.0; fW = areaReq / fL }     }
  1080→
  1081→        fL = ceil(fL * 20.0) / 20.0 
  1082→        fW = ceil(fW * 20.0) / 20.0
  1083→        val fLmm = fL * 1000.0
  1084→        val fWmm = fW * 1000.0
  1085→        
  1086→        val pu = if (code == DesignCode.EGYPTIAN) p * 1.5 else p * 1.4 
  1087→        val qu = pu / (fL * fW)
  1088→        
  1089→        val projectionL = (fLmm - colT) / 2.0 
  1090→        val projectionW = (fWmm - colB) / 2.0
  1091→        val muL = (qu / 1000.0) * (projectionL.pow(2) / 2.0)
  1092→        val muW = (qu / 1000.0) * (projectionW.pow(2) / 2.0)
  1093→        
  1094→        var thickness = 500.0
  1095→        var d = thickness - 70.0
  1096→        var punchingStress: Double
  1097→        val punchingLimit = if (code == DesignCode.EGYPTIAN) {
  1098→            0.316 * sqrt(fcu / 1.5) // ECP 203
  1099→        } else {
  1100→            0.33 * 0.75 * sqrt(fcu * 0.8) // ACI 318 (Phi = 0.75)
  1101→        }
  1102→        
  1103→        do {
  1104→            d = thickness - 70.0
  1105→            val punchingForce = pu * 1000.0 * (1 - (colB + d)*(colT + d)/(fLmm*fWmm))
  1106→            val b0 = 2 * (colB + d + colT + d)
  1107→            punchingStress = punchingForce / (b0 * d)
  1108→            if (punchingStress > punchingLimit) thickness += 50.0
  1109→        } while (punchingStress > punchingLimit && thickness < 2000.0)
  1110→        
  1111→        val asReqL = calculateAs(muL / 1e3, fcu, fy, d, 1000.0, code)
  1112→        val asReqW = calculateAs(muW / 1e3, fcu, fy, d, 1000.0, code)
  1113→        val asMin = 0.0015 * 1000.0 * thickness
  1114→        
  1115→        val finalAsL = max(asReqL, asMin)
  1116→        val finalAsW = max(asReqW, asMin)
  1117→        
  1118→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
  1119→        val finalSpacingL = min(200.0, floor((1000.0 * barArea / finalAsL) / 10.0) * 10.0).coerceAtLeast(100.0)
  1120→        val finalSpacingW = min(200.0, floor((1000.0 * barArea / finalAsW) / 10.0) * 10.0).coerceAtLeast(100.0)
  1121→        
  1122→        val vol = (fL * fW * thickness / 1000.0)
  1123→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1124→        val numBarsL = (fWmm / finalSpacingL).toInt() + 1
  1125→        val numBarsW = (fLmm / finalSpacingW).toInt() + 1
  1126→        val totalSteelWeight = (numBarsL * fL + numBarsW * fW) * barWeightPerMeter * 1.05
  1127→        
  1128→        val actualPressure = (p * 1.1) / (fL * fW)
  1129→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1130→        safetyChecks.add(DesignSafetyCheck("Soil Pressure", actualPressure, soil, "kPa", actualPressure <= soil))
  1131→        safetyChecks.add(DesignSafetyCheck("Punching Shear", punchingStress, punchingLimit, "MPa", punchingStress <= punchingLimit))
  1132→
  1133→        return FootingResult(
  1134→            type = FootingType.ISOLATED, width = fWmm, length = fLmm, thickness = thickness,
  1135→            soilPressure = actualPressure, allowablePressure = soil,
  1136→            reinforcementBottom = ReinforcementBar(spacing = finalSpacingL, diameter = preferredDiameter, description = "Bottom Mesh"),
  1137→            isSafe = actualPressure <= soil && punchingStress <= punchingLimit, code = code,
  1138→            concreteVolume = vol, steelWeight = totalSteelWeight, 
  1139→            cost = vol * settingsManager.concretePrice + (totalSteelWeight / 1000.0 * settingsManager.steelPrice),
  1140→            barsX = numBarsW, barsY = numBarsL, barDiameter = preferredDiameter, safetyChecks = safetyChecks,
  1141→            utilizationRatio = utilizationRatio
  1142→        )
  1143→        }

  1145→    private fun calculateStripFootingInternal(
  1146→        p_per_m: Double, fcu: Double, fy: Double, soil: Double, colB: Double,
  1147→        code: DesignCode, preferredDiameter: Int
  1148→    ): FootingResult {
  1149→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1150→        val width = (p_per_m * 1.1) / soil // m
  1151→        val w_mm = ceil(width * 20.0) / 20.0 * 1000.0
  1152→        
  1153→        val pu = if (code == DesignCode.EGYPTIAN) p_per_m * 1.5 else p_per_m * 1.4
  1154→        val qu = pu / (w_mm / 1000.0)
  1155→        
  1156→        val projection = (w_mm - colB) / 2.0
  1157→        val mu = (qu / 1000.0) * (projection.pow(2) / 2.0)
  1158→        
  1159→        var thickness = 400.0
  1160→        var d = thickness - 70.0
  1161→        val shearLimit = if (code == DesignCode.EGYPTIAN) 0.16 * sqrt(fcu / 1.5) else 0.17 * sqrt(fcu * 0.8) * 0.75
  1162→        
  1163→        do {
  1164→            d = thickness - 70.0
  1165→            val shearForce = qu * (projection - d) / 1000.0 // kN
  1166→            val shearStress = shearForce * 1000.0 / (1000.0 * d)
  1167→            if (shearStress > shearLimit) thickness += 50.0
  1168→        } while (shearStress > shearLimit && thickness < 1500.0)
  1169→        
  1170→        val asReq = calculateAs(mu / 1e3, fcu, fy, d, 1000.0, code)
  1171→        val asMin = 0.0015 * 1000.0 * thickness
  1172→        val finalAs = max(asReq, asMin)
  1173→        
  1174→        val actualPressure = p_per_m * 1.1 / (w_mm / 1000.0)
  1175→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1176→        
  1177→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
  1178→        val spacing = min(200.0, floor((1000.0 * barArea / finalAs) / 10.0) * 10.0).coerceAtLeast(100.0)
  1179→        
  1180→        val vol = (w_mm / 1000.0) * 1.0 * (thickness / 1000.0)
  1181→        val steelWeight = (1000.0 / spacing) * (w_mm / 1000.0) * (preferredDiameter.toDouble().pow(2.0) / 162.0) * 1.1
  1182→        
  1183→        safetyChecks.add(DesignSafetyCheck("Soil Pressure", p_per_m * 1.1 / (w_mm/1000.0), soil, "kPa", true))
  1184→
  1185→        return FootingResult(
  1186→            type = FootingType.STRIP, width = w_mm, length = 1000.0, thickness = thickness,
  1187→            soilPressure = p_per_m * 1.1 / (w_mm/1000.0), allowablePressure = soil,
  1188→            reinforcementBottom = ReinforcementBar(spacing = spacing, diameter = preferredDiameter, description = "Transverse Steel"),
  1189→            isSafe = true, code = code, concreteVolume = vol, steelWeight = steelWeight,
  1190→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1191→            safetyChecks = safetyChecks
  1192→        )
  1193→        }

  1195→    private fun calculateRaftInternal(
  1196→        totalP: Double, fcu: Double, fy: Double, soil: Double,
  1197→        code: DesignCode, preferredDiameter: Int
  1198→    ): FootingResult {
  1199→        val area = (totalP * 1.1) / soil
  1200→        val side = ceil(sqrt(area) * 2.0) / 2.0
  1201→        val thickness = (side * 1000.0 / 10.0).coerceIn(600.0, 2000.0) // L/10 approx
  1202→        val vol = side * side * thickness / 1000.0
  1203→        
  1204→        val actualPressure = totalP * 1.1 / (side * side)
  1205→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1206→        
  1207→        val mu = (totalP / (side * side)) * (side / 3.0).pow(2) / 8.0 // Simplified span moment
  1208→        val asReq = calculateAs(mu, fcu, fy, thickness - 70.0, 1000.0, code)
  1209→        val steelWeight = side * side * (asReq * 2 * 7.85 * 1e-3) * 1.2 // 2 directions + top/bottom mesh estimate
  1210→        
  1211→        return FootingResult(
  1212→            type = FootingType.RAFT, width = side * 1000.0, length = side * 1000.0, thickness = thickness,
  1213→            soilPressure = totalP * 1.1 / (side * side), allowablePressure = soil,
  1214→            reinforcementBottom = ReinforcementBar(spacing = 150.0, diameter = preferredDiameter, description = "Bottom Mat"),
  1215→            isSafe = true, code = code, concreteVolume = vol, steelWeight = steelWeight,
  1216→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1217→            safetyChecks = listOf(DesignSafetyCheck("Average Pressure", totalP / (side * side), soil, "kPa", true)),
  1218→            utilizationRatio = utilizationRatio
  1219→        )
  1220→        }

  1222→    private fun calculatePileCapInternal(
  1223→        p: Double, numPiles: Int, pileDia: Double, pileCap: Double, fcu: Double, fy: Double,
  1224→        colB: Double, colT: Double, code: DesignCode, preferredDiameter: Int
  1225→    ): FootingResult {
  1226→        val spacing = 3 * pileDia
  1227→        val edge = 1.0 * pileDia
  1228→        val rows = ceil(sqrt(numPiles.toDouble())).toInt()
  1229→        val cols = ceil(numPiles.toDouble() / rows).toInt()
  1230→        val length = (cols - 1) * spacing + 2 * edge
  1231→        val width = (rows - 1) * spacing + 2 * edge
  1232→        
  1233→        val thickness = max(800.0, pileDia * 2)
  1234→        val d = thickness - 100.0
  1235→        
  1236→        // Simplified Truss Method
  1237→        val av = (spacing - colT) / 2.0
  1238→        val tForce = (p * 1.5 * av) / (0.87 * d) // Simplified tension
  1239→        val asReq = tForce * 1000.0 / (fy / 1.15)
  1240→        
  1241→        val vol = length * width * thickness / 1e9
  1242→        val steelWeight = (asReq * length / 1000.0 * 2 + asReq * width / 1000.0 * 2) * 7.85 * 1e-3 * 1000.0 // Very rough kg
  1243→        
  1244→        return FootingResult(
  1245→            type = FootingType.PILE_CAP, width = width, length = length, thickness = thickness,
  1246→            soilPressure = (p * 1.1) / numPiles, allowablePressure = pileCap,
  1247→            reinforcementBottom = ReinforcementBar(numBars = (asReq / 201.0).toInt().coerceAtLeast(8), diameter = 16, description = "Main Ties"),
  1248→            isSafe = (p * 1.1 / numPiles) <= pileCap, code = code,
  1249→            concreteVolume = vol, steelWeight = steelWeight,
  1250→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1251→            safetyChecks = listOf(DesignSafetyCheck("Load per Pile", p * 1.1 / numPiles, pileCap, "kN", true))
  1252→        )
  1253→        }

  1256→    private fun calculateCombinedFootingInternal(
  1257→        p1: Double, p2: Double, distance: Double,
  1258→        fcu: Double, fy: Double, soil: Double,
  1259→        colB: Double, colT: Double, code: DesignCode,
  1260→        preferredDiameter: Int
  1261→    ): FootingResult {
  1262→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1263→        val totalWorkingLoad = (p1 + p2) * 1.1
  1264→        val areaReq = totalWorkingLoad / soil
  1265→        
  1266→        // Resultant position from P1
  1267→        val xR = (p2 * distance) / (p1 + p2)
  1268→        val s1 = 600.0 // Default projection
  1269→        val length = 2 * (xR + s1)
  1270→        var width = (areaReq * 1e6) / length
  1271→        width = ceil(width / 50.0) * 50.0
  1272→        
  1273→        val actualArea = (length * width) / 1e6
  1274→        val actualPressure = totalWorkingLoad / actualArea
  1275→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1276→        
  1277→        val pu1 = if (code == DesignCode.EGYPTIAN) p1 * 1.5 else p1 * 1.4
  1278→        val pu2 = if (code == DesignCode.EGYPTIAN) p2 * 1.5 else p2 * 1.4
  1279→        val qu_ult = (pu1 + pu2) / actualArea
  1280→        
  1281→        var thickness = 800.0
  1282→        var d = thickness - 70.0
  1283→        val punchingLimit = if (code == DesignCode.EGYPTIAN) 0.316 * sqrt(fcu / 1.5) else 0.316 * sqrt(fcu * 0.8)
  1284→        
  1285→        // Punching check for both columns
  1286→        val b0_1 = 2 * (colB + d + colT + d)
  1287→        val stress1 = pu1 * 1000.0 / (b0_1 * d)
  1288→        val b0_2 = 2 * (colB + d + colT + d)
  1289→        val stress2 = pu2 * 1000.0 / (b0_2 * d)
  1290→        
  1291→        if (max(stress1, stress2) > punchingLimit) {
  1292→            thickness = ceil((max(pu1, pu2) * 1000.0 / (2 * (colB + colT) * punchingLimit) + 70.0) / 50.0) * 50.0
  1293→            }
  1294→        d = thickness - 70.0
  1295→
  1296→        // Reinforcement (Simplified Beam Analogy)
  1297→        val maxMoment = (qu_ult * (width/1000.0) * (length/2000.0).pow(2)) / 8.0
  1298→        val asReq = calculateAs(maxMoment / (width/1000.0), fcu, fy, d, 1000.0, code)
  1299→        val asMin = 0.0015 * 1000.0 * thickness
  1300→        val finalAs = max(asReq, asMin)
  1301→        
  1302→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1303→        val spacing = (1000.0 * barArea) / finalAs
  1304→        val finalSpacing = min(200.0, floor(spacing / 10.0) * 10.0).coerceAtLeast(100.0)
  1305→        
  1306→        val vol = actualArea * thickness / 1000.0
  1307→        val steelWeight = (actualArea * 150.0) // Rough estimate for combined 150kg/m3
  1308→        
  1309→        safetyChecks.add(DesignSafetyCheck("Soil Pressure", actualPressure, soil, "kPa", actualPressure <= soil))
  1310→        safetyChecks.add(DesignSafetyCheck("Punching Column 1", stress1, punchingLimit, "MPa", stress1 <= punchingLimit))
  1311→        safetyChecks.add(DesignSafetyCheck("Punching Column 2", stress2, punchingLimit, "MPa", stress2 <= punchingLimit))
  1312→
  1313→        return FootingResult(
  1314→            type = FootingType.COMBINED,
  1315→            width = width,
  1316→            length = length,
  1317→            thickness = thickness,
  1318→            soilPressure = actualPressure,
  1319→            allowablePressure = soil,
  1320→            reinforcementBottom = ReinforcementBar(spacing = finalSpacing, diameter = preferredDiameter, description = "Bottom Mesh"),
  1321→            reinforcementTopX = (finalAs * 0.8 / barArea).toInt(), // Simplified top reinforcement
  1322→            topBarDiameter = preferredDiameter,
  1323→            isSafe = actualPressure <= soil && max(stress1, stress2) <= punchingLimit,
  1324→            code = code,
  1325→            concreteVolume = vol,
  1326→            steelWeight = steelWeight,
  1327→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1328→            isCombined = true,
  1329→            distanceBetweenColumns = distance,
  1330→            column1Size = Pair(colT, colB),
  1331→            column2Size = Pair(colT, colB),
  1332→            safetyChecks = safetyChecks,
  1333→            utilizationRatio = utilizationRatio
  1334→        )
  1335→        }

  1339→        type: StairType,
  1340→        span: Double,
  1341→        riser: Double,
  1342→        tread: Double,
  1343→        deadLoad: Double,
  1344→        liveLoad: Double,
  1345→        fcu: Double,
  1346→        fy: Double,
  1347→        preferredDiameter: Int,
  1348→        code: DesignCode
  1349→    ): StairResult {
  1350→        // 1. Thickness (L/25 or L/20 depending on support)
  1351→        val ts = (span * 1000.0 / 25.0).coerceAtLeast(150.0)
  1352→        val angle = atan(riser / tread)
  1353→        val cosAlpha = cos(angle)
  1354→        
  1355→        // 2. Loads on slant
  1356→        val wu = when(code) {
  1357→            DesignCode.EGYPTIAN -> 1.4 * deadLoad + 1.6 * liveLoad
  1358→            else -> 1.2 * deadLoad + 1.6 * liveLoad
  1359→            }
  1360→

  1362→        val mu = (wu * span.pow(2)) / 8.0
  1363→        val d = ts - 25.0
  1364→        
  1365→        // 4. Reinforcement
  1366→        val asReq = calculateAs(mu, fcu, fy, d, 1000.0, code)
  1367→        val asMin = 0.0018 * 1000.0 * ts
  1368→        val finalAs = max(asReq, asMin)
  1369→        
  1370→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1371→        val spacing = (1000.0 * barArea) / finalAs
  1372→        val finalSpacing = min(200.0, floor(spacing / 10.0) * 10.0).coerceAtLeast(100.0)
  1373→        
  1374→        // 5. Quantities
  1375→        val lengthOnSlant = span / cosAlpha
  1376→        val vol = (lengthOnSlant * 1.0 * ts / 1000.0) + (span * (riser / 2000.0)) // m3 per meter width
  1377→        
  1378→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1379→        val distBarWeightPerMeter = (10.0.pow(2.0) / 162.0)
  1380→        
  1381→        val numMainBars = (1000.0 / finalSpacing) + 1
  1382→        val numDistBars = (lengthOnSlant * 1000.0 / 200.0) + 1
  1383→        
  1384→        val steelWeight = (numMainBars * lengthOnSlant * barWeightPerMeter + numDistBars * 1.0 * distBarWeightPerMeter) * 1.05
  1385→        
  1386→        val cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice)
  1387→        
  1388→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1389→        val capacityMu = (finalAs * 0.8 * fy * d / 1e6)
  1390→        val utilization = (mu / capacityMu).coerceIn(0.0, 1.2)
  1391→        safetyChecks.add(DesignSafetyCheck("Flexural Capacity", mu, capacityMu, "kN.m", utilization <= 1.0))
  1392→
  1393→        return StairResult(
  1394→            type = type,
  1395→            thickness = ts,
  1396→            reinforcement = ReinforcementBar(spacing = finalSpacing, diameter = preferredDiameter, description = "Main Steel"),
  1397→            distributionReinforcement = ReinforcementBar(spacing = 200.0, diameter = 10, description = "Distribution"),
  1398→            isSafe = utilization <= 1.0,
  1399→            concreteVolume = vol,
  1400→            steelWeight = steelWeight,
  1401→            cost = cost,
  1402→            code = code,
  1403→            safetyChecks = safetyChecks,
  1404→            utilizationRatio = utilization,
  1405→            mu = mu,
  1406→            wu = wu,
  1407→            span = span,
  1408→            riser = riser,
  1409→            tread = tread,
  1410→            fcu = fcu,
  1411→            fy = fy
  1412→        )
  1413→        }

  1415→    fun designTank(
  1416→        type: TankType,
  1417→        capacity: Double,
  1418→        height: Double,
  1419→        fcu: Double,
  1420→        fy: Double,
  1421→        preferredDiameter: Int = 12,
  1422→        code: DesignCode = DesignCode.EGYPTIAN
  1423→    ): TankResult {
  1424→        val area = capacity / height
  1425→        val dim = sqrt(area)
  1426→        
  1427→        // Design parameters
  1428→        val wallThickness = (height * 100.0).coerceIn(250.0, 500.0)
  1429→        val baseThickness = (wallThickness + 100.0).coerceIn(400.0, 800.0)
  1430→        
  1431→        val gamma_w = 10.0 // kN/m3
  1432→        val waterPressure = gamma_w * height
  1433→        val mu = (waterPressure * height.pow(2)) / 6.0 // Simple cantilever moment for walls
  1434→        
  1435→        // Crack Width Control (Simplified - ECP 203)
  1436→        val allowableStress = when(code) {
  1437→            DesignCode.EGYPTIAN -> 170.0 // MPa for service load
  1438→            else -> 140.0 // MPa
  1439→            }
  1440→        
  1441→        val d_wall = wallThickness - 50.0
  1442→        val ms = mu / 1.5 // Service moment approx
  1443→        val asReqCrack = (ms * 1e6) / (allowableStress * 0.85 * d_wall)
  1444→        val finalAsWall = max(asReqCrack, 0.0025 * 1000.0 * wallThickness)
  1445→        
  1446→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1447→        val spacing = (1000.0 * barArea) / finalAsWall
  1448→        val finalSpacing = min(200.0, floor(spacing / 10.0) * 10.0).coerceAtLeast(100.0)
  1449→        
  1450→        val providedArea = (1000.0 / finalSpacing) * barArea
  1451→        val utilization = (asReqCrack / providedArea).coerceIn(0.0, 1.2)
  1452→
  1453→        val volWall = (4 * dim * height * wallThickness / 1000.0)
  1454→        val volBase = (dim * dim * baseThickness / 1000.0)
  1455→        val totalVol = volWall + volBase
  1456→        
  1457→        // Accurate Steel Calculation
  1458→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1459→        
  1460→        // Wall Steel (2 nets: inner and outer)
  1461→        val numVerticalBars = floor((4 * dim * 1000.0) / finalSpacing).toInt() + 1
  1462→        val numHorizontalBars = floor((height * 1000.0) / 200.0).toInt() + 1
  1463→        val wallSteelWeight = (numVerticalBars * height + numHorizontalBars * (4 * dim)) * barWeightPerMeter * 2 // 2 nets
  1464→        
  1465→        // Base Steel (Top and Bottom nets)
  1466→        val numBaseBarsX = floor((dim * 1000.0) / 200.0).toInt() + 1
  1467→        val numBaseBarsY = floor((dim * 1000.0) / 200.0).toInt() + 1
  1468→        val baseSteelWeight = (numBaseBarsX * dim + numBaseBarsY * dim) * barWeightPerMeter * 2 // 2 nets
  1469→        
  1470→        val totalSteelWeight = (wallSteelWeight + baseSteelWeight) * 1.15 // 15% laps/anchors/corners
  1471→        val cost = totalVol * settingsManager.concretePrice + (totalSteelWeight/1000.0 * settingsManager.steelPrice)
  1472→        
  1473→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1474→        safetyChecks.add(DesignSafetyCheck("Crack Control (Steel Area)", providedArea, asReqCrack, "mm2", utilization <= 1.0))
  1475→        safetyChecks.add(DesignSafetyCheck("Wall Thickness", wallThickness, 250.0, "mm", wallThickness >= 250.0))
  1476→
  1477→        return TankResult(
  1478→            type = type, length = dim, width = dim, height = height,
  1479→            wallThickness = wallThickness, baseThickness = baseThickness,
  1480→            wallReinforcement = ReinforcementBar(spacing = finalSpacing, diameter = preferredDiameter, description = "Walls (Vertical)"),
  1481→            baseReinforcement = ReinforcementBar(spacing = 200.0, diameter = preferredDiameter, description = "Base Mat"),
  1482→            isSafe = utilization <= 1.0, concreteVolume = totalVol, steelWeight = totalSteelWeight, 
  1483→            cost = cost, code = code,
  1484→            waterPressure = waterPressure, mu = mu, capacity = capacity,
  1485→            safetyChecks = safetyChecks, utilizationRatio = utilization,
  1486→            fcu = fcu, fy = fy
  1487→        )
  1488→        }

  1490→    fun designRetainingWall(
  1491→        height: Double,
  1492→        soilDensity: Double,
  1493→        frictionAngle: Double,
  1494→        surcharge: Double,
  1495→        fcu: Double,
  1496→        fy: Double,
  1497→        preferredDiameter: Int = 16,
  1498→        code: DesignCode = DesignCode.EGYPTIAN
  1499→    ): RetainingWallResult {
  1500→        // 1. Earth Pressure Coefficients
  1501→        val phiRad = frictionAngle * PI / 180.0
  1502→        val ka = (1 - sin(phiRad)) / (1 + sin(phiRad))
  1503→        val kp = 1 / ka
  1504→        
  1505→        // 2. Dimensions (Standard Proportions)
  1506→        val stemT = (height * 1000.0 / 12.0).coerceAtLeast(300.0) // mm
  1507→        val baseW = (height * 0.5).coerceAtLeast(1.5) // m
  1508→        val baseT = stemT // mm
  1509→        val toeW = baseW / 3.0
  1510→        val heelW = baseW - toeW - (stemT/1000.0)
  1511→        
  1512→        // 3. Stability Analysis
  1513→        val pa = 0.5 * ka * soilDensity * height.pow(2)
  1514→        val ps = ka * surcharge * height
  1515→        
  1516→        val wStem = height * (stemT / 1000.0) * 25.0
  1517→        val wBase = baseW * (baseT / 1000.0) * 25.0
  1518→        val wSoil = heelW * height * soilDensity
  1519→        
  1520→        val xStem = toeW + (stemT / 2000.0)
  1521→        val xBase = baseW / 2.0
  1522→        val xSoil = toeW + (stemT / 1000.0) + (heelW / 2.0)
  1523→        
  1524→        val resistingMoment = (wStem * xStem) + (wBase * xBase) + (wSoil * xSoil)
  1525→        val drivingMoment = (pa * height / 3.0) + (ps * height / 2.0)
  1526→        
  1527→        val totalVerticalWeight = wStem + wBase + wSoil
  1528→        val fsOverturning = resistingMoment / drivingMoment.coerceAtLeast(1.0)
  1529→        val fsSliding = (0.5 * totalVerticalWeight) / (pa + ps).coerceAtLeast(1.0)
  1530→        
  1531→        val utilizationRatio = max(1.5 / fsOverturning, 1.5 / fsSliding)
  1532→        val isSafe = fsOverturning >= 1.5 && fsSliding >= 1.5
  1533→        
  1534→        // 4. Design for Stem Moment
  1535→        val muStem = drivingMoment * 1.5 // Factored
  1536→        val d = stemT - 70.0
  1537→        val asReq = calculateAs(muStem, fcu, fy, d, 1000.0, code)
  1538→        
  1539→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1540→        val numBars = ceil(asReq / barArea).toInt().coerceAtLeast(5)
  1541→        val stemReinforcement = ReinforcementBar(numBars, preferredDiameter, description = "Stem (Vertical)")
  1542→        
  1543→        val vol = (height * (stemT / 1000.0)) + (baseW * (baseT / 1000.0))
  1544→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1545→        val steelWeight = (numBars * height + (1000.0/200.0) * baseW) * barWeightPerMeter * 1.15
  1546→        
  1547→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1548→        safetyChecks.add(DesignSafetyCheck("Overturning Stability", fsOverturning, 1.5, "", fsOverturning >= 1.5))
  1549→        safetyChecks.add(DesignSafetyCheck("Sliding Stability", fsSliding, 1.5, "", fsSliding >= 1.5))
  1550→        
  1551→        return RetainingWallResult(
  1552→            height = height,
  1553→            stemThickness = stemT,
  1554→            baseWidth = baseW * 1000.0,
  1555→            stemReinforcement = stemReinforcement,
  1556→            baseReinforcement = ReinforcementBar(spacing = 200.0, diameter = preferredDiameter, description = "Base Bottom"),
  1557→            safetyChecks = safetyChecks,
  1558→            isSafe = isSafe,
  1559→            concreteVolume = vol,
  1560→            steelWeight = steelWeight,
  1561→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1562→            code = code,
  1563→            factorOfSafetyOverturning = fsOverturning,
  1564→            factorOfSafetySliding = fsSliding,
  1565→            utilizationRatio = utilizationRatio,
  1566→            muStem = muStem,
  1567→            pa = pa,
  1568→            ps = ps,
  1569→            ka = ka,
  1570→            soilDensity = soilDensity,
  1571→            fcu = fcu,
  1572→            fy = fy
  1573→        )
  1574→        }

  1576→    fun calculateSeismicLoads(input: SeismicInput): SeismicResult {

  1585→            storyDrift = 0.005 * h/n,
