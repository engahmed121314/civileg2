    71→    onNavigateBack: () -> Unit = {}

    73→    val context = LocalContext.current

    76→    // ── Code / Zone / Soil selectors ────────────────────────────────────────
    77→    var selectedCode by remember { mutableStateOf(SeismicCodeOption.ECP_201) }
    78→    var expandedCode by remember { mutableStateOf(false) }
    79→

    81→    var expandedZone by remember { mutableStateOf(false) }

    84→    var expandedSoil by remember { mutableStateOf(false) }
    85→
    86→    // ── Building parameters ─────────────────────────────────────────────────
    87→    var numFloors by remember { mutableStateOf("5") }
    88→    var avgFloorHeight by remember { mutableStateOf("3.0") }
    89→    var totalWeight by remember { mutableStateOf("5000") }
    90→    var importanceFactor by remember { mutableStateOf("1.0") }
    91→    var reductionFactor by remember { mutableStateOf("5.0") }
    92→

   100→
   101→    // ── Helpers ─────────────────────────────────────────────────────────────
   102→    val getDesigner: () -> SeismicDesign = {
   103→        when (selectedCode) {
   104→            SeismicCodeOption.ECP_201 -> ECPSeismic()
   105→            SeismicCodeOption.ASCE_7 -> ACISeismic()
   106→            SeismicCodeOption.SBC_301 -> SBCSeismic()
   107→        }
   108→    }
   109→
   110→    val totalHeight = (numFloors.toIntOrNull() ?: 5) * (avgFloorHeight.toDoubleOrNull() ?: 3.0)
   111→
   112→    fun calculate() {
   113→        isCalculating = true
   114→        val designer = getDesigner()
   115→        val nFloors = numFloors.toIntOrNull()?.coerceIn(1, 50) ?: 5
   116→        val floorH = avgFloorHeight.toDoubleOrNull()?.coerceIn(2.0, 6.0) ?: 3.0
   117→        val w = totalWeight.toDoubleOrNull()?.coerceAtLeast(1.0) ?: 5000.0
   118→        val I = importanceFactor.toDoubleOrNull()?.coerceIn(0.5, 2.0) ?: 1.0
   119→        val R = reductionFactor.toDoubleOrNull()?.coerceAtLeast(1.0) ?: 5.0
   120→
   121→        // Floor weights distributed equally
   122→        val floorWeights = List(nFloors) { w / nFloors }
   123→        val floorHeights = List(nFloors) { (it + 1) * floorH }
   124→
   125→        val baseShearResult = designer.calculateBaseShear(
   126→            totalWeight = w,
   127→            seismicZone = selectedZone,
   128→            soilType = selectedSoil,
   129→            importanceFactor = I,
   130→            responseModificationFactor = R,
   131→            buildingHeight = totalHeight
   132→        )
   133→
   134→        // Fundamental period (approximate)
   135→        val T = 0.075 * totalHeight.pow(0.75)
   136→
   137→        // Spectrum values (generate curve)
   138→        val spectrumValues = (1..50).map { i ->
   139→            val period = i * 0.06          // 0.06s → 3.0s
   140→            designer.getResponseSpectrum(
   141→                period = period,
   142→                dampingRatio = 0.05,
   143→                soilType = selectedSoil,
   144→                importanceFactor = I
   145→            )
   146→        }
   147→
   148→        val floorForces = designer.distributeSeismicForces(
   149→            baseShear = baseShearResult.baseShear,
   150→            floorWeights = floorWeights,
   151→            floorHeights = floorHeights
   152→        )
   153→
   154→        // Sa at T
   155→        val saAtT = designer.getResponseSpectrum(
   156→            period = T, dampingRatio = 0.05,
   157→            soilType = selectedSoil, importanceFactor = I
   158→        )
   159→
   160→        result = SeismicUiResult(
   161→            baseShearResult = baseShearResult,
   162→            spectrumValues = spectrumValues,
   163→            floorForces = floorForces,
   164→            fundamentalPeriod = T,
   165→            spectralAccel = saAtT.spectralAcceleration,
   166→            code = selectedCode.designCode
   167→        )
   168→        isCalculating = false
   169→    }

   707→// ─── Seismic Spectrum Mini Canvas ────────────────────────────────────────────
   708→@Composable
   709→private fun SeismicSpectrumCanvas(
   710→    spectrumValues: List<SpectrumValue>,
   711→    designPeriod: Double,
   712→    modifier: Modifier = Modifier
   713→) {
   714→    Card(modifier = modifier, shape = RoundedCornerShape(12.dp)) {
   715→        Canvas(modifier = Modifier.fillMaxSize().background(Color.White).padding(20.dp)) {
   716→            if (spectrumValues.isEmpty()) return@Canvas
   717→
   718→            val w = size.width
   719→            val h = size.height
   720→            val pad = 50f
   721→
   722→            val maxT = spectrumValues.maxOf { it.period }
   723→            val maxSa = spectrumValues.maxOf { it.spectralAcceleration }
   724→
   725→            fun toX(t: Double) = pad + (t / maxT * (w - 2 * pad)).toFloat()
   726→            fun toY(sa: Double) = h - pad - (sa / maxSa * (h - 2 * pad)).toFloat()
   727→
   728→            // Grid
   729→            val gridColor = Color.Gray.copy(alpha = 0.15f)
   730→            for (i in 1..4) {
   731→                val x = toX(maxT * i / 4.0)
   732→                drawLine(gridColor, Offset(x, pad), Offset(x, h - pad), 1f)
   733→                val y = toY(maxSa * i / 4.0)
   734→                drawLine(gridColor, Offset(pad, y), Offset(w - pad, y), 1f)
   735→            }
   736→
   737→            // Axes
   738→            drawLine(Color.Gray, Offset(pad, h - pad), Offset(w - pad, h - pad), 2f)
   739→            drawLine(Color.Gray, Offset(pad, h - pad), Offset(pad, pad), 2f)
   740→
   741→            // Spectrum curve
   742→            val path = Path()
   743→            spectrumValues.forEachIndexed { idx, sv ->
   744→                val x = toX(sv.period)
   745→                val y = toY(sv.spectralAcceleration)
   746→                if (idx == 0) path.moveTo(x, y) else path.lineTo(x, y)
   747→            }
   748→            drawPath(path, Color(0xFF1565C0), style = Stroke(width = 3f, cap = StrokeCap.Round, join = StrokeJoin.Round))
   749→
   750→            // Design period marker
   751→            val designX = toX(designPeriod)
   752→            val nearestSa = spectrumValues.minByOrNull { kotlin.math.abs(it.period - designPeriod) }?.spectralAcceleration ?: 0.0
   753→            val designY = toY(nearestSa)
   754→
   755→            val dash = PathEffect.dashPathEffect(floatArrayOf(8f, 8f), 0f)
   756→            drawLine(Color.Red, Offset(designX, h - pad), Offset(designX, designY), 2f, pathEffect = dash)
   757→            drawLine(Color.Red, Offset(pad, designY), Offset(designX, designY), 2f, pathEffect = dash)
   758→            drawCircle(Color.Red, 6f, Offset(designX, designY))
   759→            drawCircle(Color.White, 3f, Offset(designX, designY))
   760→
   761→            // Labels
   762→            drawContext.canvas.nativeCanvas.apply {
   763→                val paint = android.graphics.Paint().apply {
   764→                    textSize = 22f
   765→                    color = android.graphics.Color.GRAY
   766→                    textAlign = android.graphics.Paint.Align.CENTER
   767→                }
   768→                drawText("T (sec)", w / 2, h - 5f, paint)
   769→                save()
   770→                rotate(-90f, 12f, h / 2)
   771→                drawText("Sa (g)", 12f, h / 2, paint)
   772→                restore()
   773→            }
   774→        }
   775→    }
   776→}
