   169→    val uniqueTypes = allDesigns.map { it.type }.distinct().size
   170→
   171→    val recentProjects = allDesigns
   172→        .sortedByDescending { it.createdAt }
   173→        .take(5)
   174→        .map { design ->
   175→            val typeLabel = when (design.type) {
   176→                DesignType.BEAM -> stringResource(R.string.type_beams)
   177→                DesignType.COLUMN -> stringResource(R.string.type_columns)
   178→                DesignType.SLAB -> stringResource(R.string.type_slabs)
   179→                DesignType.FOOTING -> stringResource(R.string.type_footings)
   180→                DesignType.STAIRCASE -> stringResource(R.string.type_stairs)
   181→                DesignType.RETAINING_WALL -> stringResource(R.string.type_retaining_walls)
   182→                DesignType.WATER_TANK -> stringResource(R.string.type_tanks)
   183→                DesignType.SEISMIC -> stringResource(R.string.type_seismic)
   184→                DesignType.STEEL_MEMBER -> stringResource(R.string.type_steel)
   185→                DesignType.PILE -> stringResource(R.string.type_piles)
   186→                DesignType.STEEL_WAREHOUSE -> stringResource(R.string.type_steel_warehouse)
   187→            }
   188→            val icon = when (design.type) {
   189→                DesignType.BEAM -> Icons.Default.AccountBalance
   190→                DesignType.COLUMN -> Icons.Default.ViewColumn
   191→                DesignType.SLAB -> Icons.Default.ViewWeek
   192→                DesignType.FOOTING -> Icons.Default.Layers
   193→                DesignType.STAIRCASE -> Icons.Default.Stairs
   194→                DesignType.RETAINING_WALL -> Icons.Default.SensorDoor
   195→                DesignType.WATER_TANK -> Icons.Default.WaterDrop
   196→                DesignType.SEISMIC -> Icons.Default.Warning
   197→                DesignType.STEEL_MEMBER -> Icons.Default.Build
   198→                else -> Icons.Default.Description
   199→            }
   200→            val dateStr = SimpleDateFormat("dd/MM/yyyy", Locale("ar")).format(design.createdAt)
   201→            RecentProject(
   202→                name = design.name,
   203→                elementType = typeLabel,
   204→                date = dateStr,
   205→                code = design.codeUsed,
   206→                icon = icon
   207→            )
   208→        }
   209→
   210→    Scaffold(
   211→        topBar = {
   212→            TopAppBar(
   213→                title = {
   214→                    Row(verticalAlignment = Alignment.CenterVertically) {
   215→                        Icon(
   216→                            painter = painterResource(id = R.drawable.ic_logo),
   217→                            contentDescription = null,
   218→                            modifier = Modifier.size(32.dp),
   219→                            tint = MaterialTheme.colorScheme.primary
   220→                        )
   221→                        Spacer(modifier = Modifier.width(12.dp))
   222→                        Column {
   223→                            Text(
   224→                                "Civil Engineer Pro",
   225→                                fontWeight = FontWeight.ExtraBold,
   226→                                fontSize = 18.sp
   227→                            )
   228→                            Text(
   229→                                "Professional Suite",
   230→                                style = MaterialTheme.typography.labelSmall,
   231→                                color = MaterialTheme.colorScheme.primary
   232→                            )
   233→                        }
   234→                    }
   235→                },
   236→                actions = {
   237→                    IconButton(onClick = onShowSettings) {
   238→                        Icon(
   239→                            Icons.Default.Settings,
   240→                            contentDescription = "Settings",
   241→                            tint = MaterialTheme.colorScheme.primary
   242→                        )
   243→                    }
   244→                },
   245→                colors = TopAppBarDefaults.topAppBarColors(
   246→                    containerColor = MaterialTheme.colorScheme.surface
   247→                )
   248→            )
   249→        }
   250→    ) { padding ->
   251→        LazyVerticalGrid(
   252→            columns = GridCells.Fixed(2),
   253→            horizontalArrangement = Arrangement.spacedBy(12.dp),
   254→            verticalArrangement = Arrangement.spacedBy(10.dp),
   255→            modifier = Modifier
   256→                .fillMaxSize()
   257→                .padding(padding)
   258→                .background(
   259→                    if (isDark) Color(0xFF0F0F1A) else Color(0xFFF5F6FA)
   260→                )
   261→                .padding(horizontal = 14.dp)
   262→        ) {
   263→            // ══════════════════════════════════════════
   264→            // A. HEADER SECTION
   265→            // ══════════════════════════════════════════
   266→            item(span = { GridItemSpan(2) }) {

   299→            item(span = { GridItemSpan(2) }) {
   300→                SectionHeader(
   301→                    icon = Icons.Default.Build,
   302→                    title = stringResource(R.string.home_quick_tools),
   303→                    subtitle = stringResource(R.string.home_section_tools_subtitle)
