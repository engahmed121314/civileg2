   285→@Composable
   286→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   287→    InteractiveDrawingScreen(
   288→        title = "تحليل الإطار الإنشائي",
   289→        subtitle = "Frame Analysis & BMD"
   290→    ) {
   291→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFF0D1117))) {
   292→            val padding = 60f
   293→            val w = size.width - 2 * padding
   294→            val h = size.height - 2 * padding
   295→            val baseY = size.height - padding
   296→
   297→            // Scale calculations
   298→            val scale = (w / inputs.span).toFloat() * 0.75f
   299→            val eh = (inputs.eaveHeight * scale).toFloat()
   300→            val rh = (inputs.ridgeHeight * scale).toFloat()
   301→            val midX = padding + w / 2
   302→            val colLX = padding + (w - inputs.span * scale) / 2
   303→            val colRX = colLX + (inputs.span * scale).toFloat()
   304→
   305→            // ── Grid ──
   306→            val gridColor = Color(0x1AFFFFFF)
   307→            for (i in 0..20) {
   308→                val x = padding + (i * w / 20)
   309→                drawLine(gridColor, Offset(x, padding), Offset(x, size.height - padding), strokeWidth = 0.5f)
   310→            }
   311→            for (i in 0..12) {
   312→                val y = padding + (i * h / 12)
   313→                drawLine(gridColor, Offset(padding, y), Offset(size.width - padding, y), strokeWidth = 0.5f)
   314→            }
   315→
   316→            // ── Ground Line with hatching ──
   317→            val groundColor = Color(0xFF4A4A4A)
   318→            drawLine(groundColor, Offset(colLX - 30f, baseY), Offset(colRX + 30f, baseY), strokeWidth = 3f)
   319→            for (i in 0..24) {
   320→                val x = (colLX - 30f) + (i * (colRX - colLX + 60f) / 24)
   321→                drawLine(groundColor, Offset(x, baseY), Offset(x - 8f, baseY + 10f), strokeWidth = 1f)
   322→            }
   323→
   324→            // ── Pin Supports ──
   325→            val supportColor = Color(0xFF8A8A8A)
   326→            // Left support
   327→            drawPath(Path().apply {
   328→                moveTo(colLX - 12f, baseY)
   329→                lineTo(colLX + 12f, baseY)
   330→                lineTo(colLX, baseY + 18f)
   331→                close()
   332→            }, supportColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   333→            // Right support
   334→            drawPath(Path().apply {
   335→                moveTo(colRX - 12f, baseY)
   336→                lineTo(colRX + 12f, baseY)
   337→                lineTo(colRX, baseY + 18f)
   338→                close()
   339→            }, supportColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   340→
   341→            // ── Main Frame with 3D effect ──
   342→            val frameColor = Color(0xFFB0BEC5)
   343→            val frameShadow = Color(0xFF546E7A)
   344→            val frameHighlight = Color(0xFFECEFF1)
   345→            val memberWidth = 6f
   346→
   347→            // Columns (draw shadow, main, highlight for 3D look)
   348→            // Left column
   349→            drawLine(frameShadow, Offset(colLX + 2f, baseY), Offset(colLX + 2f, baseY - eh), strokeWidth = memberWidth + 2f)
   350→            drawLine(frameColor, Offset(colLX, baseY), Offset(colLX, baseY - eh), strokeWidth = memberWidth)
   351→            drawLine(frameHighlight, Offset(colLX - 1f, baseY), Offset(colLX - 1f, baseY - eh), strokeWidth = 1f)
   352→
   353→            // Right column
   354→            drawLine(frameShadow, Offset(colRX + 2f, baseY), Offset(colRX + 2f, baseY - eh), strokeWidth = memberWidth + 2f)
   355→            drawLine(frameColor, Offset(colRX, baseY), Offset(colRX, baseY - eh), strokeWidth = memberWidth)
   356→            drawLine(frameHighlight, Offset(colRX - 1f, baseY), Offset(colRX - 1f, baseY - eh), strokeWidth = 1f)
   357→
   358→            // Rafters (left)
   359→            drawLine(frameShadow, Offset(colLX + 2f, baseY - eh + 2f), Offset(midX + 1f, baseY - rh + 1f), strokeWidth = memberWidth + 2f)
   360→            drawLine(frameColor, Offset(colLX, baseY - eh), Offset(midX, baseY - rh), strokeWidth = memberWidth)
   361→            // Rafters (right)
   362→            drawLine(frameShadow, Offset(colRX + 2f, baseY - eh + 2f), Offset(midX + 1f, baseY - rh + 1f), strokeWidth = memberWidth + 2f)
   363→            drawLine(frameColor, Offset(colRX, baseY - eh), Offset(midX, baseY - rh), strokeWidth = memberWidth)
   364→
   365→            // ── Joint Circles ──
   366→            val jointColor = Color(0xFFFF9800)
   367→            listOf(
   368→                Offset(colLX, baseY - eh),
   369→                Offset(colRX, baseY - eh),
   370→                Offset(midX, baseY - rh)
   371→            ).forEach { joint ->
   372→                drawCircle(Color(0x33FF9800), 14f, joint)
   373→                drawCircle(jointColor, 6f, joint)
   374→                drawCircle(Color.White, 3f, joint)
   375→            }
   376→
   377→            // ── Distributed Load Arrows (on rafters) ──
   378→            val loadColor = Color(0xFFEF5350)
   379→            for (i in 0..14) {
   380→                val t = i / 14f
   381→                // Left rafter
   382→                val xL = colLX + t * (midX - colLX)
   383→                val yL = baseY - eh + t * (rh - eh) * (-1f)
   384→                val nxL = -(baseY - eh - (baseY - rh)) // normal direction
   385→                val lenL = (midX - colLX)
   386→                val nxNorm = ((baseY - rh) - (baseY - eh)) / lenL
   387→                val nyNorm = (midX - colLX) / lenL
   388→                val arrowLen = 25f
   389→                drawLine(loadColor, Offset(xL - nxNorm * arrowLen, yL - nyNorm * arrowLen), Offset(xL, yL), strokeWidth = 1.5f)
   390→                // Arrow head
   391→                drawLine(loadColor, Offset(xL - 4f, yL - 5f), Offset(xL, yL), strokeWidth = 1.5f)
   392→                drawLine(loadColor, Offset(xL + 4f, yL - 5f), Offset(xL, yL), strokeWidth = 1.5f)
   393→
   394→                // Right rafter
   395→                val xR = midX + t * (colRX - midX)
   396→                val yR = (baseY - rh) + t * ((baseY - eh) - (baseY - rh))
   397→                drawLine(loadColor, Offset(xR + nxNorm * arrowLen, yR - nyNorm * arrowLen), Offset(xR, yR), strokeWidth = 1.5f)
   398→                drawLine(loadColor, Offset(xR - 4f, yR - 5f), Offset(xR, yR), strokeWidth = 1.5f)
   399→                drawLine(loadColor, Offset(xR + 4f, yR - 5f), Offset(xR, yR), strokeWidth = 1.5f)
   400→            }
   401→
   402→            // Load label
   403→            drawContext.canvas.nativeCanvas.drawText(
   404→                "w = %.1f kN/m".format(result.mainFrame.maxShear / max(inputs.span, 1.0)),
   405→                (colLX + colRX) / 2f - 60f, baseY - rh - 35f,
   406→                android.graphics.Paint().apply { color = loadColor.toArgb(); textSize = 28f; isFakeBoldText = true }
   407→            )
   408→
   409→            // ── Bending Moment Diagram (filled) ──
   410→            val momentColor = Color(0xFF42A5F5)
   411→            val maxMoment = result.mainFrame.maxMoment
   412→            val momentScale = min(eh * 0.35f / max(maxMoment, 1f), 0.5f)
   413→
   414→            // Left rafter moment (parabolic)
   415→            val momentPath = Path().apply {
   416→                moveTo(colLX, baseY - eh)
   417→                for (i in 0..20) {
   418→                    val t = i / 20f
   419→                    val x = colLX + t * (midX - colLX)
   420→                    val y = baseY - eh + t * (rh - eh) * (-1f)
   421→                    val m = 4 * maxMoment * t * (1 - t)
   422→                    val nxN = ((baseY - rh) - (baseY - eh)) / (midX - colLX)
   423→                    val nyN = (midX - colLX) / (midX - colLX)
   424→                    lineTo(x + nxN * m * momentScale, y + nyN * m * momentScale)
   425→                }
   426→                lineTo(midX, baseY - rh)
   427→                close()
   428→            }
   429→            drawPath(momentPath, momentColor.copy(alpha = 0.2f))
   430→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   431→
   432→            // Right rafter moment (mirror)
   433→            val momentPathR = Path().apply {
   434→                moveTo(colRX, baseY - eh)
   435→                for (i in 0..20) {
   436→                    val t = i / 20f
   437→                    val x = colRX + t * (midX - colRX)
   438→                    val y = (baseY - eh) + t * ((baseY - rh) - (baseY - eh))
   439→                    val m = 4 * maxMoment * t * (1 - t)
   440→                    val nxN = ((baseY - rh) - (baseY - eh)) / (midX - colRX)
   441→                    val nyN = (midX - colRX) / (midX - colRX)
   442→                    lineTo(x + nxN * m * momentScale, y + nyN * m * momentScale)
   443→                }
   444→                lineTo(midX, baseY - rh)
   445→                close()
   446→            }
   447→            drawPath(momentPathR, momentColor.copy(alpha = 0.2f))
   448→            drawPath(momentPathR, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   449→
   450→            // Column moment (triangular)
   451→            val colMoment = maxMoment * 0.3f
   452→            drawPath(Path().apply {
   453→                moveTo(colLX, baseY - eh)
   454→                lineTo(colLX - colMoment * momentScale * 0.5f, baseY - eh / 2)
   455→                lineTo(colLX, baseY)
   456→                close()

   824→                    drawLine(Color(0xFF5A5A5A), Offset(sideLeft - 20f, groundY), Offset(sideLeft + planL + 20f, groundY), strokeWidth = 3f)
   825→
   826→                    // Purlin lines (along side view, shown as short vertical ticks on eave)
   827→                    val numPurlins = (inputs.span / 1.5).toInt().coerceIn(2, 8)
   828→                    for (i in 1 until numPurlins) {
   829→                        val x = sideLeft + (i * planL / numPurlins)
   830→                        drawLine(Color(0xFF4FC3F7), Offset(x, eaveY - 8f), Offset(x, eaveY + 8f), strokeWidth = 2f)
   831→                    }
   832→
   833→                    // Bracing in first bay
   834→                    if (numBays >= 1) {
   835→                        val bx1 = sideLeft
   836→                        val bx2 = sideLeft + (inputs.baySpacing * scale).toFloat().coerceAtMost(planL)
   837→                        drawLine(Color(0x66FF5722), Offset(bx1, groundY), Offset(bx2, eaveY), strokeWidth = 1.5f,
   838→                            pathEffect = PathEffect.dashPathEffect(floatArrayOf(10f, 5f)))
   839→                        drawLine(Color(0x66FF5722), Offset(bx2, groundY), Offset(bx1, eaveY), strokeWidth = 1.5f,
   840→                            pathEffect = PathEffect.dashPathEffect(floatArrayOf(10f, 5f)))
   841→                    }
   842→
   843→                    // Girt markers on columns
   844→                    val numGirts = (inputs.eaveHeight / 1.5).toInt().coerceIn(1, 4)
   845→                    for (i in 1..numGirts) {
   846→                        val gy = groundY - (i * eaveH / (numGirts + 1))
   847→                        drawLine(Color(0x40FF9800), Offset(sideLeft, gy), Offset(sideLeft + planL, gy), strokeWidth = 0.5f,
   848→                            pathEffect = PathEffect.dashPathEffect(floatArrayOf(6f, 4f)))
   849→                    }
   850→
   851→                    // Dimensions
   852→                    drawContext.canvas.nativeCanvas.drawText("SIDE ELEVATION", size.width / 2, padding - 10f, textPaint)
   853→
   854→                    // Length dimension
   855→                    val dY = groundY + 35f
   856→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft, groundY + 8f), Offset(sideLeft, dY + 5f), strokeWidth = 1f)
   857→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft + planL, groundY + 8f), Offset(sideLeft + planL, dY + 5f), strokeWidth = 1f)
   858→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft, dY), Offset(sideLeft + planL, dY), strokeWidth = 1.5f)
   859→                    drawContext.canvas.nativeCanvas.drawText("%.1f m".format(inputs.length), sideLeft + planL / 2, dY - 6f, dimPaint)
   860→
   861→                    // Eave height dimension
   862→                    val dX = sideLeft - 35f
   863→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft - 8f, groundY), Offset(dX - 5f, groundY), strokeWidth = 1f)
   864→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft - 8f, eaveY), Offset(dX - 5f, eaveY), strokeWidth = 1f)
   865→                    drawLine(Color(0xFF90A4AE), Offset(dX, groundY), Offset(dX, eaveY), strokeWidth = 1.5f)
   866→                    drawContext.canvas.nativeCanvas.save()
   867→                    drawContext.canvas.nativeCanvas.rotate(-90f, dX - 10f, (groundY + eaveY) / 2f)
   868→                    drawContext.canvas.nativeCanvas.drawText("%.1f m".format(inputs.eaveHeight), dX - 10f, (groundY + eaveY) / 2f + 6f, dimPaint)
   869→                    drawContext.canvas.nativeCanvas.restore()
   870→                }
   871→
   872→                3 -> { // 3D Isometric — Professional
   873→                    val s = scale * 0.65f
   874→                    val depthOff = 40f
   875→                    val numFrames = 5
   876→
   877→                    val startX_f = startX
   878→                    val baseY_f = baseY
   879→

   881→                        val off = frameIdx * depthOff
   882→                        val x0 = startX_f + off
   883→                        val y0 = baseY_f - off
   884→                        val spanW = (inputs.span * s).toFloat()
   885→                        val eaveH = (inputs.eaveHeight * s).toFloat()
   886→                        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   887→                        val midXf = x0 + spanW / 2

   890→                        val color = Color.White.copy(alpha = alpha)
   891→                        val stroke = if (frameIdx == 0) 3f else 1.5f
   892→

   894→                        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   895→                        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   896→                        // Rafters
   897→                        drawLine(color, Offset(x0, y0 - eaveH), Offset(midXf, y0 - ridgeH), strokeWidth = stroke)
   898→                        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(midXf, y0 - ridgeH), strokeWidth = stroke)
   899→
   900→                        // Eave line (longitudinal)
   901→                        if (frameIdx > 0) {
   902→                            val prevOff = (frameIdx - 1) * depthOff
   903→                            val px = startX_f + prevOff
   904→                            val py = baseY_f - prevOff
   905→                            val lineC = Color.White.copy(alpha = 0.1f + 0.08f * frameIdx)
   906→                            drawLine(lineC, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   907→                            drawLine(lineC, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   908→                            drawLine(lineC, Offset(midXf, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   909→                        }
   910→                    }
   911→
   912→                    // Ground plane
   913→                    val groundColor = Color(0x15FFFFFF)
   914→                    val gOff = numFrames * depthOff
   915→                    drawRect(groundColor, Offset(startX_f, baseY_f - gOff - 10f), Size((inputs.span * s).toFloat(), gOff + 10f))
   916→
   917→                    drawContext.canvas.nativeCanvas.drawText("3D ISOMETRIC VIEW", size.width / 2, padding - 10f, textPaint)
   918→                }
   919→            }
   920→        }

  1031→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
  1032→    val tickSize = 15f
  1033→    canvas.drawLine(x1, y1, x2, y2, paint)
  1034→    
  1035→    // Draw Ticks (45 degree lines)
  1036→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
  1037→    val tickAngle = angle + Math.PI / 4
  1038→    
  1039→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
  1040→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
  1041→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
  1042→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
  1043→
  1044→    // Text label
  1045→    val midX = (x1 + x2) / 2
  1046→    val midY = (y1 + y2) / 2
  1047→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
  1048→    
  1049→    if (x1 == x2) { // Vertical
  1050→        canvas.save()
  1051→        canvas.rotate(-90f, midX - 10f, midY)
  1052→        canvas.drawText(text, midX - 10f, midY, textPaint)
  1053→        canvas.restore()
  1054→    } else { // Horizontal
  1055→        canvas.drawText(text, midX, midY - 10f, textPaint)
  1056→    }
  1057→}

  1059→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
  1060→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
  1061→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
  1062→    
  1063→    val s = scale * 0.6f
  1064→    val w = inputs.span.toFloat() * s
  1065→    val l = inputs.length.toFloat() * s
  1066→    val planTop = y - l
  1067→    
  1068→    canvas.drawRect(x, planTop, x + w, y, p)
  1069→    
  1070→    val baySpacing = inputs.baySpacing.toFloat() * s
  1071→    var curY = planTop
  1072→    var count = 0
  1073→    while (curY <= y + 0.1f) {
  1074→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
  1075→        
  1076→        // Column markers
  1077→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
  1078→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
  1079→        
  1080→        if (curY + baySpacing <= y + 0.1f) {
  1081→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
  1082→        }
  1083→        
  1084→        curY += baySpacing
  1085→        count++
  1086→    }
  1087→    
  1088→    // Span Dimension
  1089→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
  1090→    // Total Length Dimension
  1091→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
  1092→}

  1094→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
  1095→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
  1096→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
  1097→    
  1098→    val s = scale * 0.6f
  1099→    val l = inputs.length.toFloat() * s
  1100→    val eh = inputs.eaveHeight.toFloat() * s
  1101→    
  1102→    canvas.drawLine(x, y, x + l, y, p) // Ground
  1103→    val bay = inputs.baySpacing.toFloat() * s
  1104→    var cur = x
  1105→    while (cur <= x + l + 0.1f) {
  1106→        canvas.drawLine(cur, y, cur, y - eh, p)
  1107→        cur += bay
  1108→    }
  1109→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
  1110→    
  1111→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
  1112→}
