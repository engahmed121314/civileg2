   376→@Composable
   377→private fun StairReinforcementDrawing(res: CalculatorEngine.StairResult, modifier: Modifier) {
   378→    Card(
   379→        modifier = modifier.height(440.dp),
   380→        shape = RoundedCornerShape(12.dp),
   381→        elevation = CardDefaults.cardElevation(4.dp)
   382→    ) {
   383→        Canvas(
   384→            modifier = Modifier
   385→                .fillMaxSize()
   386→                .background(Color.White)
   387→                .padding(12.dp)
   388→        ) {
   389→            // ===== DATA EXTRACTION FROM RESULT =====
   390→            val totalSpanMM = res.span.toFloat() * 1000f
   391→            val riserMM = res.riser.toFloat()
   392→            val treadMM = res.tread.toFloat()
   393→            val thicknessMM = res.thickness.toFloat()
   394→            val numSteps = (totalSpanMM / treadMM).toInt().coerceIn(1, 30)
   395→            val totalHeightMM = numSteps * riserMM
   396→            val coverMM = 25f
   397→
   398→            // ===== LAYOUT & SCALING =====
   399→            val border = 2.5f
   400→            val marginL = 55f
   401→            val marginR = 50f
   402→            val marginT = 45f
   403→            val marginB = 55f
   404→            val drawW = size.width - marginL - marginR - border * 2
   405→            val drawH = size.height - marginT - marginB - border * 2
   406→            if (drawW <= 0f || drawH <= 0f) return@Canvas
   407→
   408→            val fitSpan = totalSpanMM
   409→            val fitHeight = totalHeightMM + thicknessMM + 30f
   410→            val scX = drawW / fitSpan
   411→            val scY = drawH / fitHeight
   412→            val sc = minOf(scX.toFloat(), scY) * 0.88f
   413→
   414→            // Origin: bottom-left corner of stair (floor level, start of first riser)
   415→            val ox = marginL + border + (drawW - totalSpanMM * sc) / 2f
   416→            val oy = size.height - marginB - border - (drawH - (totalHeightMM + thicknessMM) * sc) / 2f
   417→            // End point: top of last riser
   418→            val ex = ox + numSteps * treadMM * sc
   419→            val ey = oy - numSteps * riserMM * sc
   420→
   421→            // ===== COLORS =====
   422→            val concreteFill = Color(0xFFE6E6E6)
   423→            val concreteStroke = Color(0xFF2C2C2C)
   424→            val mainSteel = Color(0xFFC62828)
   425→            val distSteel = Color(0xFF0D47A1)
   426→            val dimClr = Color(0xFF4A4A4A)
   427→            val borderClr = Color(0xFF1A1A1A)
   428→            val supportClr = Color(0xFF444444)
   429→
   430→            // ===== GEOMETRY HELPERS =====
   431→            val slopeAngle = atan2(riserMM.toDouble(), treadMM.toDouble())
   432→            val cosA = cos(slopeAngle).toFloat()
   433→            val sinA = sin(slopeAngle).toFloat()
   434→            // Perpendicular offset for slab thickness (below slope line, into the slab)
   435→            val perpX = thicknessMM * sinA * sc
   436→            val perpY = thicknessMM * cosA * sc
   437→            // Cover offset (from slab soffit toward top surface, perpendicular to slope)
   438→            val covX = coverMM * sinA * sc
   439→            val covY = coverMM * cosA * sc
   440→
   441→            // ===== 1. PROFESSIONAL BORDER =====
   442→            drawRect(
   443→                color = borderClr,
   444→                topLeft = Offset(border / 2f, border / 2f),
   445→                size = Size(size.width - border, size.height - border),
   446→                style = Stroke(width = border)
   447→            )
   448→            // Inner border for double-line effect
   449→            drawRect(
   450→                color = borderClr.copy(alpha = 0.3f),
   451→                topLeft = Offset(border + 2f, border + 2f),
   452→                size = Size(size.width - border * 2 - 4f, size.height - border * 2 - 4f),
   453→                style = Stroke(width = 0.5f)
   454→            )
   455→
   456→            // ===== 2. STAIR PROFILE (CONCRETE SECTION) =====
   457→            val soffitStartX = ox + perpX
   458→            val soffitStartY = oy + perpY
   459→            val soffitEndX = ex + perpX
   460→            val soffitEndY = ey + perpY
   461→
   462→            val profilePath = Path()
   463→            // Start at soffit bottom-left
   464→            profilePath.moveTo(soffitStartX, soffitStartY)
   465→            // Up to floor level at stair start
   466→            profilePath.lineTo(ox, oy)
   467→            // Draw each step: riser up, then tread across
   468→            for (i in 0 until numSteps) {
   469→                val stepX = ox + i * treadMM * sc
   470→                val stepTopY = oy - (i + 1) * riserMM * sc
   471→                val nextX = ox + (i + 1) * treadMM * sc
   472→                // Riser (vertical up)
   473→                profilePath.lineTo(stepX, stepTopY)
   474→                // Tread (horizontal across)
   475→                profilePath.lineTo(nextX, stepTopY)
   476→            }
   477→            // Down to soffit at top end
   478→            profilePath.lineTo(soffitEndX, soffitEndY)
   479→            // Soffit straight back to start
   480→            profilePath.lineTo(soffitStartX, soffitStartY)
   481→            profilePath.close()
   482→
   483→            // Fill concrete section
   484→            drawPath(profilePath, concreteFill)
   485→            // Outline concrete section
   486→            drawPath(profilePath, concreteStroke, style = Stroke(width = 2.2f))
   487→
   488→            // ===== 3. STEP DIVISION LINES (risers for clarity) =====
   489→            for (i in 0 until numSteps) {
   490→                val x = ox + i * treadMM * sc
   491→                val yBot = oy - i * riserMM * sc
   492→                val yTop = oy - (i + 1) * riserMM * sc
   493→                drawLine(concreteStroke, Offset(x, yBot), Offset(x, yTop), strokeWidth = 1.2f)
   494→            }
   495→            // Last vertical line at the top riser
   496→            if (numSteps > 0) {
   497→                val lastX = ox + numSteps * treadMM * sc
   498→                val lastYTop = ey
   499→                val lastYBot = oy - (numSteps - 1) * riserMM * sc
   500→                drawLine(concreteStroke, Offset(lastX, lastYBot), Offset(lastX, lastYTop), strokeWidth = 1.2f)
   501→            }
   502→
   503→            // ===== 4. SOFFIT LINE EMPHASIS (waist slab bottom) =====
   504→            drawLine(
   505→                concreteStroke,
   506→                Offset(soffitStartX, soffitStartY),
   507→                Offset(soffitEndX, soffitEndY),
   508→                strokeWidth = 2.5f
   509→            )
   510→
   511→            // ===== 5. SUPPORT CONDITIONS =====
   512→            val supSize = 14f
   513→
   514→            // --- Bottom support: pinned (triangle + hatching) ---
   515→            val bsCx = ox
   516→            val bsCy = oy
   517→            drawLine(supportClr, Offset(bsCx, bsCy), Offset(bsCx - supSize, bsCy + supSize), 2f)
   518→            drawLine(supportClr, Offset(bsCx - supSize, bsCy + supSize), Offset(bsCx + supSize, bsCy + supSize), 2f)
   519→            drawLine(supportClr, Offset(bsCx + supSize, bsCy + supSize), Offset(bsCx, bsCy), 2f)
   520→            // Hatching below bottom support
   521→            for (h in 0..5) {
   522→                val hx = bsCx - supSize + h * (supSize * 2f / 5f)
   523→                drawLine(supportClr, Offset(hx, bsCy + supSize), Offset(hx - 5f, bsCy + supSize + 7f), 1f)
   524→            }
   525→            // Ground line
   526→            drawLine(supportClr, Offset(bsCx - supSize - 5f, bsCy + supSize), Offset(bsCx + supSize + 5f, bsCy + supSize), 1.5f)
   527→
   528→            // --- Top support: roller (triangle + circle + hatching) ---
   529→            val tsCx = ex
   530→            val tsCy = ey
   531→            drawLine(supportClr, Offset(tsCx, tsCy), Offset(tsCx, tsCy - supSize), 2f)
   532→            drawLine(supportClr, Offset(tsCx - supSize, tsCy - supSize), Offset(tsCx + supSize, tsCy - supSize), 2f)
   533→            // Roller circle
   534→            drawCircle(supportClr, radius = 5f, center = Offset(tsCx, tsCy - supSize - 5f), style = Stroke(1.8f))
   535→            drawCircle(supportClr, radius = 1.5f, center = Offset(tsCx, tsCy - supSize - 5f))
   536→            // Hatching above top support
   537→            for (h in 0..5) {
   538→                val hx = tsCx - supSize + h * (supSize * 2f / 5f)
   539→                drawLine(supportClr, Offset(hx, tsCy - supSize), Offset(hx - 5f, tsCy - supSize - 7f), 1f)
   540→            }
   541→            drawLine(supportClr, Offset(tsCx - supSize - 5f, tsCy - supSize), Offset(tsCx + supSize + 5f, tsCy - supSize), 1.5f)
   542→
   543→            // ===== 6. MAIN REINFORCEMENT (diagonal line following slope) =====
   544→            val mbStartX = soffitStartX - covX
   545→            val mbStartY = soffitStartY - covY
   546→            val mbEndX = soffitEndX - covX
   547→            val mbEndY = soffitEndY - covY
   548→
   549→            // Extend bars into supports (anchorage)
   550→            val anchLen = 25f
   551→            val mbFullStartX = mbStartX - anchLen * cosA
   552→            val mbFullStartY = mbStartY + anchLen * sinA
   553→            val mbFullEndX = mbEndX + anchLen * cosA
   554→            val mbFullEndY = mbEndY - anchLen * sinA
   555→
   556→            // Main bar line
   557→            drawLine(mainSteel, Offset(mbFullStartX, mbFullStartY), Offset(mbFullEndX, mbFullEndY), strokeWidth = 3.5f)
   558→
   559→            // 90-degree hooks at both ends
   560→            val hookLen = 14f
   561→            drawLine(mainSteel, Offset(mbFullStartX, mbFullStartY),
   562→                Offset(mbFullStartX - hookLen * sinA, mbFullStartY - hookLen * cosA), 3.5f)
   563→            drawLine(mainSteel, Offset(mbFullEndX, mbFullEndY),
   564→                Offset(mbFullEndX - hookLen * sinA, mbFullEndY - hookLen * cosA), 3.5f)
   565→
   566→            // ===== 7. DISTRIBUTION REINFORCEMENT (small circles along span) =====
   567→            val numDistBars = maxOf(4, numSteps)
   568→            for (i in 1..numDistBars) {
   569→                val t = i.toFloat() / (numDistBars + 1f)
   570→                val cx = mbStartX + t * (mbEndX - mbStartX)
   571→                val cy = mbStartY + t * (mbEndY - mbStartY)
   572→                // Open circle (typical rebar cross-section symbol)
   573→                drawCircle(distSteel, radius = 4.5f, center = Offset(cx, cy), style = Stroke(1.8f))
   574→                drawCircle(Color.White, radius = 2.5f, center = Offset(cx, cy))
   575→            }
   576→
   577→            // ===== 8. TOP COVER INDICATION =====
   578→            val cvFrac = 0.35f
   579→            val cvTopX = ox + cvFrac * (ex - ox)
   580→            val cvTopY = oy + cvFrac * (ey - oy)
   581→            val cvSoffX = cvTopX + perpX
   582→            val cvSoffY = cvTopY + perpY
   583→            val cvBarX = cvSoffX - covX
   584→            val cvBarY = cvSoffY - covY
   585→            val cvOff = 18f
   586→            val cvD1X = cvSoffX + cvOff * cosA
   587→            val cvD1Y = cvSoffY + cvOff * sinA
   588→            val cvD2X = cvBarX + cvOff * cosA
   589→            val cvD2Y = cvBarY + cvOff * sinA
   590→            drawLine(dimClr, Offset(cvD1X, cvD1Y), Offset(cvD2X, cvD2Y), strokeWidth = 0.8f)
   591→            // Tick marks at soffit and bar
   592→            drawLine(dimClr,
   593→                Offset(cvSoffX - 2f * sinA, cvSoffY - 2f * cosA),
   594→                Offset(cvSoffX + 4f * sinA, cvSoffY + 4f * cosA), 0.8f)
   595→            drawLine(dimClr,
   596→                Offset(cvBarX - 2f * sinA, cvBarY - 2f * cosA),
   597→                Offset(cvBarX + 4f * sinA, cvBarY + 4f * cosA), 0.8f)
   598→
   599→            // ===== 9. DIMENSION LINES =====
   600→            val dimPaint = android.graphics.Paint().apply {
   601→                color = android.graphics.Color.parseColor("#4A4A4A")
   602→                textSize = 17f
   603→                textAlign = android.graphics.Paint.Align.CENTER
   604→                isFakeBoldText = true
   605→            }
   606→            val dimSmPaint = android.graphics.Paint().apply {
   607→                color = android.graphics.Color.parseColor("#4A4A4A")
   608→                textSize = 13f
   609→                textAlign = android.graphics.Paint.Align.CENTER
   610→            }
   611→
   612→            fun extLine(x1: Float, y1: Float, x2: Float, y2: Float) {
   613→                drawLine(dimClr, Offset(x1, y1), Offset(x2, y2), strokeWidth = 0.7f)
   614→            }
   615→
   616→            fun drawArrow(x: Float, y: Float, angle: Float, sz: Float = 5f) {
   617→                val a1 = angle + 2.6f
   618→                val a2 = angle - 2.6f
   619→                drawLine(dimClr, Offset(x, y),
   620→                    Offset(x + sz * cos(a1.toDouble()).toFloat(), y + sz * sin(a1.toDouble()).toFloat()), 1f)
   621→                drawLine(dimClr, Offset(x, y),
   622→                    Offset(x + sz * cos(a2.toDouble()).toFloat(), y + sz * sin(a2.toDouble()).toFloat()), 1f)
   623→            }
   624→
   625→            // --- (a) Total Horizontal Span ---
   626→            val spanDimY = oy + 35f
   627→            extLine(ox, oy + 6f, ox, spanDimY + 4f)
   628→            extLine(ex, oy + 6f, ex, spanDimY + 4f)
   629→            drawLine(dimClr, Offset(ox, spanDimY), Offset(ex, spanDimY), strokeWidth = 1f)
   630→            drawArrow(ox, spanDimY, 0f)
   631→            drawArrow(ex, spanDimY, PI.toFloat())
   632→            drawContext.canvas.nativeCanvas.drawText(
   633→                "L = ${res.span} m", (ox + ex) / 2f, spanDimY + 15f, dimPaint
   634→            )
   635→
   636→            // --- (b) Total Vertical Height ---
   637→            val hDimX = ox - 38f
   638→            extLine(ox - 5f, oy, hDimX - 4f, oy)
   639→            extLine(ox - 5f, ey, hDimX - 4f, ey)
   640→            drawLine(dimClr, Offset(hDimX, oy), Offset(hDimX, ey), strokeWidth = 1f)
   641→            drawArrow(hDimX, oy, -(PI.toFloat() / 2f))
   642→            drawArrow(hDimX, ey, PI.toFloat() / 2f)
   643→            drawContext.canvas.nativeCanvas.save()
   644→            drawContext.canvas.nativeCanvas.rotate(-90f, hDimX - 14f, (oy + ey) / 2f)
   645→            drawContext.canvas.nativeCanvas.drawText(
   646→                "H = ${totalHeightMM.toInt()} mm", hDimX - 14f, (oy + ey) / 2f + 5f, dimPaint
   647→            )
   648→            drawContext.canvas.nativeCanvas.restore()
   649→
   650→            // --- (c) Riser Dimension (first step) ---
   651→            if (numSteps >= 1) {
   652→                val rDimX = ox - 14f
   653→                val rTop = oy - riserMM * sc
   654→                extLine(ox - 2f, oy, rDimX - 2f, oy)
   655→                extLine(ox - 2f, rTop, rDimX - 2f, rTop)
   656→                drawLine(dimClr, Offset(rDimX, oy), Offset(rDimX, rTop), strokeWidth = 0.8f)
   657→                drawContext.canvas.nativeCanvas.save()
   658→                drawContext.canvas.nativeCanvas.rotate(-90f, rDimX - 8f, (oy + rTop) / 2f)
   659→                drawContext.canvas.nativeCanvas.drawText(
   660→                    "R=${riserMM.toInt()}", rDimX - 8f, (oy + rTop) / 2f + 4f, dimSmPaint
   661→                )
   662→                drawContext.canvas.nativeCanvas.restore()
   663→            }
   664→
   665→            // --- (d) Tread Dimension (first step) ---
   666→            if (numSteps >= 1) {
   667→                val tDimY = oy + 17f
   668→                val tEndX = ox + treadMM * sc
   669→                extLine(ox, oy + 2f, ox, tDimY + 2f)
   670→                extLine(tEndX, oy + 2f, tEndX, tDimY + 2f)
   671→                drawLine(dimClr, Offset(ox, tDimY), Offset(tEndX, tDimY), strokeWidth = 0.8f)
   672→                drawArrow(ox, tDimY, 0f, 4f)
   673→                drawArrow(tEndX, tDimY, PI.toFloat(), 4f)
   674→                drawContext.canvas.nativeCanvas.drawText(
   675→                    "T=${treadMM.toInt()}", (ox + tEndX) / 2f, tDimY + 11f, dimSmPaint
   676→                )
   677→            }
   678→
   679→            // --- (e) Waist Slab Thickness (rotated along slope) ---
   680→            val wtFrac = 0.65f
   681→            val wtPx = ox + wtFrac * (ex - ox)
   682→            val wtPy = oy + wtFrac * (ey - oy)
   683→            val wtSx = wtPx + perpX
   684→            val wtSy = wtPy + perpY
   685→            val wtOff = 22f
   686→            val wtD1x = wtPx + wtOff * cosA
   687→            val wtD1y = wtPy + wtOff * sinA
   688→            val wtD2x = wtSx + wtOff * cosA
   689→            val wtD2y = wtSy + wtOff * sinA
   690→            drawLine(dimClr, Offset(wtD1x, wtD1y), Offset(wtD2x, wtD2y), strokeWidth = 0.8f)
   691→            drawLine(dimClr,
   692→                Offset(wtPx - 3f * sinA, wtPy - 3f * cosA),
   693→                Offset(wtPx + 5f * sinA, wtPy + 5f * cosA), 0.8f)
   694→            drawLine(dimClr,
   695→                Offset(wtSx - 3f * sinA, wtSy - 3f * cosA),
   696→                Offset(wtSx + 5f * sinA, wtSy + 5f * cosA), 0.8f)
   697→            drawContext.canvas.nativeCanvas.save()
   698→            val wtMidX = (wtD1x + wtD2x) / 2f
   699→            val wtMidY = (wtD1y + wtD2y) / 2f
   700→            val wtAngleDeg = -Math.toDegrees(slopeAngle).toFloat()
   701→            drawContext.canvas.nativeCanvas.rotate(wtAngleDeg, wtMidX, wtMidY)
   702→            drawContext.canvas.nativeCanvas.drawText(
   703→                "ts=${thicknessMM.toInt()}", wtMidX, wtMidY - 6f, dimSmPaint
   704→            )
   705→            drawContext.canvas.nativeCanvas.restore()
   706→
   707→            // ===== 10. REINFORCEMENT LABELS =====
   708→            val labelPaint = android.graphics.Paint().apply {
   709→                textSize = 19f
   710→                textAlign = android.graphics.Paint.Align.LEFT
   711→                isFakeBoldText = true
   712→            }
   713→            val labelSmPaint = android.graphics.Paint().apply {
   714→                textSize = 13f
   715→                textAlign = android.graphics.Paint.Align.LEFT
   716→            }
   717→
   718→            // Main reinforcement label with leader line
   719→            val mlX = (mbStartX + mbEndX) / 2f
   720→            val mlY = (mbStartY + mbEndY) / 2f
   721→            val mlLabelX = mlX + 18f
   722→            val mlLabelY = mlY - 12f
   723→            drawLine(mainSteel, Offset(mlX, mlY), Offset(mlLabelX, mlLabelY), strokeWidth = 1f)
   724→            drawLine(mainSteel, Offset(mlLabelX, mlLabelY), Offset(mlLabelX + 6f, mlLabelY), strokeWidth = 1f)
   725→            labelPaint.color = android.graphics.Color.parseColor("#C62828")
   726→            drawContext.canvas.nativeCanvas.drawText(
   727→                res.reinforcement.barString, mlLabelX + 10f, mlLabelY - 2f, labelPaint
   728→            )
   729→            labelSmPaint.color = android.graphics.Color.parseColor("#C62828")
   730→            drawContext.canvas.nativeCanvas.drawText(
   731→                "(Main Bottom)", mlLabelX + 10f, mlLabelY + 14f, labelSmPaint
   732→            )
   733→
   734→            // Distribution reinforcement label
   735→            val dlY = mlY + 22f
   736→            drawLine(distSteel, Offset(mlX, mlY + 5f), Offset(mlLabelX, dlY), strokeWidth = 1f)
   737→            drawLine(distSteel, Offset(mlLabelX, dlY), Offset(mlLabelX + 6f, dlY), strokeWidth = 1f)
   738→            labelPaint.color = android.graphics.Color.parseColor("#0D47A1")
   739→            drawContext.canvas.nativeCanvas.drawText(
   740→                res.distributionReinforcement.barString, mlLabelX + 10f, dlY - 2f, labelPaint
   741→            )
   742→            labelSmPaint.color = android.graphics.Color.parseColor("#0D47A1")
   743→            drawContext.canvas.nativeCanvas.drawText(
   744→                "(Distribution)", mlLabelX + 10f, dlY + 14f, labelSmPaint
   745→            )
   746→
   747→            // ===== 11. TITLE & SCALE TEXT =====
   748→            val titlePaint = android.graphics.Paint().apply {
   749→                color = android.graphics.Color.parseColor("#1A1A1A")
   750→                textSize = 15f
   751→                textAlign = android.graphics.Paint.Align.RIGHT
   752→                isFakeBoldText = true
   753→            }
   754→            drawContext.canvas.nativeCanvas.drawText(
   755→                "STAIR REINFORCEMENT DETAIL - S1", size.width - 10f, marginT - 5f, titlePaint
   756→            )
   757→
   758→            val stepsPaint = android.graphics.Paint().apply {
   759→                color = android.graphics.Color.parseColor("#333333")
   760→                textSize = 13f
   761→                textAlign = android.graphics.Paint.Align.LEFT
   762→                isFakeBoldText = true
   763→            }
   764→            drawContext.canvas.nativeCanvas.drawText(
   765→                "Steps: $numSteps  |  R x T = ${riserMM.toInt()} x ${treadMM.toInt()} mm",
   766→                marginL + 5f, marginT - 5f, stepsPaint
   767→            )
   768→
   769→            val scalePaint = android.graphics.Paint().apply {
   770→                color = android.graphics.Color.parseColor("#888888")
   771→                textSize = 12f
   772→                textAlign = android.graphics.Paint.Align.CENTER
   773→            }
   774→            drawContext.canvas.nativeCanvas.drawText(
   775→                "Scale: Not to Scale", size.width / 2f, size.height - 3f, scalePaint
   776→            )
   777→
   778→            // Safety status indicator
   779→            val safePaint = android.graphics.Paint().apply {
   780→                textSize = 12f
   781→                textAlign = android.graphics.Paint.Align.RIGHT
   782→                isFakeBoldText = true
   783→            }
   784→            if (res.isSafe) {
   785→                safePaint.color = android.graphics.Color.parseColor("#2E7D32")
   786→                drawContext.canvas.nativeCanvas.drawText(
   787→                    "SAFE", size.width - 10f, size.height - 3f, safePaint
   788→                )
   789→            } else {
   790→                safePaint.color = android.graphics.Color.parseColor("#C62828")
   791→                drawContext.canvas.nativeCanvas.drawText(
   792→                    "UNSAFE", size.width - 10f, size.height - 3f, safePaint
   793→                )
   794→            }
   795→        }
   796→    }
