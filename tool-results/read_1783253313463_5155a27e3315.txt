   278→@Composable
   279→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   280→    Card(
   281→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   282→        border = BorderStroke(1.dp, Color.LightGray)
   283→    ) {
   284→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   285→            val padding = 40.dp.toPx()
   286→            val w = size.width - 2 * padding
   287→            val h = size.height - 2 * padding
   288→            val baseY = size.height - padding
   289→            
   290→            // Scale calculations
   291→            val scale = (w / inputs.span).toFloat()
   292→            val eh = (inputs.eaveHeight * scale).toFloat()
   293→            val rh = (inputs.ridgeHeight * scale).toFloat()
   294→            val midX = padding + w / 2
   295→            
   296→            // Draw Main Frame Outline
   297→            val framePath = Path().apply {
   298→                moveTo(padding, baseY)
   299→                lineTo(padding, baseY - eh)
   300→                lineTo(midX, baseY - rh)
   301→                lineTo(padding + w, baseY - eh)
   302→                lineTo(padding + w, baseY)
   303→            }
   304→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   305→            
   306→            // Draw Distributed Load Arrows
   307→            val loadColor = Color(0xFFE53935)
   308→            for (i in 0..12) {
   309→                val x = padding + (i * w / 12)
   310→                val y = if (x <= midX) {
   311→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   312→                } else {
   313→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   314→                }
   315→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   316→                // Arrow heads
   317→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   318→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   319→            }
   320→            
   321→            // Draw Moment Diagram (Simplified)
   322→            val momentColor = Color(0xFF1E88E5)
   323→            val momentPath = Path().apply {
   324→                moveTo(padding, baseY - eh)
   325→                // Simulated parabolic curve for moment
   326→                cubicTo(
   327→                    padding + w * 0.25f, baseY - eh + 50f,
   328→                    padding + w * 0.75f, baseY - eh + 50f,
   329→                    padding + w, baseY - eh
   330→                )
   331→            }
   332→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   333→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   334→            
   335→            // Labels
   336→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   337→        }
   338→    }
   339→}

   369→    var viewMode by remember { mutableStateOf(0) } // 0: Front, 1: Plan, 2: Side, 3: 3D
   370→    val views = listOf("الواجهة الأمامية", "المسقط الأفقي", "الواجهة الجانبية", "رسم ثلاثي الأبعاد")

   374→            views.forEachIndexed { index, title ->

   386→                    (size.width - 2 * padding) / max(inputs.span, inputs.length),
   387→                    (size.height - 2 * padding) / max(inputs.ridgeHeight, inputs.span)
   388→                )
   389→                val startX = padding
   390→                val baseY = size.height - padding
   391→
   392→                when (viewMode) {
   393→                    0 -> { // Front Elevation
   394→                        drawFrontElevation(inputs, scale, startX, baseY)
   395→                    }
   396→                    1 -> { // Plan View
   397→                        drawPlanView(inputs, scale, startX, baseY)
   398→                    }
   399→                    2 -> { // Side View
   400→                        drawSideElevation(inputs, scale, startX, baseY)
   401→                    }
   402→                    3 -> { // 3D Perspective
   403→                        draw3DView(inputs, scale, startX, baseY)
   404→                    }
   405→                }
   406→            }
   407→        }
   408→    }
   409→}

   411→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawFrontElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   412→    val colLX = startX
   413→    val colRX = startX + (inputs.span * scale).toFloat()
   414→    val eaveY = baseY - (inputs.eaveHeight * scale).toFloat()
   415→    val ridgeY = baseY - (inputs.ridgeHeight * scale).toFloat()
   416→    val midX = startX + (inputs.span * scale / 2).toFloat()
   417→
   418→    // Columns
   419→    drawLine(Color.Black, Offset(colLX, baseY), Offset(colLX, eaveY), strokeWidth = 8f)
   420→    drawLine(Color.Black, Offset(colRX, baseY), Offset(colRX, eaveY), strokeWidth = 8f)
   421→    // Rafters
   422→    drawLine(Color.Black, Offset(colLX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   423→    drawLine(Color.Black, Offset(colRX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   424→    
   425→    // Foundations
   426→    drawRect(Color.Gray, Offset(colLX - 15f, baseY), Size(30f, 15f))
   427→    drawRect(Color.Gray, Offset(colRX - 15f, baseY), Size(30f, 15f))
   428→}
   429→
   430→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   431→    val planW = (inputs.span * scale).toFloat()
   432→    val planL = (inputs.length * scale).toFloat()
   433→    val planTop = baseY - planL
   434→    
   435→    // Boundary
   436→    drawRect(Color.Black, Offset(startX, planTop), Size(planW, planL), style = androidx.compose.ui.graphics.drawscope.Stroke(4f))
   437→    
   438→    // Grid Lines (Bays)
   439→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   440→    for (i in 0..numBays) {
   441→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()
   442→        drawLine(Color.LightGray, Offset(startX, y), Offset(startX + planW, y), strokeWidth = 1f)
   443→        // Columns markers
   444→        drawCircle(Color.Black, 4f, Offset(startX, y))
   445→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))
   446→    }
   447→}
   448→
   449→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSideElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   450→    val planL = (inputs.length * scale).toFloat()
   451→    val eaveH = (inputs.eaveHeight * scale).toFloat()
   452→    val eaveY = baseY - eaveH
   453→    
   454→    // Ground
   455→    drawLine(Color.Gray, Offset(startX, baseY), Offset(startX + planL, baseY), strokeWidth = 2f)
   456→    
   457→    // Columns
   458→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   459→    for (i in 0..numBays) {
   460→        val x = startX + (i * inputs.baySpacing * scale).toFloat()
   461→        drawLine(Color.Black, Offset(x, baseY), Offset(x, eaveY), strokeWidth = 6f)
   462→    }
   463→    // Eave Line
   464→    drawLine(Color.Black, Offset(startX, eaveY), Offset(startX + planL, eaveY), strokeWidth = 6f)
   465→}

   467→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   468→    val s = scale.toFloat() * 0.65f
   469→    val depthOff = 35f
   470→    
   471→    // Draw from back to front for better perspective look
   472→    for (i in 4 downTo 0) {
   473→        val off = i * depthOff
   474→        val x0 = startX + off
   475→        val y0 = baseY - off
   476→        val spanW = (inputs.span * s).toFloat()
   477→        val eaveH = (inputs.eaveHeight * s).toFloat()
   478→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   479→        
   480→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   481→        val stroke = if (i == 0) 3f else 1.5f
   482→        
   483→        // Sections of the frame
   484→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   485→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   486→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   487→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   488→        
   489→        // Purlins/Girts connections (longitudinal)
   490→        if (i > 0) {
   491→            val px = x0 - depthOff
   492→            val py = y0 + depthOff
   493→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   494→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   495→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   496→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   497→        }
   498→    }
   499→}

   533→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   534→    val p = Paint().apply { 
   535→        color = android.graphics.Color.BLACK
   536→        strokeWidth = 6f
   537→        isAntiAlias = true
   538→        style = Paint.Style.STROKE
   539→    }
   540→    val dimPaint = Paint().apply {
   541→        color = android.graphics.Color.DKGRAY
   542→        strokeWidth = 2f
   543→        textSize = 24f
   544→        isAntiAlias = true
   545→    }
   546→    val textP = Paint().apply {
   547→        color = android.graphics.Color.BLACK
   548→        textSize = 22f
   549→        isFakeBoldText = true
   550→        textAlign = Paint.Align.CENTER
   551→    }
   552→    
   553→    val s = scale * 0.7f
   554→    val w = inputs.span.toFloat() * s
   555→    val eh = inputs.eaveHeight.toFloat() * s
   556→    val rh = inputs.ridgeHeight.toFloat() * s
   557→    val midX = x + w/2
   558→    
   559→    // Draw Main Frame (Columns & Rafters)
   560→    canvas.drawLine(x, y, x, y - eh, p)
   561→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   562→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   563→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   564→    
   565→    // Foundations
   566→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   567→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   568→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   569→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   570→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   571→    p.strokeWidth = 6f
   572→    
   573→    // --- Dimension Lines ---
   574→    // Span Dimension
   575→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   576→    
   577→    // Eave Height Dimension
   578→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   579→    
   580→    // Ridge Height Dimension
   581→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   582→
   583→    // --- Section Labels ---
   584→    result?.let { res ->
   585→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   586→        
   587→        // Column Label
   588→        canvas.save()
   589→        canvas.rotate(-90f, x + 25f, y - eh/2)
   590→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   591→        canvas.restore()
   592→        
   593→        // Rafter Label
   594→        canvas.save()
   595→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   596→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   597→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   598→        canvas.restore()
   599→    }
   600→
   601→    // Connection Markers (Circles at joints)
   602→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   603→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   604→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   605→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   606→}

   608→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   609→    val tickSize = 15f
   610→    canvas.drawLine(x1, y1, x2, y2, paint)
   611→    
   612→    // Draw Ticks (45 degree lines)
   613→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   614→    val tickAngle = angle + Math.PI / 4
   615→    
   616→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   617→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   618→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   619→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   620→
   621→    // Text label
   622→    val midX = (x1 + x2) / 2
   623→    val midY = (y1 + y2) / 2
   624→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   625→    
   626→    if (x1 == x2) { // Vertical
   627→        canvas.save()
   628→        canvas.rotate(-90f, midX - 10f, midY)
   629→        canvas.drawText(text, midX - 10f, midY, textPaint)
   630→        canvas.restore()
   631→    } else { // Horizontal
   632→        canvas.drawText(text, midX, midY - 10f, textPaint)
   633→    }
   634→}

   636→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   637→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   638→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   639→    
   640→    val s = scale * 0.6f
   641→    val w = inputs.span.toFloat() * s
   642→    val l = inputs.length.toFloat() * s
   643→    val planTop = y - l
   644→    
   645→    canvas.drawRect(x, planTop, x + w, y, p)
   646→    
   647→    val baySpacing = inputs.baySpacing.toFloat() * s
   648→    var curY = planTop
   649→    var count = 0
   650→    while (curY <= y + 0.1f) {
   651→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   652→        
   653→        // Column markers
   654→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   655→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   656→        
   657→        if (curY + baySpacing <= y + 0.1f) {
   658→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   659→        }
   660→        
   661→        curY += baySpacing
   662→        count++
   663→    }
   664→    
   665→    // Span Dimension
   666→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   667→    // Total Length Dimension
   668→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   669→}

   671→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   672→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   673→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   674→    
   675→    val s = scale * 0.6f
   676→    val l = inputs.length.toFloat() * s
   677→    val eh = inputs.eaveHeight.toFloat() * s
   678→    
   679→    canvas.drawLine(x, y, x + l, y, p) // Ground
   680→    val bay = inputs.baySpacing.toFloat() * s
   681→    var cur = x
   682→    while (cur <= x + l + 0.1f) {
   683→        canvas.drawLine(cur, y, cur, y - eh, p)
   684→        cur += bay
   685→    }
   686→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   687→    
   688→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   689→}

  1314→@Composable
  1315→fun SteelSectionDrawing(section: SteelSectionType) {
  1316→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1317→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1318→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1319→                val w = size.width
  1320→                val h = size.height
  1321→                
  1322→                when (section) {
  1323→                    is SteelSectionType.ISection -> {
  1324→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1325→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1326→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1327→                        val offsetBF = (w - bf) / 2
  1328→                        
  1329→                        // Top Flange
  1330→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1331→                        // Bottom Flange
  1332→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1333→                        // Web
  1334→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1335→                        
  1336→                        // Stress Distribution (Schematic)
  1337→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1338→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1339→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1340→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1341→                    }
  1342→                    is SteelSectionType.LSection -> {
  1343→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
  1344→                        drawRect(Color.DarkGray, Offset(0f, 0f), Size(t, h))
  1345→                        drawRect(Color.DarkGray, Offset(t, h - t), Size(w - t, t))
  1346→                    }
  1347→                    is SteelSectionType.RHS -> {
  1348→                        val t = (section.thickness / section.height * h).toFloat().coerceIn(5f, 20f)
  1349→                        drawRect(Color.DarkGray, Offset(0f, 0f), size, style = androidx.compose.ui.graphics.drawscope.Stroke(t))
  1350→                    }
  1351→                    else -> {
  1352→                        drawRect(Color.DarkGray, Offset(0f, 0f), size)
  1353→                    }
  1354→                }
  1355→            }
  1356→            Spacer(Modifier.height(12.dp))
  1357→            Text("توزيع الإجهادات المرن", style = MaterialTheme.typography.labelSmall, color = Color.Gray)
  1358→        }
  1359→    }
  1360→}
