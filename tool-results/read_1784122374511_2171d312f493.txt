    90→    var numFloors by remember { mutableStateOf("5") }

    92→    var totalWeight by remember { mutableStateOf("5000") }

   104→    // ── Helpers ─────────────────────────────────────────────────────────────
   105→    val getDesigner: () -> SeismicDesign = {
   106→        when (selectedCode) {
   107→            SeismicCodeOption.ECP_201 -> ECPSeismic()
   108→            SeismicCodeOption.ASCE_7 -> ACISeismic()
   109→            SeismicCodeOption.SBC_301 -> SBCSeismic()
   110→        }
   111→    }
   112→
   113→    val totalHeight = (numFloors.toIntOrNull() ?: 5) * (avgFloorHeight.toDoubleOrNull() ?: 3.0)
   114→
   115→    fun calculate() {
   116→        isCalculating = true
   117→        val designer = getDesigner()
   118→        val nFloors = numFloors.toIntOrNull()?.coerceIn(1, 50) ?: 5
   119→        val floorH = avgFloorHeight.toDoubleOrNull()?.coerceIn(2.0, 6.0) ?: 3.0
   120→        val w = totalWeight.toDoubleOrNull()?.coerceAtLeast(1.0) ?: 5000.0
   121→        val I = importanceFactor.toDoubleOrNull()?.coerceIn(0.5, 2.0) ?: 1.0
   122→        val R = reductionFactor.toDoubleOrNull()?.coerceAtLeast(1.0) ?: 5.0
   123→
   124→        // Floor weights distributed equally
   125→        val floorWeights = List(nFloors) { w / nFloors }
   126→        val floorHeights = List(nFloors) { (it + 1) * floorH }
   127→
   128→        val baseShearResult = designer.calculateBaseShear(
   129→            totalWeight = w,
   130→            seismicZone = selectedZone,
   131→            soilType = selectedSoil,
   132→            importanceFactor = I,
   133→            responseModificationFactor = R,
   134→            buildingHeight = totalHeight
   135→        )
   136→
   137→        // Fundamental period (approximate)
   138→        val T = 0.075 * totalHeight.pow(0.75)
   139→
   140→        // Spectrum values (generate curve)
   141→        val spectrumValues = (1..50).map { i ->
   142→            val period = i * 0.06          // 0.06s → 3.0s
   143→            designer.getResponseSpectrum(
   144→                period = period,
   145→                dampingRatio = 0.05,
   146→                soilType = selectedSoil,
   147→                importanceFactor = I
   148→            )
   149→        }
   150→
   151→        val floorForces = designer.distributeSeismicForces(
   152→            baseShear = baseShearResult.baseShear,
   153→            floorWeights = floorWeights,
   154→            floorHeights = floorHeights
   155→        )
   156→
   157→        // Sa at T
   158→        val saAtT = designer.getResponseSpectrum(
   159→            period = T, dampingRatio = 0.05,
   160→            soilType = selectedSoil, importanceFactor = I
   161→        )
   162→
   163→        result = SeismicUiResult(
   164→            baseShearResult = baseShearResult,
   165→            spectrumValues = spectrumValues,
   166→            floorForces = floorForces,
   167→            fundamentalPeriod = T,
   168→            spectralAccel = saAtT.spectralAcceleration,
   169→            code = selectedCode.designCode
   170→        )
   171→        isCalculating = false
   172→    }

   710→// ─── Seismic Spectrum Mini Canvas ────────────────────────────────────────────
   711→@Composable
   712→private fun SeismicSpectrumCanvas(
   713→    spectrumValues: List<SpectrumValue>,
   714→    designPeriod: Double,
   715→    modifier: Modifier = Modifier
   716→) {
   717→    Card(modifier = modifier, shape = RoundedCornerShape(12.dp)) {
   718→        Canvas(modifier = Modifier.fillMaxSize().background(Color.White).padding(20.dp)) {
   719→            if (spectrumValues.isEmpty()) return@Canvas
   720→
   721→            val w = size.width
   722→            val h = size.height
   723→            val pad = 50f
   724→
   725→            val maxT = spectrumValues.maxOf { it.period }
   726→            val maxSa = spectrumValues.maxOf { it.spectralAcceleration }
   727→
   728→            fun toX(t: Double) = pad + (t / maxT * (w - 2 * pad)).toFloat()
   729→            fun toY(sa: Double) = h - pad - (sa / maxSa * (h - 2 * pad)).toFloat()
   730→
   731→            // Grid
   732→            val gridColor = Color.Gray.copy(alpha = 0.15f)
   733→            for (i in 1..4) {
   734→                val x = toX(maxT * i / 4.0)
   735→                drawLine(gridColor, Offset(x, pad), Offset(x, h - pad), 1f)
   736→                val y = toY(maxSa * i / 4.0)
   737→                drawLine(gridColor, Offset(pad, y), Offset(w - pad, y), 1f)
   738→            }
   739→
   740→            // Axes
   741→            drawLine(Color.Gray, Offset(pad, h - pad), Offset(w - pad, h - pad), 2f)
   742→            drawLine(Color.Gray, Offset(pad, h - pad), Offset(pad, pad), 2f)
   743→
   744→            // Spectrum curve
   745→            val path = Path()
   746→            spectrumValues.forEachIndexed { idx, sv ->
   747→                val x = toX(sv.period)
   748→                val y = toY(sv.spectralAcceleration)
   749→                if (idx == 0) path.moveTo(x, y) else path.lineTo(x, y)
   750→            }
   751→            drawPath(path, Color(0xFF1565C0), style = Stroke(width = 3f, cap = StrokeCap.Round, join = StrokeJoin.Round))
   752→
   753→            // Design period marker
   754→            val designX = toX(designPeriod)
   755→            val nearestSa = spectrumValues.minByOrNull { kotlin.math.abs(it.period - designPeriod) }?.spectralAcceleration ?: 0.0
   756→            val designY = toY(nearestSa)
   757→
   758→            val dash = PathEffect.dashPathEffect(floatArrayOf(8f, 8f), 0f)
   759→            drawLine(Color.Red, Offset(designX, h - pad), Offset(designX, designY), 2f, pathEffect = dash)
   760→            drawLine(Color.Red, Offset(pad, designY), Offset(designX, designY), 2f, pathEffect = dash)
   761→            drawCircle(Color.Red, 6f, Offset(designX, designY))
   762→            drawCircle(Color.White, 3f, Offset(designX, designY))
   763→
   764→            // Labels
   765→            drawContext.canvas.nativeCanvas.apply {
   766→                val paint = android.graphics.Paint().apply {
   767→                    textSize = 22f
   768→                    color = android.graphics.Color.GRAY
   769→                    textAlign = android.graphics.Paint.Align.CENTER
   770→                }
   771→                drawText("T (sec)", w / 2, h - 5f, paint)
   772→                save()
   773→                rotate(-90f, 12f, h / 2)
   774→                drawText("Sa (g)", 12f, h / 2, paint)
   775→                restore()
   776→            }
   777→        }
   778→    }
   779→}
