    27→data class InteractionDiagramPoint(

    29→    val phiMn: Double,           // kN.m - factored moment capacity (φ × Mn)

    69→    /**
    70→     * تصميم متقدم للأعمدة بجميع أنواعها حسب الكود المصري ECP 203
    71→     */
    72→    fun designAdvancedColumn(
    73→        columnType: ColumnType,
    74→        fcu: Double,
    75→        fy: Double,
    76→        axialLoad: Double,
    77→        momentX: Double,
    78→        momentY: Double,
    79→        unsupportedLength: Double,  // m
    80→        endConditions: ColumnEndConditions,
    81→        connectedSlab: ConnectedSlabType,
    82→        hasCap: Boolean,
    83→        inventory: RebarInventory?,
    84→        loadCombination: LoadCombination
    85→    ): AdvancedColumnResult {
    86→        val warnings = mutableListOf<String>()
    87→        val codeNotes = mutableListOf<String>()
    88→        
    89→        // المساحة الكلية
    90→        val Ag = columnType.getGrossArea()
    91→        
    92→        // طول العمود الفعال
    93→        val k = max(endConditions.topCondition.effectiveLengthFactor,
    94→                   endConditions.bottomCondition.effectiveLengthFactor)
    95→        val effectiveLength = k * unsupportedLength * 1000  // mm
    96→        
    97→        // نصف القطر الدوراني (Radius of Gyration)
    98→        val radiusOfGyration = when (columnType) {
    99→            is ColumnType.Rectangular -> min(columnType.width, columnType.depth) / sqrt(12.0)
   100→            is ColumnType.Circular -> columnType.diameter / 4
   101→            is ColumnType.LShaped -> columnType.thickness / sqrt(12.0)
   102→            else -> sqrt(Ag / 12.0)
   103→        }
   104→        
   105→        val slendernessRatio = effectiveLength / radiusOfGyration
   106→        val limit = when(columnType) {
   107→            is ColumnType.Circular -> 12.0
   108→            else -> 15.0
   109→        }
   110→        
   111→        val isSlender = slendernessRatio > limit
   112→        if (isSlender) {
   113→            warnings.add("⚠️ Slender column! Second-order effects (P-Delta) must be considered")
   114→            codeNotes.add("ECP 203-4.2.7: Slenderness ratio λ=$slendernessRatio > $limit")
   115→        }
   116→        
   117→        // حساب القدرة المحورية القصوى
   118→        val axialCapacity = calculateAxialCapacity(fcu, fy, columnType, Ag)
   119→        
   120→        // حساب التسليح
   121→        val reinforcementResult = when(columnType) {
   122→            is ColumnType.Rectangular -> baseDesign.calculateReinforcement(
   123→                fcu, fy, columnType.width, columnType.depth, axialLoad, momentX, momentY, loadCombination
   124→            )
   125→            is ColumnType.Circular -> {
   126→                // تقريب دائري لمربع مكافئ
   127→                val side = sqrt(Ag)
   128→                baseDesign.calculateReinforcement(fcu, fy, side, side, axialLoad, momentX, momentY, loadCombination)
   129→            }
   130→            else -> {
   131→                val side = sqrt(Ag)
   132→                baseDesign.calculateReinforcement(fcu, fy, side, side, axialLoad, momentX, momentY, loadCombination)
   133→            }
   134→        }
   135→        
   136→        // فحص القص الاختراقي (Punching Shear Check) - للبلاطات المسطحة فقط
   137→        val punchingResult = if (connectedSlab == ConnectedSlabType.FLAT) {
   138→            checkPunching(columnType, axialLoad, fcu, hasCap, footingDepth = 200.0)
   139→        } else null
   140→
   141→        if (punchingResult?.isSafe == false) {
   142→            warnings.add("❌ Punching shear failure! Increase section or add column head/cap.")
   143→        }
   144→
   145→        // تحليل المخزون وحساب الهالك
   146→        val inventoryAnalysis = inventory?.let { inv ->
   147→            analyzeInventory(reinforcementResult, unsupportedLength, inv)
   148→        }
   149→        
   150→        val biaxialCheck = if (abs(momentX) > 0.1 && abs(momentY) > 0.1) {
   151→            checkBiaxialLoading(columnType, axialLoad, momentX, momentY, axialCapacity)
   152→        } else null
   153→        
   154→        codeNotes.add("ECP 203-2020: Section 4.2 Design of Compression Members")
   155→        codeNotes.add("Slab Connection: ${connectedSlab.displayName}")
   156→        if (hasCap) codeNotes.add("Column Cap provided - improved punching resistance")
   157→        
   158→        val steelWeight = (inventoryAnalysis?.totalWeight ?: (Ag * 0.01 * 7850 / 1e6)) * unsupportedLength
   159→        
   160→        // حساب قدرة العزم الحقيقية حسب ECP 203 البند 4-2-2
   161→        val h_eff = when(columnType) {
   162→            is ColumnType.Rectangular -> columnType.depth
   163→            is ColumnType.Circular -> columnType.diameter
   164→            else -> sqrt(Ag)
   165→        } - 50.0 // d = h - cover(40) - ties(~10)
   166→        val fs = fy / 1.15 // fy / γs
   167→        // K-method لحساب قدرة العزم
   168→        val As = reinforcementResult.astProvided
   169→        val b = when(columnType) {
   170→            is ColumnType.Rectangular -> columnType.width
   171→            is ColumnType.Circular -> columnType.diameter
   172→            else -> sqrt(Ag)
   173→        }
   174→        val Mu_capacity = fs * As * 0.8 * h_eff / 1e6 // kN.m (تقريبي: R=0.8d)
   175→        // فحص K_bal
   176→        val K_check = Mu_capacity * 1e6 / (fcu * b * h_eff * h_eff)
   177→        val momentCapX = if (K_check <= 0.186) Mu_capacity else {
   178→            // إذا K > K_bal: نستخدم z = d*(0.5+√(0.25-K/1.25))
   179→            val z = h_eff * (0.5 + sqrt(max(0.001, 0.25 - 0.186 / 1.25)))
   180→            As * fs * z / 1e6
   181→        }
   182→
   183→        return AdvancedColumnResult(
   184→            columnType = columnType,
   185→            axialCapacity = axialCapacity,
   186→            momentCapacityX = momentCapX,
   187→            momentCapacityY = momentCapX,
   188→            slendernessRatio = slendernessRatio,
   189→            isSlender = isSlender,
   190→            effectiveLength = effectiveLength,
   191→            reinforcementResult = reinforcementResult,
   192→            inventoryAnalysis = inventoryAnalysis,
   193→            biaxialCheck = biaxialCheck,
   194→            punchingCheck = punchingResult,
   195→            warnings = warnings,
   196→            codeNotes = codeNotes,
   197→            steelWeightPerMeter = steelWeight / unsupportedLength,
   198→            concreteVolumePerMeter = Ag / 1e6
   199→        )
   200→    }

   202→    private fun checkPunching(type: ColumnType, pu: Double, fcu: Double, hasCap: Boolean, footingDepth: Double = 200.0): PunchingCheckResult {
   203→        // ECP 203 البند 4-3-2: فحص قص الاختراق
   204→        val d = footingDepth - 20.0 - 10.0  // d = سمك البلاطة - cover - bar/2
   205→        val perimeter = when (type) {
   206→            is ColumnType.Rectangular -> 2.0 * (type.width + type.depth) + 4.0 * d
   207→            is ColumnType.Circular -> PI * (type.diameter + d)
   208→            else -> 4.0 * (sqrt(type.getGrossArea()) + d)
   209→        }
   210→        
   211→        // ضغط القص المطبق = Vu / (bo × d)
   212→        val punchingStress = (pu * 0.90 * 1000.0) / (perimeter * d)
   213→        // قدرة قص الاختراق: qp = 0.316 × √(fcu/γc) (ECP 203)
   214→        val capacity = (if (hasCap) 1.5 else 1.0) * 0.316 * sqrt(fcu / 1.5)
   215→        
   216→        return PunchingCheckResult(
   217→            appliedShear = pu,
   218→            capacity = capacity * perimeter * d / 1000.0,  // kN
   219→            isSafe = punchingStress <= capacity,
   220→            hasCap = hasCap,
   221→            criticalPerimeter = perimeter
   222→        )
   223→    }

   226→    // Pu = φ × α × [0.67×fcu/γc × (Ag-Ast) + fy/γs × Ast]
   227→    private fun calculateAxialCapacity(fcu: Double, fy: Double, columnType: ColumnType, Ag: Double, Ast: Double = 0.0): Double {
   228→        val GAMMA_C = 1.5
   229→        val GAMMA_S = 1.15
   230→        val ALPHA = 0.8
   231→        val PHI = 0.65
   232→        val ast = if (Ast > 0) Ast.coerceAtMost(Ag * 0.08) else 0.008 * Ag
   233→        val concreteStress = 0.67 * fcu / GAMMA_C
   234→        val steelStress = fy / GAMMA_S
   235→        val nominalCapacity = ALPHA * (concreteStress * (Ag - ast) + steelStress * ast)
   236→        return PHI * nominalCapacity / 1000.0 // kN
   237→    }

   244→    // فحص الانحناء الثنائي بطريقة Bresler التقريبية (ECP 203-4.2.6)

   247→    private fun checkBiaxialLoading(type: ColumnType, Pu: Double, Mx: Double, My: Double, Pn: Double, fy: Double = 360.0): BiaxialCheckResult {
   248→        // حساب أبعاد المقطع
   249→        val dimensions = when (type) {
   250→            is ColumnType.Rectangular -> type.width to type.depth
   251→            is ColumnType.Circular -> type.diameter to type.diameter
   252→            else -> sqrt(type.getGrossArea()) to sqrt(type.getGrossArea())
   253→        }
   254→        val b = dimensions.first
   255→        val h = dimensions.second
   256→        
   257→        // نسبة التسليح التقريبية (1%)
   258→        val Ag = b * h
   259→        val ast = 0.01 * Ag
   260→        val d_eff = h - 40.0 - 10.0  // خصم الغطاء والكانة
   261→        
   262→        // قدرة العزم التصميمية لكل اتجاه (تقريبي)
   263→        // Mnx = φ × (fy/γs) × As × (d - d') × 0.8
   264→        val fs = fy / 1.15
   265→        val d_prime = 40.0 + 10.0  // cover + stirrup
   266→        val leverArm = (d_eff - d_prime).coerceAtLeast(d_eff * 0.5)
   267→        val Mnx = 0.65 * fs * ast * leverArm / 1e6  // kN.m (مع φ=0.65 للضغط)
   268→        
   269→        // للمقطع الدائري: Mnx = Mny (تماثل)
   270→        // للمستطيل: Mny يُحسب بنفس الطريقة لكن b و h يتبدلان
   271→        val Mny = if (type is ColumnType.Rectangular && abs(b - h) > 10.0) {
   272→            val d_eff_y = b - d_prime
   273→            val leverArmY = (d_eff_y - d_prime).coerceAtLeast(d_eff_y * 0.5)
   274→            0.65 * fs * ast * leverArmY / 1e6
   275→        } else {
   276→            Mnx  // للمربع والدائري
   277→        }
   278→        
   279→        // معامل α حسب Bresler المبسط (ECP 203 البند 4-2-6)
   280→        // α = 1.0 للعزوم المتساوية، يزيد مع عدم التماثل
   281→        val alpha = if (Mnx > 0 && Mny > 0) {
   282→            val beta = min(abs(Mx) / Mnx, abs(My) / Mny).coerceAtMost(1.0)
   283→            // α يتراوح بين 1.0 و 2.0 حسب نسبة الحمل المحوري
   284→            val r = (Pu / Pn).coerceIn(0.1, 0.9)
   285→            1.0 + (1.0 - r)  // α أكبر عندما يكون الحمل المحوري أقل
   286→        } else 1.0
   287→        
   288→        val interactionFactor = if (Mnx > 0 && Mny > 0) {
   289→            (abs(Mx) / Mnx).pow(alpha) + (abs(My) / Mny).pow(alpha)
   290→        } else 0.0
   291→        
   292→        return BiaxialCheckResult(
   293→            abs(Mx), abs(My), interactionFactor,
   294→            interactionFactor <= 1.0,
   295→            "Bresler's reciprocal approximation (ECP 203 Sec 4.2.6), α=%.2f".format(alpha)
   296→        )
   297→    }

   317→    // ثوابت الكود المصري ECP 203-2020 (ECP Constants)

   320→    companion object {
   321→        const val GAMMA_C = 1.5         // معامل أمان الخرسانة
   322→        const val GAMMA_S = 1.15        // معامل أمان الحديد
   323→        const val ALPHA   = 0.8         // عامل اختزال مقاومة الخرسانة (ثابت حسب ECP)
   324→        const val EPS_CU  = 0.003       // إنفعال الخرسانة الأقصى عند الضغط
   325→        const val ES      = 200_000.0   // معامل مرونة الحديد (MPa)
   326→        const val COVER   = 40.0        // الغطاء الخرساني (mm)
   327→        const val STIRRUP_DIA = 10.0    // قطر الكانة الافتراضي (mm)
   328→    }

   330→    // ──────────────────────────────────────────────────────────────────────────────
   331→    // 1) منحنى التفاعل P-M  —  ECP 203 البند 4-2-4
   332→    //    Interaction Diagram Generation
   333→    // ──────────────────────────────────────────────────────────────────────────────
   334→
   335→    /**
   336→     * توليد منحنى التفاعل (P-M Interaction Diagram) حسب الكود المصري ECP 203 البند 4-2-4
   337→     *
   338→     * يقوم بإنشاء 26 نقطة من الضغط البحت حتى الشد البحت
   339→     * لكل نقطة يُحسب: φPn, φMn, c, εt, isTensionControlled
   340→     *
   341→     * @param fcu مقاومة الخرسانة المميزة (MPa)
   342→     * @param fy  إجهاد خضوع الحديد (MPa)
   343→     * @param width  عرض المقطع (mm)
   344→     * @param depth  عمق المقطع (mm)
   345→     * @param totalAs مساحة الحديد الكلية المقدمة (mm²)
   346→     * @param barConfiguration توزيع الأسياخ (افتراضي: متماثل)
   347→     * @return قائمة بنقاط منحنى التفاعل
   348→     */
   349→    fun generateInteractionDiagram(
   350→        fcu: Double,
   351→        fy: Double,
   352→        width: Double,
   353→        depth: Double,
   354→        totalAs: Double,
   355→        barConfiguration: BarConfiguration = BarConfiguration.SYMMETRIC
   356→    ): List<InteractionDiagramPoint> {
   357→        val h = depth
   358→        val b = width
   359→        val dPrime = COVER + STIRRUP_DIA                            // mm
   360→        val d = h - dPrime                                         // mm to extreme tension steel
   361→        val fsYield = fy / GAMMA_S                                  // MPa
   362→
   363→        // ── تعريف طبقات التسليح (Bar Layer Model) ──
   364→        // كل طبقة: Pair<distance from extreme compression fiber (mm), area (mm²)>
   365→        val layers: List<Pair<Double, Double>> = when (barConfiguration) {
   366→            BarConfiguration.SYMMETRIC -> {
   367→                // أربع طبقات: ضغط - نصفي - نصفي - شد
   368→                val asFace = totalAs / 2.0
   369→                listOf(
   370→                    dPrime          to asFace / 2.0,   // ضغط (قريب من وجه الضغط)
   371→                    h / 2.0         to asFace / 2.0,   // نصفي (عند المحور)
   372→                    h / 2.0         to 0.0,             // (تم دمجه في الطبقة السابقة)
   373→                    d               to asFace / 2.0    // شد (الوجه البعيد)
   374→                ).filter { it.second > 0.0 }
   375→            }
   376→            BarConfiguration.ONE_FACE -> {
   377→                listOf(d to totalAs)
   378→            }
   379→            BarConfiguration.TWO_FACES -> {
   380→                val half = totalAs / 2.0
   381→                listOf(dPrime to half, d to half)
   382→            }
   383→            BarConfiguration.PERIMETER -> {
   384→                // تقريب: 4 طبقات متساوية على المحيط
   385→                val perLayer = totalAs / 4.0
   386→                listOf(
   387→                    dPrime          to perLayer,
   388→                    h * 0.25 + dPrime * 0.75  to perLayer,
   389→                    h * 0.75 + dPrime * 0.25  to perLayer,
   390→                    d               to perLayer
   391→                )
   392→            }
   393→        }
   394→
   395→        // ── نقطة الضغط البحت (c → كبير جداً) ──
   396→        val pureCompressionPn = ALPHA * (
   397→            0.67 * fcu / GAMMA_C * (b * h - totalAs) + fsYield * totalAs
   398→        )
   399→
   400→        val points = mutableListOf<InteractionDiagramPoint>()
   401→        val numPoints = 26
   402→
   403→        // c varies from h (pure compression) down to 5 mm (near pure tension)
   404→        // Use a mix of linear and fine spacing near balance
   405→        for (i in 0 until numPoints) {
   406→            val c = when (i) {
   407→                0 -> h + 50.0     // beyond section depth → pure compression
   408→                numPoints - 1 -> 5.0   // near pure tension
   409→                else -> {
   410→                    // Non-linear distribution: more points near balance region
   411→                    val t = i.toDouble() / (numPoints - 1)  // 0→1
   412→                    val cMax = h + 50.0
   413→                    val cMin = 5.0
   414→                    // Use power-law spacing for better resolution near balance
   415→                    cMax - (cMax - cMin) * t.pow(0.7)
   416→                }
   417→            }
   418→
   419→            // a = β1 × c, β1 per ECP ≈ 0.85 for fcu ≤ 40, linearly decreasing
   420→            val beta1 = if (fcu <= 30.0) 0.85
   421→                        else if (fcu <= 55.0) 0.85 - 0.05 * (fcu - 30.0) / 25.0
   422→                        else 0.80
   423→            val a = beta1 * c
   424→
   425→            // Strain in each layer: εsi = εcu × (c - di) / c
   426→            var sumFsiAsi = 0.0        // Σ(fsi × Asi)  in N
   427→            var sumMoment = 0.0         // Σ(fsi × Asi × (di - h/2))  in N.mm
   428→            var epsilonTensionExtreme = 0.0
   429→
   430→            for ((di, Asi) in layers) {
   431→                val epsilonSi = EPS_CU * (c - di) / c
   432→                // Stress: fsi = Es × εsi, capped at ±fy/γs
   433→                var fsi = ES * epsilonSi
   434→                fsi = fsi.coerceIn(-fsYield, fsYield)
   435→
   436→                sumFsiAsi += fsi * Asi
   437→                sumMoment += fsi * Asi * (di - h / 2.0)
   438→
   439→                // Track extreme tension strain (largest di)
   440→                if (di >= d - 1.0) {
   441→                    epsilonTensionExtreme = epsilonSi
   442→                }
   443→            }
   444→
   445→            // Concrete compression block
   446→            val concreteForce = 0.67 * fcu / GAMMA_C * a * b
   447→
   448→            // Nominal axial force (positive = compression)
   449→            val Pn = concreteForce + sumFsiAsi
   450→            // Nominal moment about section centroid
   451→            val Mn = sumMoment
   452→
   453→            // ECP 203 uses α = 0.8 always (unlike ACI's variable φ)
   454→            // φPn = α × Pn, φMn = α × Mn
   455→            val phiPn = ALPHA * Pn / 1000.0     // kN
   456→            val phiMn = ALPHA * Mn / 1e6          // kN.m
   457→
   458→            // Tension controlled: εt ≥ εy
   459→            val epsilonYield = fy / ES
   460→            val isTensionControlled = epsilonTensionExtreme >= epsilonYield
   461→
   462→            // Pure compression point override
   463→            val finalPhiPn = if (i == 0) ALPHA * pureCompressionPn / 1000.0 else phiPn
   464→            val finalPhiMn = if (i == 0) 0.0 else phiMn
   465→
   466→            points.add(InteractionDiagramPoint(
   467→                phiPn = finalPhiPn,
   468→                phiMn = finalPhiMn,
   469→                c = c,
   470→                epsilonT = epsilonTensionExtreme,
   471→                phi = ALPHA,
   472→                isTensionControlled = isTensionControlled
   473→            ))
   474→        }
   475→
   476→        // ── إضافة نقطة الشد البحت (Pure Tension) ──
   477→        val pureTensionPn = -ALPHA * fsYield * totalAs / 1000.0  // kN (negative = tension)
   478→        points.add(InteractionDiagramPoint(
   479→            phiPn = pureTensionPn,
   480→            phiMn = 0.0,
   481→            c = 0.0,
   482→            epsilonT = EPS_CU * 10.0,  // very large
   483→            phi = ALPHA,
   484→            isTensionControlled = true
   485→        ))
   486→
   487→        return points
   488→    }
   489→
   490→    // ──────────────────────────────────────────────────────────────────────────────
   491→    // 2) تكبير العزم (P-Delta)  —  ECP 203 البند 4-2-7
   492→    //    Moment Magnification for Slender Columns
   493→    // ──────────────────────────────────────────────────────────────────────────────
   494→
   495→    /**
   496→     * حساب تكبير العزم بسبب النحافة (P-Delta Effects) حسب ECP 203 البند 4-2-7
   497→     *
   498→     * @param Pu  الحمل المحوري التصميمي (kN)
   499→     * @param Mu  العزم التصميمي (kN.m)
   500→     * @param unsupportedLength  الطول غير المدعوم (m)
   501→     * @param effectiveLengthFactor  معامل الطول الفعال K
   502→     * @param width  عرض المقطع (mm)
   503→     * @param depth  عمق المقطع (mm)
   504→     * @param fcu  مقاومة الخرسانة (MPa)
   505→     * @param fy  إجهاد خضوع الحديد (MPa)
   506→     * @param totalAs  مساحة الحديد الكلية (mm²)
   507→     * @param Cm  معامل العزم المكافئ (default 1.0)
   508→     * @param isSwayFrame  هل الإطار مقاوم للجانبية؟
   509→     * @return MomentMagnificationResult
   510→     */
   511→    fun calculateMomentMagnification(
   512→        Pu: Double,                      // kN
   513→        Mu: Double,                      // kN.m
   514→        unsupportedLength: Double,       // m
   515→        effectiveLengthFactor: Double,   // K
   516→        width: Double,                   // mm
   517→        depth: Double,                   // mm
   518→        fcu: Double,
   519→        fy: Double,
   520→        totalAs: Double,                 // mm²
   521→        Cm: Double = 1.0,                // equivalent moment factor
   522→        isSwayFrame: Boolean = false
   523→    ): MomentMagnificationResult {
   524→        val codeNotes = mutableListOf<String>()
   525→        val L = unsupportedLength * 1000.0   // m → mm
   526→        val K = effectiveLengthFactor
   527→
   528→        // ── Section Properties ──
   529→        val b = width
   530→        val h = depth
   531→        val Ig = b * h * h * h / 12.0       // mm⁴ - gross moment of inertia
   532→
   533→        // ── Modulus of Elasticity (ECP 203: Ec = 4400 × √fcu) ──
   534→        val Ec = 4400.0 * sqrt(fcu)          // MPa
   535→
   536→        // ── Steel moment of inertia (approximate using gross section) ──
   537→        // Is ≈ As × (h/2 - d')²  (assuming symmetric reinforcement)
   538→        val dPrime = COVER + STIRRUP_DIA
   539→        val leverArm = (h / 2.0 - dPrime).coerceAtLeast(0.0)
   540→        val Is = totalAs * leverArm * leverArm   // mm⁴
   541→
   542→        // ── Effective Flexural Rigidity per ECP 203 (simplified) ──
   543→        // EI = 0.2 × Ec × Ig + Es × Is
   544→        val EI = 0.2 * Ec * Ig + ES * Is        // N.mm²
   545→
   546→        // ── Critical Buckling Load (Euler) ──
   547→        val effectiveLength = K * L             // mm
   548→        val Pc = PI * PI * EI / (effectiveLength * effectiveLength)  // N
   549→        val PcKn = Pc / 1000.0                  // kN
   550→
   551→        // ── Slenderness check ──
   552→        val r = min(b, h) / sqrt(12.0)          // mm - radius of gyration
   553→        val lambda = effectiveLength / r
   554→        val isSlender = lambda > 15.0           // rectangular column limit
   555→
   556→        codeNotes.add("ECP 203 البند 4-2-7: تكبير العزم بسبب النحافة")
   557→        codeNotes.add("Ec = 4400√fcu = %.0f MPa".format(Ec))
   558→        codeNotes.add("Ig = %.2e mm⁴, Is = %.2e mm⁴".format(Ig, Is))
   559→        codeNotes.add("EI = %.2e N.mm²".format(EI))
   560→        codeNotes.add("Pc = π²×EI/(KL)² = %.1f kN".format(PcKn))
   561→        codeNotes.add("نسبة النحافة λ = %.1f, العمود %s".format(
   562→            lambda, if (isSlender) "طويل (Long)" else "قصير (Short)"))
   563→
   564→        // ── Magnification Factor ──
   565→        val delta: Double
   566→        val magnifiedMoment: Double
   567→
   568→        if (!isSwayFrame) {
   569→            // Non-sway frame: δ = Cm / (1 - Pu/Pc) ≥ 1.0, capped at 2.5
   570→            val ratio = (Pu * 1000.0) / Pc    // dimensionless
   571→            if (ratio >= 1.0) {
   572→                codeNotes.add("⚠️ Pu/Pc = %.3f ≥ 1.0 → عدم استقرار! يجب زيادة المقطع.".format(ratio))
   573→                delta = 2.5  // cap at maximum
   574→            } else {
   575→                delta = (Cm / (1.0 - ratio)).coerceIn(1.0, 2.5)
   576→            }
   577→            magnifiedMoment = Mu * delta
   578→
   579→            codeNotes.add("إطار غير جانبي (Non-Sway): δ = Cm/(1-Pu/Pc)")
   580→            codeNotes.add("Cm = %.2f, Pu/Pc = %.3f".format(Cm, ratio))
   581→            codeNotes.add("δ = %.3f (max 2.5), M_design = %.1f kN.m".format(delta, magnifiedMoment))
   582→
   583→            if (delta > 1.05) {
   584→                codeNotes.add("⚠️ العزم مكبّر بمعامل δ = %.2f > 1.05 — يُنصح بزيادة المقطع".format(delta))
   585→            }
   586→        } else {
   587→            // Sway frame: δs = 1 / (1 - ΣPu/ΣPc)
   588→            // Here we assume this column's Pu/Pc ratio as approximation for ΣPu/ΣPc
   589→            val ratio = (Pu * 1000.0) / Pc
   590→            val isSwaySensitive = ratio > 0.05
   591→
   592→            if (ratio >= 0.6) {
   593→                codeNotes.add("⚠️ ΣPu/ΣPc = %.3f ≥ 0.6 → الإطار حساس جداً للجانبية!".format(ratio))
   594→                delta = 1.0 / (1.0 - 0.59)  // use capped ratio
   595→            } else {
   596→                delta = 1.0 / (1.0 - ratio)
   597→            }
   598→            magnifiedMoment = Mu + (delta - 1.0) * Mu  // M_design = Mu + δs × Msway (≈ δs × Mu if all sway)
   599→
   600→            codeNotes.add("إطار جانبي (Sway): δs = 1/(1-ΣPu/ΣPc)")
   601→            codeNotes.add("ΣPu/ΣPc ≈ %.3f, δs = %.3f".format(ratio, delta))
   602→
   603→            return MomentMagnificationResult(
   604→                magnifiedMoment = magnifiedMoment,
   605→                delta = delta,
   606→                EI = EI,
   607→                Pc = PcKn,
   608→                isSlender = isSlender,
   609→                isSwaySensitive = isSwaySensitive,
   610→                codeNotes = codeNotes
   611→            )
   612→        }
   613→
   614→        return MomentMagnificationResult(
   615→            magnifiedMoment = magnifiedMoment,
   616→            delta = delta,
   617→            EI = EI,
   618→            Pc = PcKn,
   619→            isSlender = isSlender,
   620→            isSwaySensitive = false,
   621→            codeNotes = codeNotes
   622→        )
   623→    }
   624→
   625→    // ──────────────────────────────────────────────────────────────────────────────
   626→    // 3) تصميم عمود حلزوني (Spiral Column)  —  ECP 203 البند 4-2-6
   627→    //    Spiral Reinforcement Design for Circular Columns
   628→    // ──────────────────────────────────────────────────────────────────────────────
   629→
   630→    /**
   631→     * تصميم عمود دائري حلزوني حسب الكود المصري ECP 203 البند 4-2-6
   632→     *
   633→     * يستخدم معامل φ = 0.75 بدلاً من 0.65 للأعمدة العادية
   634→     *
   635→     * @param fcu مقاومة الخرسانة (MPa)
   636→     * @param fy  إجهاد خضوع الحديد (MPa)
   637→     * @param diameter قطر العمود (mm)
   638→     * @param axialLoad الحمل المحوري (kN)
   639→     * @param moment العزم (kN.m)
   640→     * @param unsupportedLength الطول غير المدعوم (m)
   641→     * @param endConditions ظروف التثبيت
   642→     * @param inventory مخزون الحديد (اختياري)
   643→     * @param loadCombination مجموعة التحميل
   644→     * @return AdvancedColumnResult مع ملاحظات الحلزون
   645→     */
   646→    fun designSpiralColumn(
   647→        fcu: Double,
   648→        fy: Double,
   649→        diameter: Double,               // mm
   650→        axialLoad: Double,              // kN
   651→        moment: Double,                 // kN.m
   652→        unsupportedLength: Double,      // m
   653→        endConditions: ColumnEndConditions,
   654→        inventory: RebarInventory?,
   655→        loadCombination: LoadCombination
   656→    ): AdvancedColumnResult {
   657→        val warnings = mutableListOf<String>()
   658→        val codeNotes = mutableListOf<String>()
   659→
   660→        // معامل أمان أعلى للأعمدة الحلزونية: φ = 0.75
   661→        val PHI_SPIRAL = 0.75
   662→        codeNotes.add("ECP 203 البند 4-2-6: عمود حلزوني (Spiral Column)")
   663→        codeNotes.add("معامل التخفيض φ = %.2f (أعلى من 0.65 للأعمدة العادية)".format(PHI_SPIRAL))
   664→
   665→        val Ag = PI * diameter * diameter / 4.0
   666→        val cover = COVER
   667→        val dSpiral = 8.0  // قطر الحلزون الافتراضي (mm)
   668→        val coreDiameter = diameter - 2.0 * cover
   669→        val Ac = PI * coreDiameter * coreDiameter / 4.0  // مساحة اللب
   670→
   671→        // ── حساب التسليح الطولي المطلوب ──
   672→        val b = diameter
   673→        val h = diameter
   674→        val columnType = ColumnType.Circular(diameter)
   675→
   676→        // delegate to base for reinforcement sizing
   677→        val reinforcementResult = baseDesign.calculateReinforcement(
   678→            fcu, fy, b, h, axialLoad, moment, 0.0, loadCombination
   679→        )
   680→
   681→        // ── نسبة التسليح الحلزوني المطلوبة ──
   682→        // ρs ≥ 0.45 × (Ag/Ac - 1) × (0.67×fcu/(γc × fy/γs))
   683→        val concreteDesignStress = 0.67 * fcu / GAMMA_C
   684→        val steelDesignStress = fy / GAMMA_S
   685→        val rhoSRequired = 0.45 * (Ag / Ac - 1.0) * (concreteDesignStress / steelDesignStress)
   686→
   687→        // ── حساب مساحة الحلزون لكل لفة ──
   688→        val asSpiral = PI * dSpiral * dSpiral / 4.0  // mm²
   689→
   690→        // ── حساب أقصى تباعد للالحلزون ──
   691→        // s ≤ min(75mm, (D_core - 2×db_spiral) / 6)
   692→        val sMax1 = 75.0
   693→        val sMax2 = (coreDiameter - 2.0 * dSpiral) / 6.0
   694→        val sMax = min(sMax1, sMax2)
   695→
   696→        // ── حساب التباعد المطلوب ──
   697→        // ρs = 4 × as × π × (D_core - db) / (s × π × D_core²)
   698→        // Simplified: s = 4 × as × (D_core - db) / (ρs × D_core²)
   699→        // More directly: s = volume_of_spiral_per_turn / (ρs × Ac)
   700→        val sRequired = if (rhoSRequired > 0) {
   701→            4.0 * asSpiral * (coreDiameter - dSpiral) / (rhoSRequired * coreDiameter * coreDiameter)
   702→        } else 75.0
   703→
   704→        val sProvided = sRequired.coerceIn(25.0, sMax)
   705→
   706→        codeNotes.add("المساحة الكلية Ag = %.0f mm², مساحة اللب Ac = %.0f mm²".format(Ag, Ac))
   707→        codeNotes.add("نسبة التسليح الحلزوني المطلوبة ρs = %.4f".format(rhoSRequired))
   708→        codeNotes.add("قطر الحلزون = %.0f mm, التباعد = %.0f mm (أقصى = %.0f mm)".format(dSpiral, sProvided, sMax))
   709→
   710→        if (sRequired > sMax) {
   711→            warnings.add("⚠️ التباعد المطلوب للحلزون (%.0f mm) يزيد عن الحد الأقصى (%.0f mm)".format(sRequired, sMax))
   712→            warnings.add("   يُنصح بزيادة قطر الحلزون أو قطر العمود")
   713→        }
   714→
   715→        // ── فحص الحد الأدنى للمسافة الخالية بين الحلزونات ──
   716→        val clearSpacing = sProvided - dSpiral
   717→        if (clearSpacing < 25.0) {
   718→            codeNotes.add("⚠️ المسافة الخالية بين الحلزونات = %.0f mm < 25 mm — تقليل تباعد الحلزون".format(clearSpacing))
   719→        }
   720→
   721→        // ── Slenderness check ──
   722→        val k = max(endConditions.topCondition.effectiveLengthFactor,
   723→                   endConditions.bottomCondition.effectiveLengthFactor)
   724→        val r = diameter / 4.0
   725→        val effectiveLength = k * unsupportedLength * 1000.0
   726→        val slendernessRatio = effectiveLength / r
   727→        val isSlender = slendernessRatio > 12.0  // circular column limit
   728→        if (isSlender) {
   729→            warnings.add("⚠️ عمود حلزوني نحيف! λ = %.1f > 12 — يجب مراعاة تأثيرات الدرجة الثانية".format(slendernessRatio))
   730→        }
   731→
   732→        // ── Axial capacity with spiral φ ──
   733→        val axialCapacity = calculateAxialCapacityWithPhi(fcu, fy, Ag, reinforcementResult.astProvided, PHI_SPIRAL)
   734→
   735→        // ── Moment capacity (approximate for circular) ──
   736→        val dEff = diameter - cover - STIRRUP_DIA
   737→        val muCap = reinforcementResult.astProvided * (fy / GAMMA_S) * 0.8 * dEff / 1e6
   738→
   739→        codeNotes.add("القدرة المحورية (مع φ=%.2f) = %.1f kN".format(PHI_SPIRAL, axialCapacity))
   740→        codeNotes.add("تباعد الكانات الحلزونية s = %.0f mm".format(sProvided))
   741→
   742→        val inventoryAnalysis = inventory?.let { inv ->
   743→            analyzeInventory(reinforcementResult, unsupportedLength, inv)
   744→        }
   745→
   746→        return AdvancedColumnResult(
   747→            columnType = columnType,
   748→            axialCapacity = axialCapacity,
   749→            momentCapacityX = muCap,
   750→            momentCapacityY = muCap,
   751→            slendernessRatio = slendernessRatio,
   752→            isSlender = isSlender,
   753→            effectiveLength = effectiveLength,
   754→            reinforcementResult = reinforcementResult,
   755→            inventoryAnalysis = inventoryAnalysis,
   756→            biaxialCheck = null,  // circular columns don't have biaxial bending
   757→            punchingCheck = null,
   758→            warnings = warnings,
   759→            codeNotes = codeNotes,
   760→            steelWeightPerMeter = reinforcementResult.astProvided * 7850.0 / 1e6 +
   761→                                   (4.0 * asSpiral * PI * coreDiameter / max(sProvided, 1.0)) * 7850.0 / 1e6,
   762→            concreteVolumePerMeter = Ag / 1e6
   763→        )
   764→    }
   765→
   766→    /**
   767→     * حساب القدرة المحورية مع معامل تخفيض مخصص φ
   768→     * Axial capacity with a custom φ factor (used for spiral columns with φ=0.75)
   769→     */
   770→    private fun calculateAxialCapacityWithPhi(
   771→        fcu: Double, fy: Double, Ag: Double, Ast: Double, phi: Double
   772→    ): Double {
   773→        val ast = Ast.coerceAtMost(Ag * 0.08)
   774→        val concreteStress = 0.67 * fcu / GAMMA_C
   775→        val steelStress = fy / GAMMA_S
   776→        val nominalCapacity = ALPHA * (concreteStress * (Ag - ast) + steelStress * ast)
   777→        return phi * nominalCapacity / 1000.0  // kN
   778→    }
   779→
   780→    // ──────────────────────────────────────────────────────────────────────────────
   781→    // 4) تصنيف العمود: قصير أم طويل  —  ECP 203 البند 4-2-7
   782→    //    Short Column vs Long Column Classification
   783→    // ──────────────────────────────────────────────────────────────────────────────
   784→
   785→    /**
   786→     * تصنيف العمود إلى قصير أو طويل حسب نسبة النحافة
   787→     * Classifies the column as short or long based on slenderness ratio per ECP 203
   788→     *
   789→     * @param unsupportedLength الطول غير المدعوم (m)
   790→     * @param width  عرض المقطع (mm)
   791→     * @param depth  عمق المقطع (mm)
   792→     * @param endConditions ظروف التثبيت
   793→     * @param columnType نوع العمود
   794→     * @return ColumnSlendernessResult
   795→     */
   796→    fun checkShortColumnVsLongColumn(
   797→        unsupportedLength: Double,
   798→        width: Double,
   799→        depth: Double,
   800→        endConditions: ColumnEndConditions,
   801→        columnType: ColumnType
   802→    ): ColumnSlendernessResult {
   803→        val notes = mutableListOf<String>()
   804→
   805→        // ── K factor: max of top and bottom conditions ──
   806→        val kFactor = max(
   807→            endConditions.topCondition.effectiveLengthFactor,
   808→            endConditions.bottomCondition.effectiveLengthFactor
   809→        )
   810→        notes.add("معامل الطول الفعال K = %.2f (max of top=%.2f, bottom=%.2f)".format(
   811→            kFactor,
   812→            endConditions.topCondition.effectiveLengthFactor,
   813→            endConditions.bottomCondition.effectiveLengthFactor
   814→        ))
   815→
   816→        // ── Radius of Gyration r ──
   817→        val radiusOfGyration = when (columnType) {
   818→            is ColumnType.Circular -> columnType.diameter / 4.0
   819→            is ColumnType.Rectangular -> min(columnType.width, columnType.depth) / sqrt(12.0)
   820→            is ColumnType.LShaped -> columnType.thickness / sqrt(12.0)
   821→            is ColumnType.TShaped -> min(columnType.webWidth, columnType.flangeThickness) / sqrt(12.0)
   822→            is ColumnType.Composite -> min(columnType.concreteWidth, columnType.concreteDepth) / sqrt(12.0)
   823→            is ColumnType.Tubular -> (columnType.outerDiameter + columnType.innerDiameter) / 8.0
   824→        }
   825→
   826→        notes.add("نصف القطر الدوراني r = %.1f mm".format(radiusOfGyration))
   827→
   828→        // ── Effective length and slenderness ratio ──
   829→        val effectiveLength = kFactor * unsupportedLength * 1000.0  // mm
   830→        val slendernessRatio = effectiveLength / radiusOfGyration
   831→
   832→        // ── Short column limits per ECP 203 ──
   833→        val shortColumnLimit = when (columnType) {
   834→            is ColumnType.Circular -> 12.0
   835→            else -> 15.0
   836→        }
   837→
   838→        val isShortColumn = slendernessRatio <= shortColumnLimit
   839→        val isLongColumn = !isShortColumn
   840→
   841→        notes.add("الطول الفعال KL = %.0f mm (%.2f m)".format(effectiveLength, effectiveLength / 1000.0))
   842→        notes.add("نسبة النحافة λ = KL/r = %.1f".format(slendernessRatio))
   843→        notes.add("حد العمود القصير λ_lim = %.0f (%s)".format(
   844→            shortColumnLimit,
   845→            if (columnType is ColumnType.Circular) "دائري" else "مستطيل"
   846→        ))
   847→
   848→        if (isShortColumn) {
   849→            notes.add("✅ العمود قصير (Short Column) — لا حاجة لتكبير العزم")
   850→        } else {
   851→            notes.add("⚠️ العمود طويل (Long Column) — يجب تطبيق البند 4-2-7 لتكبير العزم (P-Delta)")
   852→        }
   853→
   854→        // ── Additional checks ──
   855→        if (slendernessRatio > 100.0) {
   856→            notes.add("⚠️ نسبة النحافة عالية جداً (λ=%.1f > 100) — يُنصح بإعادة النظر في تصميم النظام الإنشائي".format(slendernessRatio))
   857→        }
   858→
   859→        // ── End condition recommendations ──
   860→        if (isLongColumn && kFactor > 0.9) {
   861→            notes.add("💡 لتحسين الأداء: يمكن تقليل K بتثبيت أفضل للأطراف (مثال: Fixed-Fixed → K=0.65)")
   862→        }
   863→
   864→        return ColumnSlendernessResult(
   865→            slendernessRatio = slendernessRatio,
   866→            kFactor = kFactor,
   867→            radiusOfGyration = radiusOfGyration,
   868→            shortColumnLimit = shortColumnLimit,
   869→            isShortColumn = isShortColumn,
   870→            isLongColumn = isLongColumn,
   871→            effectiveLength = effectiveLength,
   872→            notes = notes
   873→        )
   874→    }
   875→
   876→    // ──────────────────────────────────────────────────────────────────────────────
   877→    // 5) فحص الانحناء الثنائي المحسّن (Enhanced Biaxial Bending Check)
   878→    //    Uses Bresler reciprocal load method + Load Contour method
   879→    // ──────────────────────────────────────────────────────────────────────────────
   880→
   881→    /**
   882→     * فحص الانحناء الثنائي المحسّن بطريقة Bresler وطريقة Load Contour
   883→     * Enhanced biaxial bending check per ECP 203 Section 4-2-6
   884→     *
   885→     * @param type نوع العمود
   886→     * @param Pu الحمل المحوري (kN)
   887→     * @param Mx العزم حول المحور X (kN.m)
   888→     * @param My العزم حول المحور Y (kN.m)
   889→     * @param Pn القدرة المحورية الكلية (kN)
   890→     * @param fy إجهاد خضوع الحديد (MPa)
   891→     * @return BiaxialCheckResult
   892→     */
   893→    private fun checkBiaxialLoading(
   894→        type: ColumnType,
   895→        Pu: Double,
   896→        Mx: Double,
   897→        My: Double,
   898→        Pn: Double,
   899→        fy: Double = 360.0
   900→    ): BiaxialCheckResult {
   901→        // ── Section dimensions ──
   902→        val (b, h) = when (type) {
   903→            is ColumnType.Rectangular -> type.width to type.depth
   904→            is ColumnType.Circular -> type.diameter to type.diameter
   905→            else -> {
   906→                // Approximate L/T shaped as equivalent square
   907→                val side = sqrt(type.getGrossArea())
   908→                side to side
   909→            }
   910→        }
   911→        val Ag = b * h
   912→        val ast = 0.01 * Ag
   913→
   914→        // ── Generate interaction diagrams for both axes ──
   915→        val diagramX = generateInteractionDiagram(
   916→            fcu = 25.0,  // default assumption when fcu not passed
   917→            fy = fy,
   918→            width = b,
   919→            depth = h,
   920→            totalAs = ast,
   921→            barConfiguration = BarConfiguration.SYMMETRIC
   922→        )
   923→
   924→        val diagramY = if (type is ColumnType.Rectangular && abs(b - h) > 10.0) {
   925→            generateInteractionDiagram(
   926→                fcu = 25.0,
   927→                fy = fy,
   928→                width = h,   // swapped
   929→                depth = b,
   930→                totalAs = ast,
   931→                barConfiguration = BarConfiguration.SYMMETRIC
   932→            )
   933→        } else {
   934→            diagramX  // symmetric section
   935→        }
   936→
   937→        // ── Interpolate to find Mnx and Mny at current Pu ──
   938→        val Mnx = interpolateMomentAtLoad(diagramX, Pu)
   939→        val Mny = interpolateMomentAtLoad(diagramY, Pu)
   940→
   941→        // ── Method 1: Bresler Reciprocal Load (ECP 203 Sec 4.2.6) ──
   942→        // 1/Pn(Biaxial) = 1/Pnx + 1/Pny - 1/Po
   943→        val Pnx = interpolateAxialAtMoment(diagramX, abs(Mx))
   944→        val Pny = interpolateAxialAtMoment(diagramY, abs(My))
   945→        val Po = diagramX.maxOfOrNull { it.phiPn } ?: 0.0
   946→
   947→        val breslerSafe = if (Pnx > 0 && Pny > 0 && Po > 0) {
   948→            val PnBiaxial = 1.0 / (1.0 / Pnx + 1.0 / Pny - 1.0 / Po)
   949→            PnBiaxial >= Pu
   950→        } else true
   951→
   952→        // ── Method 2: Load Contour (Hsu) ──
   953→        // (Mux/Mnx)^α + (Muy/Mny)^α ≤ 1
   954→        // α = 1.0 for Pu ≈ 0, α → 2.0 for Pu → P平衡
   955→        // Accurate α based on Pu/Po ratio
   956→        val r = if (Po > 0) (Pu / Po).coerceIn(0.05, 0.95) else 0.5
   957→        val alphaContour = log(0.5) / log(max(0.001, 1.0 - r))  // Hsu equation
   958→        val alphaEffective = alphaContour.coerceIn(1.0, 2.0)
   959→
   960→        val contourFactor = if (Mnx > 0 && Mny > 0) {
   961→            (abs(Mx) / Mnx).pow(alphaEffective) + (abs(My) / Mny).pow(alphaEffective)
   962→        } else 0.0
   963→
   964→        val contourSafe = contourFactor <= 1.0
   965→
   966→        // ── Use the more critical (higher) factor ──
   967→        // Combine: use load contour factor as primary, Bresler as backup
   968→        val interactionFactor = max(contourFactor,
   969→            if (Pnx > 0 && Pny > 0 && Po > 0) {
   970→                val PnBiaxial = 1.0 / (1.0 / Pnx + 1.0 / Pny - 1.0 / Po)
   971→                if (PnBiaxial > 0) Pu / PnBiaxial else 99.0
   972→            } else 0.0
   973→        )
   974→
   975→        val isSafe = interactionFactor <= 1.0
   976→
   977→        val formula = buildString {
   978→            append("Bresler + Load Contour (ECP 203 Sec 4.2.6)\n")
   979→            append("α(Hsu) = %.2f, Contour factor = %.3f\n".format(alphaEffective, contourFactor))
   980→            if (breslerSafe != contourSafe) {
   981→                append("Bresler: %s, Contour: %s".format(
   982→                    if (breslerSafe) "✅" else "❌",
   983→                    if (contourSafe) "✅" else "❌"
   984→                ))
   985→            }
   986→        }
   987→
   988→        return BiaxialCheckResult(
   989→            mxRatio = if (Mnx > 0) abs(Mx) / Mnx else 0.0,
   990→            myRatio = if (Mny > 0) abs(My) / Mny else 0.0,
   991→            interactionFactor = interactionFactor,
   992→            isSafe = isSafe,
   993→            formula = formula
   994→        )
   995→    }
   996→
   997→    /**
   998→     * Interpolate moment capacity at a given axial load from interaction diagram
   999→     * يُعطي φMn عند φPn المطلوب بالاستكمال الخطي
  1000→     */
  1001→    private fun interpolateMomentAtLoad(diagram: List<InteractionDiagramPoint>, targetPn: Double): Double {
  1002→        if (diagram.isEmpty()) return 0.0
  1003→        if (diagram.size == 1) return diagram[0].phiMn
  1004→
  1005→        // Find two bracketing points
  1006→        for (i in 0 until diagram.size - 1) {
  1007→            val p1 = diagram[i]
  1008→            val p2 = diagram[i + 1]
  1009→
  1010→            // Pn decreases as we move along the diagram (compression → tension)
  1011→            if ((p1.phiPn >= targetPn && p2.phiPn <= targetPn) ||
  1012→                (p1.phiPn <= targetPn && p2.phiPn >= targetPn)) {
  1013→                val dp = p2.phiPn - p1.phiPn
  1014→                if (abs(dp) < 0.001) return max(p1.phiMn, p2.phiMn)
  1015→                val t = (targetPn - p1.phiPn) / dp
  1016→                return p1.phiMn + t * (p2.phiMn - p1.phiMn)
  1017→            }
  1018→        }
  1019→
  1020→        // Extrapolate from nearest endpoint
  1021→        return if (targetPn > diagram.first().phiPn) {
  1022→            diagram.first().phiMn
  1023→        } else {
  1024→            diagram.last().phiMn
  1025→        }
  1026→    }
  1027→
  1028→    /**
  1029→     * Interpolate axial capacity at a given moment from interaction diagram
  1030→     * يُعطي φPn عند φMn المطلوب بالاستكمال الخطي
  1031→     */
  1032→    private fun interpolateAxialAtMoment(diagram: List<InteractionDiagramPoint>, targetMn: Double): Double {
  1033→        if (diagram.isEmpty()) return 0.0
  1034→        if (diagram.size == 1) return diagram[0].phiPn
  1035→
  1036→        for (i in 0 until diagram.size - 1) {
  1037→            val p1 = diagram[i]
  1038→            val p2 = diagram[i + 1]
  1039→
  1040→            if ((p1.phiMn <= targetMn && p2.phiMn >= targetMn) ||
  1041→                (p1.phiMn >= targetMn && p2.phiMn <= targetMn)) {
  1042→                val dm = p2.phiMn - p1.phiMn
  1043→                if (abs(dm) < 0.001) return max(p1.phiPn, p2.phiPn)
  1044→                val t = (targetMn - p1.phiMn) / dm
  1045→                return p1.phiPn + t * (p2.phiPn - p1.phiPn)
  1046→            }
  1047→        }
  1048→
  1049→        return if (targetMn > (diagram.maxOfOrNull { it.phiMn } ?: 0.0)) {
  1050→            diagram.minOfOrNull { it.phiPn } ?: 0.0
  1051→        } else {
  1052→            diagram.maxOfOrNull { it.phiPn } ?: 0.0
  1053→        }
  1054→    }

  1056→    // ──────────────────────────────────────────────────────────────────────────────
  1057→    // 6) الجساءة الجانبية للعمود (Lateral Stiffness)
  1058→    //    For frame analysis and drift calculations
  1059→    // ──────────────────────────────────────────────────────────────────────────────
  1060→
  1061→    /**
  1062→     * حساب الجساءة الجانبية للعمود (k = F/δ) لتحليل الإطارات
  1063→     * Calculates the lateral stiffness of the column for frame analysis
  1064→     *
  1065→     * المعادلات حسب ظروف التثبيت:
  1066→     *   - Fixed-Fixed:     k = 12 × EI / L³
  1067→     *   - Fixed-Pinned:    k = 3 × EI / L³
  1068→     *   - Pinned-Pinned:   k = 0 (unstable laterally, returns minimum)
  1069→     *   - Cantilever:      k = 3 × EI / L³
  1070→     *
  1071→     * @param width  عرض المقطع (mm)
  1072→     * @param depth  عمق المقطع (mm)
  1073→     * @param unsupportedLength  الطول غير المدعوم (m)
  1074→     * @param fcu  مقاومة الخرسانة (MPa)
  1075→     * @param fy  إجهاد خضوع الحديد (MPa)
  1076→     * @param totalAs  مساحة الحديد الكلية (mm²)
  1077→     * @param endConditions  ظروف التثبيت
  1078→     * @return الجساءة الجانبية (kN/mm)
  1079→     */
  1080→    fun calculateLateralStiffness(
  1081→        width: Double,
  1082→        depth: Double,
  1083→        unsupportedLength: Double,
  1084→        fcu: Double,
  1085→        fy: Double,
  1086→        totalAs: Double,
  1087→        endConditions: ColumnEndConditions
  1088→    ): Double {
  1089→        val L = unsupportedLength * 1000.0  // m → mm
  1090→
  1091→        // ── Flexural rigidity ──
  1092→        val b = width
  1093→        val h = depth
  1094→        val Ig = b * h * h * h / 12.0
  1095→        val Ec = 4400.0 * sqrt(fcu)
  1096→        val dPrime = COVER + STIRRUP_DIA
  1097→        val leverArm = (h / 2.0 - dPrime).coerceAtLeast(0.0)
  1098→        val Is = totalAs * leverArm * leverArm
  1099→
  1100→        // EI = 0.2 × Ec × Ig + Es × Is (same formula as moment magnification)
  1101→        val EI = 0.2 * Ec * Ig + ES * Is  // N.mm²
  1102→
  1103→        // ── Determine end condition factor for stiffness ──
  1104→        // Use a simplified approach based on K-factor combinations
  1105→        val kTop = endConditions.topCondition.effectiveLengthFactor
  1106→        val kBottom = endConditions.bottomCondition.effectiveLengthFactor
  1107→        val avgK = (kTop + kBottom) / 2.0
  1108→
  1109→        // Stiffness coefficient based on end conditions
  1110→        // Fixed-Fixed → 12, Fixed-Pinned → 3, Pinned-Pinned → 0 (unstable)
  1111→        val stiffnessCoeff = when {
  1112→            kTop <= 0.7 && kBottom <= 0.7 -> 12.0        // approximately Fixed-Fixed
  1113→            kTop <= 0.7 || kBottom <= 0.7 -> 3.0         // Fixed-Pinned or Pinned-Fixed
  1114→            avgK <= 0.85 -> 6.0                           // Partially fixed both ends
  1115→            else -> 1.5                                    // Pinned-Pinned (very low stiffness)
  1116→        }
  1117→
  1118→        // k = coeff × EI / L³  (N/mm)
  1119→        val stiffness = stiffnessCoeff * EI / (L * L * L)
  1120→
  1121→        // Convert to kN/mm
  1122→        return stiffness / 1000.0
  1123→    }
  1124→}
