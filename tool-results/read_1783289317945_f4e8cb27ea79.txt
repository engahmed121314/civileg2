   446→            HorizontalDivider(modifier = Modifier.padding(vertical = 8.dp), color = Color.Gray.copy(alpha = 0.1f))

   476→        )
   477→        Spacer(modifier = Modifier.width(16.dp))

   599→private fun generateBarPositions(width: Double, depth: Double, numBars: Int, diameter: Double): List<BarInfo> {
   600→    val cover = 40.0
   601→    val cx = width / 2.0
   602→    val cy = depth / 2.0
   603→    val effectiveW = width - 2 * cover
   604→    val effectiveD = depth - 2 * cover
   605→    val positions = mutableListOf<BarInfo>()
   606→    val barsPerSide = maxOf(2, numBars / 4)
   607→    val corners = 4
   608→
   609→    // Corner bars
   610→    listOf(
   611→        BarInfo(cover + diameter/2, cover + diameter/2, diameter, isCorner = true),
   612→        BarInfo(width - cover - diameter/2, cover + diameter/2, diameter, isCorner = true),
   613→        BarInfo(cover + diameter/2, depth - cover - diameter/2, diameter, isCorner = true),
   614→        BarInfo(width - cover - diameter/2, depth - cover - diameter/2, diameter, isCorner = true)
   615→    ).forEach { positions.add(it) }
   616→
   617→    // Distribute remaining bars along sides
   618→    val remaining = numBars - 4
   619→    if (remaining > 0) {
   620→        val perSide = remaining / 4
   621→        for (side in 0 until 4) {
   622→            val count = if (side < remaining % 4) perSide + 1 else perSide
   623→            for (i in 1..count) {
   624→                val t = i.toDouble() / (count + 1)
   625→                val pos = when (side) {
   626→                    0 -> BarInfo(cover + diameter/2 + t * effectiveW, cover + diameter/2, diameter)
   627→                    1 -> BarInfo(width - cover - diameter/2, cover + diameter/2 + t * effectiveD, diameter)
   628→                    2 -> BarInfo(width - cover - diameter/2 - t * effectiveW, depth - cover - diameter/2, diameter)
   629→                    else -> BarInfo(cover + diameter/2, depth - cover - diameter/2 - t * effectiveD, diameter)
   630→                }
   631→                positions.add(pos)
   632→            }
   633→        }
   634→    }
   635→    return positions
   636→}
