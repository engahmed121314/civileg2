    74тЖТ    74тЖТ    private fun Canvas.drawHDim(x1: Float, x2: Float, y: Float, text: String, offset: Float = 20f, paint: Paint? = null) {
    75тЖТ    75тЖТ        val p = paint ?: createPaint(DIM_TEXT, 1f)
    76тЖТ    76тЖТ        val dimY = y + offset
    77тЖТ    77тЖТ        val extLen = 8f
    78тЖТ    78тЖТ        val arrowSize = 5f
    79тЖТ    79тЖТ
    80тЖТ    80тЖТ        drawLine(x1, y + 2f, x1, dimY + extLen, p)
    81тЖТ    81тЖТ        drawLine(x2, y + 2f, x2, dimY + extLen, p)
    82тЖТ    82тЖТ        drawLine(x1 + arrowSize, dimY, x2 - arrowSize, dimY, p)
    83тЖТ    83тЖТ
    84тЖТ    84тЖТ        // Left arrow
    85тЖТ    85тЖТ        drawPath(Path().apply {
    86тЖТ    86тЖТ            moveTo(x1, dimY); lineTo(x1 + arrowSize * 2, dimY - arrowSize); lineTo(x1 + arrowSize * 2, dimY + arrowSize); close()
    87тЖТ    87тЖТ        }, p)
    88тЖТ    88тЖТ        // Right arrow
    89тЖТ    89тЖТ        drawPath(Path().apply {
    90тЖТ    90тЖТ            moveTo(x2, dimY); lineTo(x2 - arrowSize * 2, dimY - arrowSize); lineTo(x2 - arrowSize * 2, dimY + arrowSize); close()
    91тЖТ    91тЖТ        }, p)
    92тЖТ    92тЖТ
    93тЖТ    93тЖТ        val tp = textPaint(DIM_TEXT, 18f)
    94тЖТ    94тЖТ        drawTextCentered(text, (x1 + x2) / 2f, dimY + extLen + 16f, tp)
    95тЖТ    95тЖТ    }

   112тЖТ   112тЖТ    private fun Canvas.drawRebar(cx: Float, cy: Float, r: Float, color: Int = REBAR_BLUE) {
   113тЖТ   113тЖТ        drawCircle(cx, cy, r, fillPaint(color))
   114тЖТ   114тЖТ        drawCircle(cx, cy, r * 0.35f, fillPaint(Color.argb(128, Color.red(color), Color.green(color), Color.blue(color))))
   115тЖТ   115тЖТ    }

   118тЖТ   118тЖТ    private fun Canvas.drawHatch(x: Float, y: Float, w: Float, h: Float, spacing: Float = 12f) {
   119тЖТ   119тЖТ        val p = createPaint(HATCH, 0.8f)
   120тЖТ   120тЖТ        var i = 0
   121тЖТ   121тЖТ        while (i * spacing < w + h) {
   122тЖТ   122тЖТ            val sx = x + i * spacing
   123тЖТ   123тЖТ            val sy = y + h
   124тЖТ   124тЖТ            val ex = sx + h
   125тЖТ   125тЖТ            val ey = y
   126тЖТ   126тЖТ            drawLine(
   127тЖТ   127тЖТ                sx.coerceIn(x, x + w), maxOf(y, sy - (sx - x).coerceAtLeast(0f)).coerceAtMost(y + h),
   128тЖТ   128тЖТ                ex.coerceIn(x, x + w), minOf(y + h, ey + (x + w - ex).coerceAtLeast(0f)).coerceAtLeast(y),
   129тЖТ   129тЖТ                p
   130тЖТ   130тЖТ            )
   131тЖТ   131тЖТ            i++
   132тЖТ   132тЖТ        }
   133тЖТ   133тЖТ    }

   135тЖТ   135тЖТ    // ========== GENERATE BEAM DRAWING ==========
   136тЖТ   136тЖТ    fun generateBeamDrawing(
   137тЖТ   137тЖТ        beamWidth: Double, beamDepth: Double, span: Double,
   138тЖТ   138тЖТ        mainRebarDia: Double, mainRebarCount: Int,
   139тЖТ   139тЖТ        stirrupDia: Double, stirrupSpacing: Double,
   140тЖТ   140тЖТ        cover: Double = 50.0,
   141тЖТ   141тЖТ        hasTopSteel: Boolean = false, topRebarDia: Double = 0.0, topRebarCount: Int = 0
   142тЖТ   142тЖТ    ): Bitmap {
   143тЖТ   143тЖТ        val W = 1200; val H = 700
   144тЖТ   144тЖТ        val (bitmap, canvas) = createCanvas(W, H)
   145тЖТ   145тЖТ
   146тЖТ   146тЖТ        // Layout
   147тЖТ   147тЖТ        val marginL = 80f; val marginR = 80f; val marginT = 50f
   148тЖТ   148тЖТ        val mainBottom = H * 0.55f
   149тЖТ   149тЖТ        val availW = W - marginL - marginR - 60f
   150тЖТ   150тЖТ        val availH = mainBottom - marginT - 40f
   151тЖТ   151тЖТ        val scaleW = availW / span.toFloat()
   152тЖТ   152тЖТ        val scaleH = availH / beamDepth.toFloat()
   153тЖТ   153тЖТ        val scale = min(scaleW, scaleH) * 0.75f
   154тЖТ   154тЖТ
   155тЖТ   155тЖТ        val bDrawW = span.toFloat() * scale
   156тЖТ   156тЖТ        val bDrawH = beamDepth.toFloat() * scale
   157тЖТ   157тЖТ        val bDrawD = beamWidth.toFloat() * scale * 0.3f
   158тЖТ   158тЖТ        val bLeft = marginL + 40f
   159тЖТ   159тЖТ        val bTop = marginT + 40f + bDrawD * 0.2f
   160тЖТ   160тЖТ        val bRight = bLeft + bDrawW
   161тЖТ   161тЖТ        val bBottom = bTop + bDrawH
   162тЖТ   162тЖТ
   163тЖТ   163тЖТ        // 3D beam body
   164тЖТ   164тЖТ        // Front face
   165тЖТ   165тЖТ        canvas.drawRect(bLeft, bTop, bRight, bBottom, fillPaint(CONCRETE))
   166тЖТ   166тЖТ        // Top face (3D)
   167тЖТ   167тЖТ        canvas.drawPath(Path().apply {
   168тЖТ   168тЖТ            moveTo(bLeft, bTop); lineTo(bLeft + bDrawD * 0.3f, bTop - bDrawD * 0.2f)
   169тЖТ   169тЖТ            lineTo(bRight + bDrawD * 0.3f, bTop - bDrawD * 0.2f); lineTo(bRight, bTop); close()
   170тЖТ   170тЖТ        }, fillPaint(CONCRETE_TOP))
   171тЖТ   171тЖТ        // Right side face
   172тЖТ   172тЖТ        canvas.drawPath(Path().apply {
   173тЖТ   173тЖТ            moveTo(bRight, bTop); lineTo(bRight + bDrawD * 0.3f, bTop - bDrawD * 0.2f)
   174тЖТ   174тЖТ            lineTo(bRight + bDrawD * 0.3f, bBottom - bDrawD * 0.2f); lineTo(bRight, bBottom); close()
   175тЖТ   175тЖТ        }, fillPaint(CONCRETE_SIDE))
   176тЖТ   176тЖТ
   177тЖТ   177тЖТ        // Outline
   178тЖТ   178тЖТ        val outlineP = createPaint(Color.WHITE, 1.5f)
   179тЖТ   179тЖТ        canvas.drawRect(bLeft, bTop, bRight, bBottom, outlineP)
   180тЖТ   180тЖТ
   181тЖТ   181тЖТ        // Supports (pin left, roller right)
   182тЖТ   182тЖТ        val supportSize = 20f
   183тЖТ   183тЖТ        // Left pin
   184тЖТ   184тЖТ        canvas.drawPath(Path().apply {
   185тЖТ   185тЖТ            moveTo(bLeft - supportSize/2, bBottom); lineTo(bLeft + supportSize/2, bBottom)
   186тЖТ   186тЖТ            lineTo(bLeft, bBottom + supportSize); close()
   187тЖТ   187тЖТ        }, fillPaint(SUPPORT))
   188тЖТ   188тЖТ        canvas.drawLine(bLeft - supportSize, bBottom + supportSize, bLeft + supportSize, bBottom + supportSize, outlineP)
   189тЖТ   189тЖТ
   190тЖТ   190тЖТ        // Right roller
   191тЖТ   191тЖТ        canvas.drawPath(Path().apply {
   192тЖТ   192тЖТ            moveTo(bRight - supportSize/2, bBottom); lineTo(bRight + supportSize/2, bBottom)
   193тЖТ   193тЖТ            lineTo(bRight, bBottom + supportSize * 0.7f); close()
   194тЖТ   194тЖТ        }, fillPaint(SUPPORT))
   195тЖТ   195тЖТ        canvas.drawCircle(bRight - 6f, bBottom + supportSize * 0.7f + 4f, 3f, fillPaint(SUPPORT))
   196тЖТ   196тЖТ        canvas.drawCircle(bRight + 6f, bBottom + supportSize * 0.7f + 4f, 3f, fillPaint(SUPPORT))
   197тЖТ   197тЖТ        canvas.drawLine(bRight - supportSize, bBottom + supportSize, bRight + supportSize, bBottom + supportSize, outlineP)
   198тЖТ   198тЖТ
   199тЖТ   199тЖТ        // Main reinforcement (bottom)
   200тЖТ   200тЖТ        val rebarR = maxOf(mainRebarDia.toFloat() / 2f * scale * 0.5f, 3f)
   201тЖТ   201тЖТ        val rebarY = bBottom - cover.toFloat() * scale - rebarR
   202тЖТ   202тЖТ        val spacing = (bDrawW - 2 * cover.toFloat() * scale) / maxOf(mainRebarCount - 1, 1)
   203тЖТ   203тЖТ        for (i in 0 until mainRebarCount) {
   204тЖТ   204тЖТ            val rx = bLeft + cover.toFloat() * scale + i * spacing
   205тЖТ   205тЖТ            canvas.drawRebar(rx, rebarY, rebarR, REBAR_BLUE)
   206тЖТ   206тЖТ        }
   207тЖТ   207тЖТ
   208тЖТ   208тЖТ        // Stirrups
   209тЖТ   209тЖТ        val stirrupP = createPaint(STIRRUP, 1.5f)
   210тЖТ   210тЖТ        val stirrupY1 = bTop + cover.toFloat() * scale
   211тЖТ   211тЖТ        val stirrupY2 = bBottom - cover.toFloat() * scale
   212тЖТ   212тЖТ        val stirrupX1 = bLeft + cover.toFloat() * scale
   213тЖТ   213тЖТ        val stirrupX2 = bRight - cover.toFloat() * scale
   214тЖТ   214тЖТ        val stirrupSpacingPx = stirrupSpacing.toFloat() * scale
   215тЖТ   215тЖТ        var sx = bLeft + stirrupSpacingPx
   216тЖТ   216тЖТ        while (sx < bRight - stirrupSpacingPx) {
   217тЖТ   217тЖТ            canvas.drawRect(sx - 2f, stirrupY1, sx + 2f, stirrupY2, stirrupP)
   218тЖТ   218тЖТ            sx += stirrupSpacingPx
   219тЖТ   219тЖТ        }
   220тЖТ   220тЖТ
   221тЖТ   221тЖТ        // Top reinforcement
   222тЖТ   222тЖТ        if (hasTopSteel && topRebarCount > 0) {
   223тЖТ   223тЖТ            val topR = maxOf(topRebarDia.toFloat() / 2f * scale * 0.5f, 3f)
   224тЖТ   224тЖТ            val topY = bTop + cover.toFloat() * scale + topR
   225тЖТ   225тЖТ            val topSpacing = (bDrawW - 2 * cover.toFloat() * scale) / maxOf(topRebarCount - 1, 1)
   226тЖТ   226тЖТ            for (i in 0 until topRebarCount) {
   227тЖТ   227тЖТ                val rx = bLeft + cover.toFloat() * scale + i * topSpacing
   228тЖТ   228тЖТ                canvas.drawRebar(rx, topY, topR, TOP_REBAR)
   229тЖТ   229тЖТ            }
   230тЖТ   230тЖТ        }
   231тЖТ   231тЖТ
   232тЖТ   232тЖТ        // Dimensions
   233тЖТ   233тЖТ        canvas.drawHDim(bLeft, bRight, bBottom + supportSize + 20f, "${span.toInt()} mm")
   234тЖТ   234тЖТ        canvas.drawVDim(bTop, bBottom, bLeft - 20f, "${beamDepth.toInt()} mm")
   235тЖТ   235тЖТ        canvas.drawVDim(bTop, bTop + cover.toFloat() * scale, bRight + 20f, "cover=${cover.toInt()}")
   236тЖТ   236тЖТ
   237тЖТ   237тЖТ        // Title
   238тЖТ   238тЖТ        val titleP = textPaint(Color.WHITE, 24f, true)
   239тЖТ   239тЖТ        canvas.drawTextCentered("BEAM SECTION & ELEVATION", W / 2f, 30f, titleP)
   240тЖТ   240тЖТ
   241тЖТ   241тЖТ        // Cross-section inset (bottom-left)
   242тЖТ   242тЖТ        val csX = 80f; val csY = H * 0.60f; val csScale = 0.4f
   243тЖТ   243тЖТ        val csW = beamWidth.toFloat() * scale * csScale
   244тЖТ   244тЖТ        val csH = beamDepth.toFloat() * scale * csScale
   245тЖТ   245тЖТ
   246тЖТ   246тЖТ        canvas.drawRect(csX, csY, csX + csW, csY + csH, fillPaint(CONCRETE))
   247тЖТ   247тЖТ        canvas.drawRect(csX, csY, csX + csW, csY + csH, outlineP)
   248тЖТ   248тЖТ
   249тЖТ   249тЖТ        // Cross-section stirrup
   250тЖТ   250тЖТ        val csCover = cover.toFloat() * scale * csScale
   251тЖТ   251тЖТ        canvas.drawRect(csX + csCover, csY + csCover, csX + csW - csCover, csY + csH - csCover, createPaint(STIRRUP, 1.5f))
   252тЖТ   252тЖТ
   253тЖТ   253тЖТ        // Cross-section bars
   254тЖТ   254тЖТ        val csRebarR = maxOf(mainRebarDia.toFloat() * csScale * 0.3f, 3f)
   255тЖТ   255тЖТ        val csBarY = csY + csH - csCover - csRebarR
   256тЖТ   256тЖТ        for (i in 0 until minOf(mainRebarCount, 6)) {
   257тЖТ   257тЖТ            val rx = if (mainRebarCount <= 2) {
   258тЖТ   258тЖТ                csX + csW / 2f
   259тЖТ   259тЖТ            } else {
   260тЖТ   260тЖТ                csX + csCover + csRebarR + i * (csW - 2 * csCover - 2 * csRebarR) / maxOf(mainRebarCount - 1, 1)
   261тЖТ   261тЖТ            }
   262тЖТ   262тЖТ            canvas.drawRebar(rx, csBarY, csRebarR, REBAR_BLUE)
   263тЖТ   263тЖТ        }
   264тЖТ   264тЖТ
   265тЖТ   265тЖТ        // Section label
   266тЖТ   266тЖТ        canvas.drawTextCentered("Section A-A", csX + csW / 2f, csY - 8f, textPaint(DIM_TEXT, 16f))
   267тЖТ   267тЖТ
   268тЖТ   268тЖТ        // Reinforcement table (bottom-right)
   269тЖТ   269тЖТ        drawRebarTable(canvas, 
   270тЖТ   270тЖТ            x = W * 0.5f, y = H * 0.60f,
   271тЖТ   271тЖТ            data = listOf(
   272тЖТ   272тЖТ                listOf("Mark", "Dia", "No.", "Spacing", "Length"),
   273тЖТ   273тЖТ                listOf("B1", "${mainRebarDia.toInt()}mm", "$mainRebarCount", "-", "${span.toInt()}mm"),
   274тЖТ   274тЖТ                listOf("S1", "${stirrupDia.toInt()}mm", "-", "${stirrupSpacing.toInt()}mm", "-")
   275тЖТ   275тЖТ            ) + if (hasTopSteel && topRebarCount > 0) listOf(listOf("T1", "${topRebarDia.toInt()}mm", "$topRebarCount", "-", "${(span*0.3).toInt()}mm")) else emptyList()
   276тЖТ   276тЖТ        )
   277тЖТ   277тЖТ
   278тЖТ   278тЖТ        // Title block
   279тЖТ   279тЖТ        drawTitleBlock(canvas, W - 280f, H - 60f, 280f, 60f, "Beam Detail")
   280тЖТ   280тЖТ
   281тЖТ   281тЖТ        return bitmap
   282тЖТ   282тЖТ    }

   284тЖТ   284тЖТ    // ========== GENERATE COLUMN DRAWING ==========
   285тЖТ   285тЖТ    fun generateColumnDrawing(
   286тЖТ   286тЖТ        columnWidth: Double, columnDepth: Double, columnHeight: Double,
   287тЖТ   287тЖТ        numBars: Int, barDia: Double, tieDia: Double, tieSpacing: Double,
   288тЖТ   288тЖТ        cover: Double = 40.0
   289тЖТ   289тЖТ    ): Bitmap {
   290тЖТ   290тЖТ        val W = 1200; val H = 800
   291тЖТ   291тЖТ        val (bitmap, canvas) = createCanvas(W, H)
   292тЖТ   292тЖТ
   293тЖТ   293тЖТ        val outlineP = createPaint(Color.WHITE, 1.5f)
   294тЖТ   294тЖТ        val titleP = textPaint(Color.WHITE, 24f, true)
   295тЖТ   295тЖТ
   296тЖТ   296тЖТ        // 3D Elevation (left side)
   297тЖТ   297тЖТ        val elevLeft = 100f; val elevW = 200f
   298тЖТ   298тЖТ        val elevTop = 80f; val elevH = min(columnHeight.toFloat(), 500f)
   299тЖТ   299тЖТ        val depth3D = 40f
   300тЖТ   300тЖТ
   301тЖТ   301тЖТ        // Front face
   302тЖТ   302тЖТ        canvas.drawRect(elevLeft, elevTop, elevLeft + elevW, elevTop + elevH, fillPaint(CONCRETE))
   303тЖТ   303тЖТ        // Top face

   305тЖТ   305тЖТ            moveTo(elevLeft, elevTop); lineTo(elevLeft + depth3D, elevTop - depth3D * 0.3f)

   308тЖТ   308тЖТ        // Side face
   309тЖТ   309тЖТ        canvas.drawPath(Path().apply {
   310тЖТ   310тЖТ            moveTo(elevLeft + elevW, elevTop); lineTo(elevLeft + elevW + depth3D, elevTop - depth3D * 0.3f)

   312тЖТ   312тЖТ        }, fillPaint(CONCRETE_SIDE))
   313тЖТ   313тЖТ
   314тЖТ   314тЖТ        canvas.drawRect(elevLeft, elevTop, elevLeft + elevW, elevTop + elevH, outlineP)
   315тЖТ   315тЖТ
   316тЖТ   316тЖТ        // Visible bars on front face
   317тЖТ   317тЖТ        val barR = maxOf(barDia.toFloat() * 0.4f, 3f)
   318тЖТ   318тЖТ        val coverPx = cover.toFloat()
   319тЖТ   319тЖТ        // Corner bars (front face - 4 bars)
   320тЖТ   320тЖТ        val corners = listOf(
   321тЖТ   321тЖТ            Offset(elevLeft + coverPx, elevTop + coverPx),
   322тЖТ   322тЖТ            Offset(elevLeft + elevW - coverPx, elevTop + coverPx),
   323тЖТ   323тЖТ            Offset(elevLeft + coverPx, elevTop + elevH - coverPx),
   324тЖТ   324тЖТ            Offset(elevLeft + elevW - coverPx, elevTop + elevH - coverPx)

   327тЖТ   327тЖТ
   328тЖТ   328тЖТ        // Ties
   329тЖТ   329тЖТ        val tieP = createPaint(STIRRUP, 1.5f)
   330тЖТ   330тЖТ        var ty = elevTop + coverPx + 20f
   331тЖТ   331тЖТ        while (ty < elevTop + elevH - coverPx) {
   332тЖТ   332тЖТ            canvas.drawRect(elevLeft + coverPx, ty, elevLeft + elevW - coverPx, ty + 3f, tieP)
   333тЖТ   333тЖТ            ty += tieSpacing.toFloat() * (elevH / columnHeight.toFloat())
   334тЖТ   334тЖТ        }
   335тЖТ   335тЖТ
   336тЖТ   336тЖТ        // Floor slabs at top/bottom
   337тЖТ   337тЖТ        canvas.drawRect(elevLeft - 60f, elevTop - 15f, elevLeft + elevW + depth3D + 60f, elevTop, fillPaint(CONCRETE_TOP))
   338тЖТ   338тЖТ        canvas.drawRect(elevLeft - 60f, elevTop - 15f, elevLeft + elevW + depth3D + 60f, elevTop, outlineP)

   343тЖТ   343тЖТ

   345тЖТ   345тЖТ        val csCx = W * 0.65f; val csCy = H * 0.35f
   346тЖТ   346тЖТ        val csW2 = columnWidth.toFloat() * 0.8f
   347тЖТ   347тЖТ        val csH2 = columnDepth.toFloat() * 0.8f

   358тЖТ   358тЖТ        val effW = csW2 - 2 * csCover
   359тЖТ   359тЖТ        val effH = csH2 - 2 * csCover
   360тЖТ   360тЖТ        val barsPerSide = maxOf(2, (numBars - 4) / 4)
   361тЖТ   361тЖТ        // 4 corners always
   362тЖТ   362тЖТ        val csBarR = maxOf(barDia.toFloat() * 0.5f, 4f)
   363тЖТ   363тЖТ        val barPositions = mutableListOf<Offset>()
   364тЖТ   364тЖТ        barPositions.add(Offset(csLeft + csCover + csBarR, csTop2 + csCover + csBarR))

   366тЖТ   366тЖТ        barPositions.add(Offset(csLeft + csCover + csBarR, csTop2 + csH2 - csCover - csBarR))
   367тЖТ   367тЖТ        barPositions.add(Offset(csLeft + csW2 - csCover - csBarR, csTop2 + csH2 - csCover - csBarR))
   368тЖТ   368тЖТ
   369тЖТ   369тЖТ        val remaining = numBars - 4
   370тЖТ   370тЖТ        if (remaining > 0) {

   372тЖТ   372тЖТ            val extra = remaining % 4
   373тЖТ   373тЖТ            for (side in 0 until 4) {
   374тЖТ   374тЖТ                val count = perSide + if (side < extra) 1 else 0
   375тЖТ   375тЖТ                for (i in 1..count) {
   376тЖТ   376тЖТ                    val t = i.toFloat() / (count + 1)
   377тЖТ   377тЖТ                    when (side) {
   378тЖТ   378тЖТ                        0 -> barPositions.add(Offset(csLeft + csCover + csBarR + t * (effW - 2 * csBarR), csTop2 + csCover + csBarR))
   379тЖТ   379тЖТ                        1 -> barPositions.add(Offset(csLeft + csW2 - csCover - csBarR, csTop2 + csCover + csBarR + t * (effH - 2 * csBarR)))
   380тЖТ   380тЖТ                        2 -> barPositions.add(Offset(csLeft + csCover + csBarR + t * (effW - 2 * csBarR), csTop2 + csH2 - csCover - csBarR))
   381тЖТ   381тЖТ                        3 -> barPositions.add(Offset(csLeft + csCover + csBarR, csTop2 + csCover + csBarR + t * (effH - 2 * csBarR)))
   382тЖТ   382тЖТ                    }
   383тЖТ   383тЖТ                }
   384тЖТ   384тЖТ            }
   385тЖТ   385тЖТ        }
   386тЖТ   386тЖТ

   389тЖТ   389тЖТ            // Bar mark number
   390тЖТ   390тЖТ            val mark = getCircleNumber(idx + 1)
   391тЖТ   391тЖТ            canvas.drawTextCentered(mark, pos.x, pos.y - csBarR - 6f, textPaint(DIM_TEXT, 14f, true))
   392тЖТ   392тЖТ        }
   393тЖТ   393тЖТ
   394тЖТ   394тЖТ        // Section dimensions
   395тЖТ   395тЖТ        canvas.drawHDim(csLeft, csLeft + csW2, csTop2 + csH2 + 15f, "${columnWidth.toInt()} mm", 20f)
   396тЖТ   396тЖТ        canvas.drawVDim(csTop2, csTop2 + csH2, csLeft + csW2 + 15f, "${columnDepth.toInt()} mm", 20f)
   397тЖТ   397тЖТ        canvas.drawTextCentered("SECTION A-A", csCx, csTop2 - 20f, titleP)
   398тЖТ   398тЖТ
   399тЖТ   399тЖТ        // Rebar table
   400тЖТ   400тЖТ        drawRebarTable(canvas, 
   401тЖТ   401тЖТ            x = 100f, y = H * 0.65f,
   402тЖТ   402тЖТ            data = listOf(
   403тЖТ   403тЖТ                listOf("Mark", "Dia", "No.", "Type", "Spacing"),
   404тЖТ   404тЖТ                listOf("M1", "${barDia.toInt()}mm", "$numBars", "Main", "-"),
   405тЖТ   405тЖТ                listOf("T1", "${tieDia.toInt()}mm", "-", "Ties", "${tieSpacing.toInt()}mm")
   406тЖТ   406тЖТ            )
   407тЖТ   407тЖТ        )
   408тЖТ   408тЖТ
   409тЖТ   409тЖТ        drawTitleBlock(canvas, W - 280f, H - 60f, 280f, 60f, "Column Detail")
   410тЖТ   410тЖТ
   411тЖТ   411тЖТ        return bitmap
   412тЖТ   412тЖТ    }

   566тЖТ   566тЖТ            )
   567тЖТ   567тЖТ        )
   568тЖТ   568тЖТ
   569тЖТ   569тЖТ        drawTitleBlock(canvas, W - 280f, H - 60f, 280f, 60f, "Footing Detail")
   570тЖТ   570тЖТ        return bitmap
   571тЖТ   571тЖТ    }

   573тЖТ   573тЖТ    // ========== GENERATE STAIR DRAWING ==========
   574тЖТ   574тЖТ    fun generateStairDrawing(
   575тЖТ   575тЖТ        totalHeight: Double, totalLength: Double, stairWidth: Double,
   576тЖТ   576тЖТ        riserHeight: Double, treadWidth: Double, slabThickness: Double,
   577тЖТ   577тЖТ        mainDia: Double, mainSpacing: Double, distDia: Double = 8.0, distSpacing: Double = 200.0
   578тЖТ   578тЖТ    ): Bitmap {
   579тЖТ   579тЖТ        val W = 1200; val H = 700
   580тЖТ   580тЖТ        val (bitmap, canvas) = createCanvas(W, H)
   581тЖТ   581тЖТ        val outlineP = createPaint(Color.WHITE, 1.5f)
   582тЖТ   582тЖТ
   583тЖТ   583тЖТ        val nRisers = (totalHeight / riserHeight).toInt().coerceAtLeast(3)
   584тЖТ   584тЖТ        val nTreads = nRisers - 1
   585тЖТ   585тЖТ        val angle = atan2(totalHeight.toFloat(), totalLength.toFloat())
   586тЖТ   586тЖТ
   587тЖТ   587тЖТ        // Elevation
   588тЖТ   588тЖТ        val elevL = 80f; val elevT = 80f
   589тЖТ   589тЖТ        val scale = min(500f / totalLength.toFloat(), 350f / totalHeight.toFloat()) * 0.8f
   590тЖТ   590тЖТ        val drawL = totalLength.toFloat() * scale; val drawH = totalHeight.toFloat() * scale
   591тЖТ   591тЖТ        val treadH = riserHeight.toFloat() * scale; val treadW = treadWidth.toFloat() * scale
   592тЖТ   592тЖТ
   593тЖТ   593тЖТ        // Stair outline
   594тЖТ   594тЖТ        val path = Path()
   595тЖТ   595тЖТ        path.moveTo(elevL, elevT + drawH) // bottom-left
   596тЖТ   596тЖТ        for (i in 0 until nRisers) {
   597тЖТ   597тЖТ            path.lineTo(elevL + i * treadW, elevT + drawH - (i + 1) * treadH)
   598тЖТ   598тЖТ            if (i < nTreads) path.lineTo(elevL + (i + 1) * treadW, elevT + drawH - (i + 1) * treadH)
   599тЖТ   599тЖТ        }
   600тЖТ   600тЖТ        // Soffit line
   601тЖТ   601тЖТ        val soffitOffset = slabThickness.toFloat() * scale
   602тЖТ   602тЖТ        path.lineTo(elevL + drawL + soffitOffset * sin(angle), elevT + drawH - drawH - soffitOffset * cos(angle))
   603тЖТ   603тЖТ        path.lineTo(elevL - soffitOffset * sin(angle), elevT + drawH - soffitOffset)
   604тЖТ   604тЖТ        path.close()
   605тЖТ   605тЖТ
   606тЖТ   606тЖТ        canvas.drawPath(path, fillPaint(CONCRETE))
   607тЖТ   607тЖТ        canvas.drawPath(path, outlineP)
   608тЖТ   608тЖТ
   609тЖТ   609тЖТ        // Main reinforcement (following slope)
   610тЖТ   610тЖТ        val rebarP = createPaint(REBAR_BLUE, 2f)
   611тЖТ   611тЖТ        val nBars = 4
   612тЖТ   612тЖТ        for (i in 0 until nBars) {
   613тЖТ   613тЖТ            val offset = (i + 1) * soffitOffset / (nBars + 1)
   614тЖТ   614тЖТ            val yOff = offset * cos(angle)
   615тЖТ   615тЖТ            val xOff = offset * sin(angle)
   616тЖТ   616тЖТ            canvas.drawLine(
   617тЖТ   617тЖТ                elevL + xOff + 15f, elevT + drawH - yOff - 15f,
   618тЖТ   618тЖТ                elevL + drawL - xOff - 15f, elevT + drawH - drawH + yOff + 15f, rebarP
   619тЖТ   619тЖТ            )
   620тЖТ   620тЖТ        }
   621тЖТ   621тЖТ
   622тЖТ   622тЖТ        canvas.drawHDim(elevL, elevL + drawL, elevT + drawH + 20f, "${totalLength.toInt()} mm")
   623тЖТ   623тЖТ        canvas.drawVDim(elevT, elevT + drawH, elevL - 25f, "${totalHeight.toInt()} mm")
   624тЖТ   624тЖТ
   625тЖТ   625тЖТ        val titleP = textPaint(Color.WHITE, 22f, true)
   626тЖТ   626тЖТ        canvas.drawTextCentered("STAIRCASE ELEVATION", elevL + drawL / 2f, elevT - 25f, titleP)
   627тЖТ   627тЖТ
   628тЖТ   628тЖТ        // Section (right side)
   629тЖТ   629тЖТ        val secL = W * 0.55f; val secT = 100f
   630тЖТ   630тЖТ        val secW2 = 250f; val secH2 = slabThickness.toFloat() * scale * 3f
   631тЖТ   631тЖТ        canvas.drawRect(secL, secT, secL + secW2, secT + secH2, fillPaint(CONCRETE))
   632тЖТ   632тЖТ        canvas.drawRect(secL, secT, secL + secW2, secT + secH2, outlineP)
   633тЖТ   633тЖТ
   634тЖТ   634тЖТ        val bR = maxOf(mainDia.toFloat() * 0.3f, 3f)
   635тЖТ   635тЖТ        for (i in 0 until 5) {
   636тЖТ   636тЖТ            canvas.drawRebar(secL + 30f + i * 40f, secT + secH2 - 10f, bR, REBAR_BLUE)
   637тЖТ   637тЖТ        }
   638тЖТ   638тЖТ
   639тЖТ   639тЖТ        canvas.drawTextCentered("SECTION A-A", secL + secW2 / 2f, secT - 15f, textPaint(DIM_TEXT, 18f, true))
   640тЖТ   640тЖТ
   641тЖТ   641тЖТ        // Table
   642тЖТ   642тЖТ        drawRebarTable(canvas,
   643тЖТ   643тЖТ            x = 80f, y = H * 0.58f,
   644тЖТ   644тЖТ            data = listOf(
   645тЖТ   645тЖТ                listOf("Mark", "Dia", "Spacing", "Length"),
   646тЖТ   646тЖТ                listOf("B1", "${mainDia.toInt()}mm", "${mainSpacing.toInt()}mm c/c", "-"),
   647тЖТ   647тЖТ                listOf("D1", "${distDia.toInt()}mm", "${distSpacing.toInt()}mm c/c", "-")
   648тЖТ   648тЖТ            )
   649тЖТ   649тЖТ        )
   650тЖТ   650тЖТ
   651тЖТ   651тЖТ        drawTitleBlock(canvas, W - 280f, H - 60f, 280f, 60f, "Stair Detail")
   652тЖТ   652тЖТ        return bitmap
   653тЖТ   653тЖТ    }

   655тЖТ   655тЖТ    // ========== HELPER: Table ==========
   656тЖТ   656тЖТ    private fun drawRebarTable(canvas: Canvas, x: Float, y: Float, data: List<List<String>>) {
   657тЖТ   657тЖТ        if (data.isEmpty()) return
   658тЖТ   658тЖТ        val cols = data[0].size
   659тЖТ   659тЖТ        val colW = 130f
   660тЖТ   660тЖТ        val rowH = 28f
   661тЖТ   661тЖТ        val headerH = 32f
   662тЖТ   662тЖТ        val totalW = cols * colW
   663тЖТ   663тЖТ        val totalH = headerH + (data.size - 1) * rowH
   664тЖТ   664тЖТ
   665тЖТ   665тЖТ        // Background
   666тЖТ   666тЖТ        canvas.drawRect(x, y, x + totalW, y + totalH, fillPaint(Color.parseColor("#11FFFFFF")))
   667тЖТ   667тЖТ        // Header
   668тЖТ   668тЖТ        canvas.drawRect(x, y, x + totalW, y + headerH, fillPaint(TABLE_HEADER))
   669тЖТ   669тЖТ
   670тЖТ   670тЖТ        val tp = textPaint(DIM_TEXT, 16f)
   671тЖТ   671тЖТ        val boldTp = textPaint(DIM_TEXT, 16f, true)
   672тЖТ   672тЖТ
   673тЖТ   673тЖТ        // Headers
   674тЖТ   674тЖТ        for (c in 0 until cols) {
   675тЖТ   675тЖТ            canvas.drawTextCentered(data[0][c], x + c * colW + colW / 2f, y + headerH / 2f + 5f, boldTp)
   676тЖТ   676тЖТ        }
   677тЖТ   677тЖТ
   678тЖТ   678тЖТ        // Rows
   679тЖТ   679тЖТ        for (r in 1 until data.size) {
   680тЖТ   680тЖТ            val ry = y + headerH + (r - 1) * rowH
   681тЖТ   681тЖТ            if (r % 2 == 0) canvas.drawRect(x, ry, x + totalW, ry + rowH, fillPaint(TABLE_ALT))
   682тЖТ   682тЖТ            for (c in 0 until cols) {
   683тЖТ   683тЖТ                canvas.drawTextCentered(data[r][c], x + c * colW + colW / 2f, ry + rowH / 2f + 5f, tp)
   684тЖТ   684тЖТ            }
   685тЖТ   685тЖТ        }
   686тЖТ   686тЖТ
   687тЖТ   687тЖТ        // Border
   688тЖТ   688тЖТ        canvas.drawRect(x, y, x + totalW, y + totalH, createPaint(Color.parseColor("#66FFFFFF"), 1.5f))
   689тЖТ   689тЖТ    }
