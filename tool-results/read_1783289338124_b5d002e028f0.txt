   280→   280→@Composable
   281→   281→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   282→   282→    Card(
   283→   283→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   284→   284→        border = BorderStroke(1.dp, Color.LightGray)
   285→   285→    ) {
   286→   286→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   287→   287→            val padding = 40.dp.toPx()
   288→   288→            val w = size.width - 2 * padding
   289→   289→            val h = size.height - 2 * padding
   290→   290→            val baseY = size.height - padding
   291→   291→            
   292→   292→            // Scale calculations
   293→   293→            val scale = (w / inputs.span).toFloat()
   294→   294→            val eh = (inputs.eaveHeight * scale).toFloat()
   295→   295→            val rh = (inputs.ridgeHeight * scale).toFloat()
   296→   296→            val midX = padding + w / 2
   297→   297→            
   298→   298→            // Draw Main Frame Outline
   299→   299→            val framePath = Path().apply {
   300→   300→                moveTo(padding, baseY)
   301→   301→                lineTo(padding, baseY - eh)
   302→   302→                lineTo(midX, baseY - rh)
   303→   303→                lineTo(padding + w, baseY - eh)
   304→   304→                lineTo(padding + w, baseY)
   305→   305→            }
   306→   306→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   307→   307→            
   308→   308→            // Draw Distributed Load Arrows
   309→   309→            val loadColor = Color(0xFFE53935)
   310→   310→            for (i in 0..12) {
   311→   311→                val x = padding + (i * w / 12)
   312→   312→                val y = if (x <= midX) {
   313→   313→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   314→   314→                } else {
   315→   315→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   316→   316→                }
   317→   317→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   318→   318→                // Arrow heads
   319→   319→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   320→   320→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   321→   321→            }
   322→   322→            
   323→   323→            // Draw Moment Diagram (Simplified)
   324→   324→            val momentColor = Color(0xFF1E88E5)
   325→   325→            val momentPath = Path().apply {
   326→   326→                moveTo(padding, baseY - eh)
   327→   327→                // Simulated parabolic curve for moment
   328→   328→                cubicTo(
   329→   329→                    padding + w * 0.25f, baseY - eh + 50f,
   330→   330→                    padding + w * 0.75f, baseY - eh + 50f,
   331→   331→                    padding + w, baseY - eh
   332→   332→                )
   333→   333→            }
   334→   334→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   335→   335→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   336→   336→            
   337→   337→            // Labels
   338→   338→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   339→   339→        }
   340→   340→    }
   341→   341→}

   432→   432→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   433→   433→    val planW = (inputs.span * scale).toFloat()
   434→   434→    val planL = (inputs.length * scale).toFloat()
   435→   435→    val planTop = baseY - planL

   443→   443→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()

   445→   445→        // Columns markers
   446→   446→        drawCircle(Color.Black, 4f, Offset(startX, y))
   447→   447→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))

   455→   455→

   469→   469→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   470→   470→    val s = scale.toFloat() * 0.65f
   471→   471→    val depthOff = 35f
   472→   472→    
   473→   473→    // Draw from back to front for better perspective look
   474→   474→    for (i in 4 downTo 0) {
   475→   475→        val off = i * depthOff
   476→   476→        val x0 = startX + off
   477→   477→        val y0 = baseY - off
   478→   478→        val spanW = (inputs.span * s).toFloat()
   479→   479→        val eaveH = (inputs.eaveHeight * s).toFloat()
   480→   480→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   481→   481→        
   482→   482→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   483→   483→        val stroke = if (i == 0) 3f else 1.5f
   484→   484→        
   485→   485→        // Sections of the frame
   486→   486→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   487→   487→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   488→   488→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   489→   489→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   490→   490→        
   491→   491→        // Purlins/Girts connections (longitudinal)
   492→   492→        if (i > 0) {
   493→   493→            val px = x0 - depthOff
   494→   494→            val py = y0 + depthOff
   495→   495→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   496→   496→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   497→   497→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   498→   498→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   499→   499→        }
   500→   500→    }
   501→   501→}

   535→   535→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   536→   536→    val p = Paint().apply { 
   537→   537→        color = android.graphics.Color.BLACK
   538→   538→        strokeWidth = 6f
   539→   539→        isAntiAlias = true
   540→   540→        style = Paint.Style.STROKE
   541→   541→    }
   542→   542→    val dimPaint = Paint().apply {
   543→   543→        color = android.graphics.Color.DKGRAY
   544→   544→        strokeWidth = 2f
   545→   545→        textSize = 24f
   546→   546→        isAntiAlias = true
   547→   547→    }
   548→   548→    val textP = Paint().apply {
   549→   549→        color = android.graphics.Color.BLACK
   550→   550→        textSize = 22f
   551→   551→        isFakeBoldText = true
   552→   552→        textAlign = Paint.Align.CENTER
   553→   553→    }
   554→   554→    
   555→   555→    val s = scale * 0.7f
   556→   556→    val w = inputs.span.toFloat() * s
   557→   557→    val eh = inputs.eaveHeight.toFloat() * s
   558→   558→    val rh = inputs.ridgeHeight.toFloat() * s
   559→   559→    val midX = x + w/2
   560→   560→    
   561→   561→    // Draw Main Frame (Columns & Rafters)
   562→   562→    canvas.drawLine(x, y, x, y - eh, p)
   563→   563→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   564→   564→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   565→   565→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   566→   566→    
   567→   567→    // Foundations
   568→   568→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   569→   569→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   570→   570→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   571→   571→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   572→   572→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   573→   573→    p.strokeWidth = 6f
   574→   574→    
   575→   575→    // --- Dimension Lines ---
   576→   576→    // Span Dimension
   577→   577→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   578→   578→    
   579→   579→    // Eave Height Dimension
   580→   580→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   581→   581→    
   582→   582→    // Ridge Height Dimension
   583→   583→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   584→   584→
   585→   585→    // --- Section Labels ---
   586→   586→    result?.let { res ->
   587→   587→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   588→   588→        
   589→   589→        // Column Label
   590→   590→        canvas.save()
   591→   591→        canvas.rotate(-90f, x + 25f, y - eh/2)
   592→   592→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   593→   593→        canvas.restore()
   594→   594→        
   595→   595→        // Rafter Label
   596→   596→        canvas.save()
   597→   597→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   598→   598→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   599→   599→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   600→   600→        canvas.restore()
   601→   601→    }
   602→   602→
   603→   603→    // Connection Markers (Circles at joints)
   604→   604→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   605→   605→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   606→   606→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   607→   607→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   608→   608→}

   610→   610→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   611→   611→    val tickSize = 15f
   612→   612→    canvas.drawLine(x1, y1, x2, y2, paint)
   613→   613→    
   614→   614→    // Draw Ticks (45 degree lines)
   615→   615→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   616→   616→    val tickAngle = angle + Math.PI / 4
   617→   617→    
   618→   618→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   619→   619→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   620→   620→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   621→   621→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   622→   622→
   623→   623→    // Text label
   624→   624→    val midX = (x1 + x2) / 2
   625→   625→    val midY = (y1 + y2) / 2
   626→   626→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   627→   627→    
   628→   628→    if (x1 == x2) { // Vertical
   629→   629→        canvas.save()
   630→   630→        canvas.rotate(-90f, midX - 10f, midY)
   631→   631→        canvas.drawText(text, midX - 10f, midY, textPaint)
   632→   632→        canvas.restore()
   633→   633→    } else { // Horizontal
   634→   634→        canvas.drawText(text, midX, midY - 10f, textPaint)
   635→   635→    }
   636→   636→}

   638→   638→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   639→   639→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   640→   640→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   641→   641→    
   642→   642→    val s = scale * 0.6f
   643→   643→    val w = inputs.span.toFloat() * s
   644→   644→    val l = inputs.length.toFloat() * s
   645→   645→    val planTop = y - l
   646→   646→    
   647→   647→    canvas.drawRect(x, planTop, x + w, y, p)
   648→   648→    
   649→   649→    val baySpacing = inputs.baySpacing.toFloat() * s
   650→   650→    var curY = planTop
   651→   651→    var count = 0
   652→   652→    while (curY <= y + 0.1f) {
   653→   653→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   654→   654→        
   655→   655→        // Column markers
   656→   656→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   657→   657→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   658→   658→        
   659→   659→        if (curY + baySpacing <= y + 0.1f) {
   660→   660→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   661→   661→        }
   662→   662→        
   663→   663→        curY += baySpacing
   664→   664→        count++
   665→   665→    }
   666→   666→    
   667→   667→    // Span Dimension
   668→   668→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   669→   669→    // Total Length Dimension
   670→   670→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   671→   671→}

   673→   673→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   674→   674→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   675→   675→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   676→   676→    
   677→   677→    val s = scale * 0.6f
   678→   678→    val l = inputs.length.toFloat() * s
   679→   679→    val eh = inputs.eaveHeight.toFloat() * s
   680→   680→    
   681→   681→    canvas.drawLine(x, y, x + l, y, p) // Ground
   682→   682→    val bay = inputs.baySpacing.toFloat() * s
   683→   683→    var cur = x
   684→   684→    while (cur <= x + l + 0.1f) {
   685→   685→        canvas.drawLine(cur, y, cur, y - eh, p)
   686→   686→        cur += bay
   687→   687→    }
   688→   688→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   689→   689→    
   690→   690→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   691→   691→}

  1344→  1344→fun SteelSectionDrawing(section: SteelSectionType) {
  1345→  1345→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1346→  1346→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1347→  1347→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1348→  1348→                val w = size.width
  1349→  1349→                val h = size.height
  1350→  1350→                
  1351→  1351→                when (section) {
  1352→  1352→                    is SteelSectionType.ISection -> {
  1353→  1353→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1354→  1354→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1355→  1355→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1356→  1356→                        val offsetBF = (w - bf) / 2
  1357→  1357→                        
  1358→  1358→                        // Top Flange
  1359→  1359→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1360→  1360→                        // Bottom Flange
  1361→  1361→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1362→  1362→                        // Web
  1363→  1363→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1364→  1364→                        
  1365→  1365→                        // Stress Distribution (Schematic)
  1366→  1366→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1367→  1367→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1368→  1368→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1369→  1369→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1370→  1370→                    }
  1371→  1371→                    is SteelSectionType.LSection -> {
  1372→  1372→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
  1373→  1373→                        drawRect(Color.DarkGray, Offset(0f, 0f), Size(t, h))
  1374→  1374→                        drawRect(Color.DarkGray, Offset(t, h - t), Size(w - t, t))
  1375→  1375→                    }
  1376→  1376→                    is SteelSectionType.RHS -> {
  1377→  1377→                        val t = (section.thickness / section.height * h).toFloat().coerceIn(5f, 20f)
  1378→  1378→                        drawRect(Color.DarkGray, Offset(0f, 0f), size, style = androidx.compose.ui.graphics.drawscope.Stroke(t))
  1379→  1379→                    }
  1380→  1380→                    else -> {
  1381→  1381→                        drawRect(Color.DarkGray, Offset(0f, 0f), size)
  1382→  1382→                    }
  1383→  1383→                }
  1384→  1384→            }
  1385→  1385→            Spacer(Modifier.height(12.dp))
  1386→  1386→            Text("توزيع الإجهادات المرن", style = MaterialTheme.typography.labelSmall, color = Color.Gray)
  1387→  1387→        }
  1388→  1388→    }
