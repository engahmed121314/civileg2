   284→   284→@Composable
   285→   285→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   286→   286→    Card(
   287→   287→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   288→   288→        border = BorderStroke(1.dp, Color.LightGray)
   289→   289→    ) {
   290→   290→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   291→   291→            val padding = 40.dp.toPx()
   292→   292→            val w = size.width - 2 * padding
   293→   293→            val h = size.height - 2 * padding
   294→   294→            val baseY = size.height - padding
   295→   295→            
   296→   296→            // Scale calculations
   297→   297→            val scale = (w / inputs.span).toFloat()
   298→   298→            val eh = (inputs.eaveHeight * scale).toFloat()
   299→   299→            val rh = (inputs.ridgeHeight * scale).toFloat()
   300→   300→            val midX = padding + w / 2
   301→   301→            
   302→   302→            // Draw Main Frame Outline
   303→   303→            val framePath = Path().apply {
   304→   304→                moveTo(padding, baseY)
   305→   305→                lineTo(padding, baseY - eh)
   306→   306→                lineTo(midX, baseY - rh)
   307→   307→                lineTo(padding + w, baseY - eh)
   308→   308→                lineTo(padding + w, baseY)
   309→   309→            }
   310→   310→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   311→   311→            
   312→   312→            // Draw Distributed Load Arrows
   313→   313→            val loadColor = Color(0xFFE53935)
   314→   314→            for (i in 0..12) {
   315→   315→                val x = padding + (i * w / 12)
   316→   316→                val y = if (x <= midX) {
   317→   317→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   318→   318→                } else {
   319→   319→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   320→   320→                }
   321→   321→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   322→   322→                // Arrow heads
   323→   323→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   324→   324→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   325→   325→            }
   326→   326→            
   327→   327→            // Draw Moment Diagram (Simplified)
   328→   328→            val momentColor = Color(0xFF1E88E5)
   329→   329→            val momentPath = Path().apply {
   330→   330→                moveTo(padding, baseY - eh)
   331→   331→                // Simulated parabolic curve for moment
   332→   332→                cubicTo(
   333→   333→                    padding + w * 0.25f, baseY - eh + 50f,
   334→   334→                    padding + w * 0.75f, baseY - eh + 50f,
   335→   335→                    padding + w, baseY - eh
   336→   336→                )
   337→   337→            }
   338→   338→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   339→   339→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   340→   340→            
   341→   341→            // Labels
   342→   342→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   343→   343→        }
   344→   344→    }
   345→   345→}

   376→   376→    val views = listOf("الواجهة الأمامية", "المسقط الأفقي", "الواجهة الجانبية", "رسم ثلاثي الأبعاد")

   390→   390→                val padding = 50.dp.toPx()
   391→   391→                val scale = minOf(
   392→   392→                    (size.width - 2 * padding) / max(inputs.span, inputs.length),
   393→   393→                    (size.height - 2 * padding) / max(inputs.ridgeHeight, inputs.span)
   394→   394→                )
   395→   395→                val startX = padding
   396→   396→                val baseY = size.height - padding
   397→   397→
   398→   398→                when (viewMode) {
   399→   399→                    0 -> { // Front Elevation
   400→   400→                        drawFrontElevation(inputs, scale, startX, baseY)
   401→   401→                    }
   402→   402→                    1 -> { // Plan View

   404→   404→                    }
   405→   405→                    2 -> { // Side View

   417→   417→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawFrontElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {

   419→   419→    val colRX = startX + (inputs.span * scale).toFloat()
   420→   420→    val eaveY = baseY - (inputs.eaveHeight * scale).toFloat()
   421→   421→    val ridgeY = baseY - (inputs.ridgeHeight * scale).toFloat()

   429→   429→    drawLine(Color.Black, Offset(colRX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   430→   430→    
   431→   431→    // Foundations
   432→   432→    drawRect(Color.Gray, Offset(colLX - 15f, baseY), Size(30f, 15f))
   433→   433→    drawRect(Color.Gray, Offset(colRX - 15f, baseY), Size(30f, 15f))
   434→   434→}

   436→   436→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   437→   437→    val planW = (inputs.span * scale).toFloat()
   438→   438→    val planL = (inputs.length * scale).toFloat()
   439→   439→    val planTop = baseY - planL
   440→   440→    
   441→   441→    // Boundary
   442→   442→    drawRect(Color.Black, Offset(startX, planTop), Size(planW, planL), style = androidx.compose.ui.graphics.drawscope.Stroke(4f))
   443→   443→    
   444→   444→    // Grid Lines (Bays)
   445→   445→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   446→   446→    for (i in 0..numBays) {
   447→   447→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()
   448→   448→        drawLine(Color.LightGray, Offset(startX, y), Offset(startX + planW, y), strokeWidth = 1f)
   449→   449→        // Columns markers
   450→   450→        drawCircle(Color.Black, 4f, Offset(startX, y))
   451→   451→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))
   452→   452→    }
   453→   453→}

   455→   455→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSideElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   456→   456→    val planL = (inputs.length * scale).toFloat()
   457→   457→    val eaveH = (inputs.eaveHeight * scale).toFloat()
   458→   458→    val eaveY = baseY - eaveH
   459→   459→    
   460→   460→    // Ground
   461→   461→    drawLine(Color.Gray, Offset(startX, baseY), Offset(startX + planL, baseY), strokeWidth = 2f)
   462→   462→    
   463→   463→    // Columns
   464→   464→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   465→   465→    for (i in 0..numBays) {
   466→   466→        val x = startX + (i * inputs.baySpacing * scale).toFloat()
   467→   467→        drawLine(Color.Black, Offset(x, baseY), Offset(x, eaveY), strokeWidth = 6f)
   468→   468→    }
   469→   469→    // Eave Line
   470→   470→    drawLine(Color.Black, Offset(startX, eaveY), Offset(startX + planL, eaveY), strokeWidth = 6f)
   471→   471→}

   473→   473→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   474→   474→    val s = scale.toFloat() * 0.65f
   475→   475→    val depthOff = 35f
   476→   476→    
   477→   477→    // Draw from back to front for better perspective look
   478→   478→    for (i in 4 downTo 0) {
   479→   479→        val off = i * depthOff
   480→   480→        val x0 = startX + off
   481→   481→        val y0 = baseY - off
   482→   482→        val spanW = (inputs.span * s).toFloat()
   483→   483→        val eaveH = (inputs.eaveHeight * s).toFloat()
   484→   484→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   485→   485→        
   486→   486→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   487→   487→        val stroke = if (i == 0) 3f else 1.5f
   488→   488→        
   489→   489→        // Sections of the frame
   490→   490→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   491→   491→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   492→   492→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   493→   493→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   494→   494→        
   495→   495→        // Purlins/Girts connections (longitudinal)
   496→   496→        if (i > 0) {
   497→   497→            val px = x0 - depthOff
   498→   498→            val py = y0 + depthOff
   499→   499→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   500→   500→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   501→   501→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   502→   502→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   503→   503→        }
   504→   504→    }
   505→   505→}

   539→   539→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   540→   540→    val p = Paint().apply { 
   541→   541→        color = android.graphics.Color.BLACK
   542→   542→        strokeWidth = 6f
   543→   543→        isAntiAlias = true
   544→   544→        style = Paint.Style.STROKE
   545→   545→    }
   546→   546→    val dimPaint = Paint().apply {
   547→   547→        color = android.graphics.Color.DKGRAY
   548→   548→        strokeWidth = 2f
   549→   549→        textSize = 24f
   550→   550→        isAntiAlias = true
   551→   551→    }
   552→   552→    val textP = Paint().apply {
   553→   553→        color = android.graphics.Color.BLACK
   554→   554→        textSize = 22f
   555→   555→        isFakeBoldText = true
   556→   556→        textAlign = Paint.Align.CENTER
   557→   557→    }
   558→   558→    
   559→   559→    val s = scale * 0.7f
   560→   560→    val w = inputs.span.toFloat() * s
   561→   561→    val eh = inputs.eaveHeight.toFloat() * s
   562→   562→    val rh = inputs.ridgeHeight.toFloat() * s
   563→   563→    val midX = x + w/2
   564→   564→    
   565→   565→    // Draw Main Frame (Columns & Rafters)
   566→   566→    canvas.drawLine(x, y, x, y - eh, p)
   567→   567→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   568→   568→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   569→   569→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   570→   570→    
   571→   571→    // Foundations
   572→   572→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   573→   573→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   574→   574→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   575→   575→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   576→   576→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   577→   577→    p.strokeWidth = 6f
   578→   578→    
   579→   579→    // --- Dimension Lines ---
   580→   580→    // Span Dimension
   581→   581→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   582→   582→    
   583→   583→    // Eave Height Dimension
   584→   584→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   585→   585→    
   586→   586→    // Ridge Height Dimension
   587→   587→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   588→   588→
   589→   589→    // --- Section Labels ---
   590→   590→    result?.let { res ->
   591→   591→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   592→   592→        
   593→   593→        // Column Label
   594→   594→        canvas.save()
   595→   595→        canvas.rotate(-90f, x + 25f, y - eh/2)
   596→   596→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   597→   597→        canvas.restore()
   598→   598→        
   599→   599→        // Rafter Label
   600→   600→        canvas.save()
   601→   601→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   602→   602→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   603→   603→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   604→   604→        canvas.restore()
   605→   605→    }
   606→   606→
   607→   607→    // Connection Markers (Circles at joints)
   608→   608→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   609→   609→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   610→   610→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   611→   611→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   612→   612→}

   614→   614→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   615→   615→    val tickSize = 15f
   616→   616→    canvas.drawLine(x1, y1, x2, y2, paint)
   617→   617→    
   618→   618→    // Draw Ticks (45 degree lines)
   619→   619→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   620→   620→    val tickAngle = angle + Math.PI / 4
   621→   621→    
   622→   622→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   623→   623→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   624→   624→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   625→   625→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   626→   626→
   627→   627→    // Text label
   628→   628→    val midX = (x1 + x2) / 2
   629→   629→    val midY = (y1 + y2) / 2
   630→   630→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   631→   631→    
   632→   632→    if (x1 == x2) { // Vertical
   633→   633→        canvas.save()
   634→   634→        canvas.rotate(-90f, midX - 10f, midY)
   635→   635→        canvas.drawText(text, midX - 10f, midY, textPaint)
   636→   636→        canvas.restore()
   637→   637→    } else { // Horizontal
   638→   638→        canvas.drawText(text, midX, midY - 10f, textPaint)
   639→   639→    }
   640→   640→}

   642→   642→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   643→   643→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   644→   644→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   645→   645→    
   646→   646→    val s = scale * 0.6f
   647→   647→    val w = inputs.span.toFloat() * s
   648→   648→    val l = inputs.length.toFloat() * s
   649→   649→    val planTop = y - l
   650→   650→    
   651→   651→    canvas.drawRect(x, planTop, x + w, y, p)
   652→   652→    
   653→   653→    val baySpacing = inputs.baySpacing.toFloat() * s
   654→   654→    var curY = planTop
   655→   655→    var count = 0
   656→   656→    while (curY <= y + 0.1f) {
   657→   657→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   658→   658→        
   659→   659→        // Column markers
   660→   660→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   661→   661→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   662→   662→        
   663→   663→        if (curY + baySpacing <= y + 0.1f) {
   664→   664→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   665→   665→        }
   666→   666→        
   667→   667→        curY += baySpacing
   668→   668→        count++
   669→   669→    }
   670→   670→    
   671→   671→    // Span Dimension
   672→   672→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   673→   673→    // Total Length Dimension
   674→   674→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   675→   675→}

   677→   677→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   678→   678→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   679→   679→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   680→   680→    
   681→   681→    val s = scale * 0.6f
   682→   682→    val l = inputs.length.toFloat() * s
   683→   683→    val eh = inputs.eaveHeight.toFloat() * s
   684→   684→    
   685→   685→    canvas.drawLine(x, y, x + l, y, p) // Ground
   686→   686→    val bay = inputs.baySpacing.toFloat() * s
   687→   687→    var cur = x
   688→   688→    while (cur <= x + l + 0.1f) {
   689→   689→        canvas.drawLine(cur, y, cur, y - eh, p)
   690→   690→        cur += bay
   691→   691→    }
   692→   692→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   693→   693→    
   694→   694→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   695→   695→}

  1348→  1348→fun SteelSectionDrawing(section: SteelSectionType) {
  1349→  1349→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1350→  1350→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1351→  1351→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1352→  1352→                val w = size.width
  1353→  1353→                val h = size.height
  1354→  1354→                
  1355→  1355→                when (section) {
  1356→  1356→                    is SteelSectionType.ISection -> {
  1357→  1357→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1358→  1358→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1359→  1359→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1360→  1360→                        val offsetBF = (w - bf) / 2
  1361→  1361→                        
  1362→  1362→                        // Top Flange
  1363→  1363→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1364→  1364→                        // Bottom Flange
  1365→  1365→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1366→  1366→                        // Web
  1367→  1367→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1368→  1368→                        
  1369→  1369→                        // Stress Distribution (Schematic)
  1370→  1370→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1371→  1371→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1372→  1372→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1373→  1373→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1374→  1374→                    }
  1375→  1375→                    is SteelSectionType.LSection -> {
  1376→  1376→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
  1377→  1377→                        drawRect(Color.DarkGray, Offset(0f, 0f), Size(t, h))
  1378→  1378→                        drawRect(Color.DarkGray, Offset(t, h - t), Size(w - t, t))
  1379→  1379→                    }
  1380→  1380→                    is SteelSectionType.RHS -> {
  1381→  1381→                        val t = (section.thickness / section.height * h).toFloat().coerceIn(5f, 20f)
  1382→  1382→                        drawRect(Color.DarkGray, Offset(0f, 0f), size, style = androidx.compose.ui.graphics.drawscope.Stroke(t))
  1383→  1383→                    }
  1384→  1384→                    else -> {
  1385→  1385→                        drawRect(Color.DarkGray, Offset(0f, 0f), size)
  1386→  1386→                    }
  1387→  1387→                }
  1388→  1388→            }
  1389→  1389→            Spacer(Modifier.height(12.dp))
  1390→  1390→            Text("توزيع الإجهادات المرن", style = MaterialTheme.typography.labelSmall, color = Color.Gray)
  1391→  1391→        }
  1392→  1392→    }
