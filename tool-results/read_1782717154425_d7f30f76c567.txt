   272→@Composable
   273→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   274→    Card(
   275→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   276→        border = BorderStroke(1.dp, Color.LightGray)
   277→    ) {
   278→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   279→            val padding = 40.dp.toPx()
   280→            val w = size.width - 2 * padding
   281→            val h = size.height - 2 * padding
   282→            val baseY = size.height - padding
   283→            
   284→            // Scale calculations
   285→            val scale = (w / inputs.span).toFloat()
   286→            val eh = (inputs.eaveHeight * scale).toFloat()
   287→            val rh = (inputs.ridgeHeight * scale).toFloat()
   288→            val midX = padding + w / 2
   289→            
   290→            // Draw Main Frame Outline
   291→            val framePath = Path().apply {
   292→                moveTo(padding, baseY)
   293→                lineTo(padding, baseY - eh)
   294→                lineTo(midX, baseY - rh)
   295→                lineTo(padding + w, baseY - eh)
   296→                lineTo(padding + w, baseY)
   297→            }
   298→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   299→            
   300→            // Draw Distributed Load Arrows
   301→            val loadColor = Color(0xFFE53935)
   302→            for (i in 0..12) {
   303→                val x = padding + (i * w / 12)
   304→                val y = if (x <= midX) {
   305→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   306→                } else {
   307→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   308→                }
   309→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   310→                // Arrow heads
   311→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   312→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   313→            }
   314→            
   315→            // Draw Moment Diagram (Simplified)
   316→            val momentColor = Color(0xFF1E88E5)
   317→            val momentPath = Path().apply {
   318→                moveTo(padding, baseY - eh)
   319→                // Simulated parabolic curve for moment
   320→                cubicTo(
   321→                    padding + w * 0.25f, baseY - eh + 50f,
   322→                    padding + w * 0.75f, baseY - eh + 50f,
   323→                    padding + w, baseY - eh
   324→                )
   325→            }
   326→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   327→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   328→            
   329→            // Labels
   330→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   331→        }
   332→    }
   333→}

   362→fun SteelWarehouseVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   363→    var viewMode by remember { mutableStateOf(0) } // 0: Front, 1: Plan, 2: Side, 3: 3D
   364→    val views = listOf("Front Elevation", "Plan View", "Side View", "3D Sketch")
   365→
   366→    Column {
   367→        ScrollableTabRow(selectedTabIndex = viewMode, edgePadding = 0.dp, containerColor = Color.Transparent) {
   368→            views.forEachIndexed { index, title ->
   369→                Tab(selected = viewMode == index, onClick = { viewMode = index }, text = { Text(title, fontSize = 10.sp) })
   370→            }
   371→        }
   372→
   373→        Card(
   374→            modifier = Modifier.fillMaxWidth().height(350.dp).padding(vertical = 8.dp),
   375→            border = BorderStroke(1.dp, Color.LightGray)
   376→        ) {
   377→            ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color.White)) {
   378→                val padding = 50.dp.toPx()
   379→                val scale = minOf(
   380→                    (size.width - 2 * padding) / max(inputs.span, inputs.length),
   381→                    (size.height - 2 * padding) / max(inputs.ridgeHeight, inputs.span)
   382→                )
   383→                val startX = padding
   384→                val baseY = size.height - padding
   385→
   386→                when (viewMode) {
   387→                    0 -> { // Front Elevation
   388→                        drawFrontElevation(inputs, scale, startX, baseY)
   389→                    }
   390→                    1 -> { // Plan View
   391→                        drawPlanView(inputs, scale, startX, baseY)
   392→                    }
   393→                    2 -> { // Side View
   394→                        drawSideElevation(inputs, scale, startX, baseY)
   395→                    }
   396→                    3 -> { // 3D Perspective
   397→                        draw3DView(inputs, scale, startX, baseY)
   398→                    }
   399→                }
   400→            }
   401→        }
   402→    }
   403→}

   405→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawFrontElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   406→    val colLX = startX
   407→    val colRX = startX + (inputs.span * scale).toFloat()
   408→    val eaveY = baseY - (inputs.eaveHeight * scale).toFloat()
   409→    val ridgeY = baseY - (inputs.ridgeHeight * scale).toFloat()
   410→    val midX = startX + (inputs.span * scale / 2).toFloat()
   411→
   412→    // Columns
   413→    drawLine(Color.Black, Offset(colLX, baseY), Offset(colLX, eaveY), strokeWidth = 8f)
   414→    drawLine(Color.Black, Offset(colRX, baseY), Offset(colRX, eaveY), strokeWidth = 8f)
   415→    // Rafters
   416→    drawLine(Color.Black, Offset(colLX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   417→    drawLine(Color.Black, Offset(colRX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   418→    
   419→    // Foundations
   420→    drawRect(Color.Gray, Offset(colLX - 15f, baseY), Size(30f, 15f))
   421→    drawRect(Color.Gray, Offset(colRX - 15f, baseY), Size(30f, 15f))
   422→}
   423→
   424→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   425→    val planW = (inputs.span * scale).toFloat()
   426→    val planL = (inputs.length * scale).toFloat()
   427→    val planTop = baseY - planL
   428→    
   429→    // Boundary
   430→    drawRect(Color.Black, Offset(startX, planTop), Size(planW, planL), style = androidx.compose.ui.graphics.drawscope.Stroke(4f))
   431→    
   432→    // Grid Lines (Bays)
   433→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   434→    for (i in 0..numBays) {
   435→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()
   436→        drawLine(Color.LightGray, Offset(startX, y), Offset(startX + planW, y), strokeWidth = 1f)
   437→        // Columns markers
   438→        drawCircle(Color.Black, 4f, Offset(startX, y))
   439→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))
   440→    }
   441→}
   442→
   443→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSideElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   444→    val planL = (inputs.length * scale).toFloat()
   445→    val eaveH = (inputs.eaveHeight * scale).toFloat()
   446→    val eaveY = baseY - eaveH
   447→    
   448→    // Ground
   449→    drawLine(Color.Gray, Offset(startX, baseY), Offset(startX + planL, baseY), strokeWidth = 2f)
   450→    
   451→    // Columns
   452→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   453→    for (i in 0..numBays) {
   454→        val x = startX + (i * inputs.baySpacing * scale).toFloat()
   455→        drawLine(Color.Black, Offset(x, baseY), Offset(x, eaveY), strokeWidth = 6f)
   456→    }
   457→    // Eave Line
   458→    drawLine(Color.Black, Offset(startX, eaveY), Offset(startX + planL, eaveY), strokeWidth = 6f)
   459→}

   461→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   462→    val s = scale.toFloat() * 0.65f
   463→    val depthOff = 35f
   464→    
   465→    // Draw from back to front for better perspective look
   466→    for (i in 4 downTo 0) {
   467→        val off = i * depthOff
   468→        val x0 = startX + off
   469→        val y0 = baseY - off
   470→        val spanW = (inputs.span * s).toFloat()
   471→        val eaveH = (inputs.eaveHeight * s).toFloat()
   472→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   473→        
   474→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   475→        val stroke = if (i == 0) 3f else 1.5f
   476→        
   477→        // Sections of the frame
   478→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   479→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   480→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   481→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   482→        
   483→        // Purlins/Girts connections (longitudinal)
   484→        if (i > 0) {
   485→            val px = x0 - depthOff
   486→            val py = y0 + depthOff
   487→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   488→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   489→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   490→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   491→        }
   492→    }

   527→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   528→    val p = Paint().apply { 
   529→        color = android.graphics.Color.BLACK
   530→        strokeWidth = 6f
   531→        isAntiAlias = true
   532→        style = Paint.Style.STROKE
   533→    }
   534→    val dimPaint = Paint().apply {
   535→        color = android.graphics.Color.DKGRAY
   536→        strokeWidth = 2f
   537→        textSize = 24f
   538→        isAntiAlias = true
   539→    }
   540→    val textP = Paint().apply {
   541→        color = android.graphics.Color.BLACK
   542→        textSize = 22f
   543→        isFakeBoldText = true
   544→        textAlign = Paint.Align.CENTER
   545→    }
   546→    
   547→    val s = scale * 0.7f
   548→    val w = inputs.span.toFloat() * s
   549→    val eh = inputs.eaveHeight.toFloat() * s
   550→    val rh = inputs.ridgeHeight.toFloat() * s
   551→    val midX = x + w/2
   552→    
   553→    // Draw Main Frame (Columns & Rafters)
   554→    canvas.drawLine(x, y, x, y - eh, p)
   555→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   556→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   557→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   558→    
   559→    // Foundations
   560→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   561→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   562→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   563→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   564→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   565→    p.strokeWidth = 6f
   566→    
   567→    // --- Dimension Lines ---
   568→    // Span Dimension
   569→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   570→    
   571→    // Eave Height Dimension
   572→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   573→    
   574→    // Ridge Height Dimension
   575→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   576→
   577→    // --- Section Labels ---
   578→    result?.let { res ->
   579→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   580→        
   581→        // Column Label
   582→        canvas.save()
   583→        canvas.rotate(-90f, x + 25f, y - eh/2)
   584→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   585→        canvas.restore()
   586→        
   587→        // Rafter Label
   588→        canvas.save()
   589→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   590→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   591→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   592→        canvas.restore()
   593→    }
   594→
   595→    // Connection Markers (Circles at joints)
   596→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   597→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   598→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   599→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   600→}

   602→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   603→    val tickSize = 15f
   604→    canvas.drawLine(x1, y1, x2, y2, paint)
   605→    
   606→    // Draw Ticks (45 degree lines)
   607→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   608→    val tickAngle = angle + Math.PI / 4
   609→    
   610→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   611→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   612→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   613→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   614→
   615→    // Text label
   616→    val midX = (x1 + x2) / 2
   617→    val midY = (y1 + y2) / 2
   618→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   619→    
   620→    if (x1 == x2) { // Vertical
   621→        canvas.save()
   622→        canvas.rotate(-90f, midX - 10f, midY)
   623→        canvas.drawText(text, midX - 10f, midY, textPaint)
   624→        canvas.restore()
   625→    } else { // Horizontal
   626→        canvas.drawText(text, midX, midY - 10f, textPaint)
   627→    }
   628→}

   630→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   631→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   632→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   633→    
   634→    val s = scale * 0.6f
   635→    val w = inputs.span.toFloat() * s
   636→    val l = inputs.length.toFloat() * s
   637→    val planTop = y - l
   638→    
   639→    canvas.drawRect(x, planTop, x + w, y, p)
   640→    
   641→    val baySpacing = inputs.baySpacing.toFloat() * s
   642→    var curY = planTop
   643→    var count = 0
   644→    while (curY <= y + 0.1f) {
   645→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   646→        
   647→        // Column markers
   648→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   649→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   650→        
   651→        if (curY + baySpacing <= y + 0.1f) {
   652→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   653→        }
   654→        
   655→        curY += baySpacing
   656→        count++
   657→    }
   658→    
   659→    // Span Dimension
   660→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "Span: ${inputs.span}m", dimPaint)
   661→    // Total Length Dimension
   662→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "Total: ${inputs.length}m", dimPaint)
   663→}

   665→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   666→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   667→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   668→    
   669→    val s = scale * 0.6f
   670→    val l = inputs.length.toFloat() * s
   671→    val eh = inputs.eaveHeight.toFloat() * s
   672→    
   673→    canvas.drawLine(x, y, x + l, y, p) // Ground
   674→    val bay = inputs.baySpacing.toFloat() * s
   675→    var cur = x
   676→    while (cur <= x + l + 0.1f) {
   677→        canvas.drawLine(cur, y, cur, y - eh, p)
   678→        cur += bay
   679→    }
   680→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   681→    
   682→    canvas.drawText("Total Length: ${inputs.length}m", x + l/2, y + 35f, textP)
   683→}

  1217→@Composable
  1218→fun SteelSectionDrawing(section: SteelSectionType) {
  1219→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1220→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1221→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1222→                val w = size.width
  1223→                val h = size.height
  1224→                
  1225→                when (section) {
  1226→                    is SteelSectionType.ISection -> {
  1227→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1228→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1229→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1230→                        val offsetBF = (w - bf) / 2
  1231→                        
  1232→                        // Top Flange
  1233→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1234→                        // Bottom Flange
  1235→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1236→                        // Web
  1237→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1238→                        
  1239→                        // Stress Distribution (Schematic)
  1240→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1241→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1242→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1243→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1244→                    }
  1245→                    is SteelSectionType.LSection -> {
  1246→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
  1247→                        drawRect(Color.DarkGray, Offset(0f, 0f), Size(t, h))
  1248→                        drawRect(Color.DarkGray, Offset(t, h - t), Size(w - t, t))
  1249→                    }
  1250→                    is SteelSectionType.RHS -> {
  1251→                        val t = (section.thickness / section.height * h).toFloat().coerceIn(5f, 20f)
  1252→                        drawRect(Color.DarkGray, Offset(0f, 0f), size, style = androidx.compose.ui.graphics.drawscope.Stroke(t))
  1253→                    }
  1254→                    else -> {
  1255→                        drawRect(Color.DarkGray, Offset(0f, 0f), size)
  1256→                    }
  1257→                }
  1258→            }
  1259→            Spacer(Modifier.height(12.dp))
  1260→            Text("توزيع الإجهادات المرن (Elastic Stress)", style = MaterialTheme.typography.labelSmall, color = Color.Gray)
  1261→        }
  1262→    }
  1263→}
