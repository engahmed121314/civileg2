   391→@Composable
   392→private fun BeamDiagramCanvas(result: CalculatorEngine.BeamResult, modifier: Modifier) {
   393→    Card(modifier = modifier, shape = RoundedCornerShape(12.dp)) {
   394→        Canvas(modifier = Modifier.fillMaxSize().background(Color.White).padding(20.dp)) {
   395→            val w = size.width
   396→            val h = size.height
   397→            val baseline = h * 0.4f
   398→            
   399→            // Draw baseline
   400→            drawLine(Color.DarkGray, Offset(0f, baseline), Offset(w, baseline), strokeWidth = 3f)
   401→            
   402→            // Draw Supports based on type
   403→            drawSupport(result.supportType, 0f, baseline)
   404→            drawSupport(result.supportType, w, baseline, isEnd = true)
   405→
   406→            // Draw BMD Path
   407→            val path = Path()
   408→            path.moveTo(0f, baseline)
   409→            val segments = 50
   410→            for (i in 0..segments) {
   411→                val x = (i.toFloat() / segments) * w
   412→                val normX = i.toFloat() / segments
   413→                val y = when(result.supportType) {
   414→                    CalculatorEngine.SupportType.CANTILEVER -> baseline - (h * 0.5f * normX * normX)
   415→                    else -> baseline + (h * 0.5f * 4 * normX * (1 - normX))
   416→                }
   417→                path.lineTo(x, y)
   418→            }
   419→            drawPath(path, Color.Red, style = Stroke(4f))
   420→            
   421→            // Text for Max Moment
   422→            drawContext.canvas.nativeCanvas.drawText(
   423→                "M max = ${"%.1f".format(result.appliedMoment)} kN.m",
   424→                w/2, h - 5f,
   425→                android.graphics.Paint().apply {
   426→                    color = android.graphics.Color.RED
   427→                    textSize = 28f
   428→                    textAlign = android.graphics.Paint.Align.CENTER
   429→                    isFakeBoldText = true
   430→                }
   431→            )
   432→        }
   433→    }
   434→}
   435→
   436→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSupport(type: CalculatorEngine.SupportType, x: Float, y: Float, isEnd: Boolean = false) {
   437→    val size = 20f
   438→    when (type) {
   439→        CalculatorEngine.SupportType.HINGED_HINGED, CalculatorEngine.SupportType.ROLLER_HINGED -> {
   440→            val path = Path()
   441→            path.moveTo(x, y)
   442→            path.lineTo(x - size, y + size)
   443→            path.lineTo(x + size, y + size)
   444→            path.close()
   445→            drawPath(path, Color.Black)
   446→            if (type == CalculatorEngine.SupportType.ROLLER_HINGED && isEnd) {
   447→                drawLine(Color.Black, Offset(x - size, y + size + 5f), Offset(x + size, y + size + 5f), strokeWidth = 2f)
   448→            }
   449→        }
   450→        CalculatorEngine.SupportType.FIXED_HINGED, CalculatorEngine.SupportType.FIXED_FIXED -> {
   451→            if ((!isEnd && (type == CalculatorEngine.SupportType.FIXED_HINGED || type == CalculatorEngine.SupportType.FIXED_FIXED)) || 
   452→                (isEnd && type == CalculatorEngine.SupportType.FIXED_FIXED)) {
   453→                drawLine(Color.Black, Offset(x, y - size), Offset(x, y + size), strokeWidth = 6f)
   454→                for (i in -2..2) {
   455→                    val offset = i * 8f
   456→                    drawLine(Color.Black, Offset(x, y + offset), Offset(x + (if(isEnd) -10f else 10f), y + offset - 10f), strokeWidth = 2f)
   457→                }
   458→            } else {
   459→                // Hinged part
   460→                val path = Path()
   461→                path.moveTo(x, y)
   462→                path.lineTo(x - size, y + size)
   463→                path.lineTo(x + size, y + size)
   464→                path.close()
   465→                drawPath(path, Color.Black)
   466→            }
   467→        }
   468→        CalculatorEngine.SupportType.CANTILEVER -> {
   469→            if (!isEnd) {
   470→                drawLine(Color.Black, Offset(x, y - size*2), Offset(x, y + size*2), strokeWidth = 8f)
   471→            }
   472→        }
   473→        else -> {}
   474→    }
   475→}

   477→@Composable
   478→private fun BeamSectionDrawing(result: CalculatorEngine.BeamResult, modifier: Modifier) {
   479→    Card(modifier = modifier, shape = RoundedCornerShape(12.dp)) {
   480→        Canvas(modifier = Modifier.fillMaxSize().background(Color.White)) {
   481→            val scale = minOf((size.width - 120f)/result.width, (size.height - 120f)/result.depth).toFloat()
   482→            val w = (result.width * scale).toFloat()
   483→            val h = (result.depth * scale).toFloat()
   484→            val left = (size.width - w) / 2
   485→            val top = (size.height - h) / 2
   486→            
   487→            // Concrete
   488→            drawRect(Color.LightGray, Offset(left, top), Size(w, h))
   489→            drawRect(Color.DarkGray, Offset(left, top), Size(w, h), style = Stroke(4f))
   490→            
   491→            // Stirrup (Ties)
   492→            val cover = 25f * scale
   493→            drawRect(Color.Blue.copy(alpha = 0.8f), Offset(left + cover, top + cover), Size(w - 2*cover, h - 2*cover), style = Stroke(3f))
   494→            
   495→            // Main Reinforcement (Bottom)
   496→            val barR = 8f
   497→            val nBarsBottom = result.reinforcementBottom.numBars
   498→            for (i in 0 until nBarsBottom) {
   499→                val x = left + cover + 20f + i * (w - 2*cover - 40f) / (if(nBarsBottom > 1) nBarsBottom - 1 else 1)
   500→                drawCircle(Color.Black, barR, Offset(x, top + h - cover - 20f))
   501→            }
   502→            
   503→            // Top Reinforcement (Hangers)
   504→            val nBarsTop = result.reinforcementTop.numBars
   505→            for (i in 0 until nBarsTop) {
   506→                val x = left + cover + 20f + i * (w - 2*cover - 40f) / (if(nBarsTop > 1) nBarsTop - 1 else 1)
   507→                drawCircle(Color.DarkGray, barR * 0.8f, Offset(x, top + cover + 20f))
   508→            }
   509→            
   510→            // Dimensions
   511→            drawContext.canvas.nativeCanvas.apply {
   512→                val paint = android.graphics.Paint().apply { 
   513→                    color = android.graphics.Color.BLACK
   514→                    textSize = 30f
   515→                    textAlign = android.graphics.Paint.Align.CENTER
   516→                }
   517→                drawText("${result.width.toInt()} mm", size.width/2, top - 15f, paint)
   518→                save()
   519→                rotate(-90f, left - 20f, size.height/2)
   520→                drawText("${result.depth.toInt()} mm", left - 20f, size.height/2, paint)
   521→                restore()
   522→            }
   523→        }
   524→    }
   525→}
