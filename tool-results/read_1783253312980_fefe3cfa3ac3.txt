   352→@Composable
   353→private fun StairReinforcementDrawing(res: CalculatorEngine.StairResult, modifier: Modifier) {
   354→    Card(
   355→        modifier = modifier.height(440.dp),
   356→        shape = RoundedCornerShape(12.dp),
   357→        elevation = CardDefaults.cardElevation(4.dp)
   358→    ) {
   359→        Canvas(
   360→            modifier = Modifier
   361→                .fillMaxSize()
   362→                .background(Color.White)
   363→                .padding(12.dp)
   364→        ) {
   365→            // ===== DATA EXTRACTION FROM RESULT =====
   366→            val totalSpanMM = res.span * 1000f
   367→            val riserMM = res.riser.toFloat()
   368→            val treadMM = res.tread.toFloat()
   369→            val thicknessMM = res.thickness.toFloat()
   370→            val numSteps = (totalSpanMM / treadMM).toInt().coerceIn(1, 30)
   371→            val totalHeightMM = numSteps * riserMM
   372→            val coverMM = 25f
   373→
   374→            // ===== LAYOUT & SCALING =====
   375→            val border = 2.5f
   376→            val marginL = 55f
   377→            val marginR = 50f
   378→            val marginT = 45f
   379→            val marginB = 55f
   380→            val drawW = size.width - marginL - marginR - border * 2
   381→            val drawH = size.height - marginT - marginB - border * 2
   382→            if (drawW <= 0f || drawH <= 0f) return@Canvas
   383→
   384→            val fitSpan = totalSpanMM
   385→            val fitHeight = totalHeightMM + thicknessMM + 30f
   386→            val scX = drawW / fitSpan
   387→            val scY = drawH / fitHeight
   388→            val sc = minOf(scX, scY) * 0.88f
   389→
   390→            // Origin: bottom-left corner of stair (floor level, start of first riser)
   391→            val ox = marginL + border + (drawW - totalSpanMM * sc) / 2f
   392→            val oy = size.height - marginB - border - (drawH - (totalHeightMM + thicknessMM) * sc) / 2f
   393→            // End point: top of last riser
   394→            val ex = ox + numSteps * treadMM * sc
   395→            val ey = oy - numSteps * riserMM * sc
   396→
   397→            // ===== COLORS =====
   398→            val concreteFill = Color(0xFFE6E6E6)
   399→            val concreteStroke = Color(0xFF2C2C2C)
   400→            val mainSteel = Color(0xFFC62828)
   401→            val distSteel = Color(0xFF0D47A1)
   402→            val dimClr = Color(0xFF4A4A4A)
   403→            val borderClr = Color(0xFF1A1A1A)
   404→            val supportClr = Color(0xFF444444)
   405→
   406→            // ===== GEOMETRY HELPERS =====
   407→            val slopeAngle = atan2(riserMM.toDouble(), treadMM.toDouble())
   408→            val cosA = cos(slopeAngle).toFloat()
   409→            val sinA = sin(slopeAngle).toFloat()
   410→            // Perpendicular offset for slab thickness (below slope line, into the slab)
   411→            val perpX = thicknessMM * sinA * sc
   412→            val perpY = thicknessMM * cosA * sc
   413→            // Cover offset (from slab soffit toward top surface, perpendicular to slope)
   414→            val covX = coverMM * sinA * sc
   415→            val covY = coverMM * cosA * sc
   416→
   417→            // ===== 1. PROFESSIONAL BORDER =====
   418→            drawRect(
   419→                color = borderClr,
   420→                topLeft = Offset(border / 2f, border / 2f),
   421→                size = Size(size.width - border, size.height - border),
   422→                style = Stroke(width = border)
   423→            )
   424→            // Inner border for double-line effect
   425→            drawRect(
   426→                color = borderClr.copy(alpha = 0.3f),
   427→                topLeft = Offset(border + 2f, border + 2f),
   428→                size = Size(size.width - border * 2 - 4f, size.height - border * 2 - 4f),
   429→                style = Stroke(width = 0.5f)
   430→            )
   431→
   432→            // ===== 2. STAIR PROFILE (CONCRETE SECTION) =====
   433→            val soffitStartX = ox + perpX
   434→            val soffitStartY = oy + perpY
   435→            val soffitEndX = ex + perpX
   436→            val soffitEndY = ey + perpY
   437→
   438→            val profilePath = Path()
   439→            // Start at soffit bottom-left
   440→            profilePath.moveTo(soffitStartX, soffitStartY)
   441→            // Up to floor level at stair start
   442→            profilePath.lineTo(ox, oy)
   443→            // Draw each step: riser up, then tread across
   444→            for (i in 0 until numSteps) {
   445→                val stepX = ox + i * treadMM * sc
   446→                val stepTopY = oy - (i + 1) * riserMM * sc
   447→                val nextX = ox + (i + 1) * treadMM * sc
   448→                // Riser (vertical up)
   449→                profilePath.lineTo(stepX, stepTopY)
   450→                // Tread (horizontal across)
   451→                profilePath.lineTo(nextX, stepTopY)
   452→            }
   453→            // Down to soffit at top end
   454→            profilePath.lineTo(soffitEndX, soffitEndY)
   455→            // Soffit straight back to start
   456→            profilePath.lineTo(soffitStartX, soffitStartY)
   457→            profilePath.close()
   458→
   459→            // Fill concrete section
   460→            drawPath(profilePath, concreteFill)
   461→            // Outline concrete section
   462→            drawPath(profilePath, concreteStroke, style = Stroke(width = 2.2f))
   463→
   464→            // ===== 3. STEP DIVISION LINES (risers for clarity) =====
   465→            for (i in 0 until numSteps) {
   466→                val x = ox + i * treadMM * sc
   467→                val yBot = oy - i * riserMM * sc
   468→                val yTop = oy - (i + 1) * riserMM * sc
   469→                drawLine(concreteStroke, Offset(x, yBot), Offset(x, yTop), strokeWidth = 1.2f)
   470→            }
   471→            // Last vertical line at the top riser
   472→            if (numSteps > 0) {
   473→                val lastX = ox + numSteps * treadMM * sc
   474→                val lastYTop = ey
   475→                val lastYBot = oy - (numSteps - 1) * riserMM * sc
   476→                drawLine(concreteStroke, Offset(lastX, lastYBot), Offset(lastX, lastYTop), strokeWidth = 1.2f)
   477→            }
   478→
   479→            // ===== 4. SOFFIT LINE EMPHASIS (waist slab bottom) =====
   480→            drawLine(
   481→                concreteStroke,
   482→                Offset(soffitStartX, soffitStartY),
   483→                Offset(soffitEndX, soffitEndY),
   484→                strokeWidth = 2.5f
   485→            )
   486→
   487→            // ===== 5. SUPPORT CONDITIONS =====
   488→            val supSize = 14f
   489→
   490→            // --- Bottom support: pinned (triangle + hatching) ---
   491→            val bsCx = ox
   492→            val bsCy = oy
   493→            drawLine(supportClr, Offset(bsCx, bsCy), Offset(bsCx - supSize, bsCy + supSize), 2f)
   494→            drawLine(supportClr, Offset(bsCx - supSize, bsCy + supSize), Offset(bsCx + supSize, bsCy + supSize), 2f)
   495→            drawLine(supportClr, Offset(bsCx + supSize, bsCy + supSize), Offset(bsCx, bsCy), 2f)
   496→            // Hatching below bottom support
   497→            for (h in 0..5) {
   498→                val hx = bsCx - supSize + h * (supSize * 2f / 5f)
   499→                drawLine(supportClr, Offset(hx, bsCy + supSize), Offset(hx - 5f, bsCy + supSize + 7f), 1f)
   500→            }
   501→            // Ground line
   502→            drawLine(supportClr, Offset(bsCx - supSize - 5f, bsCy + supSize), Offset(bsCx + supSize + 5f, bsCy + supSize), 1.5f)
   503→
   504→            // --- Top support: roller (triangle + circle + hatching) ---
   505→            val tsCx = ex
   506→            val tsCy = ey
   507→            drawLine(supportClr, Offset(tsCx, tsCy), Offset(tsCx, tsCy - supSize), 2f)
   508→            drawLine(supportClr, Offset(tsCx - supSize, tsCy - supSize), Offset(tsCx + supSize, tsCy - supSize), 2f)
   509→            // Roller circle
   510→            drawCircle(supportClr, radius = 5f, center = Offset(tsCx, tsCy - supSize - 5f), style = Stroke(1.8f))
   511→            drawCircle(supportClr, radius = 1.5f, center = Offset(tsCx, tsCy - supSize - 5f))
   512→            // Hatching above top support
   513→            for (h in 0..5) {
   514→                val hx = tsCx - supSize + h * (supSize * 2f / 5f)
   515→                drawLine(supportClr, Offset(hx, tsCy - supSize), Offset(hx - 5f, tsCy - supSize - 7f), 1f)
   516→            }
   517→            drawLine(supportClr, Offset(tsCx - supSize - 5f, tsCy - supSize), Offset(tsCx + supSize + 5f, tsCy - supSize), 1.5f)
   518→
   519→            // ===== 6. MAIN REINFORCEMENT (diagonal line following slope) =====
   520→            val mbStartX = soffitStartX - covX
   521→            val mbStartY = soffitStartY - covY
   522→            val mbEndX = soffitEndX - covX
   523→            val mbEndY = soffitEndY - covY
   524→
   525→            // Extend bars into supports (anchorage)
   526→            val anchLen = 25f
   527→            val mbFullStartX = mbStartX - anchLen * cosA
   528→            val mbFullStartY = mbStartY + anchLen * sinA
   529→            val mbFullEndX = mbEndX + anchLen * cosA
   530→            val mbFullEndY = mbEndY - anchLen * sinA
   531→
   532→            // Main bar line
   533→            drawLine(mainSteel, Offset(mbFullStartX, mbFullStartY), Offset(mbFullEndX, mbFullEndY), strokeWidth = 3.5f)
   534→
   535→            // 90-degree hooks at both ends
   536→            val hookLen = 14f
   537→            drawLine(mainSteel, Offset(mbFullStartX, mbFullStartY),
   538→                Offset(mbFullStartX - hookLen * sinA, mbFullStartY - hookLen * cosA), 3.5f)
   539→            drawLine(mainSteel, Offset(mbFullEndX, mbFullEndY),
   540→                Offset(mbFullEndX - hookLen * sinA, mbFullEndY - hookLen * cosA), 3.5f)
   541→
   542→            // ===== 7. DISTRIBUTION REINFORCEMENT (small circles along span) =====
   543→            val numDistBars = maxOf(4, numSteps)
   544→            for (i in 1..numDistBars) {
   545→                val t = i.toFloat() / (numDistBars + 1f)
   546→                val cx = mbStartX + t * (mbEndX - mbStartX)
   547→                val cy = mbStartY + t * (mbEndY - mbStartY)
   548→                // Open circle (typical rebar cross-section symbol)
   549→                drawCircle(distSteel, radius = 4.5f, center = Offset(cx, cy), style = Stroke(1.8f))
   550→                drawCircle(Color.White, radius = 2.5f, center = Offset(cx, cy))
   551→            }
   552→
   553→            // ===== 8. TOP COVER INDICATION =====
   554→            val cvFrac = 0.35f
   555→            val cvTopX = ox + cvFrac * (ex - ox)
   556→            val cvTopY = oy + cvFrac * (ey - oy)
   557→            val cvSoffX = cvTopX + perpX
   558→            val cvSoffY = cvTopY + perpY
   559→            val cvBarX = cvSoffX - covX
   560→            val cvBarY = cvSoffY - covY
   561→            val cvOff = 18f
   562→            val cvD1X = cvSoffX + cvOff * cosA
   563→            val cvD1Y = cvSoffY + cvOff * sinA
   564→            val cvD2X = cvBarX + cvOff * cosA
   565→            val cvD2Y = cvBarY + cvOff * sinA
   566→            drawLine(dimClr, Offset(cvD1X, cvD1Y), Offset(cvD2X, cvD2Y), strokeWidth = 0.8f)
   567→            // Tick marks at soffit and bar
   568→            drawLine(dimClr,
   569→                Offset(cvSoffX - 2f * sinA, cvSoffY - 2f * cosA),
   570→                Offset(cvSoffX + 4f * sinA, cvSoffY + 4f * cosA), 0.8f)
   571→            drawLine(dimClr,
   572→                Offset(cvBarX - 2f * sinA, cvBarY - 2f * cosA),
   573→                Offset(cvBarX + 4f * sinA, cvBarY + 4f * cosA), 0.8f)
   574→
   575→            // ===== 9. DIMENSION LINES =====
   576→            val dimPaint = android.graphics.Paint().apply {
   577→                color = android.graphics.Color.parseColor("#4A4A4A")
   578→                textSize = 17f
   579→                textAlign = android.graphics.Paint.Align.CENTER
   580→                isFakeBoldText = true
   581→            }
   582→            val dimSmPaint = android.graphics.Paint().apply {
   583→                color = android.graphics.Color.parseColor("#4A4A4A")
   584→                textSize = 13f
   585→                textAlign = android.graphics.Paint.Align.CENTER
   586→            }
   587→
   588→            fun extLine(x1: Float, y1: Float, x2: Float, y2: Float) {
   589→                drawLine(dimClr, Offset(x1, y1), Offset(x2, y2), strokeWidth = 0.7f)
   590→            }
   591→
   592→            fun drawArrow(x: Float, y: Float, angle: Float, sz: Float = 5f) {
   593→                val a1 = angle + 2.6f
   594→                val a2 = angle - 2.6f
   595→                drawLine(dimClr, Offset(x, y),
   596→                    Offset(x + sz * cos(a1.toDouble()).toFloat(), y + sz * sin(a1.toDouble()).toFloat()), 1f)
   597→                drawLine(dimClr, Offset(x, y),
   598→                    Offset(x + sz * cos(a2.toDouble()).toFloat(), y + sz * sin(a2.toDouble()).toFloat()), 1f)
   599→            }
   600→
   601→            // --- (a) Total Horizontal Span ---
   602→            val spanDimY = oy + 35f
   603→            extLine(ox, oy + 6f, ox, spanDimY + 4f)
   604→            extLine(ex, oy + 6f, ex, spanDimY + 4f)
   605→            drawLine(dimClr, Offset(ox, spanDimY), Offset(ex, spanDimY), strokeWidth = 1f)
   606→            drawArrow(ox, spanDimY, 0f)
   607→            drawArrow(ex, spanDimY, PI.toFloat())
   608→            drawContext.canvas.nativeCanvas.drawText(
   609→                "L = ${res.span} m", (ox + ex) / 2f, spanDimY + 15f, dimPaint
   610→            )
   611→
   612→            // --- (b) Total Vertical Height ---
   613→            val hDimX = ox - 38f
   614→            extLine(ox - 5f, oy, hDimX - 4f, oy)
   615→            extLine(ox - 5f, ey, hDimX - 4f, ey)
   616→            drawLine(dimClr, Offset(hDimX, oy), Offset(hDimX, ey), strokeWidth = 1f)
   617→            drawArrow(hDimX, oy, -(PI.toFloat() / 2f))
   618→            drawArrow(hDimX, ey, PI.toFloat() / 2f)
   619→            drawContext.canvas.nativeCanvas.save()
   620→            drawContext.canvas.nativeCanvas.rotate(-90f, hDimX - 14f, (oy + ey) / 2f)
   621→            drawContext.canvas.nativeCanvas.drawText(
   622→                "H = ${totalHeightMM.toInt()} mm", hDimX - 14f, (oy + ey) / 2f + 5f, dimPaint
   623→            )
   624→            drawContext.canvas.nativeCanvas.restore()
   625→
   626→            // --- (c) Riser Dimension (first step) ---
   627→            if (numSteps >= 1) {
   628→                val rDimX = ox - 14f
   629→                val rTop = oy - riserMM * sc
   630→                extLine(ox - 2f, oy, rDimX - 2f, oy)
   631→                extLine(ox - 2f, rTop, rDimX - 2f, rTop)
   632→                drawLine(dimClr, Offset(rDimX, oy), Offset(rDimX, rTop), strokeWidth = 0.8f)
   633→                drawContext.canvas.nativeCanvas.save()
   634→                drawContext.canvas.nativeCanvas.rotate(-90f, rDimX - 8f, (oy + rTop) / 2f)
   635→                drawContext.canvas.nativeCanvas.drawText(
   636→                    "R=${riserMM.toInt()}", rDimX - 8f, (oy + rTop) / 2f + 4f, dimSmPaint
   637→                )
   638→                drawContext.canvas.nativeCanvas.restore()
   639→            }
   640→
   641→            // --- (d) Tread Dimension (first step) ---
   642→            if (numSteps >= 1) {
   643→                val tDimY = oy + 17f
   644→                val tEndX = ox + treadMM * sc
   645→                extLine(ox, oy + 2f, ox, tDimY + 2f)
   646→                extLine(tEndX, oy + 2f, tEndX, tDimY + 2f)
   647→                drawLine(dimClr, Offset(ox, tDimY), Offset(tEndX, tDimY), strokeWidth = 0.8f)
   648→                drawArrow(ox, tDimY, 0f, 4f)
   649→                drawArrow(tEndX, tDimY, PI.toFloat(), 4f)
   650→                drawContext.canvas.nativeCanvas.drawText(
   651→                    "T=${treadMM.toInt()}", (ox + tEndX) / 2f, tDimY + 11f, dimSmPaint
   652→                )
   653→            }
   654→
   655→            // --- (e) Waist Slab Thickness (rotated along slope) ---
   656→            val wtFrac = 0.65f
   657→            val wtPx = ox + wtFrac * (ex - ox)
   658→            val wtPy = oy + wtFrac * (ey - oy)
   659→            val wtSx = wtPx + perpX
   660→            val wtSy = wtPy + perpY
   661→            val wtOff = 22f
   662→            val wtD1x = wtPx + wtOff * cosA
   663→            val wtD1y = wtPy + wtOff * sinA
   664→            val wtD2x = wtSx + wtOff * cosA
   665→            val wtD2y = wtSy + wtOff * sinA
   666→            drawLine(dimClr, Offset(wtD1x, wtD1y), Offset(wtD2x, wtD2y), strokeWidth = 0.8f)
   667→            drawLine(dimClr,
   668→                Offset(wtPx - 3f * sinA, wtPy - 3f * cosA),
   669→                Offset(wtPx + 5f * sinA, wtPy + 5f * cosA), 0.8f)
   670→            drawLine(dimClr,
   671→                Offset(wtSx - 3f * sinA, wtSy - 3f * cosA),
   672→                Offset(wtSx + 5f * sinA, wtSy + 5f * cosA), 0.8f)
   673→            drawContext.canvas.nativeCanvas.save()
   674→            val wtMidX = (wtD1x + wtD2x) / 2f
   675→            val wtMidY = (wtD1y + wtD2y) / 2f
   676→            val wtAngleDeg = -toDegrees(slopeAngle).toFloat()
   677→            drawContext.canvas.nativeCanvas.rotate(wtAngleDeg, wtMidX, wtMidY)
   678→            drawContext.canvas.nativeCanvas.drawText(
   679→                "ts=${thicknessMM.toInt()}", wtMidX, wtMidY - 6f, dimSmPaint
   680→            )
   681→            drawContext.canvas.nativeCanvas.restore()
   682→
   683→            // ===== 10. REINFORCEMENT LABELS =====
   684→            val labelPaint = android.graphics.Paint().apply {
   685→                textSize = 19f
   686→                textAlign = android.graphics.Paint.Align.LEFT
   687→                isFakeBoldText = true
   688→            }
   689→            val labelSmPaint = android.graphics.Paint().apply {
   690→                textSize = 13f
   691→                textAlign = android.graphics.Paint.Align.LEFT
   692→            }
   693→
   694→            // Main reinforcement label with leader line
   695→            val mlX = (mbStartX + mbEndX) / 2f
   696→            val mlY = (mbStartY + mbEndY) / 2f
   697→            val mlLabelX = mlX + 18f
   698→            val mlLabelY = mlY - 12f
   699→            drawLine(mainSteel, Offset(mlX, mlY), Offset(mlLabelX, mlLabelY), strokeWidth = 1f)
   700→            drawLine(mainSteel, Offset(mlLabelX, mlLabelY), Offset(mlLabelX + 6f, mlLabelY), strokeWidth = 1f)
   701→            labelPaint.color = android.graphics.Color.parseColor("#C62828")
   702→            drawContext.canvas.nativeCanvas.drawText(
   703→                res.reinforcement.barString, mlLabelX + 10f, mlLabelY - 2f, labelPaint
   704→            )
   705→            labelSmPaint.color = android.graphics.Color.parseColor("#C62828")
   706→            drawContext.canvas.nativeCanvas.drawText(
   707→                "(Main Bottom)", mlLabelX + 10f, mlLabelY + 14f, labelSmPaint
   708→            )
   709→
   710→            // Distribution reinforcement label
   711→            val dlY = mlY + 22f
   712→            drawLine(distSteel, Offset(mlX, mlY + 5f), Offset(mlLabelX, dlY), strokeWidth = 1f)
   713→            drawLine(distSteel, Offset(mlLabelX, dlY), Offset(mlLabelX + 6f, dlY), strokeWidth = 1f)
   714→            labelPaint.color = android.graphics.Color.parseColor("#0D47A1")
   715→            drawContext.canvas.nativeCanvas.drawText(
   716→                res.distributionReinforcement.barString, mlLabelX + 10f, dlY - 2f, labelPaint
   717→            )
   718→            labelSmPaint.color = android.graphics.Color.parseColor("#0D47A1")
   719→            drawContext.canvas.nativeCanvas.drawText(
   720→                "(Distribution)", mlLabelX + 10f, dlY + 14f, labelSmPaint
   721→            )
   722→
   723→            // ===== 11. TITLE & SCALE TEXT =====
   724→            val titlePaint = android.graphics.Paint().apply {
   725→                color = android.graphics.Color.parseColor("#1A1A1A")
   726→                textSize = 15f
   727→                textAlign = android.graphics.Paint.Align.RIGHT
   728→                isFakeBoldText = true
   729→            }
   730→            drawContext.canvas.nativeCanvas.drawText(
   731→                "STAIR REINFORCEMENT DETAIL - S1", size.width - 10f, marginT - 5f, titlePaint
   732→            )
   733→
   734→            val stepsPaint = android.graphics.Paint().apply {
   735→                color = android.graphics.Color.parseColor("#333333")
   736→                textSize = 13f
   737→                textAlign = android.graphics.Paint.Align.LEFT
   738→                isFakeBoldText = true
   739→            }
   740→            drawContext.canvas.nativeCanvas.drawText(
   741→                "Steps: $numSteps  |  R x T = ${riserMM.toInt()} x ${treadMM.toInt()} mm",
   742→                marginL + 5f, marginT - 5f, stepsPaint
   743→            )
   744→
   745→            val scalePaint = android.graphics.Paint().apply {
   746→                color = android.graphics.Color.parseColor("#888888")
   747→                textSize = 12f
   748→                textAlign = android.graphics.Paint.Align.CENTER
   749→            }
   750→            drawContext.canvas.nativeCanvas.drawText(
   751→                "Scale: Not to Scale", size.width / 2f, size.height - 3f, scalePaint
   752→            )
   753→
   754→            // Safety status indicator
   755→            val safePaint = android.graphics.Paint().apply {
   756→                textSize = 12f
   757→                textAlign = android.graphics.Paint.Align.RIGHT
   758→                isFakeBoldText = true
   759→            }
   760→            if (res.isSafe) {
   761→                safePaint.color = android.graphics.Color.parseColor("#2E7D32")
   762→                drawContext.canvas.nativeCanvas.drawText(
   763→                    "SAFE", size.width - 10f, size.height - 3f, safePaint
   764→                )
   765→            } else {
   766→                safePaint.color = android.graphics.Color.parseColor("#C62828")
   767→                drawContext.canvas.nativeCanvas.drawText(
   768→                    "UNSAFE", size.width - 10f, size.height - 3f, safePaint
   769→                )
   770→            }
   771→        }
   772→    }
   773→}
