   280→@Composable
   281→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   282→    Card(
   283→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   284→        border = BorderStroke(1.dp, Color.LightGray)
   285→    ) {
   286→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   287→            val padding = 40.dp.toPx()
   288→            val w = size.width - 2 * padding
   289→            val h = size.height - 2 * padding
   290→            val baseY = size.height - padding
   291→            
   292→            // Scale calculations
   293→            val scale = (w / inputs.span).toFloat()
   294→            val eh = (inputs.eaveHeight * scale).toFloat()
   295→            val rh = (inputs.ridgeHeight * scale).toFloat()
   296→            val midX = padding + w / 2
   297→            
   298→            // Draw Main Frame Outline
   299→            val framePath = Path().apply {
   300→                moveTo(padding, baseY)
   301→                lineTo(padding, baseY - eh)
   302→                lineTo(midX, baseY - rh)
   303→                lineTo(padding + w, baseY - eh)
   304→                lineTo(padding + w, baseY)
   305→            }
   306→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   307→            
   308→            // Draw Distributed Load Arrows
   309→            val loadColor = Color(0xFFE53935)
   310→            for (i in 0..12) {
   311→                val x = padding + (i * w / 12)
   312→                val y = if (x <= midX) {
   313→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   314→                } else {
   315→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   316→                }
   317→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   318→                // Arrow heads
   319→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   320→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   321→            }
   322→            
   323→            // Draw Moment Diagram (Simplified)
   324→            val momentColor = Color(0xFF1E88E5)
   325→            val momentPath = Path().apply {
   326→                moveTo(padding, baseY - eh)
   327→                // Simulated parabolic curve for moment
   328→                cubicTo(
   329→                    padding + w * 0.25f, baseY - eh + 50f,
   330→                    padding + w * 0.75f, baseY - eh + 50f,
   331→                    padding + w, baseY - eh
   332→                )
   333→            }
   334→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   335→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   336→            
   337→            // Labels
   338→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   339→        }
   340→    }
   341→}

   393→

   395→                    0 -> { // Front Elevation

   398→                    1 -> { // Plan View

   401→                    2 -> { // Side View
   402→                        drawSideElevation(inputs, scale, startX, baseY)
   403→                    }
   404→                    3 -> { // 3D Perspective
   405→                        draw3DView(inputs, scale, startX, baseY)
   406→                    }
   407→                }
   408→            }
   409→        }
   410→    }

   413→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawFrontElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   414→    val colLX = startX
   415→    val colRX = startX + (inputs.span * scale).toFloat()
   416→    val eaveY = baseY - (inputs.eaveHeight * scale).toFloat()
   417→    val ridgeY = baseY - (inputs.ridgeHeight * scale).toFloat()
   418→    val midX = startX + (inputs.span * scale / 2).toFloat()
   419→
   420→    // Columns
   421→    drawLine(Color.Black, Offset(colLX, baseY), Offset(colLX, eaveY), strokeWidth = 8f)
   422→    drawLine(Color.Black, Offset(colRX, baseY), Offset(colRX, eaveY), strokeWidth = 8f)
   423→    // Rafters
   424→    drawLine(Color.Black, Offset(colLX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   425→    drawLine(Color.Black, Offset(colRX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   426→    
   427→    // Foundations
   428→    drawRect(Color.Gray, Offset(colLX - 15f, baseY), Size(30f, 15f))
   429→    drawRect(Color.Gray, Offset(colRX - 15f, baseY), Size(30f, 15f))
   430→}
   431→
   432→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   433→    val planW = (inputs.span * scale).toFloat()
   434→    val planL = (inputs.length * scale).toFloat()
   435→    val planTop = baseY - planL
   436→    
   437→    // Boundary
   438→    drawRect(Color.Black, Offset(startX, planTop), Size(planW, planL), style = androidx.compose.ui.graphics.drawscope.Stroke(4f))
   439→    
   440→    // Grid Lines (Bays)
   441→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   442→    for (i in 0..numBays) {
   443→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()
   444→        drawLine(Color.LightGray, Offset(startX, y), Offset(startX + planW, y), strokeWidth = 1f)
   445→        // Columns markers
   446→        drawCircle(Color.Black, 4f, Offset(startX, y))
   447→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))
   448→    }
   449→}
   450→
   451→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSideElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   452→    val planL = (inputs.length * scale).toFloat()
   453→    val eaveH = (inputs.eaveHeight * scale).toFloat()
   454→    val eaveY = baseY - eaveH
   455→    
   456→    // Ground
   457→    drawLine(Color.Gray, Offset(startX, baseY), Offset(startX + planL, baseY), strokeWidth = 2f)
   458→    
   459→    // Columns
   460→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   461→    for (i in 0..numBays) {
   462→        val x = startX + (i * inputs.baySpacing * scale).toFloat()
   463→        drawLine(Color.Black, Offset(x, baseY), Offset(x, eaveY), strokeWidth = 6f)
   464→    }
   465→    // Eave Line
   466→    drawLine(Color.Black, Offset(startX, eaveY), Offset(startX + planL, eaveY), strokeWidth = 6f)
   467→}

   469→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   470→    val s = scale.toFloat() * 0.65f
   471→    val depthOff = 35f
   472→    
   473→    // Draw from back to front for better perspective look
   474→    for (i in 4 downTo 0) {
   475→        val off = i * depthOff
   476→        val x0 = startX + off
   477→        val y0 = baseY - off
   478→        val spanW = (inputs.span * s).toFloat()
   479→        val eaveH = (inputs.eaveHeight * s).toFloat()
   480→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   481→        
   482→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   483→        val stroke = if (i == 0) 3f else 1.5f
   484→        
   485→        // Sections of the frame
   486→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   487→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   488→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   489→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   490→        
   491→        // Purlins/Girts connections (longitudinal)
   492→        if (i > 0) {
   493→            val px = x0 - depthOff
   494→            val py = y0 + depthOff
   495→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   496→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   497→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   498→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   499→        }
   500→    }
   501→}

   535→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   536→    val p = Paint().apply { 
   537→        color = android.graphics.Color.BLACK
   538→        strokeWidth = 6f
   539→        isAntiAlias = true
   540→        style = Paint.Style.STROKE
   541→    }
   542→    val dimPaint = Paint().apply {
   543→        color = android.graphics.Color.DKGRAY
   544→        strokeWidth = 2f
   545→        textSize = 24f
   546→        isAntiAlias = true
   547→    }
   548→    val textP = Paint().apply {
   549→        color = android.graphics.Color.BLACK
   550→        textSize = 22f
   551→        isFakeBoldText = true
   552→        textAlign = Paint.Align.CENTER
   553→    }
   554→    
   555→    val s = scale * 0.7f
   556→    val w = inputs.span.toFloat() * s
   557→    val eh = inputs.eaveHeight.toFloat() * s
   558→    val rh = inputs.ridgeHeight.toFloat() * s
   559→    val midX = x + w/2
   560→    
   561→    // Draw Main Frame (Columns & Rafters)
   562→    canvas.drawLine(x, y, x, y - eh, p)
   563→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   564→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   565→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   566→    
   567→    // Foundations
   568→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   569→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   570→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   571→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   572→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   573→    p.strokeWidth = 6f
   574→    
   575→    // --- Dimension Lines ---
   576→    // Span Dimension
   577→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   578→    
   579→    // Eave Height Dimension
   580→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   581→    
   582→    // Ridge Height Dimension
   583→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   584→
   585→    // --- Section Labels ---
   586→    result?.let { res ->
   587→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   588→        
   589→        // Column Label
   590→        canvas.save()
   591→        canvas.rotate(-90f, x + 25f, y - eh/2)
   592→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   593→        canvas.restore()
   594→        
   595→        // Rafter Label
   596→        canvas.save()
   597→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   598→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   599→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   600→        canvas.restore()
   601→    }
   602→
   603→    // Connection Markers (Circles at joints)
   604→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   605→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   606→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   607→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   608→}

   610→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   611→    val tickSize = 15f
   612→    canvas.drawLine(x1, y1, x2, y2, paint)
   613→    
   614→    // Draw Ticks (45 degree lines)
   615→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   616→    val tickAngle = angle + Math.PI / 4
   617→    
   618→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   619→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   620→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   621→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   622→
   623→    // Text label
   624→    val midX = (x1 + x2) / 2
   625→    val midY = (y1 + y2) / 2
   626→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   627→    
   628→    if (x1 == x2) { // Vertical
   629→        canvas.save()
   630→        canvas.rotate(-90f, midX - 10f, midY)
   631→        canvas.drawText(text, midX - 10f, midY, textPaint)
   632→        canvas.restore()
   633→    } else { // Horizontal
   634→        canvas.drawText(text, midX, midY - 10f, textPaint)
   635→    }
   636→}

   638→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   639→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   640→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   641→    
   642→    val s = scale * 0.6f
   643→    val w = inputs.span.toFloat() * s
   644→    val l = inputs.length.toFloat() * s
   645→    val planTop = y - l
   646→    
   647→    canvas.drawRect(x, planTop, x + w, y, p)
   648→    
   649→    val baySpacing = inputs.baySpacing.toFloat() * s
   650→    var curY = planTop
   651→    var count = 0
   652→    while (curY <= y + 0.1f) {
   653→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   654→        
   655→        // Column markers
   656→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   657→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   658→        
   659→        if (curY + baySpacing <= y + 0.1f) {
   660→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   661→        }
   662→        
   663→        curY += baySpacing
   664→        count++
   665→    }
   666→    
   667→    // Span Dimension
   668→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   669→    // Total Length Dimension
   670→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   671→}

   673→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   674→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   675→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   676→    
   677→    val s = scale * 0.6f
   678→    val l = inputs.length.toFloat() * s
   679→    val eh = inputs.eaveHeight.toFloat() * s
   680→    
   681→    canvas.drawLine(x, y, x + l, y, p) // Ground
   682→    val bay = inputs.baySpacing.toFloat() * s
   683→    var cur = x
   684→    while (cur <= x + l + 0.1f) {
   685→        canvas.drawLine(cur, y, cur, y - eh, p)
   686→        cur += bay
   687→    }
   688→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   689→    
   690→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   691→}

  1343→@Composable
  1344→fun SteelSectionDrawing(section: SteelSectionType) {
  1345→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1346→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1347→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1348→                val w = size.width
  1349→                val h = size.height
  1350→                
  1351→                when (section) {
  1352→                    is SteelSectionType.ISection -> {
  1353→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1354→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1355→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1356→                        val offsetBF = (w - bf) / 2
  1357→                        
  1358→                        // Top Flange
  1359→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1360→                        // Bottom Flange
  1361→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1362→                        // Web
  1363→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1364→                        
  1365→                        // Stress Distribution (Schematic)
  1366→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1367→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1368→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1369→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1370→                    }
  1371→                    is SteelSectionType.LSection -> {
  1372→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
  1373→                        drawRect(Color.DarkGray, Offset(0f, 0f), Size(t, h))
  1374→                        drawRect(Color.DarkGray, Offset(t, h - t), Size(w - t, t))
