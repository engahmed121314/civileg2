    70→                },

    76→                actions = {

   170→                for ((index, title) in tabs.withIndex()) {
   171→                    Tab(
   172→                        selected = selectedTab == index,
   173→                        onClick = { selectedTab = index },
   174→                        text = { Text(title, fontSize = 13.sp) }
   175→                    )
   176→                }
   177→            }
   178→
   179→            // Tab Content
   180→            when (selectedTab) {
   181→                0 -> DrawingTab(nodes, members, memberLoads, nodalLoads, result, diagramType, selectedMemberId, viewModel)
   182→                1 -> NodesTab(nodes, members, viewModel)
   183→                2 -> MembersTab(nodes, members, settings, viewModel)
   184→                3 -> LoadsTab(nodes, members, nodalLoads, memberLoads, viewModel)
   185→                4 -> ResultsTab(result, concreteResults, steelResults, selectedMemberId, viewModel)
   186→            }
   187→        }
   188→    }
   189→}

   196→    nodes: List<FrameNode>,
   197→    members: List<FrameMember>,
   198→    memberLoads: List<MemberLoad>,
   199→    nodalLoads: List<NodalLoad>,

   212→            diagramType = diagramType,
   213→            selectedMemberId = selectedMemberId,
   214→            onMemberTap = { viewModel.setSelectedMember(it) }
   215→        )

   230→                    Text("حل", fontSize = 10.sp, fontWeight = FontWeight.Bold)
   231→                }
   232→            }
   233→        }
   234→
   235→        // Info card
   236→        Card(
   237→            modifier = Modifier
   238→                .align(Alignment.TopStart)
   239→                .padding(8.dp),
   240→            colors = CardDefaults.cardColors(containerColor = Color.White.copy(alpha = 0.9f))
   241→        ) {
   242→            Column(modifier = Modifier.padding(8.dp)) {
   243→                Text("العقد: ${nodes.size}  |  الأعضاء: ${members.size}", fontSize = 11.sp, color = Color.Gray)
   244→                if (result?.hasResults == true) {
   245→                    Text(
   246→                        "تم الحل ✓  |  الحمولات: ${memberLoads.size + nodalLoads.size}",
   247→                        fontSize = 11.sp, color = Color(0xFF4CAF50), fontWeight = FontWeight.Bold
   248→                    )
   249→                }
   250→            }
   251→        }
   252→    }
   253→}

   255→// ============================================================================
   256→// Tab 1: Nodes
   257→// ============================================================================
   258→@Composable
   259→private fun NodesTab(
   260→    nodes: List<FrameNode>,
   261→    members: List<FrameMember>,
   262→    viewModel: FrameAnalysisViewModel
   263→) {
   264→    var showAddDialog by remember { mutableStateOf(false) }
   265→    var editingNode by remember { mutableStateOf<FrameNode?>(null) }
   266→    var editX by remember { mutableStateOf("") }
   267→    var editY by remember { mutableStateOf("") }
   268→    var editSupport by remember { mutableStateOf(SupportType.Free) }
   269→
   270→    Column(modifier = Modifier.fillMaxSize()) {
   271→        // Header
   272→        Row(
   273→            modifier = Modifier
   274→                .fillMaxWidth()
   275→                .padding(12.dp),
   276→            horizontalArrangement = Arrangement.SpaceBetween,
   277→            verticalAlignment = Alignment.CenterVertically
   278→        ) {
   279→            Text("العقد (${nodes.size})", fontWeight = FontWeight.Bold, fontSize = 16.sp)
   280→            Button(
   281→                onClick = {
   282→                    editX = ""
   283→                    editY = ""
   284→                    editSupport = SupportType.Free
   285→                    editingNode = null
   286→                    showAddDialog = true
   287→                },
   288→                colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF1565C0))
   289→            ) {
   290→                Icon(Icons.Default.Add, contentDescription = null, modifier = Modifier.size(18.dp))
   291→                Spacer(modifier = Modifier.width(4.dp))
   292→                Text("إضافة عقدة")
   293→            }
   294→        }
   295→
   296→        if (nodes.isEmpty()) {
   297→            Box(modifier = Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
   298→                Column(horizontalAlignment = Alignment.CenterHorizontally) {
   299→                    Icon(Icons.Default.AddLocation, contentDescription = null, modifier = Modifier.size(48.dp), tint = Color.Gray)
   300→                    Spacer(modifier = Modifier.height(8.dp))
   301→                    Text("لا توجد عقد بعد", color = Color.Gray)
   302→                    Text("أضف عقد أو استخدم قالب جاهز", color = Color.Gray, fontSize = 12.sp)
   303→                }
   304→            }
   305→            return@Column
   306→        }
   307→
   308→        // Node list header
   309→        Row(
   310→            modifier = Modifier
   311→                .fillMaxWidth()
   312→                .background(Color(0xFFE3F2FD))
   313→                .padding(horizontal = 12.dp, vertical = 8.dp)
   314→        ) {
   315→            Text("ID", modifier = Modifier.width(40.dp), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   316→            Text("X (m)", modifier = Modifier.width(60.dp), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   317→            Text("Y (m)", modifier = Modifier.width(60.dp), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   318→            Text("الارتكاز", modifier = Modifier.weight(1f), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   319→            Text("", modifier = Modifier.width(60.dp))
   320→        }
   321→
   322→        LazyColumn(modifier = Modifier.fillMaxSize()) {
   323→            items(nodes, key = { it.id }) { node ->
   324→                val isConnected = members.any { it.nodeI == node.id || it.nodeJ == node.id }
   325→                Row(
   326→                    modifier = Modifier
   327→                        .fillMaxWidth()
   328→                        .then(if (isConnected) Modifier else Modifier.background(Color(0xFFFFF8E1)))
   329→                        .padding(horizontal = 12.dp, vertical = 8.dp),
   330→                    verticalAlignment = Alignment.CenterVertically
   331→                ) {
   332→                    Text("${node.id}", modifier = Modifier.width(40.dp), fontWeight = FontWeight.Bold)
   333→                    Text("${node.x}", modifier = Modifier.width(60.dp))
   334→                    Text("${node.y}", modifier = Modifier.width(60.dp))
   335→                    Text(node.support.displayNameAr, modifier = Modifier.weight(1f), fontSize = 12.sp)
   336→                    IconButton(onClick = {
   337→                        editingNode = node
   338→                        editX = node.x.toString()
   339→                        editY = node.y.toString()
   340→                        editSupport = node.support
   341→                        showAddDialog = true
   342→                    }) {
   343→                        Icon(Icons.Default.Edit, "تعديل", modifier = Modifier.size(18.dp), tint = Color(0xFF1565C0))
   344→                    }
   345→                }
   346→                HorizontalDivider()
   347→            }
   348→        }
   349→    }
   350→
   351→    // Add/Edit Dialog
   352→    if (showAddDialog) {
   353→        AlertDialog(
   354→            onDismissRequest = { showAddDialog = false },
   355→            title = Text(if (editingNode != null) "تعديل عقدة" else "إضافة عقدة"),
   356→            text = {
   357→                Column {
   358→                    OutlinedTextField(
   359→                        value = editX, onValueChange = { editX = it },
   360→                        label = { Text("X (متر)") },
   361→                        keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
   362→                        modifier = Modifier.fillMaxWidth()
   363→                    )
   364→                    Spacer(modifier = Modifier.height(8.dp))
   365→                    OutlinedTextField(
   366→                        value = editY, onValueChange = { editY = it },
   367→                        label = { Text("Y (متر)") },
   368→                        keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
   369→                        modifier = Modifier.fillMaxWidth()
   370→                    )
   371→                    Spacer(modifier = Modifier.height(8.dp))
   372→                    Text("نوع الارتكاز:", fontWeight = FontWeight.Bold, fontSize = 13.sp)
   373→                    Spacer(modifier = Modifier.height(4.dp))
   374→                    for (st in SupportType.entries) {
   375→                        Row(verticalAlignment = Alignment.CenterVertically, modifier = Modifier.padding(2.dp)) {
   376→                            RadioButton(
   377→                                selected = editSupport == st,
   378→                                onClick = { editSupport = st }
   379→                            )
   380→                            Text(st.displayNameAr, fontSize = 13.sp)
   381→                        }
   382→                    }
   383→                }
   384→            },
   385→            confirmButton = {
   386→                TextButton(onClick = {
   387→                    val x = editX.toDoubleOrNull() ?: return@TextButton
   388→                    val y = editY.toDoubleOrNull() ?: return@TextButton
   389→                    if (editingNode != null) {
   390→                        viewModel.updateNode(editingNode!!.id, x, y, editSupport)
   391→                    } else {
   392→                        viewModel.addNode(x, y, editSupport)
   393→                    }
   394→                    showAddDialog = false
   395→                }) { Text("حفظ", color = Color(0xFF1565C0)) }
   396→            },
   397→            dismissButton = {
   398→                TextButton(onClick = { showAddDialog = false }) { Text("إلغاء") }
   399→            }
   400→        )
   401→    }
   402→}

  1001→@Composable
  1002→private fun UtilizationBar(ratio: Double, modifier: Modifier = Modifier) {
  1003→    val clampedRatio = ratio.coerceIn(0.0, 1.5)
  1004→    val color = when {
  1005→        clampedRatio <= 0.7 -> Color(0xFF4CAF50)
  1006→        clampedRatio <= 1.0 -> Color(0xFFFF9800)
  1007→        else -> Color(0xFFF44336)
  1008→    }
  1009→    Row(modifier = modifier.height(16.dp), verticalAlignment = Alignment.CenterVertically) {
  1010→        LinearProgressIndicator(
  1011→            progress = { clampedRatio.toFloat() / 1.5f },
  1012→            modifier = Modifier
  1013→                .weight(1f)
  1014→                .height(8.dp)
  1015→                .clip(RoundedCornerShape(4.dp)),
  1016→            color = color,
  1017→            trackColor = Color.LightGray
  1018→        )
  1019→        Spacer(modifier = Modifier.width(8.dp))
  1020→        Text(
  1021→            "${String.format("%.0f", ratio * 100)}%",
  1022→            fontSize = 11.sp,
  1023→            fontWeight = FontWeight.Bold,
  1024→            color = color
  1025→        )
  1026→    }
