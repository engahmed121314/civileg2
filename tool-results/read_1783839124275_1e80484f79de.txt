   970→   970→@Composable
   971→   971→private fun UtilizationBar(ratio: Double, modifier: Modifier = Modifier) {
   972→   972→    val clampedRatio = ratio.coerceIn(0.0, 1.5)
   973→   973→    val color = when {
   974→   974→        clampedRatio <= 0.7 -> Color(0xFF4CAF50)
   975→   975→        clampedRatio <= 1.0 -> Color(0xFFFF9800)
   976→   976→        else -> Color(0xFFF44336)
   977→   977→    }
   978→   978→    Row(modifier = modifier.height(16.dp), verticalAlignment = Alignment.CenterVertically) {
   979→   979→        LinearProgressIndicator(
   980→   980→            progress = { clampedRatio.toFloat() / 1.5f },
   981→   981→            modifier = Modifier
   982→   982→                .weight(1f)
   983→   983→                .height(8.dp)
   984→   984→                .clip(RoundedCornerShape(4.dp)),
   985→   985→            color = color,
   986→   986→            trackColor = Color.LightGray
   987→   987→        )
   988→   988→        Spacer(modifier = Modifier.width(8.dp))
   989→   989→        Text(
   990→   990→            "${String.format("%.0f", ratio * 100)}%",
   991→   991→            fontSize = 11.sp,
   992→   992→            fontWeight = FontWeight.Bold,
   993→   993→            color = color
   994→   994→        )
   995→   995→    }
