    32→    private val PRIMARY_COLOR = DeviceRgb(21, 101, 192) // Professional Blue
    33→    private val SECONDARY_COLOR = DeviceRgb(69, 90, 100)
    34→    private val SUCCESS_COLOR = DeviceRgb(46, 125, 50)
    35→    private val ERROR_COLOR = DeviceRgb(198, 40, 40)

    39→    private fun getArabicFont(context: Context): PdfFont? {
    40→        return ArabicFontProvider.getArabicPdfFont(context)
    41→    }
    42→
    43→    private fun containsArabic(text: String): Boolean {
    44→        return ArabicFontProvider.containsArabic(text)
    45→    }
    46→
    47→    private fun createStyledParagraph(text: String, font: PdfFont?, fontSize: Float = 12f, isBold: Boolean = false): Paragraph {
    48→        val p = Paragraph(text).setFontSize(fontSize)
    49→        if (isBold) p.setBold()
    50→        if (font != null && containsArabic(text)) {
    51→            p.setFont(font)
    52→            p.setBaseDirection(BaseDirection.RIGHT_TO_LEFT)
    53→        }
    54→        return p
    55→    }
    56→
    57→    private fun addArabicCell(table: Table, text: String, font: PdfFont?, isBold: Boolean = false, fontSize: Float = 9f) {
    58→        val p = createStyledParagraph(text, font, fontSize, isBold)
    59→        table.addCell(Cell().add(p))
    60→    }
    61→
    62→    private fun createDataCell(label: String, value: String, font: PdfFont?): Cell {
    63→        val table = Table(UnitValue.createPercentArray(floatArrayOf(60f, 40f))).useAllAvailableWidth()
    64→        val pLabel = createStyledParagraph(label, font, 9f)
    65→        table.addCell(Cell().add(pLabel).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
    66→        val pValue = createStyledParagraph(value, font, 9f, isBold = true)
    67→        table.addCell(Cell().add(pValue).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
    68→        return Cell().add(table)
    69→    }
    70→
    71→    private fun addMemberRow(table: Table, type: String, section: String, status: String, font: PdfFont?) {
    72→        table.addCell(Cell().add(createStyledParagraph(type, font, 9f)))
    73→        table.addCell(Cell().add(createStyledParagraph(section, font, 9f, isBold = true)))
    74→        val statusP = createStyledParagraph(status, font, 9f, isBold = true)
    75→        if (status.contains("Safe", true) || status.contains("آمن", true)) {
    76→            statusP.setFontColor(SUCCESS_COLOR)
    77→        } else {
    78→            statusP.setFontColor(ERROR_COLOR)
    79→        }
    80→        table.addCell(Cell().add(statusP))
    81→    }
    82→
    83→    private fun createSummaryCell(label: String, value: String, font: PdfFont?): Cell {
    84→        val cell = Cell().setPadding(10f).setTextAlignment(TextAlignment.CENTER)
    85→        cell.add(createStyledParagraph(label, font, 10f).setFontColor(ColorConstants.GRAY))
    86→        cell.add(createStyledParagraph(value, font, 14f, isBold = true).setFontColor(PRIMARY_COLOR))
    87→        return cell
    88→    }

    90→    fun generateProfessionalReport(
    91→        context: Context,
    92→        title: String,
    93→        designType: String,
    94→        inputs: Map<String, String>,
    95→        results: Map<String, String>,
    96→        safetyChecks: List<CalculatorEngine.DesignSafetyCheck>,
    97→        isSafe: Boolean,
    98→        drawingBitmap: Bitmap? = null
    99→    ): File {
   100→        val fileName = "CivilEngPro_${designType}_${System.currentTimeMillis()}.pdf"
   101→        val file = File(context.getExternalFilesDir(null), fileName)
   102→
   103→        val writer = PdfWriter(file)
   104→        val pdf = PdfDocument(writer)
   105→        val document = Document(pdf)
   106→        val arabicFont = getArabicFont(context)
   107→
   108→        // --- Header Section ---
   109→        val header = Table(UnitValue.createPercentArray(floatArrayOf(70f, 30f))).useAllAvailableWidth()
   110→        val appNamePara = createStyledParagraph(context.getString(R.string.app_name), arabicFont, 26f, isBold = true)
   111→            .setFontColor(PRIMARY_COLOR)
   112→        header.addCell(Cell().add(appNamePara).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
   113→        
   114→        header.addCell(Cell().add(Paragraph("STRUCTURAL REPORT")
   115→            .setTextAlignment(TextAlignment.RIGHT).setFontSize(10f).setFontColor(ColorConstants.GRAY)).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
   116→        document.add(header)
   117→        
   118→        document.add(LineSeparator(SolidLine(1f)))
   119→        document.add(Paragraph("\n"))
   120→
   121→        // --- Title & Status ---
   122→        document.add(createStyledParagraph(title, arabicFont, 18f, isBold = true))
   123→
   124→        val statusText = if (isSafe) "STATUS: SAFE (مطابق للأكواد)" else "STATUS: UNSAFE (غير آمن - مراجعة التصميم)"
   125→        val statusPara = createStyledParagraph(statusText, arabicFont, 12f, isBold = true)
   126→            .setFontColor(if (isSafe) SUCCESS_COLOR else ERROR_COLOR)
   127→            .setPadding(10f)
   128→            .setBorder(com.itextpdf.layout.borders.SolidBorder(if (isSafe) SUCCESS_COLOR else ERROR_COLOR, 1f))
   129→        document.add(statusPara)
   130→        
   131→        document.add(Paragraph("\n"))
   132→
   133→        // --- Split Layout: Info vs Drawing ---
   134→        val mainTable = Table(UnitValue.createPercentArray(floatArrayOf(50f, 50f))).useAllAvailableWidth()
   135→
   136→        // Column 1: Inputs & Results
   137→        val infoCell = Cell().setBorder(com.itextpdf.layout.borders.Border.NO_BORDER)
   138→        infoCell.add(createStyledParagraph("DESIGN DATA", arabicFont, 12f, isBold = true).setFontColor(PRIMARY_COLOR))
   139→        
   140→        val dataTable = Table(UnitValue.createPercentArray(floatArrayOf(60f, 40f))).useAllAvailableWidth()
   141→        inputs.forEach { (k, v) ->
   142→            addArabicCell(dataTable, k, arabicFont, fontSize = 9f)
   143→            addArabicCell(dataTable, v, arabicFont, isBold = true, fontSize = 9f)
   144→        }
   145→        results.forEach { (k, v) ->
   146→            addArabicCell(dataTable, k, arabicFont, fontSize = 9f)
   147→            addArabicCell(dataTable, v, arabicFont, isBold = true, fontSize = 9f)
   148→        }
   149→        infoCell.add(dataTable)
   150→        mainTable.addCell(infoCell)
   151→
   152→        // Column 2: Drawing
   153→        val drawingCell = Cell().setBorder(com.itextpdf.layout.borders.Border.NO_BORDER).setPaddingLeft(10f)
   154→        drawingCell.add(createStyledParagraph("SECTION DRAWING (رسم تسليح القطاع)", arabicFont, 12f, isBold = true).setFontColor(PRIMARY_COLOR))
   155→        if (drawingBitmap != null) {
   156→            val stream = ByteArrayOutputStream()
   157→            drawingBitmap.compress(Bitmap.CompressFormat.PNG, 100, stream)
   158→            val img = Image(ImageDataFactory.create(stream.toByteArray()))
   159→            img.setAutoScale(true)
   160→            drawingCell.add(img)
   161→        } else {
   162→            drawingCell.add(Paragraph("[Drawing Placeholder]").setItalic().setFontColor(ColorConstants.GRAY))
   163→        }
   164→        mainTable.addCell(drawingCell)
   165→        
   166→        document.add(mainTable)
   167→        document.add(Paragraph("\n"))
   168→
   169→        // --- Safety Checks Table ---
   170→        if (safetyChecks.isNotEmpty()) {
   171→            document.add(createStyledParagraph("SAFETY VERIFICATIONS (تحققات الأمان)", arabicFont, 12f, isBold = true).setFontColor(PRIMARY_COLOR))
   172→            val checkTable = Table(UnitValue.createPercentArray(floatArrayOf(40f, 20f, 20f, 20f))).useAllAvailableWidth()
   173→            
   174→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Check Type", arabicFont, 9f, true)))
   175→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Calculated", arabicFont, 9f, true)))
   176→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Limit", arabicFont, 9f, true)))
   177→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Result", arabicFont, 9f, true)))
   178→
   179→            safetyChecks.forEach { check ->
   180→                addArabicCell(checkTable, check.name, arabicFont, fontSize = 9f)
   181→                checkTable.addCell(Cell().add(Paragraph(String.format(Locale.US, "%.2f %s", check.value, check.unit)).setFontSize(9f)))
   182→                checkTable.addCell(Cell().add(Paragraph(String.format(Locale.US, "%.2f %s", check.limit, check.unit)).setFontSize(9f)))
   183→                val resText = if (check.isSafe) "PASS" else "FAIL"
   184→                val resPara = createStyledParagraph(resText, arabicFont, 9f, isBold = true)
   185→                    .setFontColor(if (check.isSafe) SUCCESS_COLOR else ERROR_COLOR)
   186→                checkTable.addCell(Cell().add(resPara))
   187→            }
   188→            document.add(checkTable)
   189→        }
   190→
   191→        // --- Footer ---
   192→        document.add(Paragraph("\n\n"))
   193→        document.add(Paragraph("This report is generated based on Structural Design Codes (ECP/ACI/SBC). Professional review is mandatory before execution.")
   194→            .setFontSize(8f).setItalic().setFontColor(ColorConstants.GRAY).setTextAlignment(TextAlignment.CENTER))
   195→        
   196→        document.add(Paragraph(java.util.Date().toString())
   197→            .setFontSize(8f).setTextAlignment(TextAlignment.RIGHT))
   198→
   199→        document.close()
   200→        return file
   201→    }
