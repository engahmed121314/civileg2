   284→@Composable
   285→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   286→    InteractiveDrawingScreen(
   287→        title = "تحليل الإطار الإنشائي",
   288→        subtitle = "Frame Analysis & BMD"
   289→    ) {
   290→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFF0D1117))) {
   291→            val padding = 60f
   292→            val w = size.width - 2 * padding
   293→            val h = size.height - 2 * padding
   294→            val baseY = size.height - padding
   295→
   296→            // Scale calculations
   297→            val scale = (w / inputs.span).toFloat() * 0.75f
   298→            val eh = (inputs.eaveHeight * scale).toFloat()
   299→            val rh = (inputs.ridgeHeight * scale).toFloat()
   300→            val midX = padding + w / 2
   301→            val colLX = padding + (w - inputs.span * scale) / 2
   302→            val colRX = colLX + (inputs.span * scale).toFloat()
   303→
   304→            // ── Grid ──
   305→            val gridColor = Color(0x1AFFFFFF)
   306→            for (i in 0..20) {
   307→                val x = padding + (i * w / 20)
   308→                drawLine(gridColor, Offset(x, padding), Offset(x, size.height - padding), strokeWidth = 0.5f)
   309→            }
   310→            for (i in 0..12) {
   311→                val y = padding + (i * h / 12)
   312→                drawLine(gridColor, Offset(padding, y), Offset(size.width - padding, y), strokeWidth = 0.5f)
   313→            }
   314→
   315→            // ── Ground Line with hatching ──
   316→            val groundColor = Color(0xFF4A4A4A)
   317→            drawLine(groundColor, Offset(colLX - 30f, baseY), Offset(colRX + 30f, baseY), strokeWidth = 3f)
   318→            for (i in 0..24) {
   319→                val x = (colLX - 30f) + (i * (colRX - colLX + 60f) / 24)
   320→                drawLine(groundColor, Offset(x, baseY), Offset(x - 8f, baseY + 10f), strokeWidth = 1f)
   321→            }
   322→
   323→            // ── Pin Supports ──
   324→            val supportColor = Color(0xFF8A8A8A)
   325→            // Left support
   326→            drawPath(Path().apply {
   327→                moveTo(colLX - 12f, baseY)
   328→                lineTo(colLX + 12f, baseY)
   329→                lineTo(colLX, baseY + 18f)
   330→                close()
   331→            }, supportColor, style = Stroke(2f))
   332→            // Right support
   333→            drawPath(Path().apply {
   334→                moveTo(colRX - 12f, baseY)
   335→                lineTo(colRX + 12f, baseY)
   336→                lineTo(colRX, baseY + 18f)
   337→                close()
   338→            }, supportColor, style = Stroke(2f))
   339→
   340→            // ── Main Frame with 3D effect ──
   341→            val frameColor = Color(0xFFB0BEC5)
   342→            val frameShadow = Color(0xFF546E7A)
   343→            val frameHighlight = Color(0xFFECEFF1)
   344→            val memberWidth = 6f
   345→
   346→            // Columns (draw shadow, main, highlight for 3D look)
   347→            // Left column
   348→            drawLine(frameShadow, Offset(colLX + 2f, baseY), Offset(colLX + 2f, baseY - eh), strokeWidth = memberWidth + 2f)
   349→            drawLine(frameColor, Offset(colLX, baseY), Offset(colLX, baseY - eh), strokeWidth = memberWidth)
   350→            drawLine(frameHighlight, Offset(colLX - 1f, baseY), Offset(colLX - 1f, baseY - eh), strokeWidth = 1f)
   351→
   352→            // Right column
   353→            drawLine(frameShadow, Offset(colRX + 2f, baseY), Offset(colRX + 2f, baseY - eh), strokeWidth = memberWidth + 2f)
   354→            drawLine(frameColor, Offset(colRX, baseY), Offset(colRX, baseY - eh), strokeWidth = memberWidth)
   355→            drawLine(frameHighlight, Offset(colRX - 1f, baseY), Offset(colRX - 1f, baseY - eh), strokeWidth = 1f)
   356→
   357→            // Rafters (left)
   358→            drawLine(frameShadow, Offset(colLX + 2f, baseY - eh + 2f), Offset(midX + 1f, baseY - rh + 1f), strokeWidth = memberWidth + 2f)
   359→            drawLine(frameColor, Offset(colLX, baseY - eh), Offset(midX, baseY - rh), strokeWidth = memberWidth)
   360→            // Rafters (right)
   361→            drawLine(frameShadow, Offset(colRX + 2f, baseY - eh + 2f), Offset(midX + 1f, baseY - rh + 1f), strokeWidth = memberWidth + 2f)
   362→            drawLine(frameColor, Offset(colRX, baseY - eh), Offset(midX, baseY - rh), strokeWidth = memberWidth)
   363→
   364→            // ── Joint Circles ──
   365→            val jointColor = Color(0xFFFF9800)
   366→            listOf(
   367→                Offset(colLX, baseY - eh),
   368→                Offset(colRX, baseY - eh),
   369→                Offset(midX, baseY - rh)
   370→            ).forEach { joint ->
   371→                drawCircle(Color(0x33FF9800), 14f, joint)
   372→                drawCircle(jointColor, 6f, joint)
   373→                drawCircle(Color.White, 3f, joint)
   374→            }
   375→
   376→            // ── Distributed Load Arrows (on rafters) ──
   377→            val loadColor = Color(0xFFEF5350)
   378→            for (i in 0..14) {
   379→                val t = i / 14f
   380→                // Left rafter
   381→                val xL = colLX + t * (midX - colLX)
   382→                val yL = baseY - eh + t * (rh - eh) * (-1f)
   383→                val nxL = -(baseY - eh - (baseY - rh)) // normal direction
   384→                val lenL = (midX - colLX)
   385→                val nxNorm = ((baseY - rh) - (baseY - eh)) / lenL
   386→                val nyNorm = (midX - colLX) / lenL
   387→                val arrowLen = 25f
   388→                drawLine(loadColor, Offset(xL - nxNorm * arrowLen, yL - nyNorm * arrowLen), Offset(xL, yL), strokeWidth = 1.5f)
   389→                // Arrow head
   390→                drawLine(loadColor, Offset(xL - 4f, yL - 5f), Offset(xL, yL), strokeWidth = 1.5f)
   391→                drawLine(loadColor, Offset(xL + 4f, yL - 5f), Offset(xL, yL), strokeWidth = 1.5f)
   392→
   393→                // Right rafter
   394→                val xR = midX + t * (colRX - midX)
   395→                val yR = (baseY - rh) + t * ((baseY - eh) - (baseY - rh))
   396→                drawLine(loadColor, Offset(xR + nxNorm * arrowLen, yR - nyNorm * arrowLen), Offset(xR, yR), strokeWidth = 1.5f)
   397→                drawLine(loadColor, Offset(xR - 4f, yR - 5f), Offset(xR, yR), strokeWidth = 1.5f)
   398→                drawLine(loadColor, Offset(xR + 4f, yR - 5f), Offset(xR, yR), strokeWidth = 1.5f)
   399→            }
   400→
   401→            // Load label
   402→            drawContext.canvas.nativeCanvas.drawText(
   403→                "w = %.1f kN/m".format(result.mainFrame.maxShear / max(inputs.span, 1.0)),
   404→                (colLX + colRX) / 2f - 60f, baseY - rh - 35f,
   405→                android.graphics.Paint().apply { color = loadColor.toArgb(); textSize = 28f; isFakeBoldText = true }
   406→            )
   407→
   408→            // ── Bending Moment Diagram (filled) ──
   409→            val momentColor = Color(0xFF42A5F5)
   410→            val maxMoment = result.mainFrame.maxMoment
   411→            val momentScale = min(eh * 0.35f / max(maxMoment, 1f), 0.5f)
   412→
   413→            // Left rafter moment (parabolic)
   414→            val momentPath = Path().apply {
   415→                moveTo(colLX, baseY - eh)
   416→                for (i in 0..20) {
   417→                    val t = i / 20f
   418→                    val x = colLX + t * (midX - colLX)
   419→                    val y = baseY - eh + t * (rh - eh) * (-1f)
   420→                    val m = 4 * maxMoment * t * (1 - t)
   421→                    val nxN = ((baseY - rh) - (baseY - eh)) / (midX - colLX)
   422→                    val nyN = (midX - colLX) / (midX - colLX)
   423→                    lineTo(x + nxN * m * momentScale, y + nyN * m * momentScale)
   424→                }
   425→                lineTo(midX, baseY - rh)
   426→                close()
   427→            }
   428→            drawPath(momentPath, momentColor.copy(alpha = 0.2f))
   429→            drawPath(momentPath, momentColor, style = Stroke(2f))
   430→
   431→            // Right rafter moment (mirror)
   432→            val momentPathR = Path().apply {
   433→                moveTo(colRX, baseY - eh)
   434→                for (i in 0..20) {
   435→                    val t = i / 20f
   436→                    val x = colRX + t * (midX - colRX)
   437→                    val y = (baseY - eh) + t * ((baseY - rh) - (baseY - eh))
   438→                    val m = 4 * maxMoment * t * (1 - t)
   439→                    val nxN = ((baseY - rh) - (baseY - eh)) / (midX - colRX)
   440→                    val nyN = (midX - colRX) / (midX - colRX)
   441→                    lineTo(x + nxN * m * momentScale, y + nyN * m * momentScale)
   442→                }
   443→                lineTo(midX, baseY - rh)
   444→                close()
   445→            }
   446→            drawPath(momentPathR, momentColor.copy(alpha = 0.2f))
   447→            drawPath(momentPathR, momentColor, style = Stroke(2f))
   448→
   449→            // Column moment (triangular)
   450→            val colMoment = maxMoment * 0.3f
   451→            drawPath(Path().apply {
   452→                moveTo(colLX, baseY - eh)
   453→                lineTo(colLX - colMoment * momentScale * 0.5f, baseY - eh / 2)
   454→                lineTo(colLX, baseY)
   455→                close()
   456→            }, momentColor.copy(alpha = 0.15f))
   457→            drawPath(Path().apply {
   458→                moveTo(colLX, baseY - eh)
   459→                lineTo(colLX - colMoment * momentScale * 0.5f, baseY - eh / 2)

   829→                        drawLine(Color(0xFF4FC3F7), Offset(x, eaveY - 8f), Offset(x, eaveY + 8f), strokeWidth = 2f)
   830→                    }
   831→
   832→                    // Bracing in first bay
   833→                    if (numBays >= 1) {
   834→                        val bx1 = sideLeft
   835→                        val bx2 = sideLeft + (inputs.baySpacing * scale).toFloat().coerceAtMost(planL)
   836→                        drawLine(Color(0x66FF5722), Offset(bx1, groundY), Offset(bx2, eaveY), strokeWidth = 1.5f,
   837→                            pathEffect = PathEffect.dashPathEffect(floatArrayOf(10f, 5f)))
   838→                        drawLine(Color(0x66FF5722), Offset(bx2, groundY), Offset(bx1, eaveY), strokeWidth = 1.5f,
   839→                            pathEffect = PathEffect.dashPathEffect(floatArrayOf(10f, 5f)))
   840→                    }
   841→
   842→                    // Girt markers on columns
   843→                    val numGirts = (inputs.eaveHeight / 1.5).toInt().coerceIn(1, 4)
   844→                    for (i in 1..numGirts) {
   845→                        val gy = groundY - (i * eaveH / (numGirts + 1))
   846→                        drawLine(Color(0x40FF9800), Offset(sideLeft, gy), Offset(sideLeft + planL, gy), strokeWidth = 0.5f,
   847→                            pathEffect = PathEffect.dashPathEffect(floatArrayOf(6f, 4f)))
   848→                    }
   849→
   850→                    // Dimensions
   851→                    drawContext.canvas.nativeCanvas.drawText("SIDE ELEVATION", size.width / 2, padding - 10f, textPaint)
   852→
   853→                    // Length dimension
   854→                    val dY = groundY + 35f
   855→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft, groundY + 8f), Offset(sideLeft, dY + 5f), strokeWidth = 1f)
   856→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft + planL, groundY + 8f), Offset(sideLeft + planL, dY + 5f), strokeWidth = 1f)
   857→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft, dY), Offset(sideLeft + planL, dY), strokeWidth = 1.5f)
   858→                    drawContext.canvas.nativeCanvas.drawText("%.1f m".format(inputs.length), sideLeft + planL / 2, dY - 6f, dimPaint)
   859→
   860→                    // Eave height dimension
   861→                    val dX = sideLeft - 35f
   862→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft - 8f, groundY), Offset(dX - 5f, groundY), strokeWidth = 1f)
   863→                    drawLine(Color(0xFF90A4AE), Offset(sideLeft - 8f, eaveY), Offset(dX - 5f, eaveY), strokeWidth = 1f)
   864→                    drawLine(Color(0xFF90A4AE), Offset(dX, groundY), Offset(dX, eaveY), strokeWidth = 1.5f)
   865→                    drawContext.canvas.nativeCanvas.save()
   866→                    drawContext.canvas.nativeCanvas.rotate(-90f, dX - 10f, (groundY + eaveY) / 2)
   867→                    drawContext.canvas.nativeCanvas.drawText("%.1f m".format(inputs.eaveHeight), dX - 10f, (groundY + eaveY) / 2 + 6f, dimPaint)
   868→                    drawContext.canvas.nativeCanvas.restore()
   869→                }
   870→
   871→                3 -> { // 3D Isometric — Professional
   872→                    val s = scale * 0.65f
   873→                    val depthOff = 40f
   874→                    val numFrames = 5
   875→
   876→                    for (frameIdx in numFrames downTo 0) {
   877→                        val off = frameIdx * depthOff
   878→                        val x0 = startX + off
   879→                        val y0 = baseY - off
   880→                        val spanW = (inputs.span * s).toFloat()
   881→                        val eaveH = (inputs.eaveHeight * s).toFloat()
   882→                        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   883→                        val midXf = x0 + spanW / 2
   884→

   886→                        val color = Color.White.copy(alpha = alpha)
   887→                        val stroke = if (frameIdx == 0) 3f else 1.5f
   888→
   889→                        // Columns
   890→                        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   891→                        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   892→                        // Rafters
   893→                        drawLine(color, Offset(x0, y0 - eaveH), Offset(midXf, y0 - ridgeH), strokeWidth = stroke)
   894→                        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(midXf, y0 - ridgeH), strokeWidth = stroke)
   895→
   896→                        // Eave line (longitudinal)
   897→                        if (frameIdx > 0) {
   898→                            val prevOff = (frameIdx - 1) * depthOff
   899→                            val px = startX + prevOff
   900→                            val py = baseY - prevOff
   901→                            val lineC = Color.White.copy(alpha = 0.1f + 0.08f * frameIdx)
   902→                            drawLine(lineC, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   903→                            drawLine(lineC, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   904→                            drawLine(lineC, Offset(midXf, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   905→                        }
   906→                    }
   907→
   908→                    // Ground plane
   909→                    val groundColor = Color(0x15FFFFFF)
   910→                    val gOff = numFrames * depthOff
   911→                    drawRect(groundColor, Offset(startX, baseY - gOff - 10f), Size((inputs.span * s).toFloat(), gOff + 10f))
   912→
   913→                    drawContext.canvas.nativeCanvas.drawText("3D ISOMETRIC VIEW", size.width / 2, padding - 10f, textPaint)
   914→                }
   915→            }
   916→        }
   917→    }

   952→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {

   954→        color = android.graphics.Color.BLACK

   956→        isAntiAlias = true

  1005→

  1022→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
  1023→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
  1024→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint

  1027→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
  1028→    val tickSize = 15f
  1029→    canvas.drawLine(x1, y1, x2, y2, paint)
  1030→    
  1031→    // Draw Ticks (45 degree lines)
  1032→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
  1033→    val tickAngle = angle + Math.PI / 4
  1034→    
  1035→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
  1036→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
  1037→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
  1038→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
  1039→
  1040→    // Text label
  1041→    val midX = (x1 + x2) / 2
  1042→    val midY = (y1 + y2) / 2
  1043→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
  1044→    
  1045→    if (x1 == x2) { // Vertical
  1046→        canvas.save()
  1047→        canvas.rotate(-90f, midX - 10f, midY)
  1048→        canvas.drawText(text, midX - 10f, midY, textPaint)
  1049→        canvas.restore()
  1050→    } else { // Horizontal
  1051→        canvas.drawText(text, midX, midY - 10f, textPaint)
  1052→    }
  1053→}

  1055→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
  1056→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
  1057→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
  1058→    
  1059→    val s = scale * 0.6f
  1060→    val w = inputs.span.toFloat() * s
  1061→    val l = inputs.length.toFloat() * s
  1062→    val planTop = y - l
  1063→    
  1064→    canvas.drawRect(x, planTop, x + w, y, p)
  1065→    
  1066→    val baySpacing = inputs.baySpacing.toFloat() * s
  1067→    var curY = planTop
  1068→    var count = 0
  1069→    while (curY <= y + 0.1f) {
  1070→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
  1071→        
  1072→        // Column markers
  1073→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
  1074→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
  1075→        
  1076→        if (curY + baySpacing <= y + 0.1f) {
  1077→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
  1078→        }
  1079→        
  1080→        curY += baySpacing
  1081→        count++
  1082→    }
  1083→    
  1084→    // Span Dimension
  1085→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
  1086→    // Total Length Dimension
  1087→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
  1088→}

  1090→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
  1091→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
  1092→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
  1093→    
  1094→    val s = scale * 0.6f
  1095→    val l = inputs.length.toFloat() * s
  1096→    val eh = inputs.eaveHeight.toFloat() * s
  1097→    
  1098→    canvas.drawLine(x, y, x + l, y, p) // Ground
  1099→    val bay = inputs.baySpacing.toFloat() * s
  1100→    var cur = x
  1101→    while (cur <= x + l + 0.1f) {
  1102→        canvas.drawLine(cur, y, cur, y - eh, p)
  1103→        cur += bay
  1104→    }
  1105→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
  1106→    
  1107→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
  1108→}
