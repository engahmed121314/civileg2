   281→@Composable
   282→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   283→    Card(
   284→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   285→        border = BorderStroke(1.dp, Color.LightGray)
   286→    ) {
   287→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   288→            val padding = 40.dp.toPx()
   289→            val w = size.width - 2 * padding
   290→            val h = size.height - 2 * padding
   291→            val baseY = size.height - padding
   292→            
   293→            // Scale calculations
   294→            val scale = (w / inputs.span).toFloat()
   295→            val eh = (inputs.eaveHeight * scale).toFloat()
   296→            val rh = (inputs.ridgeHeight * scale).toFloat()
   297→            val midX = padding + w / 2
   298→            
   299→            // Draw Main Frame Outline
   300→            val framePath = Path().apply {
   301→                moveTo(padding, baseY)
   302→                lineTo(padding, baseY - eh)
   303→                lineTo(midX, baseY - rh)
   304→                lineTo(padding + w, baseY - eh)
   305→                lineTo(padding + w, baseY)
   306→            }
   307→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   308→            
   309→            // Draw Distributed Load Arrows
   310→            val loadColor = Color(0xFFE53935)
   311→            for (i in 0..12) {
   312→                val x = padding + (i * w / 12)
   313→                val y = if (x <= midX) {
   314→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   315→                } else {
   316→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   317→                }
   318→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   319→                // Arrow heads
   320→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   321→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   322→            }
   323→            
   324→            // Draw Moment Diagram (Simplified)
   325→            val momentColor = Color(0xFF1E88E5)
   326→            val momentPath = Path().apply {
   327→                moveTo(padding, baseY - eh)
   328→                // Simulated parabolic curve for moment
   329→                cubicTo(
   330→                    padding + w * 0.25f, baseY - eh + 50f,
   331→                    padding + w * 0.75f, baseY - eh + 50f,
   332→                    padding + w, baseY - eh
   333→                )
   334→            }
   335→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   336→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   337→            
   338→            // Labels
   339→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   340→        }
   341→    }
   342→}

   393→                val baseY = size.height - padding
   394→
   395→                when (viewMode) {
   396→                    0 -> { // Front Elevation

   399→                    1 -> { // Plan View

   401→                    }
   402→                    2 -> { // Side View
   403→                        drawSideElevation(inputs, scale, startX, baseY)
   404→                    }
   405→                    3 -> { // 3D Perspective
   406→                        draw3DView(inputs, scale, startX, baseY)
   407→                    }
   408→                }
   409→            }
   410→        }
   411→    }
   412→}

   414→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawFrontElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   415→    val colLX = startX
   416→    val colRX = startX + (inputs.span * scale).toFloat()
   417→    val eaveY = baseY - (inputs.eaveHeight * scale).toFloat()
   418→    val ridgeY = baseY - (inputs.ridgeHeight * scale).toFloat()
   419→    val midX = startX + (inputs.span * scale / 2).toFloat()
   420→
   421→    // Columns
   422→    drawLine(Color.Black, Offset(colLX, baseY), Offset(colLX, eaveY), strokeWidth = 8f)
   423→    drawLine(Color.Black, Offset(colRX, baseY), Offset(colRX, eaveY), strokeWidth = 8f)
   424→    // Rafters
   425→    drawLine(Color.Black, Offset(colLX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   426→    drawLine(Color.Black, Offset(colRX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   427→    
   428→    // Foundations
   429→    drawRect(Color.Gray, Offset(colLX - 15f, baseY), Size(30f, 15f))
   430→    drawRect(Color.Gray, Offset(colRX - 15f, baseY), Size(30f, 15f))
   431→}
   432→
   433→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   434→    val planW = (inputs.span * scale).toFloat()
   435→    val planL = (inputs.length * scale).toFloat()
   436→    val planTop = baseY - planL
   437→    
   438→    // Boundary
   439→    drawRect(Color.Black, Offset(startX, planTop), Size(planW, planL), style = androidx.compose.ui.graphics.drawscope.Stroke(4f))
   440→    
   441→    // Grid Lines (Bays)
   442→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   443→    for (i in 0..numBays) {
   444→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()
   445→        drawLine(Color.LightGray, Offset(startX, y), Offset(startX + planW, y), strokeWidth = 1f)
   446→        // Columns markers
   447→        drawCircle(Color.Black, 4f, Offset(startX, y))
   448→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))
   449→    }
   450→}
   451→
   452→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSideElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   453→    val planL = (inputs.length * scale).toFloat()
   454→    val eaveH = (inputs.eaveHeight * scale).toFloat()
   455→    val eaveY = baseY - eaveH
   456→    
   457→    // Ground
   458→    drawLine(Color.Gray, Offset(startX, baseY), Offset(startX + planL, baseY), strokeWidth = 2f)
   459→    
   460→    // Columns
   461→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   462→    for (i in 0..numBays) {
   463→        val x = startX + (i * inputs.baySpacing * scale).toFloat()
   464→        drawLine(Color.Black, Offset(x, baseY), Offset(x, eaveY), strokeWidth = 6f)
   465→    }
   466→    // Eave Line
   467→    drawLine(Color.Black, Offset(startX, eaveY), Offset(startX + planL, eaveY), strokeWidth = 6f)
   468→}

   470→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   471→    val s = scale.toFloat() * 0.65f
   472→    val depthOff = 35f
   473→    
   474→    // Draw from back to front for better perspective look
   475→    for (i in 4 downTo 0) {
   476→        val off = i * depthOff
   477→        val x0 = startX + off
   478→        val y0 = baseY - off
   479→        val spanW = (inputs.span * s).toFloat()
   480→        val eaveH = (inputs.eaveHeight * s).toFloat()
   481→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   482→        
   483→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   484→        val stroke = if (i == 0) 3f else 1.5f
   485→        
   486→        // Sections of the frame
   487→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   488→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   489→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   490→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   491→        
   492→        // Purlins/Girts connections (longitudinal)
   493→        if (i > 0) {
   494→            val px = x0 - depthOff
   495→            val py = y0 + depthOff
   496→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   497→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   498→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   499→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   500→        }
   501→    }
   502→}

   536→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   537→    val p = Paint().apply { 
   538→        color = android.graphics.Color.BLACK
   539→        strokeWidth = 6f
   540→        isAntiAlias = true
   541→        style = Paint.Style.STROKE
   542→    }
   543→    val dimPaint = Paint().apply {
   544→        color = android.graphics.Color.DKGRAY
   545→        strokeWidth = 2f
   546→        textSize = 24f
   547→        isAntiAlias = true
   548→    }
   549→    val textP = Paint().apply {
   550→        color = android.graphics.Color.BLACK
   551→        textSize = 22f
   552→        isFakeBoldText = true
   553→        textAlign = Paint.Align.CENTER
   554→    }
   555→    
   556→    val s = scale * 0.7f
   557→    val w = inputs.span.toFloat() * s
   558→    val eh = inputs.eaveHeight.toFloat() * s
   559→    val rh = inputs.ridgeHeight.toFloat() * s
   560→    val midX = x + w/2
   561→    
   562→    // Draw Main Frame (Columns & Rafters)
   563→    canvas.drawLine(x, y, x, y - eh, p)
   564→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   565→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   566→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   567→    
   568→    // Foundations
   569→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   570→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   571→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   572→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   573→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   574→    p.strokeWidth = 6f
   575→    
   576→    // --- Dimension Lines ---
   577→    // Span Dimension
   578→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   579→    
   580→    // Eave Height Dimension
   581→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   582→    
   583→    // Ridge Height Dimension
   584→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   585→
   586→    // --- Section Labels ---
   587→    result?.let { res ->
   588→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   589→        
   590→        // Column Label
   591→        canvas.save()
   592→        canvas.rotate(-90f, x + 25f, y - eh/2)
   593→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   594→        canvas.restore()
   595→        
   596→        // Rafter Label
   597→        canvas.save()
   598→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   599→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   600→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   601→        canvas.restore()
   602→    }
   603→
   604→    // Connection Markers (Circles at joints)
   605→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   606→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   607→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   608→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   609→}

   611→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   612→    val tickSize = 15f
   613→    canvas.drawLine(x1, y1, x2, y2, paint)
   614→    
   615→    // Draw Ticks (45 degree lines)
   616→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   617→    val tickAngle = angle + Math.PI / 4
   618→    
   619→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   620→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   621→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   622→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   623→
   624→    // Text label
   625→    val midX = (x1 + x2) / 2
   626→    val midY = (y1 + y2) / 2
   627→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   628→    
   629→    if (x1 == x2) { // Vertical
   630→        canvas.save()
   631→        canvas.rotate(-90f, midX - 10f, midY)
   632→        canvas.drawText(text, midX - 10f, midY, textPaint)
   633→        canvas.restore()
   634→    } else { // Horizontal
   635→        canvas.drawText(text, midX, midY - 10f, textPaint)
   636→    }
   637→}

   639→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   640→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   641→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   642→    
   643→    val s = scale * 0.6f
   644→    val w = inputs.span.toFloat() * s
   645→    val l = inputs.length.toFloat() * s
   646→    val planTop = y - l
   647→    
   648→    canvas.drawRect(x, planTop, x + w, y, p)
   649→    
   650→    val baySpacing = inputs.baySpacing.toFloat() * s
   651→    var curY = planTop
   652→    var count = 0
   653→    while (curY <= y + 0.1f) {
   654→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   655→        
   656→        // Column markers
   657→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   658→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   659→        
   660→        if (curY + baySpacing <= y + 0.1f) {
   661→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   662→        }
   663→        
   664→        curY += baySpacing
   665→        count++
   666→    }
   667→    
   668→    // Span Dimension
   669→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   670→    // Total Length Dimension
   671→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   672→}

   674→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   675→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   676→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   677→    
   678→    val s = scale * 0.6f
   679→    val l = inputs.length.toFloat() * s
   680→    val eh = inputs.eaveHeight.toFloat() * s
   681→    
   682→    canvas.drawLine(x, y, x + l, y, p) // Ground
   683→    val bay = inputs.baySpacing.toFloat() * s
   684→    var cur = x
   685→    while (cur <= x + l + 0.1f) {
   686→        canvas.drawLine(cur, y, cur, y - eh, p)
   687→        cur += bay
   688→    }
   689→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   690→    
   691→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   692→}

  1344→@Composable
  1345→fun SteelSectionDrawing(section: SteelSectionType) {
  1346→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1347→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1348→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1349→                val w = size.width
  1350→                val h = size.height
  1351→                
  1352→                when (section) {
  1353→                    is SteelSectionType.ISection -> {
  1354→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1355→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1356→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1357→                        val offsetBF = (w - bf) / 2
  1358→                        
  1359→                        // Top Flange
  1360→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1361→                        // Bottom Flange
  1362→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1363→                        // Web
  1364→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1365→                        
  1366→                        // Stress Distribution (Schematic)
  1367→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1368→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1369→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1370→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1371→                    }
  1372→                    is SteelSectionType.LSection -> {
  1373→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
