    35→@OptIn(ExperimentalMaterial3Api::class)
    36→@Composable
    37→fun FrameAnalysisScreen(
    38→    viewModel: FrameAnalysisViewModel = hiltViewModel(),
    39→    onNavigateBack: () -> Unit = {}
    40→) {
    41→    val nodes by viewModel.nodes.observeAsState(emptyList())
    42→    val members by viewModel.members.observeAsState(emptyList())

    44→    val memberLoads by viewModel.memberLoads.observeAsState(emptyList())
    45→    val settings by viewModel.settings.observeAsState(FrameAnalysisSettings())
    46→    val result by viewModel.result.observeAsState()

    48→    val errorMsg by viewModel.errorMessage.observeAsState()
    49→    val diagramType by viewModel.activeDiagramType.observeAsState(DiagramType.BMD)
    50→    val selectedMemberId by viewModel.selectedMemberId.observeAsState()
    51→    val concreteResults by viewModel.concreteResults.observeAsState(emptyList())
    52→    val steelResults by viewModel.steelResults.observeAsState(emptyList())
    53→    val context = LocalContext.current
    54→
    55→    var selectedTab by remember { mutableIntStateOf(0) }
    56→    val tabs = listOf("الرسم", "العقد", "الأعضاء", "الحمولات", "النتائج")
    57→
    58→    // Show errors
    59→    LaunchedEffect(errorMsg) {
    60→        errorMsg?.let { Toast.makeText(context, it, Toast.LENGTH_LONG).show() }
    61→    }
    62→
    63→    Scaffold(
    64→        topBar = {
    65→            TopAppBar(
    66→                title = { Text("تحليل وتصميم الإطارات", fontWeight = FontWeight.Bold) },
    67→                navigationIcon = {
    68→                    IconButton(onClick = onNavigateBack) {
    69→                        Icon(Icons.AutoMirrored.Filled.ArrowBack, contentDescription = "رجوع")
    70→                    }
    71→                },
    72→                colors = TopAppBarDefaults.topAppBarColors(
    73→                    containerColor = Color(0xFF1565C0),
    74→                    titleContentColor = Color.White,
    75→                    navigationIconContentColor = Color.White
    76→                ),
    77→                actions = {
    78→                    // Template buttons
    79→                    IconButton(onClick = {
    80→                        var span = 6.0; var h = 4.0; var udl = 20.0
    81→                        viewModel.loadSimplePortalFrame(span, h, udl)
    82→                    }) {
    83→                        Icon(Icons.Default.ViewInAr, "قالب إطار بسيط", tint = Color.White)
    84→                    }
    85→                    IconButton(onClick = { viewModel.clearAll() }) {
    86→                        Icon(Icons.Default.DeleteSweep, "مسح الكل", tint = Color.White)
    87→                    }
    88→                    // PDF Export
    89→                    if (result?.hasResults == true) {
    90→                        IconButton(onClick = {
    91→                            val inputs = viewModel.getStoredInputs()
    92→                            try {
    93→                                val file = FrameAnalysisPdfExporter.generateFrameAnalysisPdf(
    94→                                    context, inputs.nodes, inputs.members, inputs.nodalLoads,
    95→                                    inputs.memberLoads, inputs.settings, inputs.result
    96→                                )

   101→                                    type = "application/pdf"
   102→                                    putExtra(Intent.EXTRA_STREAM, uri)
   103→                                    putExtra(Intent.EXTRA_SUBJECT, "Frame Analysis Report")
   104→                                    addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
   105→                                }
   106→                                context.startActivity(Intent.createChooser(intent, "مشاركة التقرير"))
   107→                            } catch (e: Exception) {
   108→                                Toast.makeText(context, "خطأ في إنشاء PDF: ${e.message}", Toast.LENGTH_LONG).show()
   109→                            }
   110→                        }) {
   111→                            Icon(Icons.Default.PictureAsPdf, "تصدير PDF", tint = Color.White)
   112→                        }
   113→                    }
   114→                }
   115→            )
   116→        },
   117→        bottomBar = {
   118→            // Diagram type selector (when solved)
   119→            if (result?.hasResults == true) {
   120→                Surface(
   121→                    tonalElevation = 8.dp,
   122→                    shadowElevation = 8.dp,
   123→                    color = MaterialTheme.colorScheme.surface

   127→                            .fillMaxWidth()

   129→                        horizontalArrangement = Arrangement.Center

   133→                            FilterChip(
   134→                                selected = isSelected,

   136→                                label = { Text(type.displayNameAr, fontSize = 13.sp) },
   137→                                colors = FilterChipDefaults.filterChipColors(
   138→                                    selectedContainerColor = when (type) {
   139→                                        DiagramType.BMD -> Color(0xFF2196F3)

   141→                                        DiagramType.AFD -> Color(0xFFFF9800)

   144→                                ),
   145→                                modifier = Modifier.padding(horizontal = 4.dp)

   147→                        }

   149→                }
   150→            }
   151→        }
   152→    ) { padding ->
   153→        Column(modifier = Modifier.padding(padding)) {
   154→            // Tab Row

   157→                containerColor = Color(0xFF1565C0).copy(alpha = 0.1f),
   158→                contentColor = Color(0xFF1565C0)
   159→            ) {
   160→                tabs.forEachIndexed { index, title ->
   161→                    Tab(
   162→                        selected = selectedTab == index,
   163→                        onClick = { selectedTab = index },
   164→                        text = { Text(title, fontSize = 13.sp) }
   165→                    )
   166→                }
   167→            }
   168→
   169→            // Tab Content
   170→            when (selectedTab) {
   171→                0 -> DrawingTab(nodes, members, memberLoads, nodalLoads, result, diagramType, selectedMemberId, viewModel)
   172→                1 -> NodesTab(nodes, members, viewModel)
   173→                2 -> MembersTab(nodes, members, settings, viewModel)
   174→                3 -> LoadsTab(nodes, members, nodalLoads, memberLoads, viewModel)
   175→                4 -> ResultsTab(result, concreteResults, steelResults, selectedMemberId, viewModel)
   176→            }
   177→        }
   178→    }
   179→}

   248→@Composable
   249→private fun NodesTab(
   250→    nodes: List<FrameNode>,
   251→    members: List<FrameMember>,
   252→    viewModel: FrameAnalysisViewModel
   253→) {

   255→    var editingNode by remember { mutableStateOf<FrameNode?>(null) }
   256→    var editX by remember { mutableStateOf("") }
   257→    var editY by remember { mutableStateOf("") }
   258→    var editSupport by remember { mutableStateOf(SupportType.Free) }
   259→
   260→    Column(modifier = Modifier.fillMaxSize()) {
   261→        // Header
   262→        Row(
   263→            modifier = Modifier
   264→                .fillMaxWidth()
   265→                .padding(12.dp),
   266→            horizontalArrangement = Arrangement.SpaceBetween,

   268→        ) {
   269→            Text("العقد (${nodes.size})", fontWeight = FontWeight.Bold, fontSize = 16.sp)
   270→            Button(
   271→                onClick = {
   272→                    editX = ""
   273→                    editY = ""
   274→                    editSupport = SupportType.Free
   275→                    editingNode = null
   276→                    showAddDialog = true
   277→                },
   278→                colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF1565C0))
   279→            ) {
   280→                Icon(Icons.Default.Add, contentDescription = null, modifier = Modifier.size(18.dp))
   281→                Spacer(modifier = Modifier.width(4.dp))
   282→                Text("إضافة عقدة")

   284→        }

   286→        if (nodes.isEmpty()) {

   288→                Column(horizontalAlignment = Alignment.CenterHorizontally) {
   289→                    Icon(Icons.Default.AddLocation, contentDescription = null, modifier = Modifier.size(48.dp), tint = Color.Gray)
   290→                    Spacer(modifier = Modifier.height(8.dp))
   291→                    Text("لا توجد عقد بعد", color = Color.Gray)

   296→        }
   297→
   298→        // Node list header
   299→        Row(
   300→            modifier = Modifier
   301→                .fillMaxWidth()
   302→                .background(Color(0xFFE3F2FD))
   303→                .padding(horizontal = 12.dp, vertical = 8.dp)

   306→            Text("X (m)", modifier = Modifier.width(60.dp), fontWeight = FontWeight.Bold, fontSize = 12.sp)

   316→                    modifier = Modifier
   317→                        .fillMaxWidth()

   323→                    Text("${node.x}", modifier = Modifier.width(60.dp))
   324→                    Text("${node.y}", modifier = Modifier.width(60.dp))
   325→                    Text(node.support.displayNameAr, modifier = Modifier.weight(1f), fontSize = 12.sp)
   326→                    IconButton(onClick = {
   327→                        editingNode = node
   328→                        editX = node.x.toString()
   329→                        editY = node.y.toString()
   330→                        editSupport = node.support
   331→                        showAddDialog = true
   332→                    }) {
   333→                        Icon(Icons.Default.Edit, "تعديل", modifier = Modifier.size(18.dp), tint = Color(0xFF1565C0))
   334→                    }
   335→                }
   336→                HorizontalDivider()
   337→            }
   338→        }
   339→    }
   340→
   341→    // Add/Edit Dialog
   342→    if (showAddDialog) {
   343→        AlertDialog(
   344→            onDismissRequest = { showAddDialog = false },
   345→            title = { Text(if (editingNode != null) "تعديل عقدة" else "إضافة عقدة") },
   346→            text = {
   347→                Column {
   348→                    OutlinedTextField(
   349→                        value = editX, onValueChange = { editX = it },
   350→                        label = { Text("X (متر)") },
   351→                        keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
   352→                        modifier = Modifier.fillMaxWidth()
   353→                    )
   354→                    Spacer(modifier = Modifier.height(8.dp))
   355→                    OutlinedTextField(
   356→                        value = editY, onValueChange = { editY = it },
   357→                        label = { Text("Y (متر)") },
   358→                        keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
   359→                        modifier = Modifier.fillMaxWidth()
   360→                    )
   361→                    Spacer(modifier = Modifier.height(8.dp))
   362→                    Text("نوع الارتكاز:", fontWeight = FontWeight.Bold, fontSize = 13.sp)
   363→                    Spacer(modifier = Modifier.height(4.dp))
   364→                    SupportType.entries.forEach { st ->
   365→                        Row(verticalAlignment = Alignment.CenterVertically, modifier = Modifier.padding(2.dp)) {
   366→                            RadioButton(
   367→                                selected = editSupport == st,
   368→                                onClick = { editSupport = st }
   369→                            )
   370→                            Text(st.displayNameAr, fontSize = 13.sp)
   371→                        }
   372→                    }
   373→                }
   374→            },
   375→            confirmButton = {
   376→                TextButton(onClick = {
   377→                    val x = editX.toDoubleOrNull() ?: return@TextButton
   378→                    val y = editY.toDoubleOrNull() ?: return@TextButton
   379→                    if (editingNode != null) {
   380→                        viewModel.updateNode(editingNode!!.id, x, y, editSupport)
   381→                    } else {
   382→                        viewModel.addNode(x, y, editSupport)
   383→                    }
   384→                    showAddDialog = false
   385→                }) { Text("حفظ", color = Color(0xFF1565C0)) }
   386→            },
   387→            dismissButton = {
   388→                TextButton(onClick = { showAddDialog = false }) { Text("إلغاء") }
   389→            }
   390→        )
   391→    }
   392→}

   980→    val color = when {
   981→        clampedRatio <= 0.7 -> Color(0xFF4CAF50)
   982→        clampedRatio <= 1.0 -> Color(0xFFFF9800)
   983→        else -> Color(0xFFF44336)
   984→    }
   985→    Row(modifier = modifier.height(16.dp), verticalAlignment = Alignment.CenterVertically) {
   986→        LinearProgressIndicator(
   987→            progress = { clampedRatio.toFloat() / 1.5f },
   988→            modifier = Modifier
   989→                .weight(1f)
   990→                .height(8.dp)
   991→                .clip(RoundedCornerShape(4.dp)),
   992→            color = color,
   993→            trackColor = Color.LightGray
   994→        )
   995→        Spacer(modifier = Modifier.width(8.dp))
   996→        Text(
   997→            "${String.format("%.0f", ratio * 100)}%",
   998→            fontSize = 11.sp,
   999→            fontWeight = FontWeight.Bold,
  1000→            color = color
  1001→        )
  1002→    }
