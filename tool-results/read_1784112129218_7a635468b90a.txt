   283→@Composable
   284→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   285→    Card(
   286→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   287→        border = BorderStroke(1.dp, Color.LightGray)
   288→    ) {
   289→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   290→            val padding = 40.dp.toPx()
   291→            val w = size.width - 2 * padding
   292→            val h = size.height - 2 * padding
   293→            val baseY = size.height - padding
   294→            
   295→            // Scale calculations
   296→            val scale = (w / inputs.span).toFloat()
   297→            val eh = (inputs.eaveHeight * scale).toFloat()
   298→            val rh = (inputs.ridgeHeight * scale).toFloat()
   299→            val midX = padding + w / 2
   300→            
   301→            // Draw Main Frame Outline
   302→            val framePath = Path().apply {
   303→                moveTo(padding, baseY)
   304→                lineTo(padding, baseY - eh)
   305→                lineTo(midX, baseY - rh)
   306→                lineTo(padding + w, baseY - eh)
   307→                lineTo(padding + w, baseY)
   308→            }
   309→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   310→            
   311→            // Draw Distributed Load Arrows
   312→            val loadColor = Color(0xFFE53935)
   313→            for (i in 0..12) {
   314→                val x = padding + (i * w / 12)
   315→                val y = if (x <= midX) {
   316→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   317→                } else {
   318→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   319→                }
   320→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   321→                // Arrow heads
   322→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   323→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   324→            }
   325→            
   326→            // Draw Moment Diagram (Simplified)
   327→            val momentColor = Color(0xFF1E88E5)
   328→            val momentPath = Path().apply {
   329→                moveTo(padding, baseY - eh)
   330→                // Simulated parabolic curve for moment
   331→                cubicTo(
   332→                    padding + w * 0.25f, baseY - eh + 50f,
   333→                    padding + w * 0.75f, baseY - eh + 50f,
   334→                    padding + w, baseY - eh
   335→                )
   336→            }
   337→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   338→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   339→            
   340→            // Labels
   341→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   342→        }
   343→    }
   344→}

   410→                }

   416→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawFrontElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   417→    val colLX = startX
   418→    val colRX = startX + (inputs.span * scale).toFloat()
   419→    val eaveY = baseY - (inputs.eaveHeight * scale).toFloat()
   420→    val ridgeY = baseY - (inputs.ridgeHeight * scale).toFloat()
   421→    val midX = startX + (inputs.span * scale / 2).toFloat()
   422→
   423→    // Columns
   424→    drawLine(Color.Black, Offset(colLX, baseY), Offset(colLX, eaveY), strokeWidth = 8f)
   425→    drawLine(Color.Black, Offset(colRX, baseY), Offset(colRX, eaveY), strokeWidth = 8f)
   426→    // Rafters
   427→    drawLine(Color.Black, Offset(colLX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   428→    drawLine(Color.Black, Offset(colRX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   429→    
   430→    // Foundations
   431→    drawRect(Color.Gray, Offset(colLX - 15f, baseY), Size(30f, 15f))
   432→    drawRect(Color.Gray, Offset(colRX - 15f, baseY), Size(30f, 15f))
   433→}
   434→
   435→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   436→    val planW = (inputs.span * scale).toFloat()
   437→    val planL = (inputs.length * scale).toFloat()
   438→    val planTop = baseY - planL
   439→    
   440→    // Boundary
   441→    drawRect(Color.Black, Offset(startX, planTop), Size(planW, planL), style = androidx.compose.ui.graphics.drawscope.Stroke(4f))
   442→    
   443→    // Grid Lines (Bays)
   444→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   445→    for (i in 0..numBays) {
   446→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()
   447→        drawLine(Color.LightGray, Offset(startX, y), Offset(startX + planW, y), strokeWidth = 1f)
   448→        // Columns markers
   449→        drawCircle(Color.Black, 4f, Offset(startX, y))
   450→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))
   451→    }
   452→}
   453→
   454→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSideElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   455→    val planL = (inputs.length * scale).toFloat()
   456→    val eaveH = (inputs.eaveHeight * scale).toFloat()
   457→    val eaveY = baseY - eaveH
   458→    
   459→    // Ground
   460→    drawLine(Color.Gray, Offset(startX, baseY), Offset(startX + planL, baseY), strokeWidth = 2f)
   461→    
   462→    // Columns
   463→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   464→    for (i in 0..numBays) {
   465→        val x = startX + (i * inputs.baySpacing * scale).toFloat()
   466→        drawLine(Color.Black, Offset(x, baseY), Offset(x, eaveY), strokeWidth = 6f)
   467→    }
   468→    // Eave Line
   469→    drawLine(Color.Black, Offset(startX, eaveY), Offset(startX + planL, eaveY), strokeWidth = 6f)
   470→}

   472→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   473→    val s = scale.toFloat() * 0.65f
   474→    val depthOff = 35f
   475→    
   476→    // Draw from back to front for better perspective look
   477→    for (i in 4 downTo 0) {
   478→        val off = i * depthOff
   479→        val x0 = startX + off
   480→        val y0 = baseY - off
   481→        val spanW = (inputs.span * s).toFloat()
   482→        val eaveH = (inputs.eaveHeight * s).toFloat()
   483→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   484→        
   485→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   486→        val stroke = if (i == 0) 3f else 1.5f
   487→        
   488→        // Sections of the frame
   489→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   490→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   491→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   492→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   493→        
   494→        // Purlins/Girts connections (longitudinal)
   495→        if (i > 0) {
   496→            val px = x0 - depthOff
   497→            val py = y0 + depthOff
   498→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   499→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   500→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   501→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   502→        }
   503→    }
   504→}

   538→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   539→    val p = Paint().apply { 
   540→        color = android.graphics.Color.BLACK
   541→        strokeWidth = 6f
   542→        isAntiAlias = true
   543→        style = Paint.Style.STROKE
   544→    }
   545→    val dimPaint = Paint().apply {
   546→        color = android.graphics.Color.DKGRAY
   547→        strokeWidth = 2f
   548→        textSize = 24f
   549→        isAntiAlias = true
   550→    }
   551→    val textP = Paint().apply {
   552→        color = android.graphics.Color.BLACK
   553→        textSize = 22f
   554→        isFakeBoldText = true
   555→        textAlign = Paint.Align.CENTER
   556→    }
   557→    
   558→    val s = scale * 0.7f
   559→    val w = inputs.span.toFloat() * s
   560→    val eh = inputs.eaveHeight.toFloat() * s
   561→    val rh = inputs.ridgeHeight.toFloat() * s
   562→    val midX = x + w/2
   563→    
   564→    // Draw Main Frame (Columns & Rafters)
   565→    canvas.drawLine(x, y, x, y - eh, p)
   566→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   567→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   568→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   569→    
   570→    // Foundations
   571→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   572→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   573→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   574→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   575→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   576→    p.strokeWidth = 6f
   577→    
   578→    // --- Dimension Lines ---
   579→    // Span Dimension
   580→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   581→    
   582→    // Eave Height Dimension
   583→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   584→    
   585→    // Ridge Height Dimension
   586→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   587→
   588→    // --- Section Labels ---
   589→    result?.let { res ->
   590→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   591→        
   592→        // Column Label
   593→        canvas.save()
   594→        canvas.rotate(-90f, x + 25f, y - eh/2)
   595→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   596→        canvas.restore()
   597→        
   598→        // Rafter Label
   599→        canvas.save()
   600→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   601→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   602→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   603→        canvas.restore()
   604→    }
   605→
   606→    // Connection Markers (Circles at joints)
   607→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   608→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   609→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   610→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   611→}

   613→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   614→    val tickSize = 15f
   615→    canvas.drawLine(x1, y1, x2, y2, paint)
   616→    
   617→    // Draw Ticks (45 degree lines)
   618→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   619→    val tickAngle = angle + Math.PI / 4
   620→    
   621→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   622→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   623→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   624→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   625→
   626→    // Text label
   627→    val midX = (x1 + x2) / 2
   628→    val midY = (y1 + y2) / 2
   629→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   630→    
   631→    if (x1 == x2) { // Vertical
   632→        canvas.save()
   633→        canvas.rotate(-90f, midX - 10f, midY)
   634→        canvas.drawText(text, midX - 10f, midY, textPaint)
   635→        canvas.restore()
   636→    } else { // Horizontal
   637→        canvas.drawText(text, midX, midY - 10f, textPaint)
   638→    }
   639→}

   641→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   642→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   643→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   644→    
   645→    val s = scale * 0.6f
   646→    val w = inputs.span.toFloat() * s
   647→    val l = inputs.length.toFloat() * s
   648→    val planTop = y - l
   649→    
   650→    canvas.drawRect(x, planTop, x + w, y, p)
   651→    
   652→    val baySpacing = inputs.baySpacing.toFloat() * s
   653→    var curY = planTop
   654→    var count = 0
   655→    while (curY <= y + 0.1f) {
   656→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   657→        
   658→        // Column markers
   659→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   660→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   661→        
   662→        if (curY + baySpacing <= y + 0.1f) {
   663→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   664→        }
   665→        
   666→        curY += baySpacing
   667→        count++
   668→    }
   669→    
   670→    // Span Dimension
   671→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   672→    // Total Length Dimension
   673→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   674→}

   676→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   677→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   678→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   679→    
   680→    val s = scale * 0.6f
   681→    val l = inputs.length.toFloat() * s
   682→    val eh = inputs.eaveHeight.toFloat() * s
   683→    
   684→    canvas.drawLine(x, y, x + l, y, p) // Ground
   685→    val bay = inputs.baySpacing.toFloat() * s
   686→    var cur = x
   687→    while (cur <= x + l + 0.1f) {
   688→        canvas.drawLine(cur, y, cur, y - eh, p)
   689→        cur += bay
   690→    }
   691→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   692→    
   693→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   694→}

  1346→@Composable
  1347→fun SteelSectionDrawing(section: SteelSectionType) {
  1348→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1349→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1350→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1351→                val w = size.width
  1352→                val h = size.height
  1353→                
  1354→                when (section) {
  1355→                    is SteelSectionType.ISection -> {
  1356→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1357→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1358→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1359→                        val offsetBF = (w - bf) / 2
  1360→                        
  1361→                        // Top Flange
  1362→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1363→                        // Bottom Flange
  1364→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1365→                        // Web
  1366→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1367→                        
  1368→                        // Stress Distribution (Schematic)
  1369→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1370→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1371→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1372→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1373→                    }
  1374→                    is SteelSectionType.LSection -> {
  1375→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
