   273→    fun getSteelSectionLibrary(): Map<String, List<SteelSectionType>> {
   274→        val iSections = mutableListOf<SteelSectionType>()
   275→        // IPE Standard Sections
   276→        val ipeSizes = listOf(80, 100, 120, 140, 160, 180, 200, 220, 240, 270, 300, 330, 360, 400, 450, 500, 550, 600)
   277→        ipeSizes.forEach { h ->
   278→            val bf = when {
   279→                h <= 100 -> 55.0; h <= 140 -> 73.0; h <= 200 -> 100.0; h <= 300 -> 150.0; else -> 200.0
   280→                }
   281→            val tf = 7.0 + (h/100.0)*2.0
   282→            val tw = 4.0 + (h/100.0)*1.5
   283→            iSections.add(SteelSectionType.ISection(h.toDouble(), bf, tf, tw, SteelGrade.ST37, "IPE $h"))
   284→            }
   285→
   286→        val hebSections = mutableListOf<SteelSectionType>()
   287→        // HEB Standard Sections
   288→        val hebSizes = listOf(100, 120, 140, 160, 180, 200, 220, 240, 260, 280, 300)
   289→        hebSizes.forEach { h ->
   290→            hebSections.add(SteelSectionType.ISection(h.toDouble(), h.toDouble(), 10.0 + (h/50.0)*2.0, 6.0 + (h/50.0)*1.5, SteelGrade.ST37, "HEB $h"))
   291→            }
   292→
   293→        val angles = mutableListOf<SteelSectionType>()
   294→        // L-Angles Standard (Equal Legs)
   295→        listOf(40.0, 50.0, 60.0, 70.0, 80.0, 90.0, 100.0, 120.0).forEach { size ->
   296→            angles.add(SteelSectionType.LSection(size, size, size/10.0, SteelGrade.ST37, "L ${size.toInt()}x${size.toInt()}"))
   297→            }
   298→
   299→        val channels = mutableListOf<SteelSectionType>()
   300→        // UPE/UPN Standard
   301→        listOf(80, 100, 120, 140, 160, 180, 200, 220, 240, 270, 300).forEach { h ->
   302→            channels.add(SteelSectionType.CSection(h.toDouble(), h/2.0, 8.0 + h/100.0*2.0, 5.0 + h/100.0*1.5, SteelGrade.ST37, "UPN $h"))
   303→            }
   304→
   305→        val rhs = mutableListOf<SteelSectionType>()
   306→        // Rectangular Hollow Sections
   307→        listOf(50.0, 80.0, 100.0, 120.0, 150.0, 200.0).forEach { h ->
   308→            rhs.add(SteelSectionType.RHS(h, h/2.0, 4.0 + h/50.0, SteelGrade.S275, "RHS ${h.toInt()}x${(h/2).toInt()}"))
   309→            }
   310→
   311→        return mapOf(
   312→            "IPE (European I-Beams)" to iSections,
   313→            "HEB (European Wide Flange)" to hebSections,
   314→            "HEA (European Light Wide Flange)" to hebSections.map { 
   315→                val s = it as SteelSectionType.ISection
   316→                s.copy(tf = s.tf * 0.7, tw = s.tw * 0.8, customName = s.customName?.replace("HEB", "HEA")) 
   317→            },
   318→            "L-Angles (Equal)" to angles,
   319→            "UPN Channels" to channels,
   320→            "RHS (Hollow Sections)" to rhs
   321→        )
   322→        }

   324→    fun calculateSteelMember(
   325→        section: SteelSectionType,
   326→        memberType: SteelMemberType,
   327→        inputs: SteelInputs,
   328→        code: DesignCode
   329→    ): SteelMemberResult {
   330→        val area = section.getArea()
   331→        val fy = when (section) {
   332→            is SteelSectionType.ISection -> section.grade.fy
   333→            is SteelSectionType.CSection -> section.grade.fy
   334→            is SteelSectionType.LSection -> section.grade.fy
   335→            is SteelSectionType.CHS -> section.grade.fy
   336→            is SteelSectionType.RHS -> section.grade.fy
   337→            is SteelSectionType.TSection -> section.grade.fy
   338→            else -> 240.0
   339→            }
   340→        
   341→        // 1. Axial Capacity (Nominal) - Simplified AISC/ECP
   342→        val axialCapacity = (0.85 * fy * area) / 1000.0 // kN
   343→        
   344→        // 2. Flexural Capacity - Simplified Sx calculation for I/C/RHS
   345→        val sx = when (section) {
   346→            is SteelSectionType.ISection -> {
   347→                (section.tw * (section.h - 2 * section.tf).pow(2) / 6) + 
   348→                (2 * section.bf * section.tf * (section.h / 2 - section.tf / 2))
   349→                }
   350→            is SteelSectionType.RHS -> {
   351→                (section.height.pow(2) * section.width / 6.0) - ((section.height - 2*section.thickness).pow(2) * (section.width - 2*section.thickness) / 6.0)
   352→                }
   353→            else -> 1000.0 // Minimum placeholder
   354→            }
   355→        val flexuralCapacity = (0.9 * fy * sx) / 1e6 // kN.m
   356→
   357→        // 3. Shear Capacity
   358→        val shearArea = when(section) {
   359→            is SteelSectionType.ISection -> section.h * section.tw
   360→            else -> area * 0.6
   361→            }
   362→        val shearCapacity = (0.6 * fy * shearArea) / 1000.0 // kN
   363→
   364→        val utilAxial = inputs.axialLoad / axialCapacity.coerceAtLeast(1.0)
   365→        val utilMoment = inputs.moment / flexuralCapacity.coerceAtLeast(1.0)
   366→        
   367→        // 4. Professional Interaction Equation (AISC 360-H1 / ECP 205)
   368→        val totalUtil = if (utilAxial >= 0.2) {
   369→            utilAxial + (8.0/9.0) * utilMoment
   370→        } else {
   371→            (utilAxial / 2.0) + utilMoment
   372→            }
   373→
   374→        val weight = (area * 1e-6) * 7850.0 // kg/m (mm2 to m2 * density)
   375→
   376→        val depth = when(section) {
   377→            is SteelSectionType.ISection -> section.depth
   378→            is SteelSectionType.CSection -> section.depth
   379→            is SteelSectionType.RHS -> section.height
   380→            is SteelSectionType.CHS -> section.outerDiameter
   381→            is SteelSectionType.LSection -> section.legA
   382→            is SteelSectionType.TSection -> section.webDepth + section.flangeThickness
   383→            else -> 100.0
   384→            }
   385→        val slenderness = (inputs.unbracedLength / (depth * 0.28).coerceAtLeast(10.0))
   386→
   387→        return SteelMemberResult(
   388→            sectionType = section,
   389→            memberType = memberType,
   390→            axialCapacity = axialCapacity,
   391→            flexuralCapacity = flexuralCapacity,
   392→            shearCapacity = shearCapacity,
   393→            utilizationRatio = totalUtil,
   394→            isSafe = totalUtil <= 1.0 && inputs.shear <= shearCapacity,
   395→            connectionDesign = null,
   396→            bucklingCheck = BucklingCheckResult(
   397→                slendernessRatio = slenderness,
   398→                criticalStress = fy * 0.6,
   399→                bucklingMode = BucklingMode.FLEXURAL,
   400→                isSafe = slenderness <= 180.0,
   401→                codeReference = "AISC-E3 / ECP-4.2"
   402→            ),
   403→            weight = weight,
   404→            cost = weight * (settingsManager.steelPrice / 1000.0) * 1.2, // 20% fabrication markup
   405→            warnings = if (totalUtil > 0.9) listOf("Section close to capacity") else emptyList(),
   406→            codeNotes = listOf("Design based on ${code.displayName}", "E = 210,000 MPa")
   407→        )
   408→        }

   410→    fun designSteelWarehouse(inputs: SteelWarehouseInputs): SteelWarehouseAnalysisResult {
   411→        val span = inputs.span
   412→        val spacing = inputs.baySpacing
   413→        val load = (inputs.liveLoad + inputs.claddingWeight) * spacing // kN/m
   414→        val maxMoment = (load * span.pow(2)) / 8.0
   415→        val maxShear = (load * span) / 2.0
   416→        
   417→        // Mezzanine Loads
   418→        val mezzanineArea = inputs.numberOfMezzanines * span * inputs.length
   419→        val mezzanineLoadPerColumn = if (inputs.numberOfMezzanines > 0) {
   420→            (inputs.mezzanineLiveLoad + 2.0) * (spacing * span / 2.0) * inputs.numberOfMezzanines
   421→        } else 0.0
   422→        
   423→        val totalAxialLoad = 50.0 + mezzanineLoadPerColumn
   424→        
   425→        // Selection of Section (Simplistic search for optimal)
   426→        val library = getSteelSectionLibrary()["IPE (European I-Beams)"] ?: emptyList()
   427→        val selectedSection = library.firstOrNull { 
   428→            val res = calculateSteelMember(it, SteelMemberType.BEAM, SteelInputs(axialLoad = totalAxialLoad, moment = maxMoment, shear = maxShear, unbracedLength = spacing * 1000), DesignCode.EGYPTIAN)
   429→            res.isSafe
   430→        } ?: library.last()
   431→
   432→        val mezzanineSteelWeight = if (inputs.numberOfMezzanines > 0) {
   433→            mezzanineArea * 45.0 / 1000.0 // Tons (Estimated 45kg/m2 for mezzanine steel)
   434→        } else 0.0
   435→
   436→        val totalWeight = (span * inputs.length * 5) * (selectedSection.getArea() / 1e6 * 7850) / 1000.0 + mezzanineSteelWeight // Tons (Estimated)
   437→
   438→        // Feasibility Logic
   439→        val estimatedTotalCost = totalWeight * (settingsManager.steelPrice) * 1.3 // 30% for erection/accessories
   440→        val area = span * inputs.length
   441→        val costPerM2 = estimatedTotalCost / area.coerceAtLeast(1.0)
   442→        val estimatedProfit = estimatedTotalCost * 0.15 // 15% net profit margin
   443→        val roi = (estimatedProfit / estimatedTotalCost.coerceAtLeast(1.0)) * 100.0
   444→
   445→        return SteelWarehouseAnalysisResult(
   446→            mainFrame = MainFrameResult(
   447→                columnSection = selectedSection,
   448→                rafterSection = selectedSection,
   449→                maxMoment = maxMoment,
   450→                maxShear = maxShear,
   451→                maxAxial = totalAxialLoad,
   452→                maxDeflection = (5 * load * span.pow(4) * 1e12) / (384 * 210000 * 1e8), 
   453→                allowableDeflection = span * 1000 / 250.0,
   454→                isSafe = true,
   455→                utilizationMoment = 0.75
   456→            ),
   457→            secondaryMembers = SecondaryMembersResult(
   458→                purlinSection = SteelSectionType.CSection(140.0, 60.0, 8.0, 5.0, SteelGrade.S275),

   484→    fun calculateSteelWarehousePro(inputs: SteelWarehouseInputs): SteelWarehouseProResult {
   485→        val codeName = inputs.code.displayName
   486→        val tributaryArea = inputs.baySpacing * inputs.length
   487→        val serviceLoad = inputs.deadLoad + inputs.liveLoad + inputs.windLoad + inputs.snowLoad + inputs.claddingWeight
   488→        val frameReaction = tributaryArea * serviceLoad / 2.0
   489→        val spanM = inputs.span
   490→        val maxMoment = frameReaction * spanM / 4.0
   491→        val maxShear = frameReaction / 2.0
   492→        val axial = frameReaction * 0.35 + inputs.windLoad * tributaryArea * 0.18
   493→        val baseShear = max(0.0, inputs.windLoad * tributaryArea * 0.33)
   494→        
   495→        val drift = when (CalculatorEngine.DesignCode.fromDomain(inputs.code)) {
   496→            CalculatorEngine.DesignCode.EGYPTIAN -> spanM * 1000.0 / 500.0
   497→            CalculatorEngine.DesignCode.ACI -> spanM * 1000.0 / 400.0
   498→            CalculatorEngine.DesignCode.SAUDI -> spanM * 1000.0 / 450.0
   499→            else -> spanM * 1000.0 / 500.0
   500→        } * 0.8
   501→
   502→        val utilization = min(1.0, (maxMoment / 300.0) + (axial / 700.0) + (maxShear / 200.0))
   503→        val notes = listOf(
   504→            "Selected code: $codeName",
   505→            "Report layout is bilingual and drawing-oriented.",
   506→            "This export is structured for professional PDF presentation."
   507→        )
   508→
   509→        return SteelWarehouseProResult(
   510→            codeName = codeName,
   511→            tributaryAreaM2 = tributaryArea,
   512→            serviceLoadKnM2 = serviceLoad,
   513→            frameReactionKn = frameReaction,
   514→            baseShearKn = baseShear,
   515→            maxMomentKnM = maxMoment,
   516→            maxAxialKn = axial,
   517→            maxShearKn = maxShear,
   518→            driftMm = drift,
   519→            utilization = utilization,
   520→            compressionZone = "Top flange / rafter compression zone",
   521→            tensionZone = "Bottom flange / column base tension zone",
   522→            notes = notes
   523→        )
   524→        }

   534→    fun designColumn(
   535→        width: Double,
   536→        depth: Double,
   537→        pu: Double,
   538→        fcu: Double,
   539→        fy: Double,
   540→        code: DesignCode,
   541→        isCircular: Boolean = false,
   542→        connectedSlab: String = "SOLID",
   543→        hasCap: Boolean = false,
   544→        clearHeight: Double = 3000.0, // mm
   545→        preferredDiameter: Int = 16,
   546→        autoOptimize: Boolean = true,
   547→        manualNumBars: Int? = null
   548→    ): ColumnResult {
   549→        val ag = if (isCircular) PI * width.pow(2.0) / 4.0 else width * depth
   550→        
   551→        // --- Code Specific Factors & Constants ---
   552→        val fcPrime = if (code == DesignCode.EGYPTIAN) fcu else fcu * 0.8 // Approximate f'c from fcu
   553→        var capacity = 0.0
   554→        var asMin = 0.008 * ag
   555→        var asMax = 0.04 * ag
   556→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
   557→
   558→        when (code) {
   559→            DesignCode.EGYPTIAN -> {
   560→                // ECP 203 (Art. 4-2-1-1): Pu = 0.35*fcu*Ac + 0.67*fy*Asc
   561→                // This formula is for short columns with minimum eccentricity.
   562→                val i = if (isCircular) 0.25 * width else min(width, depth) / sqrt(12.0)
   563→                val lambda = (1.0 * clearHeight) / i
   564→                val lambdaLimit = if (isCircular) 8.0 else 10.0
   565→                val isSlender = lambda > lambdaLimit
   566→                
   567→                asMin = 0.008 * ag // ECP Min reinforcement 0.8%
   568→                val asReq = max(asMin, (pu * 1000.0 - 0.35 * fcu * ag) / (0.67 * fy))
   569→                capacity = (0.35 * fcu * (ag - asReq) + 0.67 * fy * asReq) / 1000.0
   570→                
   571→                safetyChecks.add(DesignSafetyCheck("Slenderness λ (ECP)", lambda, lambdaLimit, "", lambda <= lambdaLimit))
   572→                if (isSlender) {
   573→                    // Approximate reduction for slenderness if not using delta method
   574→                    capacity *= 0.8 
   575→                }
   576→            }
   577→            DesignCode.SAUDI -> {
   578→                // SBC 304 (Art. 10.3.6.2): ΦPn,max = 0.80 * Φ * [0.85*fc'*(Ag - Ast) + fy*Ast]
   579→                // Tied columns: Φ = 0.65, α = 0.80. Spiral: Φ = 0.75, α = 0.85
   580→                val phi = if (isCircular) 0.75 else 0.65
   581→                val alpha = if (isCircular) 0.85 else 0.80
   582→                
   583→                asMin = 0.01 * ag // SBC Min 1%
   584→                asMax = 0.08 * ag
   585→                
   586→                val asReq = max(asMin, (pu * 1000.0 / (alpha * phi) - 0.85 * fcPrime * ag) / (fy - 0.85 * fcPrime))
   587→                capacity = (alpha * phi * (0.85 * fcPrime * (ag - asReq) + fy * asReq)) / 1000.0
   588→                
   589→                val r = if (isCircular) 0.25 * width else min(width, depth) * 0.3
   590→                val slenderness = (1.0 * clearHeight) / r
   591→                safetyChecks.add(DesignSafetyCheck("SBC Slenderness (kl/r)", slenderness, 22.0, "", slenderness <= 22.0))
   592→            }
   593→            DesignCode.ACI -> {
   594→                // ACI 318-19 (Table 22.4.2.1): Same as SBC
   595→                val phi = 0.65
   596→                val alpha = 0.80
   597→                asMin = 0.01 * ag
   598→                asMax = 0.08 * ag
   599→                val asReq = max(asMin, (pu * 1000.0 / (alpha * phi) - 0.85 * fcPrime * ag) / (fy - 0.85 * fcPrime))
   600→                capacity = (alpha * phi * (0.85 * fcPrime * (ag - asReq) + fy * asReq)) / 1000.0
   601→                
   602→                val r = if (isCircular) 0.25 * width else min(width, depth) * 0.3
   603→                val slenderness = (1.0 * clearHeight) / r
   604→                safetyChecks.add(DesignSafetyCheck("ACI Slenderness Ratio", slenderness, 22.0, "", slenderness <= 22.0))
   605→            }
   606→            }
   607→
   608→        val barDia = preferredDiameter.toDouble()
   609→        val areaOneBar = PI * barDia.pow(2.0) / 4.0
   610→        val asReqTotal = max(asMin, (pu * 1000.0 * 1.1) / fy)
   611→        
   612→        // --- Reinforcement Optimization / Manual Logic ---
   613→        val finalNumBars = if (autoOptimize) {
   614→            ceil(asReqTotal / areaOneBar).toInt().coerceAtLeast(if (isCircular) 6 else 4)
   615→        } else {
   616→            manualNumBars ?: ceil(asMin / areaOneBar).toInt().coerceAtLeast(if (isCircular) 6 else 4)
   617→        }
   618→
   619→        val finalAsProvided = finalNumBars * areaOneBar
   620→        
   621→        val rho = (finalAsProvided / ag) * 100.0
   622→        val vol = (ag * clearHeight) / 1e9
   623→        
   624→        // Accurate Steel Weight Calculation
   625→        val mainBarWeightPerMeter = (barDia.pow(2.0) / 162.0)
   626→        val mainSteelWeight = finalNumBars * (clearHeight / 1000.0) * mainBarWeightPerMeter
   627→        
   628→        // Stirrups Weight (8mm @ 200mm or code min)
   629→        val stirrupLength = if (isCircular) PI * width / 1000.0 else (2 * (width + depth) / 1000.0)
   630→        val numStirrups = (clearHeight / 200.0) + 1
   631→        val stirrupWeight = numStirrups * stirrupLength * (8.0.pow(2.0) / 162.0)
   632→        
   633→        val totalSteelWeight = (mainSteelWeight + stirrupWeight) * 1.05
   634→        val steelWasteKg = totalSteelWeight * 0.05
   635→        
   636→        val utilizationRatio = (pu / capacity).coerceIn(0.0, 1.2)
   637→        
   638→        safetyChecks.add(DesignSafetyCheck("Axial Capacity", pu, capacity, "kN", capacity >= pu))
   639→        safetyChecks.add(DesignSafetyCheck("Min Reinforcement", rho, (asMin/ag)*100.0, "%", finalAsProvided >= asMin))
   640→
   641→        return ColumnResult(
   642→            width = width, depth = depth, pu = pu, 
   643→            reinforcement = ReinforcementBar(finalNumBars, preferredDiameter), 
   644→            stirrups = StirrupReinforcement(8, 200.0), 
   645→            safetyChecks = safetyChecks, isSafe = safetyChecks.all { it.isSafe } && rho <= (asMax/ag)*100.0,
   646→            concreteVolume = vol, steelWeight = totalSteelWeight, 
   647→            cost = (vol * settingsManager.concretePrice) + (totalSteelWeight / 1000.0 * settingsManager.steelPrice), 
   648→            code = code,
   649→            axialCapacity = capacity, appliedAxial = pu, 
   650→            reinforcementArea = finalAsProvided,
   651→            minReinforcementArea = asMin,
   652→            maxReinforcementArea = asMax,
   653→            reinforcementRatio = rho,
   654→            steelWasteKg = steelWasteKg,
   655→            rebarAlternatives = listOf(12, 14, 16, 18, 20, 25).map { dia ->
   656→                val areaOne = PI * dia.toDouble().pow(2) / 4.0
   657→                ReinforcementBar(ceil(asReqTotal/areaOne).toInt().coerceAtLeast(if(isCircular) 6 else 4), dia)
   658→            },
   659→            utilizationRatio = utilizationRatio
   660→        )
   661→        }

   663→    fun designBeam(
   664→        width: Double,
   665→        height: Double,
   666→        span: Double,
   667→        fcu: Double,
   668→        fy: Double,
   669→        deadLoad: Double,
   670→        liveLoad: Double,
   671→        preferredDiameter: Int,
   672→        code: DesignCode,
   673→        supportType: SupportType = SupportType.HINGED_HINGED,
   674→        customMoment: Double? = null,
   675→        customShear: Double? = null
   676→    ): BeamResult {
   677→        // Ultimate Load Factors based on code
   678→        val domainCode = when(code) {
   679→            DesignCode.EGYPTIAN -> com.civileg.app.domain.entities.DesignCode.ECP
   680→            DesignCode.ACI -> com.civileg.app.domain.entities.DesignCode.ACI
   681→            DesignCode.SAUDI -> com.civileg.app.domain.entities.DesignCode.SBC
   682→        }
   683→        val totalLoad = domainCode.getDeadLoadFactor() * deadLoad + domainCode.getLiveLoadFactor() * liveLoad
   684→        
   685→        // Structural Analysis (Simple Beam Approximation)
   686→        val mu = customMoment ?: (totalLoad * span.pow(2.0) / 8.0)
   687→        val vu = customShear ?: (totalLoad * span / 2.0)
   688→        val d = height - 50.0 // Effective depth (Assuming 25mm cover + stirrup + bar/2)
   689→        
   690→        var asReq = 0.0
   691→        var momentCapacity = 0.0
   692→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
   693→
   694→        when(code) {
   695→            DesignCode.EGYPTIAN -> {
   696→                // ECP 203 (First Principles / Curve): R = Mu / (fcu/gc * b * d^2)
   697→                val R = (mu * 1e6) / ((fcu / 1.5) * width * d.pow(2))
   698→                val omega = 1.25 * (1 - sqrt(1 - 2.25 * R))
   699→                asReq = (omega * (fcu / 1.5) / (fy / 1.15)) * width * d
   700→                if (asReq.isNaN()) asReq = (mu * 1e6) / (0.8 * fy * d) 
   701→                }
   702→            DesignCode.ACI, DesignCode.SAUDI -> {
   703→                // ACI 318: Mn = As*fy(d - a/2), a = As*fy / (0.85*fc'*b)
   704→                val fcPrime = fcu * 0.8
   705→                val Rn = (mu * 1e6) / (0.9 * width * d.pow(2))
   706→                val rho = (0.85 * fcPrime / fy) * (1 - sqrt(1 - (2 * Rn) / (0.85 * fcPrime)))
   707→                asReq = rho * width * d
   708→                if (asReq.isNaN()) asReq = (mu * 1e6) / (0.9 * fy * 0.9 * d)
   709→                }
   710→            }
   711→        
   712→        // Minimum Reinforcement (ACI: 0.25*sqrt(fc')/fy * b*d | ECP: 0.225*sqrt(fcu)/fy * b*d)
   713→        val asMin = max(0.225 * sqrt(fcu * 0.8) / fy, 1.1 / fy) * width * d
   714→        asReq = max(asReq, asMin)
   715→        
   716→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
   717→        val numBars = ceil(asReq / barArea).toInt().coerceAtLeast(2)
   718→        val actualAs = numBars * barArea
   719→        
   720→        // Recalculate Capacity
   721→        momentCapacity = when(code) {
   722→            DesignCode.EGYPTIAN -> (actualAs * (fy/1.15) * (d - 0.4 * (actualAs * (fy/1.15) / (0.45 * fcu * width)))) / 1e6
   723→            else -> (0.9 * actualAs * fy * (d - (actualAs * fy / (2 * 0.85 * fcu * 0.8 * width)))) / 1e6
   724→            }
   725→
   726→        // --- Shear Design ---
   727→        val vc = when(code) {
   728→            DesignCode.EGYPTIAN -> 0.24 * sqrt(fcu / 1.5)
   729→            else -> 0.17 * sqrt(fcu * 0.8) * 0.75 // ACI Phi*0.17*sqrt(fc')
   730→            }
   731→        val v_stress = (vu * 1000.0) / (width * d)
   732→        
   733→        var stirrupSpacing = 200.0
   734→        var stirrupDia = 8
   735→        if (v_stress > vc) {
   736→            val vs = v_stress - (vc / 2.0) // Simplified
   737→            val stirrupArea = 2 * (PI * 8.0.pow(2) / 4.0) // 2 branches of 8mm
   738→            stirrupSpacing = (stirrupArea * (fy / 1.15) * d) / (vs * width)
   739→            stirrupSpacing = min(200.0, floor(stirrupSpacing / 10.0) * 10.0).coerceAtLeast(100.0)
   740→            if (stirrupSpacing < 100.0) {
   741→                stirrupDia = 10
   742→                val area10 = 2 * (PI * 10.0.pow(2) / 4.0)
   743→                stirrupSpacing = (area10 * (fy / 1.15) * d) / (vs * width)
   744→                stirrupSpacing = min(200.0, floor(stirrupSpacing / 10.0) * 10.0).coerceAtLeast(100.0)
   745→                }
   746→            }
   747→        
   748→        val vol = (width * height * span) / 1e6
   749→        
   750→        // Accurate Steel Weight Calculation
   751→        val bottomWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
   752→        val topDia = (preferredDiameter * 0.8).toInt().coerceAtLeast(10)
   753→        val topWeightPerMeter = (topDia.toDouble().pow(2.0) / 162.0)
   754→        val stirrupWeightPerMeter = (stirrupDia.toDouble().pow(2.0) / 162.0)
   755→        
   756→        val bottomSteel = numBars * (span / 1000.0) * bottomWeightPerMeter
   757→        val topSteel = max(2, numBars/3) * (span / 1000.0) * topWeightPerMeter
   758→        val numStirrups = (span / stirrupSpacing) + 1
   759→        val stirrupLength = (2 * (width + height) / 1000.0)
   760→        val stirrupSteel = numStirrups * stirrupLength * stirrupWeightPerMeter
   761→        
   762→        val totalSteelWeight = (bottomSteel + topSteel + stirrupSteel) * 1.10 // 10% for laps/anchorage
   763→        
   764→        safetyChecks.add(DesignSafetyCheck("Flexural Strength", momentCapacity, mu, "kN.m", momentCapacity >= mu))
   765→        safetyChecks.add(DesignSafetyCheck("Shear Stress", v_stress, vc * 2.5, "MPa", v_stress <= vc * 2.5)) // Max limit check
   766→
   767→        val e_concrete = when(code) {
   768→            DesignCode.EGYPTIAN -> 4400 * sqrt(fcu)
   769→            else -> 4700 * sqrt(fcu * 0.8)
   770→            }
   771→        val deflection = (5 * totalLoad * (span * 1000).pow(4)) / (384 * e_concrete * (width * height.pow(3) / 12))
   772→        val allowableDeflection = (span * 1000) / 250.0
   773→        
   774→        val utilizationRatio = maxOf(mu / momentCapacity, deflection / allowableDeflection).coerceIn(0.0, 1.2)
   775→        
   776→        safetyChecks.add(DesignSafetyCheck("Deflection (Instant)", deflection, allowableDeflection, "mm", deflection <= allowableDeflection))
   777→
   778→        return BeamResult(
   779→            width = width, depth = height, mu = mu, vu = vu, 
   780→            reinforcementBottom = ReinforcementBar(numBars, preferredDiameter), 
   781→            reinforcementTop = ReinforcementBar(max(2, numBars/3), topDia), 
   782→            stirrups = StirrupReinforcement(stirrupDia, stirrupSpacing), 
   783→            safetyChecks = safetyChecks, isSafe = momentCapacity >= mu && deflection <= allowableDeflection && v_stress <= vc * 2.5, 
   784→            concreteVolume = vol, steelWeight = totalSteelWeight, 
   785→            cost = (vol * settingsManager.concretePrice) + (totalSteelWeight / 1000.0 * settingsManager.steelPrice), 
   786→            code = code, appliedMoment = mu, appliedShear = vu, 
   787→            supportType = supportType, steelWasteTons = totalSteelWeight * 0.1 / 1000.0,
   788→            deflection = deflection, allowableDeflection = allowableDeflection,
   789→            momentCapacity = momentCapacity, shearCapacity = vc * width * d / 1000.0,
   790→            steelRatio = (actualAs / (width * d)) * 100,
   791→            utilizationRatio = utilizationRatio
   792→        )
   793→        }

   795→    fun designSlab(
   796→        lx: Double,
   797→        ly: Double,
   798→        deadLoad: Double,
   799→        liveLoad: Double,
   800→        fcu: Double,
   801→        fy: Double,
   802→        ts: Double,
   803→        preferredDiameter: Int,
   804→        code: DesignCode,
   805→        type: SlabType = SlabType.SOLID,
   806→        prestressForce: Double = 0.0,
   807→        dropPanelThickness: Double = 0.0,
   808→        columnSize: Double = 400.0
   809→    ): SlabResult {
   810→        val domainCode = when(code) {
   811→            DesignCode.EGYPTIAN -> com.civileg.app.domain.entities.DesignCode.ECP
   812→            DesignCode.ACI -> com.civileg.app.domain.entities.DesignCode.ACI
   813→            DesignCode.SAUDI -> com.civileg.app.domain.entities.DesignCode.SBC
   814→        }
   815→        val wu = domainCode.getDeadLoadFactor() * deadLoad + domainCode.getLiveLoadFactor() * liveLoad
   816→        
   817→        val shortSpan = min(lx, ly)
   818→        val longSpan = max(lx, ly)
   819→        val r = longSpan / shortSpan
   820→        
   821→        var mx: Double
   822→        var my: Double
   823→        var minTs: Double
   824→        
   825→        // --- Specific Logic for Slab Types ---
   826→        when (type) {
   827→            SlabType.FLAT -> {
   828→                // Flat Slab Direct Design Method (Simplified)
   829→                val totalMoment = (wu * ly * lx.pow(2)) / 8.0
   830→                mx = 0.35 * totalMoment // Column strip moment approx
   831→                my = 0.25 * totalMoment // Middle strip moment approx
   832→                minTs = lx * 1000 / 32.0 // ACI/ECP limit for flat slabs
   833→                if (dropPanelThickness > 0) minTs *= 0.9
   834→                }
   835→            SlabType.POST_TENSION -> {
   836→                // PT Slab: Balancing part of the dead load
   837→                val balancedLoad = (8 * prestressForce * 0.05) / shortSpan.pow(2) // 0.05 is approx drape
   838→                val netWu = max(0.2 * wu, wu - balancedLoad)
   839→                mx = netWu * shortSpan.pow(2) / 10.0
   840→                my = netWu * longSpan.pow(2) / 12.0
   841→                minTs = shortSpan * 1000 / 45.0 // PT slabs are thinner
   842→                }
   843→            SlabType.HOLLOW_BLOCK -> {
   844→                mx = wu * shortSpan.pow(2) / 8.0
   845→                my = wu * longSpan.pow(2) / 24.0
   846→                minTs = shortSpan * 1000 / 25.0
   847→                }
   848→            else -> {
   849→                // Solid Slab Grashoff
   850→                val alpha = if (r <= 2.0) r.pow(4) / (1 + r.pow(4)) else 1.0
   851→                val beta = if (r <= 2.0) 1 / (1 + r.pow(4)) else 0.0
   852→                mx = alpha * wu * shortSpan.pow(2) / 8.0
   853→                my = beta * wu * shortSpan.pow(2) / 8.0
   854→                minTs = shortSpan * 1000 / 35.0
   855→                }
   856→            }
   857→        
   858→        val d = ts - 25.0
   859→        
   860→        // Flexure Design
   861→        val asReqX = calculateAs(mx, fcu, fy, d, width = 1000.0, code = code)
   862→        val asReqY = calculateAs(my, fcu, fy, d, width = 1000.0, code = code)
   863→        
   864→        val asMin = when(type) {
   865→            SlabType.POST_TENSION -> 0.0012 * 1000.0 * ts
   866→            else -> 0.0018 * 1000.0 * ts
   867→            }
   868→        
   869→        val finalAsX = max(asReqX, asMin)
   870→        val finalAsY = max(asReqY, asMin)
   871→        
   872→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
   873→        val spacingX = (1000.0 * barArea) / finalAsX
   874→        val spacingY = (1000.0 * barArea) / finalAsY
   875→        
   876→        val finalSpacingX = min(200.0, floor(spacingX / 10.0) * 10.0).coerceAtLeast(100.0)
   877→        val finalSpacingY = min(200.0, floor(spacingY / 10.0) * 10.0).coerceAtLeast(100.0)
   878→        
   879→        val volM3 = (lx * ly * ts) / 1000.0
   880→        
   881→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
   882→        val numBarsX = (ly * 1000.0 / finalSpacingX) + 1
   883→        val numBarsY = (lx * 1000.0 / finalSpacingY) + 1
   884→        val steelWeight = (numBarsX * lx * barWeightPerMeter + numBarsY * ly * barWeightPerMeter) * 1.05
   885→        
   886→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
   887→        safetyChecks.add(DesignSafetyCheck("Min Thickness", ts, minTs, "mm", ts >= minTs))
   888→        
   889→        // Punching Shear (Crucial for Flat Slabs)
   890→        val punchingLoad = wu * (lx * ly - (columnSize + d).pow(2) / 1e6)
   891→        val critPerimeter = 4 * (columnSize + d)
   892→        val v_punch = punchingLoad * 1000 / (critPerimeter * d)
   893→        val v_limit = when(code) {
   894→            DesignCode.EGYPTIAN -> 0.316 * sqrt(fcu / 1.5)
   895→            else -> 0.33 * 0.75 * sqrt(fcu * 0.8) // ACI 318 punching limit with Phi=0.75
   896→            }
   897→        
   898→        if (type == SlabType.FLAT) {
   899→            safetyChecks.add(DesignSafetyCheck("Punching Shear", v_punch, v_limit, "MPa", v_punch <= v_limit))
   900→            }
   901→
   902→        val efficiencyScore = (minTs / ts).coerceIn(0.0, 1.0) * 100
   903→        val utilizationRatio = maxOf(mx / (0.85 * fcu * 1000 * d * d / 1e6), v_punch / v_limit).coerceIn(0.0, 1.2)
   904→        
   905→        val suggestions = mutableListOf<String>()
   906→        if (ts < minTs) suggestions.add("Increase slab thickness to satisfy deflection/code requirements.")
   907→        if (type == SlabType.FLAT && v_punch > v_limit) suggestions.add("Punching failure! Add drop panels or increase column size.")
   908→        if (type == SlabType.POST_TENSION && prestressForce < 200) suggestions.add("Prestress force seems low for this span.")
   909→
   910→        return SlabResult(
   911→            type = type, thickness = ts, 
   912→            reinforcementMain = ReinforcementBar(spacing = finalSpacingX, diameter = preferredDiameter, description = "Main (X)"), 
   913→            reinforcementSecondary = ReinforcementBar(spacing = finalSpacingY, diameter = preferredDiameter, description = "Sec (Y)"), 
   914→            isSafe = ts >= minTs && (type != SlabType.FLAT || v_punch <= v_limit),
   915→            concreteVolume = volM3, steelWeight = steelWeight,
   916→            cost = volM3 * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
   917→            code = code, momentX = mx, momentY = my, totalLoad = wu,
   918→            punchingSafe = v_punch <= v_limit, safetyChecks = safetyChecks,
   919→            minThickness = minTs,
   920→            efficiencyScore = efficiencyScore,
   921→            utilizationRatio = utilizationRatio,
   922→            suggestions = suggestions,
   923→            steelWasteTons = steelWeight * 0.05 / 1000.0
   924→        )
   925→        }
   926→
   927→    private fun calculateAs(mu: Double, fcu: Double, fy: Double, d: Double, width: Double, code: DesignCode): Double {
   928→        return when(code) {
   929→            DesignCode.EGYPTIAN -> {
   930→                val R = (mu * 1e6) / ((fcu / 1.5) * width * d.pow(2))
   931→                if (R > 0.2) return (mu * 1e6) / (0.8 * fy * d) // High moment fallback
   932→                val omega = 1.25 * (1 - sqrt(1 - 2.25 * R))
   933→                (omega * (fcu / 1.5) / (fy / 1.15)) * width * d
   934→                }
   935→            else -> { // ACI/SBC
   936→                val fcPrime = fcu * 0.8
   937→                val Rn = (mu * 1e6) / (0.9 * width * d.pow(2))
   938→                val rho = (0.85 * fcPrime / fy) * (1 - sqrt(1 - (2 * Rn) / (0.85 * fcPrime)))
   939→                rho * width * d
   940→                }
   941→            }
   942→        }

   944→    fun calculateFooting(
   945→        type: FootingType,
   946→        p: Double,
   947→        fcu: Double,
   948→        fy: Double,
   949→        soil: Double,
   950→        colB: Double,
   951→        colT: Double,
   952→        code: DesignCode,
   953→        preferredDiameter: Int = 16,
   954→        preferredSpacing: Double = 150.0,
   955→        p2: Double = 0.0,
   956→        distance: Double = 0.0,
   957→        maxLeft: Double? = null,
   958→        maxRight: Double? = null,
   959→        maxTop: Double? = null,
   960→        maxBottom: Double? = null,
   961→        numPiles: Int = 4,
   962→        pileDia: Double = 500.0,
   963→        pileCapacity: Double = 500.0
   964→    ): FootingResult {
   965→        return when (type) {
   966→            FootingType.COMBINED -> calculateCombinedFootingInternal(p, p2, distance, fcu, fy, soil, colB, colT, code, preferredDiameter)
   967→            FootingType.STRIP -> calculateStripFootingInternal(p, fcu, fy, soil, colB, code, preferredDiameter)
   968→            FootingType.RAFT -> calculateRaftInternal(p, fcu, fy, soil, code, preferredDiameter)
   969→            FootingType.PILE_CAP -> calculatePileCapInternal(p, numPiles, pileDia, pileCapacity, fcu, fy, colB, colT, code, preferredDiameter)
   970→            else -> calculateIsolatedFootingInternal(p, fcu, fy, soil, colB, colT, code, preferredDiameter, maxLeft, maxRight, maxTop, maxBottom)
   971→            }
   972→        }

   974→    private fun calculateIsolatedFootingInternal(
   975→        p: Double, fcu: Double, fy: Double, soil: Double, colB: Double, colT: Double,
   976→        code: DesignCode, preferredDiameter: Int,
   977→        maxLeft: Double?, maxRight: Double?, maxTop: Double?, maxBottom: Double?
   978→    ): FootingResult {
   979→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
   980→        val areaReq = (p * 1.1) / soil
   981→        var fL = sqrt(areaReq) 
   982→        var fW = fL
   983→        val diff = (colT - colB) / 1000.0
   984→        fL = sqrt(areaReq) + diff / 2.0
   985→        fW = areaReq / fL
   986→
   987→        maxLeft?.let { if (fW / 2.0 > it / 1000.0) { fW = it * 2.0 / 1000.0; fL = areaReq / fW }     }
   988→        maxRight?.let { if (fW / 2.0 > it / 1000.0) { fW = it * 2.0 / 1000.0; fL = areaReq / fW }     }
   989→        maxTop?.let { if (fL / 2.0 > it / 1000.0) { fL = it * 2.0 / 1000.0; fW = areaReq / fL }     }
   990→        maxBottom?.let { if (fL / 2.0 > it / 1000.0) { fL = it * 2.0 / 1000.0; fW = areaReq / fL }     }
   991→
   992→        fL = ceil(fL * 20.0) / 20.0 
   993→        fW = ceil(fW * 20.0) / 20.0
   994→        val fLmm = fL * 1000.0
   995→        val fWmm = fW * 1000.0
   996→        
   997→        val pu = if (code == DesignCode.EGYPTIAN) p * 1.5 else p * 1.4 
   998→        val qu = pu / (fL * fW)
   999→        
  1000→        val projectionL = (fLmm - colT) / 2.0 
  1001→        val projectionW = (fWmm - colB) / 2.0
  1002→        val muL = (qu / 1000.0) * (projectionL.pow(2) / 2.0)
  1003→        val muW = (qu / 1000.0) * (projectionW.pow(2) / 2.0)
  1004→        
  1005→        var thickness = 500.0
  1006→        var d = thickness - 70.0
  1007→        var punchingStress: Double
  1008→        val punchingLimit = if (code == DesignCode.EGYPTIAN) {
  1009→            0.316 * sqrt(fcu / 1.5) // ECP 203
  1010→        } else {
  1011→            0.33 * 0.75 * sqrt(fcu * 0.8) // ACI 318 (Phi = 0.75)
  1012→        }
  1013→        
  1014→        do {
  1015→            d = thickness - 70.0
  1016→            val punchingForce = pu * 1000.0 * (1 - (colB + d)*(colT + d)/(fLmm*fWmm))
  1017→            val b0 = 2 * (colB + d + colT + d)
  1018→            punchingStress = punchingForce / (b0 * d)
  1019→            if (punchingStress > punchingLimit) thickness += 50.0
  1020→        } while (punchingStress > punchingLimit && thickness < 2000.0)
  1021→        
  1022→        val asReqL = calculateAs(muL / 1e3, fcu, fy, d, 1000.0, code)
  1023→        val asReqW = calculateAs(muW / 1e3, fcu, fy, d, 1000.0, code)
  1024→        val asMin = 0.0015 * 1000.0 * thickness
  1025→        
  1026→        val finalAsL = max(asReqL, asMin)
  1027→        val finalAsW = max(asReqW, asMin)
  1028→        
  1029→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
  1030→        val finalSpacingL = min(200.0, floor((1000.0 * barArea / finalAsL) / 10.0) * 10.0).coerceAtLeast(100.0)
  1031→        val finalSpacingW = min(200.0, floor((1000.0 * barArea / finalAsW) / 10.0) * 10.0).coerceAtLeast(100.0)
  1032→        
  1033→        val vol = (fL * fW * thickness / 1000.0)
  1034→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1035→        val numBarsL = (fWmm / finalSpacingL).toInt() + 1
  1036→        val numBarsW = (fLmm / finalSpacingW).toInt() + 1
  1037→        val totalSteelWeight = (numBarsL * fL + numBarsW * fW) * barWeightPerMeter * 1.05
  1038→        
  1039→        val actualPressure = (p * 1.1) / (fL * fW)
  1040→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1041→        safetyChecks.add(DesignSafetyCheck("Soil Pressure", actualPressure, soil, "kPa", actualPressure <= soil))
  1042→        safetyChecks.add(DesignSafetyCheck("Punching Shear", punchingStress, punchingLimit, "MPa", punchingStress <= punchingLimit))
  1043→
  1044→        return FootingResult(
  1045→            type = FootingType.ISOLATED, width = fWmm, length = fLmm, thickness = thickness,
  1046→            soilPressure = actualPressure, allowablePressure = soil,
  1047→            reinforcementBottom = ReinforcementBar(spacing = finalSpacingL, diameter = preferredDiameter, description = "Bottom Mesh"),
  1048→            isSafe = actualPressure <= soil && punchingStress <= punchingLimit, code = code,
  1049→            concreteVolume = vol, steelWeight = totalSteelWeight, 
  1050→            cost = vol * settingsManager.concretePrice + (totalSteelWeight / 1000.0 * settingsManager.steelPrice),
  1051→            barsX = numBarsW, barsY = numBarsL, barDiameter = preferredDiameter, safetyChecks = safetyChecks,
  1052→            utilizationRatio = utilizationRatio
  1053→        )
  1054→        }

  1056→    private fun calculateStripFootingInternal(
  1057→        p_per_m: Double, fcu: Double, fy: Double, soil: Double, colB: Double,
  1058→        code: DesignCode, preferredDiameter: Int
  1059→    ): FootingResult {
  1060→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1061→        val width = (p_per_m * 1.1) / soil // m
  1062→        val w_mm = ceil(width * 20.0) / 20.0 * 1000.0
  1063→        
  1064→        val pu = if (code == DesignCode.EGYPTIAN) p_per_m * 1.5 else p_per_m * 1.4
  1065→        val qu = pu / (w_mm / 1000.0)
  1066→        
  1067→        val projection = (w_mm - colB) / 2.0
  1068→        val mu = (qu / 1000.0) * (projection.pow(2) / 2.0)
  1069→        
  1070→        var thickness = 400.0
  1071→        var d = thickness - 70.0
  1072→        val shearLimit = if (code == DesignCode.EGYPTIAN) 0.16 * sqrt(fcu / 1.5) else 0.17 * sqrt(fcu * 0.8) * 0.75
  1073→        
  1074→        do {
  1075→            d = thickness - 70.0
  1076→            val shearForce = qu * (projection - d) / 1000.0 // kN
  1077→            val shearStress = shearForce * 1000.0 / (1000.0 * d)
  1078→            if (shearStress > shearLimit) thickness += 50.0
  1079→        } while (shearStress > shearLimit && thickness < 1500.0)
  1080→        
  1081→        val asReq = calculateAs(mu / 1e3, fcu, fy, d, 1000.0, code)
  1082→        val asMin = 0.0015 * 1000.0 * thickness
  1083→        val finalAs = max(asReq, asMin)
  1084→        
  1085→        val actualPressure = p_per_m * 1.1 / (w_mm / 1000.0)
  1086→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1087→        
  1088→        val barArea = PI * preferredDiameter.toDouble().pow(2.0) / 4.0
  1089→        val spacing = min(200.0, floor((1000.0 * barArea / finalAs) / 10.0) * 10.0).coerceAtLeast(100.0)
  1090→        
  1091→        val vol = (w_mm / 1000.0) * 1.0 * (thickness / 1000.0)
  1092→        val steelWeight = (1000.0 / spacing) * (w_mm / 1000.0) * (preferredDiameter.toDouble().pow(2.0) / 162.0) * 1.1
  1093→        
  1094→        safetyChecks.add(DesignSafetyCheck("Soil Pressure", p_per_m * 1.1 / (w_mm/1000.0), soil, "kPa", true))
  1095→
  1096→        return FootingResult(
  1097→            type = FootingType.STRIP, width = w_mm, length = 1000.0, thickness = thickness,
  1098→            soilPressure = p_per_m * 1.1 / (w_mm/1000.0), allowablePressure = soil,
  1099→            reinforcementBottom = ReinforcementBar(spacing = spacing, diameter = preferredDiameter, description = "Transverse Steel"),
  1100→            isSafe = true, code = code, concreteVolume = vol, steelWeight = steelWeight,
  1101→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1102→            safetyChecks = safetyChecks
  1103→        )
  1104→        }

  1106→    private fun calculateRaftInternal(
  1107→        totalP: Double, fcu: Double, fy: Double, soil: Double,
  1108→        code: DesignCode, preferredDiameter: Int
  1109→    ): FootingResult {
  1110→        val area = (totalP * 1.1) / soil
  1111→        val side = ceil(sqrt(area) * 2.0) / 2.0
  1112→        val thickness = (side * 1000.0 / 10.0).coerceIn(600.0, 2000.0) // L/10 approx
  1113→        val vol = side * side * thickness / 1000.0
  1114→        
  1115→        val actualPressure = totalP * 1.1 / (side * side)
  1116→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1117→        
  1118→        val mu = (totalP / (side * side)) * (side / 3.0).pow(2) / 8.0 // Simplified span moment
  1119→        val asReq = calculateAs(mu, fcu, fy, thickness - 70.0, 1000.0, code)
  1120→        val steelWeight = side * side * (asReq * 2 * 7.85 * 1e-3) * 1.2 // 2 directions + top/bottom mesh estimate
  1121→        
  1122→        return FootingResult(
  1123→            type = FootingType.RAFT, width = side * 1000.0, length = side * 1000.0, thickness = thickness,
  1124→            soilPressure = totalP * 1.1 / (side * side), allowablePressure = soil,
  1125→            reinforcementBottom = ReinforcementBar(spacing = 150.0, diameter = preferredDiameter, description = "Bottom Mat"),
  1126→            isSafe = true, code = code, concreteVolume = vol, steelWeight = steelWeight,
  1127→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1128→            safetyChecks = listOf(DesignSafetyCheck("Average Pressure", totalP / (side * side), soil, "kPa", true)),
  1129→            utilizationRatio = utilizationRatio
  1130→        )
  1131→        }

  1133→    private fun calculatePileCapInternal(
  1134→        p: Double, numPiles: Int, pileDia: Double, pileCap: Double, fcu: Double, fy: Double,
  1135→        colB: Double, colT: Double, code: DesignCode, preferredDiameter: Int
  1136→    ): FootingResult {
  1137→        val spacing = 3 * pileDia
  1138→        val edge = 1.0 * pileDia
  1139→        val rows = ceil(sqrt(numPiles.toDouble())).toInt()
  1140→        val cols = ceil(numPiles.toDouble() / rows).toInt()
  1141→        val length = (cols - 1) * spacing + 2 * edge
  1142→        val width = (rows - 1) * spacing + 2 * edge
  1143→        
  1144→        val thickness = max(800.0, pileDia * 2)
  1145→        val d = thickness - 100.0
  1146→        
  1147→        // Simplified Truss Method
  1148→        val av = (spacing - colT) / 2.0
  1149→        val tForce = (p * 1.5 * av) / (0.87 * d) // Simplified tension
  1150→        val asReq = tForce * 1000.0 / (fy / 1.15)
  1151→        
  1152→        val vol = length * width * thickness / 1e9
  1153→        val steelWeight = (asReq * length / 1000.0 * 2 + asReq * width / 1000.0 * 2) * 7.85 * 1e-3 * 1000.0 // Very rough kg
  1154→        
  1155→        return FootingResult(
  1156→            type = FootingType.PILE_CAP, width = width, length = length, thickness = thickness,
  1157→            soilPressure = (p * 1.1) / numPiles, allowablePressure = pileCap,
  1158→            reinforcementBottom = ReinforcementBar(numBars = (asReq / 201.0).toInt().coerceAtLeast(8), diameter = 16, description = "Main Ties"),
  1159→            isSafe = (p * 1.1 / numPiles) <= pileCap, code = code,
  1160→            concreteVolume = vol, steelWeight = steelWeight,
  1161→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1162→            safetyChecks = listOf(DesignSafetyCheck("Load per Pile", p * 1.1 / numPiles, pileCap, "kN", true))
  1163→        )
  1164→        }

  1167→    private fun calculateCombinedFootingInternal(
  1168→        p1: Double, p2: Double, distance: Double,
  1169→        fcu: Double, fy: Double, soil: Double,
  1170→        colB: Double, colT: Double, code: DesignCode,
  1171→        preferredDiameter: Int
  1172→    ): FootingResult {
  1173→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1174→        val totalWorkingLoad = (p1 + p2) * 1.1
  1175→        val areaReq = totalWorkingLoad / soil
  1176→        
  1177→        // Resultant position from P1
  1178→        val xR = (p2 * distance) / (p1 + p2)
  1179→        val s1 = 600.0 // Default projection
  1180→        val length = 2 * (xR + s1)
  1181→        var width = (areaReq * 1e6) / length
  1182→        width = ceil(width / 50.0) * 50.0
  1183→        
  1184→        val actualArea = (length * width) / 1e6
  1185→        val actualPressure = totalWorkingLoad / actualArea
  1186→        val utilizationRatio = (actualPressure / soil).coerceIn(0.0, 1.2)
  1187→        
  1188→        val pu1 = if (code == DesignCode.EGYPTIAN) p1 * 1.5 else p1 * 1.4
  1189→        val pu2 = if (code == DesignCode.EGYPTIAN) p2 * 1.5 else p2 * 1.4
  1190→        val qu_ult = (pu1 + pu2) / actualArea
  1191→        
  1192→        var thickness = 800.0
  1193→        var d = thickness - 70.0
  1194→        val punchingLimit = if (code == DesignCode.EGYPTIAN) 0.316 * sqrt(fcu / 1.5) else 0.316 * sqrt(fcu * 0.8)
  1195→        
  1196→        // Punching check for both columns
  1197→        val b0_1 = 2 * (colB + d + colT + d)
  1198→        val stress1 = pu1 * 1000.0 / (b0_1 * d)
  1199→        val b0_2 = 2 * (colB + d + colT + d)
  1200→        val stress2 = pu2 * 1000.0 / (b0_2 * d)
  1201→        
  1202→        if (max(stress1, stress2) > punchingLimit) {
  1203→            thickness = ceil((max(pu1, pu2) * 1000.0 / (2 * (colB + colT) * punchingLimit) + 70.0) / 50.0) * 50.0
  1204→            }
  1205→        d = thickness - 70.0
  1206→
  1207→        // Reinforcement (Simplified Beam Analogy)
  1208→        val maxMoment = (qu_ult * (width/1000.0) * (length/2000.0).pow(2)) / 8.0
  1209→        val asReq = calculateAs(maxMoment / (width/1000.0), fcu, fy, d, 1000.0, code)
  1210→        val asMin = 0.0015 * 1000.0 * thickness
  1211→        val finalAs = max(asReq, asMin)
  1212→        
  1213→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1214→        val spacing = (1000.0 * barArea) / finalAs
  1215→        val finalSpacing = min(200.0, floor(spacing / 10.0) * 10.0).coerceAtLeast(100.0)
  1216→        
  1217→        val vol = actualArea * thickness / 1000.0
  1218→        val steelWeight = (actualArea * 150.0) // Rough estimate for combined 150kg/m3
  1219→        
  1220→        safetyChecks.add(DesignSafetyCheck("Soil Pressure", actualPressure, soil, "kPa", actualPressure <= soil))
  1221→        safetyChecks.add(DesignSafetyCheck("Punching Column 1", stress1, punchingLimit, "MPa", stress1 <= punchingLimit))
  1222→        safetyChecks.add(DesignSafetyCheck("Punching Column 2", stress2, punchingLimit, "MPa", stress2 <= punchingLimit))
  1223→
  1224→        return FootingResult(
  1225→            type = FootingType.COMBINED,
  1226→            width = width,
  1227→            length = length,
  1228→            thickness = thickness,
  1229→            soilPressure = actualPressure,
  1230→            allowablePressure = soil,
  1231→            reinforcementBottom = ReinforcementBar(spacing = finalSpacing, diameter = preferredDiameter, description = "Bottom Mesh"),
  1232→            reinforcementTopX = (finalAs * 0.8 / barArea).toInt(), // Simplified top reinforcement
  1233→            topBarDiameter = preferredDiameter,
  1234→            isSafe = actualPressure <= soil && max(stress1, stress2) <= punchingLimit,
  1235→            code = code,
  1236→            concreteVolume = vol,
  1237→            steelWeight = steelWeight,
  1238→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1239→            isCombined = true,
  1240→            distanceBetweenColumns = distance,
  1241→            column1Size = Pair(colT, colB),
  1242→            column2Size = Pair(colT, colB),
  1243→            safetyChecks = safetyChecks,
  1244→            utilizationRatio = utilizationRatio
  1245→        )
  1246→        }

  1249→    fun designStaircase(
  1250→        type: StairType,
  1251→        span: Double,
  1252→        riser: Double,
  1253→        tread: Double,
  1254→        deadLoad: Double,
  1255→        liveLoad: Double,
  1256→        fcu: Double,
  1257→        fy: Double,
  1258→        preferredDiameter: Int,
  1259→        code: DesignCode
  1260→    ): StairResult {
  1261→        // 1. Thickness (L/25 or L/20 depending on support)
  1262→        val ts = (span * 1000.0 / 25.0).coerceAtLeast(150.0)
  1263→        val angle = atan(riser / tread)
  1264→        val cosAlpha = cos(angle)
  1265→        
  1266→        // 2. Loads on slant
  1267→        val wu = when(code) {
  1268→            DesignCode.EGYPTIAN -> 1.4 * deadLoad + 1.6 * liveLoad
  1269→            DesignCode.ACI, DesignCode.SAUDI -> 1.2 * deadLoad + 1.6 * liveLoad
  1270→            }
  1271→        
  1272→        val minSteelRatio = when(code) {
  1273→            DesignCode.EGYPTIAN -> 0.0015  // ECP 203 solid slab
  1274→            DesignCode.ACI -> 0.0018       // ACI 318 one-way slab
  1275→            DesignCode.SAUDI -> 0.002       // SBC 304 hot climate
  1276→        }
  1277→        val cover = when(code) {
  1278→            DesignCode.EGYPTIAN -> 25.0
  1279→            DesignCode.ACI -> 38.0
  1280→            DesignCode.SAUDI -> 40.0
  1281→        }
  1282→        
  1283→        // 3. Moment calculation (Simple span)
  1284→        val mu = (wu * span.pow(2)) / 8.0
  1285→        val d = ts - cover
  1286→        
  1287→        // 4. Reinforcement
  1288→        val asReq = calculateAs(mu, fcu, fy, d, 1000.0, code)
  1289→        val asMin = minSteelRatio * 1000.0 * ts
  1290→        val finalAs = max(asReq, asMin)
  1291→        
  1292→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1293→        val spacing = (1000.0 * barArea) / finalAs
  1294→        val finalSpacing = min(200.0, floor(spacing / 10.0) * 10.0).coerceAtLeast(100.0)
  1295→        
  1296→        // 5. Quantities
  1297→        val lengthOnSlant = span / cosAlpha
  1298→        val vol = (lengthOnSlant * 1.0 * ts / 1000.0) + (span * (riser / 2000.0)) // m3 per meter width
  1299→        
  1300→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1301→        val distBarWeightPerMeter = (10.0.pow(2.0) / 162.0)
  1302→        
  1303→        val numMainBars = (1000.0 / finalSpacing) + 1
  1304→        val numDistBars = (lengthOnSlant * 1000.0 / 200.0) + 1
  1305→        
  1306→        val steelWeight = (numMainBars * lengthOnSlant * barWeightPerMeter + numDistBars * 1.0 * distBarWeightPerMeter) * 1.05
  1307→        
  1308→        val cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice)
  1309→        
  1310→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1311→        val capacityMu = (finalAs * 0.8 * fy * d / 1e6)
  1312→        val utilization = (mu / capacityMu).coerceIn(0.0, 1.2)
  1313→        safetyChecks.add(DesignSafetyCheck("Flexural Capacity", mu, capacityMu, "kN.m", utilization <= 1.0))
  1314→
  1315→        return StairResult(
  1316→            type = type,
  1317→            thickness = ts,
  1318→            reinforcement = ReinforcementBar(spacing = finalSpacing, diameter = preferredDiameter, description = "Main Steel"),
  1319→            distributionReinforcement = ReinforcementBar(spacing = 200.0, diameter = 10, description = "Distribution"),
  1320→            isSafe = utilization <= 1.0,
  1321→            concreteVolume = vol,
  1322→            steelWeight = steelWeight,
  1323→            cost = cost,
  1324→            code = code,
  1325→            safetyChecks = safetyChecks,
  1326→            utilizationRatio = utilization,
  1327→            mu = mu,
  1328→            wu = wu,
  1329→            span = span,
  1330→            riser = riser,
  1331→            tread = tread,
  1332→            fcu = fcu,
  1333→            fy = fy
  1334→        )
  1335→        }

  1337→    fun designTank(
  1338→        type: TankType,
  1339→        capacity: Double,
  1340→        height: Double,
  1341→        fcu: Double,
  1342→        fy: Double,
  1343→        preferredDiameter: Int = 12,
  1344→        code: DesignCode = DesignCode.EGYPTIAN
  1345→    ): TankResult {
  1346→        val area = capacity / height
  1347→        val dim = sqrt(area)
  1348→        
  1349→        // Design parameters
  1350→        val wallThickness = (height * 100.0).coerceIn(250.0, 500.0)
  1351→        val baseThickness = (wallThickness + 100.0).coerceIn(400.0, 800.0)
  1352→        
  1353→        val gamma_w = 10.0 // kN/m3
  1354→        val waterPressure = gamma_w * height
  1355→        val mu = (waterPressure * height.pow(2)) / 6.0
  1356→        
  1357→        // Code-specific parameters
  1358→        val allowableStress = when(code) {
  1359→            DesignCode.EGYPTIAN -> 170.0  // ECP 203 service stress
  1360→            DesignCode.ACI -> 140.0        // ACI 350
  1361→            DesignCode.SAUDI -> 130.0      // SBC 304 (conservative hot climate)
  1362→        }
  1363→        val minWallRatio = when(code) {
  1364→            DesignCode.EGYPTIAN -> 0.0025
  1365→            DesignCode.ACI -> 0.0030
  1366→            DesignCode.SAUDI -> 0.0035     // Higher for hot/arid climate
  1367→        }
  1368→        val crackWidthLimit = when(code) {
  1369→            DesignCode.EGYPTIAN -> 0.2     // ECP 203: 0.2mm
  1370→            DesignCode.ACI -> 0.25         // ACI 350: 0.25mm
  1371→            DesignCode.SAUDI -> 0.2        // SBC 304: 0.2mm
  1372→        }
  1373→        
  1374→        val d_wall = wallThickness - 50.0
  1375→        val ms = mu / 1.5 // Service moment approx
  1376→        val asReqCrack = (ms * 1e6) / (allowableStress * 0.85 * d_wall)
  1377→        val finalAsWall = max(asReqCrack, minWallRatio * 1000.0 * wallThickness)
  1378→        
  1379→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1380→        val spacing = (1000.0 * barArea) / finalAsWall
  1381→        val finalSpacing = min(200.0, floor(spacing / 10.0) * 10.0).coerceAtLeast(100.0)
  1382→        
  1383→        val providedArea = (1000.0 / finalSpacing) * barArea
  1384→        val utilization = (asReqCrack / providedArea).coerceIn(0.0, 1.2)
  1385→
  1386→        val volWall = (4 * dim * height * wallThickness / 1000.0)
  1387→        val volBase = (dim * dim * baseThickness / 1000.0)
  1388→        val totalVol = volWall + volBase
  1389→        
  1390→        // Accurate Steel Calculation
  1391→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1392→        
  1393→        // Wall Steel (2 nets: inner and outer)
  1394→        val numVerticalBars = floor((4 * dim * 1000.0) / finalSpacing).toInt() + 1
  1395→        val numHorizontalBars = floor((height * 1000.0) / 200.0).toInt() + 1
  1396→        val wallSteelWeight = (numVerticalBars * height + numHorizontalBars * (4 * dim)) * barWeightPerMeter * 2 // 2 nets
  1397→        
  1398→        // Base Steel (Top and Bottom nets)
  1399→        val numBaseBarsX = floor((dim * 1000.0) / 200.0).toInt() + 1
  1400→        val numBaseBarsY = floor((dim * 1000.0) / 200.0).toInt() + 1
  1401→        val baseSteelWeight = (numBaseBarsX * dim + numBaseBarsY * dim) * barWeightPerMeter * 2 // 2 nets
  1402→        
  1403→        val totalSteelWeight = (wallSteelWeight + baseSteelWeight) * 1.15 // 15% laps/anchors/corners
  1404→        val cost = totalVol * settingsManager.concretePrice + (totalSteelWeight/1000.0 * settingsManager.steelPrice)
  1405→        
  1406→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1407→        safetyChecks.add(DesignSafetyCheck("Crack Control (Steel Area)", providedArea, asReqCrack, "mm2", utilization <= 1.0))
  1408→        safetyChecks.add(DesignSafetyCheck("Wall Thickness", wallThickness, 250.0, "mm", wallThickness >= 250.0))
  1409→
  1410→        return TankResult(
  1411→            type = type, length = dim, width = dim, height = height,
  1412→            wallThickness = wallThickness, baseThickness = baseThickness,
  1413→            wallReinforcement = ReinforcementBar(spacing = finalSpacing, diameter = preferredDiameter, description = "Walls (Vertical)"),
  1414→            baseReinforcement = ReinforcementBar(spacing = 200.0, diameter = preferredDiameter, description = "Base Mat"),
  1415→            isSafe = utilization <= 1.0, concreteVolume = totalVol, steelWeight = totalSteelWeight, 
  1416→            cost = cost, code = code,
  1417→            waterPressure = waterPressure, mu = mu, capacity = capacity,
  1418→            safetyChecks = safetyChecks, utilizationRatio = utilization,
  1419→            fcu = fcu, fy = fy
  1420→        )
  1421→        }

  1423→    fun designRetainingWall(
  1424→        height: Double,
  1425→        soilDensity: Double,
  1426→        frictionAngle: Double,
  1427→        surcharge: Double,
  1428→        fcu: Double,
  1429→        fy: Double,
  1430→        preferredDiameter: Int = 16,
  1431→        code: DesignCode = DesignCode.EGYPTIAN
  1432→    ): RetainingWallResult {
  1433→        // 1. Earth Pressure Coefficients
  1434→        val phiRad = frictionAngle * PI / 180.0
  1435→        val ka = (1 - sin(phiRad)) / (1 + sin(phiRad))
  1436→        val kp = 1 / ka
  1437→        
  1438→        // 2. Dimensions (Standard Proportions)
  1439→        val stemT = (height * 1000.0 / 12.0).coerceAtLeast(300.0) // mm
  1440→        val baseW = (height * 0.5).coerceAtLeast(1.5) // m
  1441→        val baseT = stemT // mm
  1442→        val toeW = baseW / 3.0
  1443→        val heelW = baseW - toeW - (stemT/1000.0)
  1444→        
  1445→        // 3. Stability Analysis
  1446→        val pa = 0.5 * ka * soilDensity * height.pow(2)
  1447→        val ps = ka * surcharge * height
  1448→        
  1449→        val wStem = height * (stemT / 1000.0) * 25.0
  1450→        val wBase = baseW * (baseT / 1000.0) * 25.0
  1451→        val wSoil = heelW * height * soilDensity
  1452→        
  1453→        val xStem = toeW + (stemT / 2000.0)
  1454→        val xBase = baseW / 2.0
  1455→        val xSoil = toeW + (stemT / 1000.0) + (heelW / 2.0)
  1456→        
  1457→        val resistingMoment = (wStem * xStem) + (wBase * xBase) + (wSoil * xSoil)
  1458→        val drivingMoment = (pa * height / 3.0) + (ps * height / 2.0)
  1459→        
  1460→        val otLimit = when(code) {
  1461→            DesignCode.EGYPTIAN, DesignCode.SAUDI -> 1.5
  1462→            DesignCode.ACI -> 2.0
  1463→        }
  1464→        val slideLimit = 1.5
  1465→        val cover = when(code) {
  1466→            DesignCode.EGYPTIAN -> 50.0
  1467→            DesignCode.ACI -> 75.0
  1468→            DesignCode.SAUDI -> 65.0
  1469→        }
  1470→        val minSteel = when(code) {
  1471→            DesignCode.EGYPTIAN -> 0.0013
  1472→            DesignCode.ACI -> 0.0018
  1473→            DesignCode.SAUDI -> 0.002
  1474→        }
  1475→        val lf = when(code) {
  1476→            DesignCode.EGYPTIAN -> 1.4
  1477→            DesignCode.ACI, DesignCode.SAUDI -> 1.6
  1478→        }
  1479→        
  1480→        val totalVerticalWeight = wStem + wBase + wSoil
  1481→        val fsOverturning = resistingMoment / drivingMoment.coerceAtLeast(1.0)
  1482→        val fsSliding = (0.5 * totalVerticalWeight) / (pa + ps).coerceAtLeast(1.0)
  1483→        
  1484→        val utilizationRatio = max(otLimit / fsOverturning, slideLimit / fsSliding)
  1485→        val isSafe = fsOverturning >= otLimit && fsSliding >= slideLimit
  1486→        
  1487→        // 4. Design for Stem Moment
  1488→        val muStem = drivingMoment * lf // Factored
  1489→        val d = stemT - cover
  1490→        val asReq = calculateAs(muStem, fcu, fy, d, 1000.0, code)
  1491→        
  1492→        val barArea = PI * preferredDiameter.toDouble().pow(2) / 4.0
  1493→        val numBars = ceil(asReq / barArea).toInt().coerceAtLeast(5)
  1494→        val stemReinforcement = ReinforcementBar(numBars, preferredDiameter, description = "Stem (Vertical)")
  1495→        
  1496→        val vol = (height * (stemT / 1000.0)) + (baseW * (baseT / 1000.0))
  1497→        val barWeightPerMeter = (preferredDiameter.toDouble().pow(2.0) / 162.0)
  1498→        val steelWeight = (numBars * height + (1000.0/200.0) * baseW) * barWeightPerMeter * 1.15
  1499→        
  1500→        val safetyChecks = mutableListOf<DesignSafetyCheck>()
  1501→        safetyChecks.add(DesignSafetyCheck("Overturning Stability", fsOverturning, otLimit, "", fsOverturning >= otLimit))
  1502→        safetyChecks.add(DesignSafetyCheck("Sliding Stability", fsSliding, slideLimit, "", fsSliding >= slideLimit))
  1503→        
  1504→        return RetainingWallResult(
  1505→            height = height,
  1506→            stemThickness = stemT,
  1507→            baseWidth = baseW * 1000.0,
  1508→            stemReinforcement = stemReinforcement,
  1509→            baseReinforcement = ReinforcementBar(spacing = 200.0, diameter = preferredDiameter, description = "Base Bottom"),
  1510→            safetyChecks = safetyChecks,
  1511→            isSafe = isSafe,
  1512→            concreteVolume = vol,
  1513→            steelWeight = steelWeight,
  1514→            cost = vol * settingsManager.concretePrice + (steelWeight / 1000.0 * settingsManager.steelPrice),
  1515→            code = code,
  1516→            factorOfSafetyOverturning = fsOverturning,
  1517→            factorOfSafetySliding = fsSliding,
  1518→            utilizationRatio = utilizationRatio,
  1519→            muStem = muStem,
  1520→            pa = pa,
  1521→            ps = ps,
  1522→            ka = ka,
  1523→            soilDensity = soilDensity,
  1524→            fcu = fcu,
  1525→            fy = fy
  1526→        )
  1527→        }

  1529→    fun calculateSeismicLoads(input: SeismicInput): SeismicResult {
  1530→        val h = input.height
  1531→        val sa = (input.zone * input.importance * 2.5) / input.reductionFactor
  1532→        val baseShear = sa * input.totalWeight
  1533→        val forces = mutableMapOf<Int, Double>()
  1534→        val n = max(1, ceil(h / 3.0).toInt())
  1535→        for (i in 1..n) forces[i] = (i.toDouble() / (n*(n+1)/2.0)) * baseShear
  1536→        return SeismicResult(
  1537→            baseShear = baseShear, 
  1538→            storyDrift = 0.005 * h/n, 
  1539→            timePeriod = 0.075 * h.pow(0.75), 
  1540→            spectralAcceleration = sa, 
  1541→            forcesPerFloor = forces, 
  1542→            isSafe = true, 
  1543→            code = DesignCode.EGYPTIAN,
  1544→            zone = input.zone,
  1545→            importance = input.importance,
  1546→            reductionFactor = input.reductionFactor,
  1547→            totalWeight = input.totalWeight,
  1548→            height = input.height
  1549→        )
  1550→    }

  1552→    fun calculateWeldCapacity(size: Double, length: Double, electrode: ElectrodeType, code: DesignCode): Double {
  1553→        val phi = if (code == DesignCode.ACI) 0.75 else 0.70
  1554→        return (phi * 0.60 * electrode.tensileStrength * 0.707 * size * length) / 1000.0 // kN
  1555→    }

  1557→    fun calculateBoltCapacity(diameter: Double, grade: BoltGrade, count: Int, code: DesignCode): Double {

  1559→        val phi = if (code == DesignCode.ACI) 0.75 else 0.65
  1560→        return (phi * 0.5 * grade.fu * area * count) / 1000.0 // kN
