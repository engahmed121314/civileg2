    34→@OptIn(ExperimentalMaterial3Api::class)
    35→@Composable
    36→fun FrameAnalysisScreen(
    37→    viewModel: FrameAnalysisViewModel = hiltViewModel(),
    38→    onNavigateBack: () -> Unit = {}
    39→) {
    40→    val nodes by viewModel.nodes.observeAsState(emptyList())

    45→    val result by viewModel.result.observeAsState()

    47→    val errorMsg by viewModel.errorMessage.observeAsState()
    48→    val diagramType by viewModel.activeDiagramType.observeAsState(DiagramType.BMD)
    49→    val selectedMemberId by viewModel.selectedMemberId.observeAsState()
    50→    val concreteResults by viewModel.concreteResults.observeAsState(emptyList())
    51→    val steelResults by viewModel.steelResults.observeAsState(emptyList())
    52→    val context = LocalContext.current
    53→
    54→    var selectedTab by remember { mutableIntStateOf(0) }
    55→    val tabs = listOf("الرسم", "العقد", "الأعضاء", "الحمولات", "النتائج")
    56→
    57→    // Show errors
    58→    LaunchedEffect(errorMsg) {
    59→        errorMsg?.let { Toast.makeText(context, it, Toast.LENGTH_LONG).show() }
    60→    }
    61→
    62→    Scaffold(
    63→        topBar = {
    64→            TopAppBar(
    65→                title = { Text("تحليل وتصميم الإطارات", fontWeight = FontWeight.Bold) },
    66→                navigationIcon = {
    67→                    IconButton(onClick = onNavigateBack) {
    68→                        Icon(Icons.AutoMirrored.Filled.ArrowBack, contentDescription = "رجوع")
    69→                    }
    70→                },
    71→                colors = TopAppBarDefaults.topAppBarColors(
    72→                    containerColor = Color(0xFF1565C0),
    73→                    titleContentColor = Color.White,
    74→                    navigationIconContentColor = Color.White
    75→                ),
    76→                actions = {
    77→                    // Template buttons
    78→                    IconButton(onClick = {
    79→                        var span = 6.0; var h = 4.0; var udl = 20.0
    80→                        viewModel.loadSimplePortalFrame(span, h, udl)
    81→                    }) {
    82→                        Icon(Icons.Default.ViewInAr, "قالب إطار بسيط", tint = Color.White)
    83→                    }
    84→                    IconButton(onClick = { viewModel.clearAll() }) {
    85→                        Icon(Icons.Default.DeleteSweep, "مسح الكل", tint = Color.White)
    86→                    }
    87→                },
    88→                actions = {
    89→                    // PDF Export
    90→                    if (result?.hasResults == true) {
    91→                        IconButton(onClick = {
    92→                            val inputs = viewModel.getStoredInputs()

    94→                                val file = FrameAnalysisPdfExporter.generateFrameAnalysisPdf(
    95→                                    context, inputs.nodes, inputs.members, inputs.nodalLoads,

   102→                                    type = "application/pdf"
   103→                                    putExtra(android.content.Intent.EXTRA_STREAM, uri)
   104→                                    putExtra(android.content.Intent.EXTRA_SUBJECT, "Frame Analysis Report")
   105→                                    addFlags(android.content.Intent.FLAG_GRANT_READ_URI_PERMISSION)
   106→                                }
   107→                                context.startActivity(Intent.createChooser(intent, "مشاركة التقرير"))
   108→                            } catch (e: Exception) {
   109→                                Toast.makeText(context, "خطأ في إنشاء PDF: ${e.message}", Toast.LENGTH_LONG).show()
   110→                            }
   111→                        }) {
   112→                            Icon(Icons.Default.PictureAsPdf, "تصدير PDF", tint = Color.White)
   113→                        }
   114→                    }
   115→                }
   116→            )
   117→        },
   118→        bottomBar = {
   119→            // Diagram type selector (when solved)
   120→            if (result?.hasResults == true) {

   122→                    tonalElevation = 8.dp,
   123→                    shadowElevation = 8.dp,

   126→                    Row(

   128→                            .fillMaxWidth()

   130→                        horizontalArrangement = Arrangement.Center

   137→                                label = { Text(type.displayNameAr, fontSize = 13.sp) },
   138→                                colors = FilterChipDefaults.filterChipColors(
   139→                                    selectedContainerColor = when (type) {

   141→                                        DiagramType.SFD -> Color(0xFF4CAF50)
   142→                                        DiagramType.AFD -> Color(0xFFFF9800)

   145→                                ),
   146→                                modifier = Modifier.padding(horizontal = 4.dp)

   148→                        }

   159→                contentColor = Color(0xFF1565C0)
   160→            ) {
   161→                for ((index, title) in tabs.withIndex()) {
   162→                    Tab(
   163→                        selected = selectedTab == index,
   164→                        onClick = { selectedTab = index },
   165→                        text = { Text(title, fontSize = 13.sp) }
   166→                    )
   167→                }
   168→            }
   169→
   170→            // Tab Content
   171→            when (selectedTab) {
   172→                0 -> DrawingTab(nodes, members, memberLoads, nodalLoads, result, diagramType, selectedMemberId, viewModel)
   173→                1 -> NodesTab(nodes, members, viewModel)
   174→                2 -> MembersTab(nodes, members, settings, viewModel)
   175→                3 -> LoadsTab(nodes, members, nodalLoads, memberLoads, viewModel)
   176→                4 -> ResultsTab(result, concreteResults, steelResults, selectedMemberId, viewModel)
   177→            }
   178→        }
   179→    }
   180→}

   187→    nodes: List<FrameNode>,

   190→    nodalLoads: List<NodalLoad>,

   228→            modifier = Modifier
   229→                .align(Alignment.TopStart)

   239→                    )
   240→                }
   241→            }
   242→        }
   243→    }
   244→}

   249→@Composable
   250→private fun NodesTab(
   251→    nodes: List<FrameNode>,
   252→    members: List<FrameMember>,
   253→    viewModel: FrameAnalysisViewModel
   254→) {
   255→    var showAddDialog by remember { mutableStateOf(false) }
   256→    var editingNode by remember { mutableStateOf<FrameNode?>(null) }
   257→    var editX by remember { mutableStateOf("") }
   258→    var editY by remember { mutableStateOf("") }
   259→    var editSupport by remember { mutableStateOf(SupportType.Free) }
   260→
   261→    Column(modifier = Modifier.fillMaxSize()) {
   262→        // Header
   263→        Row(
   264→            modifier = Modifier
   265→                .fillMaxWidth()
   266→                .padding(12.dp),
   267→            horizontalArrangement = Arrangement.SpaceBetween,
   268→            verticalAlignment = Alignment.CenterVertically
   269→        ) {
   270→            Text("العقد (${nodes.size})", fontWeight = FontWeight.Bold, fontSize = 16.sp)
   271→            Button(
   272→                onClick = {
   273→                    editX = ""
   274→                    editY = ""
   275→                    editSupport = SupportType.Free
   276→                    editingNode = null
   277→                    showAddDialog = true
   278→                },
   279→                colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF1565C0))
   280→            ) {
   281→                Icon(Icons.Default.Add, contentDescription = null, modifier = Modifier.size(18.dp))
   282→                Spacer(modifier = Modifier.width(4.dp))
   283→                Text("إضافة عقدة")
   284→            }
   285→        }
   286→
   287→        if (nodes.isEmpty()) {
   288→            Box(modifier = Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
   289→                Column(horizontalAlignment = Alignment.CenterHorizontally) {
   290→                    Icon(Icons.Default.AddLocation, contentDescription = null, modifier = Modifier.size(48.dp), tint = Color.Gray)
   291→                    Spacer(modifier = Modifier.height(8.dp))
   292→                    Text("لا توجد عقد بعد", color = Color.Gray)
   293→                    Text("أضف عقد أو استخدم قالب جاهز", color = Color.Gray, fontSize = 12.sp)
   294→                }
   295→            }
   296→            return@Column
   297→        }
   298→
   299→        // Node list header
   300→        Row(
   301→            modifier = Modifier
   302→                .fillMaxWidth()
   303→                .background(Color(0xFFE3F2FD))
   304→                .padding(horizontal = 12.dp, vertical = 8.dp)
   305→        ) {
   306→            Text("ID", modifier = Modifier.width(40.dp), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   307→            Text("X (m)", modifier = Modifier.width(60.dp), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   308→            Text("Y (m)", modifier = Modifier.width(60.dp), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   309→            Text("الارتكاز", modifier = Modifier.weight(1f), fontWeight = FontWeight.Bold, fontSize = 12.sp)
   310→            Text("", modifier = Modifier.width(60.dp))
   311→        }
   312→
   313→        LazyColumn(modifier = Modifier.fillMaxSize()) {
   314→            items(nodes, key = { it.id }) { node ->
   315→                val isConnected = members.any { it.nodeI == node.id || it.nodeJ == node.id }
   316→                Row(
   317→                    modifier = Modifier
   318→                        .fillMaxWidth()
   319→                        .then(if (isConnected) Modifier else Modifier.background(Color(0xFFFFF8E1)))
   320→                        .padding(horizontal = 12.dp, vertical = 8.dp),
   321→                    verticalAlignment = Alignment.CenterVertically
   322→                ) {
   323→                    Text("${node.id}", modifier = Modifier.width(40.dp), fontWeight = FontWeight.Bold)
   324→                    Text("${node.x}", modifier = Modifier.width(60.dp))
   325→                    Text("${node.y}", modifier = Modifier.width(60.dp))
   326→                    Text(node.support.displayNameAr, modifier = Modifier.weight(1f), fontSize = 12.sp)
   327→                    IconButton(onClick = {
   328→                        editingNode = node
   329→                        editX = node.x.toString()
   330→                        editY = node.y.toString()
   331→                        editSupport = node.support
   332→                        showAddDialog = true
   333→                    }) {
   334→                        Icon(Icons.Default.Edit, "تعديل", modifier = Modifier.size(18.dp), tint = Color(0xFF1565C0))
   335→                    }
   336→                }
   337→                HorizontalDivider()
   338→            }
   339→        }
   340→    }
   341→
   342→    // Add/Edit Dialog
   343→    if (showAddDialog) {
   344→        AlertDialog(
   345→            onDismissRequest = { showAddDialog = false },
   346→            title = Text(if (editingNode != null) "تعديل عقدة" else "إضافة عقدة"),
   347→            text = {
   348→                Column {
   349→                    OutlinedTextField(
   350→                        value = editX, onValueChange = { editX = it },
   351→                        label = { Text("X (متر)") },
   352→                        keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
   353→                        modifier = Modifier.fillMaxWidth()
   354→                    )
   355→                    Spacer(modifier = Modifier.height(8.dp))
   356→                    OutlinedTextField(
   357→                        value = editY, onValueChange = { editY = it },
   358→                        label = { Text("Y (متر)") },
   359→                        keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Decimal),
   360→                        modifier = Modifier.fillMaxWidth()
   361→                    )
   362→                    Spacer(modifier = Modifier.height(8.dp))
   363→                    Text("نوع الارتكاز:", fontWeight = FontWeight.Bold, fontSize = 13.sp)
   364→                    Spacer(modifier = Modifier.height(4.dp))
   365→                    for (st in SupportType.entries) {
   366→                        Row(verticalAlignment = Alignment.CenterVertically, modifier = Modifier.padding(2.dp)) {
   367→                            RadioButton(
   368→                                selected = editSupport == st,
   369→                                onClick = { editSupport = st }
   370→                            )
   371→                            Text(st.displayNameAr, fontSize = 13.sp)
   372→                        }
   373→                    }
   374→                }
   375→            },
   376→            confirmButton = {
   377→                TextButton(onClick = {
   378→                    val x = editX.toDoubleOrNull() ?: return@TextButton
   379→                    val y = editY.toDoubleOrNull() ?: return@TextButton
   380→                    if (editingNode != null) {
   381→                        viewModel.updateNode(editingNode!!.id, x, y, editSupport)
   382→                    } else {
   383→                        viewModel.addNode(x, y, editSupport)
   384→                    }
   385→                    showAddDialog = false
   386→                }) { Text("حفظ", color = Color(0xFF1565C0)) }
   387→            },
   388→            dismissButton = {
   389→                TextButton(onClick = { showAddDialog = false }) { Text("إلغاء") }
   390→            }
   391→        )
   392→    }
   393→}

   971→private fun UtilizationBar(ratio: Double, modifier: Modifier = Modifier) {
   972→    val clampedRatio = ratio.coerceIn(0.0, 1.5)
   973→    val color = when {
   974→        clampedRatio <= 0.7 -> Color(0xFF4CAF50)
   975→        clampedRatio <= 1.0 -> Color(0xFFFF9800)
   976→        else -> Color(0xFFF44336)
   977→    }
   978→    Row(modifier = modifier.height(16.dp), verticalAlignment = Alignment.CenterVertically) {
   979→        LinearProgressIndicator(
   980→            progress = { clampedRatio.toFloat() / 1.5f },
   981→            modifier = Modifier
   982→                .weight(1f)
   983→                .height(8.dp)
   984→                .clip(RoundedCornerShape(4.dp)),
   985→            color = color,
   986→            trackColor = Color.LightGray
   987→        )
   988→        Spacer(modifier = Modifier.width(8.dp))
   989→        Text(
   990→            "${String.format("%.0f", ratio * 100)}%",
   991→            fontSize = 11.sp,
   992→            fontWeight = FontWeight.Bold,
   993→            color = color
   994→        )
   995→    }
