   592→   592→private fun generateBarPositions(width: Double, depth: Double, numBars: Int, diameter: Double): List<BarInfo> {
   593→   593→    val cover = 40.0
   594→   594→    val cx = width / 2.0
   595→   595→    val cy = depth / 2.0
   596→   596→    val effectiveW = width - 2 * cover
   597→   597→    val effectiveD = depth - 2 * cover
   598→   598→    val positions = mutableListOf<BarInfo>()
   599→   599→    val barsPerSide = maxOf(2, numBars / 4)
   600→   600→    val corners = 4
   601→   601→
   602→   602→    // Corner bars
   603→   603→    listOf(
   604→   604→        BarInfo(cover + diameter/2, cover + diameter/2, diameter, isCorner = true),
   605→   605→        BarInfo(width - cover - diameter/2, cover + diameter/2, diameter, isCorner = true),
   606→   606→        BarInfo(cover + diameter/2, depth - cover - diameter/2, diameter, isCorner = true),
   607→   607→        BarInfo(width - cover - diameter/2, depth - cover - diameter/2, diameter, isCorner = true)
   608→   608→    ).forEach { positions.add(it) }
   609→   609→
   610→   610→    // Distribute remaining bars along sides
   611→   611→    val remaining = numBars - 4
   612→   612→    if (remaining > 0) {
   613→   613→        val perSide = remaining / 4
   614→   614→        for (side in 0 until 4) {
   615→   615→            val count = if (side < remaining % 4) perSide + 1 else perSide
   616→   616→            for (i in 1..count) {
   617→   617→                val t = i.toDouble() / (count + 1)
   618→   618→                val pos = when (side) {
   619→   619→                    0 -> BarInfo(cover + diameter/2 + t * effectiveW, cover + diameter/2, diameter)
   620→   620→                    1 -> BarInfo(width - cover - diameter/2, cover + diameter/2 + t * effectiveD, diameter)
   621→   621→                    2 -> BarInfo(width - cover - diameter/2 - t * effectiveW, depth - cover - diameter/2, diameter)
   622→   622→                    else -> BarInfo(cover + diameter/2, depth - cover - diameter/2 - t * effectiveD, diameter)
   623→   623→                }
   624→   624→                positions.add(pos)
   625→   625→            }
   626→   626→        }
   627→   627→    }
   628→   628→    return positions
   629→   629→}
