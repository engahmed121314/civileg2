   278→   278→@Composable
   279→   279→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   280→   280→    Card(
   281→   281→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   282→   282→        border = BorderStroke(1.dp, Color.LightGray)
   283→   283→    ) {
   284→   284→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   285→   285→            val padding = 40.dp.toPx()
   286→   286→            val w = size.width - 2 * padding
   287→   287→            val h = size.height - 2 * padding
   288→   288→            val baseY = size.height - padding
   289→   289→            
   290→   290→            // Scale calculations
   291→   291→            val scale = (w / inputs.span).toFloat()
   292→   292→            val eh = (inputs.eaveHeight * scale).toFloat()
   293→   293→            val rh = (inputs.ridgeHeight * scale).toFloat()
   294→   294→            val midX = padding + w / 2
   295→   295→            
   296→   296→            // Draw Main Frame Outline
   297→   297→            val framePath = Path().apply {
   298→   298→                moveTo(padding, baseY)
   299→   299→                lineTo(padding, baseY - eh)
   300→   300→                lineTo(midX, baseY - rh)
   301→   301→                lineTo(padding + w, baseY - eh)
   302→   302→                lineTo(padding + w, baseY)
   303→   303→            }
   304→   304→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   305→   305→            
   306→   306→            // Draw Distributed Load Arrows
   307→   307→            val loadColor = Color(0xFFE53935)
   308→   308→            for (i in 0..12) {
   309→   309→                val x = padding + (i * w / 12)
   310→   310→                val y = if (x <= midX) {
   311→   311→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   312→   312→                } else {
   313→   313→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   314→   314→                }
   315→   315→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   316→   316→                // Arrow heads
   317→   317→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   318→   318→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   319→   319→            }
   320→   320→            
   321→   321→            // Draw Moment Diagram (Simplified)
   322→   322→            val momentColor = Color(0xFF1E88E5)
   323→   323→            val momentPath = Path().apply {
   324→   324→                moveTo(padding, baseY - eh)
   325→   325→                // Simulated parabolic curve for moment
   326→   326→                cubicTo(
   327→   327→                    padding + w * 0.25f, baseY - eh + 50f,
   328→   328→                    padding + w * 0.75f, baseY - eh + 50f,
   329→   329→                    padding + w, baseY - eh
   330→   330→                )
   331→   331→            }
   332→   332→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   333→   333→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   334→   334→            
   335→   335→            // Labels
   336→   336→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   337→   337→        }
   338→   338→    }
   339→   339→}

   413→   413→    val colRX = startX + (inputs.span * scale).toFloat()
   414→   414→    val eaveY = baseY - (inputs.eaveHeight * scale).toFloat()
   415→   415→    val ridgeY = baseY - (inputs.ridgeHeight * scale).toFloat()
   416→   416→    val midX = startX + (inputs.span * scale / 2).toFloat()
   417→   417→
   418→   418→    // Columns
   419→   419→    drawLine(Color.Black, Offset(colLX, baseY), Offset(colLX, eaveY), strokeWidth = 8f)
   420→   420→    drawLine(Color.Black, Offset(colRX, baseY), Offset(colRX, eaveY), strokeWidth = 8f)
   421→   421→    // Rafters
   422→   422→    drawLine(Color.Black, Offset(colLX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   423→   423→    drawLine(Color.Black, Offset(colRX, eaveY), Offset(midX, ridgeY), strokeWidth = 8f)
   424→   424→    
   425→   425→    // Foundations
   426→   426→    drawRect(Color.Gray, Offset(colLX - 15f, baseY), Size(30f, 15f))
   427→   427→    drawRect(Color.Gray, Offset(colRX - 15f, baseY), Size(30f, 15f))
   428→   428→}

   430→   430→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   431→   431→    val planW = (inputs.span * scale).toFloat()
   432→   432→    val planL = (inputs.length * scale).toFloat()
   433→   433→    val planTop = baseY - planL
   434→   434→    
   435→   435→    // Boundary
   436→   436→    drawRect(Color.Black, Offset(startX, planTop), Size(planW, planL), style = androidx.compose.ui.graphics.drawscope.Stroke(4f))
   437→   437→    
   438→   438→    // Grid Lines (Bays)
   439→   439→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   440→   440→    for (i in 0..numBays) {
   441→   441→        val y = planTop + (i * inputs.baySpacing * scale).toFloat()
   442→   442→        drawLine(Color.LightGray, Offset(startX, y), Offset(startX + planW, y), strokeWidth = 1f)
   443→   443→        // Columns markers
   444→   444→        drawCircle(Color.Black, 4f, Offset(startX, y))
   445→   445→        drawCircle(Color.Black, 4f, Offset(startX + planW, y))
   446→   446→    }
   447→   447→}

   449→   449→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawSideElevation(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   450→   450→    val planL = (inputs.length * scale).toFloat()
   451→   451→    val eaveH = (inputs.eaveHeight * scale).toFloat()
   452→   452→    val eaveY = baseY - eaveH
   453→   453→    
   454→   454→    // Ground
   455→   455→    drawLine(Color.Gray, Offset(startX, baseY), Offset(startX + planL, baseY), strokeWidth = 2f)
   456→   456→    
   457→   457→    // Columns
   458→   458→    val numBays = ceil(inputs.length / inputs.baySpacing).toInt()
   459→   459→    for (i in 0..numBays) {
   460→   460→        val x = startX + (i * inputs.baySpacing * scale).toFloat()
   461→   461→        drawLine(Color.Black, Offset(x, baseY), Offset(x, eaveY), strokeWidth = 6f)
   462→   462→    }
   463→   463→    // Eave Line
   464→   464→    drawLine(Color.Black, Offset(startX, eaveY), Offset(startX + planL, eaveY), strokeWidth = 6f)
   465→   465→}

   467→   467→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   468→   468→    val s = scale.toFloat() * 0.65f
   469→   469→    val depthOff = 35f
   470→   470→    
   471→   471→    // Draw from back to front for better perspective look
   472→   472→    for (i in 4 downTo 0) {
   473→   473→        val off = i * depthOff
   474→   474→        val x0 = startX + off
   475→   475→        val y0 = baseY - off
   476→   476→        val spanW = (inputs.span * s).toFloat()
   477→   477→        val eaveH = (inputs.eaveHeight * s).toFloat()
   478→   478→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   479→   479→        
   480→   480→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   481→   481→        val stroke = if (i == 0) 3f else 1.5f
   482→   482→        
   483→   483→        // Sections of the frame
   484→   484→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   485→   485→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   486→   486→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   487→   487→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   488→   488→        
   489→   489→        // Purlins/Girts connections (longitudinal)
   490→   490→        if (i > 0) {
   491→   491→            val px = x0 - depthOff
   492→   492→            val py = y0 + depthOff
   493→   493→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   494→   494→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   495→   495→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   496→   496→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   497→   497→        }
   498→   498→    }
   499→   499→}

   533→   533→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   534→   534→    val p = Paint().apply { 
   535→   535→        color = android.graphics.Color.BLACK
   536→   536→        strokeWidth = 6f
   537→   537→        isAntiAlias = true
   538→   538→        style = Paint.Style.STROKE
   539→   539→    }
   540→   540→    val dimPaint = Paint().apply {
   541→   541→        color = android.graphics.Color.DKGRAY
   542→   542→        strokeWidth = 2f
   543→   543→        textSize = 24f
   544→   544→        isAntiAlias = true
   545→   545→    }
   546→   546→    val textP = Paint().apply {
   547→   547→        color = android.graphics.Color.BLACK
   548→   548→        textSize = 22f
   549→   549→        isFakeBoldText = true
   550→   550→        textAlign = Paint.Align.CENTER
   551→   551→    }
   552→   552→    
   553→   553→    val s = scale * 0.7f
   554→   554→    val w = inputs.span.toFloat() * s
   555→   555→    val eh = inputs.eaveHeight.toFloat() * s
   556→   556→    val rh = inputs.ridgeHeight.toFloat() * s
   557→   557→    val midX = x + w/2
   558→   558→    
   559→   559→    // Draw Main Frame (Columns & Rafters)
   560→   560→    canvas.drawLine(x, y, x, y - eh, p)
   561→   561→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   562→   562→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   563→   563→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   564→   564→    
   565→   565→    // Foundations
   566→   566→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   567→   567→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   568→   568→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   569→   569→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   570→   570→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   571→   571→    p.strokeWidth = 6f
   572→   572→    
   573→   573→    // --- Dimension Lines ---
   574→   574→    // Span Dimension
   575→   575→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   576→   576→    
   577→   577→    // Eave Height Dimension
   578→   578→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   579→   579→    
   580→   580→    // Ridge Height Dimension
   581→   581→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   582→   582→
   583→   583→    // --- Section Labels ---
   584→   584→    result?.let { res ->
   585→   585→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   586→   586→        
   587→   587→        // Column Label
   588→   588→        canvas.save()
   589→   589→        canvas.rotate(-90f, x + 25f, y - eh/2)
   590→   590→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   591→   591→        canvas.restore()
   592→   592→        
   593→   593→        // Rafter Label
   594→   594→        canvas.save()
   595→   595→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   596→   596→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   597→   597→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   598→   598→        canvas.restore()
   599→   599→    }
   600→   600→
   601→   601→    // Connection Markers (Circles at joints)
   602→   602→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   603→   603→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   604→   604→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   605→   605→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   606→   606→}

   608→   608→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   609→   609→    val tickSize = 15f
   610→   610→    canvas.drawLine(x1, y1, x2, y2, paint)
   611→   611→    
   612→   612→    // Draw Ticks (45 degree lines)
   613→   613→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   614→   614→    val tickAngle = angle + Math.PI / 4
   615→   615→    
   616→   616→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   617→   617→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   618→   618→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   619→   619→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   620→   620→
   621→   621→    // Text label
   622→   622→    val midX = (x1 + x2) / 2
   623→   623→    val midY = (y1 + y2) / 2
   624→   624→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   625→   625→    
   626→   626→    if (x1 == x2) { // Vertical
   627→   627→        canvas.save()
   628→   628→        canvas.rotate(-90f, midX - 10f, midY)
   629→   629→        canvas.drawText(text, midX - 10f, midY, textPaint)
   630→   630→        canvas.restore()
   631→   631→    } else { // Horizontal
   632→   632→        canvas.drawText(text, midX, midY - 10f, textPaint)
   633→   633→    }
   634→   634→}

   636→   636→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   637→   637→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   638→   638→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   639→   639→    
   640→   640→    val s = scale * 0.6f
   641→   641→    val w = inputs.span.toFloat() * s
   642→   642→    val l = inputs.length.toFloat() * s
   643→   643→    val planTop = y - l
   644→   644→    
   645→   645→    canvas.drawRect(x, planTop, x + w, y, p)
   646→   646→    
   647→   647→    val baySpacing = inputs.baySpacing.toFloat() * s
   648→   648→    var curY = planTop
   649→   649→    var count = 0
   650→   650→    while (curY <= y + 0.1f) {
   651→   651→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   652→   652→        
   653→   653→        // Column markers
   654→   654→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   655→   655→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   656→   656→        
   657→   657→        if (curY + baySpacing <= y + 0.1f) {
   658→   658→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   659→   659→        }
   660→   660→        
   661→   661→        curY += baySpacing
   662→   662→        count++
   663→   663→    }
   664→   664→    
   665→   665→    // Span Dimension
   666→   666→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   667→   667→    // Total Length Dimension
   668→   668→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   669→   669→}

   671→   671→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   672→   672→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   673→   673→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   674→   674→    
   675→   675→    val s = scale * 0.6f
   676→   676→    val l = inputs.length.toFloat() * s
   677→   677→    val eh = inputs.eaveHeight.toFloat() * s
   678→   678→    
   679→   679→    canvas.drawLine(x, y, x + l, y, p) // Ground
   680→   680→    val bay = inputs.baySpacing.toFloat() * s
   681→   681→    var cur = x
   682→   682→    while (cur <= x + l + 0.1f) {
   683→   683→        canvas.drawLine(cur, y, cur, y - eh, p)
   684→   684→        cur += bay
   685→   685→    }
   686→   686→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   687→   687→    
   688→   688→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   689→   689→}

  1315→  1315→fun SteelSectionDrawing(section: SteelSectionType) {
  1316→  1316→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1317→  1317→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1318→  1318→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1319→  1319→                val w = size.width
  1320→  1320→                val h = size.height
  1321→  1321→                
  1322→  1322→                when (section) {
  1323→  1323→                    is SteelSectionType.ISection -> {
  1324→  1324→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1325→  1325→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1326→  1326→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1327→  1327→                        val offsetBF = (w - bf) / 2
  1328→  1328→                        
  1329→  1329→                        // Top Flange
  1330→  1330→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1331→  1331→                        // Bottom Flange
  1332→  1332→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1333→  1333→                        // Web
  1334→  1334→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1335→  1335→                        
  1336→  1336→                        // Stress Distribution (Schematic)
  1337→  1337→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1338→  1338→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1339→  1339→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1340→  1340→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1341→  1341→                    }
  1342→  1342→                    is SteelSectionType.LSection -> {
  1343→  1343→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
  1344→  1344→                        drawRect(Color.DarkGray, Offset(0f, 0f), Size(t, h))
  1345→  1345→                        drawRect(Color.DarkGray, Offset(t, h - t), Size(w - t, t))
  1346→  1346→                    }
  1347→  1347→                    is SteelSectionType.RHS -> {
  1348→  1348→                        val t = (section.thickness / section.height * h).toFloat().coerceIn(5f, 20f)
  1349→  1349→                        drawRect(Color.DarkGray, Offset(0f, 0f), size, style = androidx.compose.ui.graphics.drawscope.Stroke(t))
  1350→  1350→                    }
  1351→  1351→                    else -> {
  1352→  1352→                        drawRect(Color.DarkGray, Offset(0f, 0f), size)
  1353→  1353→                    }
  1354→  1354→                }
  1355→  1355→            }
  1356→  1356→            Spacer(Modifier.height(12.dp))
  1357→  1357→            Text("توزيع الإجهادات المرن", style = MaterialTheme.typography.labelSmall, color = Color.Gray)
  1358→  1358→        }
  1359→  1359→    }
