   484→@Composable
   485→private fun ColumnSectionCanvas(result: CalculatorEngine.ColumnResult) {
   486→    Card(
   487→        shape = RoundedCornerShape(16.dp),
   488→        modifier = Modifier.fillMaxWidth().height(300.dp),
   489→        elevation = CardDefaults.cardElevation(4.dp)
   490→    ) {
   491→        Canvas(modifier = Modifier.fillMaxSize().background(Color.White)) {
   492→            val padding = 60f
   493→            val scale = minOf((size.width - 2*padding)/result.width, (size.height - 2*padding)/result.depth).toFloat()
   494→            
   495→            val w = (result.width * scale).toFloat()
   496→            val h = (result.depth * scale).toFloat()
   497→            val left = (size.width - w) / 2
   498→            val top = (size.height - h) / 2
   499→            
   500→            // الخرسانة
   501→            drawRect(Color.LightGray, Offset(left, top), Size(w, h))
   502→            drawRect(Color.DarkGray, Offset(left, top), Size(w, h), style = Stroke(4f))
   503→            
   504→            // الكانات
   505→            val cover = 25f * scale
   506→            drawRect(Color.Blue.copy(alpha = 0.7f), Offset(left + cover, top + cover), Size(w - 2*cover, h - 2*cover), style = Stroke(3f))
   507→            
   508→            // حديد التسليح
   509→            val barRadius = 8f
   510→            val nBars = result.reinforcement.numBars
   511→            val barsSideX = max(2, (result.width / (result.width + result.depth) * nBars / 2).toInt() + 1)
   512→            val barsSideY = max(2, (result.depth / (result.width + result.depth) * nBars / 2).toInt() + 1)
   513→            
   514→            for (i in 0 until barsSideX) {
   515→                val x = left + cover + i * (w - 2*cover) / (barsSideX - 1)
   516→                drawCircle(Color.Black, barRadius, Offset(x, top + cover))
   517→                drawCircle(Color.Black, barRadius, Offset(x, top + h - cover))
   518→            }
   519→            for (i in 1 until barsSideY - 1) {
   520→                val y = top + cover + i * (h - 2*cover) / (barsSideY - 1)
   521→                drawCircle(Color.Black, barRadius, Offset(left + cover, y))
   522→                drawCircle(Color.Black, barRadius, Offset(left + w - cover, y))
   523→            }
   524→            
   525→            drawContext.canvas.nativeCanvas.apply {
   526→                val paint = android.graphics.Paint().apply {
   527→                    color = android.graphics.Color.BLACK
   528→                    textSize = 35f
   529→                    textAlign = android.graphics.Paint.Align.CENTER
   530→                }
   531→                drawText("${result.width.toInt()} mm", size.width/2, top - 20f, paint)
   532→                
   533→                val midY = size.height/2
   534→                val paintVert = android.graphics.Paint().apply {
   535→                    color = android.graphics.Color.BLACK
   536→                    textSize = 35f
   537→                    textAlign = android.graphics.Paint.Align.CENTER
   538→                }
   539→                save()
   540→                rotate(-90f, left - 30f, midY)
   541→                drawText("${result.depth.toInt()} mm", left - 30f, midY, paintVert)
   542→                restore()
   543→            }
   544→        }
   545→    }
   546→}
