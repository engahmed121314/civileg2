   977→   977→@Composable
   978→   978→private fun UtilizationBar(ratio: Double, modifier: Modifier = Modifier) {
   979→   979→    val clampedRatio = ratio.coerceIn(0.0, 1.5)
   980→   980→    val color = when {
   981→   981→        clampedRatio <= 0.7 -> Color(0xFF4CAF50)
   982→   982→        clampedRatio <= 1.0 -> Color(0xFFFF9800)
   983→   983→        else -> Color(0xFFF44336)
   984→   984→    }
   985→   985→    Row(modifier = modifier.height(16.dp), verticalAlignment = Alignment.CenterVertically) {
   986→   986→        LinearProgressIndicator(
   987→   987→            progress = { clampedRatio.toFloat() / 1.5f },
   988→   988→            modifier = Modifier
   989→   989→                .weight(1f)
   990→   990→                .height(8.dp)
   991→   991→                .clip(RoundedCornerShape(4.dp)),
   992→   992→            color = color,
   993→   993→            trackColor = Color.LightGray
   994→   994→        )
   995→   995→        Spacer(modifier = Modifier.width(8.dp))
   996→   996→        Text(
   997→   997→            "${String.format("%.0f", ratio * 100)}%",
   998→   998→            fontSize = 11.sp,
   999→   999→            fontWeight = FontWeight.Bold,
  1000→  1000→            color = color
  1001→  1001→        )
  1002→  1002→    }
