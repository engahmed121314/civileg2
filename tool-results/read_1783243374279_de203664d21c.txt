    32→    private val PRIMARY_COLOR = DeviceRgb(21, 101, 192) // Professional Blue
    33→    private val SECONDARY_COLOR = DeviceRgb(69, 90, 100)
    34→    private val SUCCESS_COLOR = DeviceRgb(46, 125, 50)
    35→    private val ERROR_COLOR = DeviceRgb(198, 40, 40)

    39→    private fun getArabicFont(): PdfFont? {
    40→        val paths = arrayOf(
    41→            "/system/fonts/NotoNaskhArabic-Regular.ttf",
    42→            "/system/fonts/DroidSansArabic.ttf",
    43→            "/system/fonts/Arbutus-Regular.ttf"
    44→        )
    45→        for (path in paths) {
    46→            try {
    47→                if (File(path).exists()) {
    48→                    return PdfFontFactory.createFont(path, PdfEncodings.IDENTITY_H)
    49→                }
    50→            } catch (e: Exception) {
    51→                e.printStackTrace()
    52→            }
    53→        }
    54→        return try {
    55→            PdfFontFactory.createFont(StandardFonts.HELVETICA, PdfEncodings.CP1252)
    56→        } catch (e: Exception) {
    57→            null
    58→        }
    59→    }
    60→
    61→    private fun containsArabic(text: String): Boolean {
    62→        return text.any { it.code in 0x0600..0x06FF }
    63→    }
    64→
    65→    private fun createStyledParagraph(text: String, font: PdfFont?, fontSize: Float = 12f, isBold: Boolean = false): Paragraph {
    66→        val p = Paragraph(text).setFontSize(fontSize)
    67→        if (isBold) p.setBold()
    68→        if (font != null && containsArabic(text)) {
    69→            p.setFont(font)
    70→            p.setBaseDirection(BaseDirection.RIGHT_TO_LEFT)
    71→        }
    72→        return p
    73→    }
    74→
    75→    private fun addArabicCell(table: Table, text: String, font: PdfFont?, isBold: Boolean = false, fontSize: Float = 9f) {
    76→        val p = createStyledParagraph(text, font, fontSize, isBold)
    77→        table.addCell(Cell().add(p))
    78→    }
    79→
    80→    private fun createDataCell(label: String, value: String, font: PdfFont?): Cell {
    81→        val table = Table(UnitValue.createPercentArray(floatArrayOf(60f, 40f))).useAllAvailableWidth()
    82→        val pLabel = createStyledParagraph(label, font, 9f)
    83→        table.addCell(Cell().add(pLabel).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
    84→        val pValue = createStyledParagraph(value, font, 9f, isBold = true)
    85→        table.addCell(Cell().add(pValue).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
    86→        return Cell().add(table)
    87→    }
    88→
    89→    private fun addMemberRow(table: Table, type: String, section: String, status: String, font: PdfFont?) {
    90→        table.addCell(Cell().add(createStyledParagraph(type, font, 9f)))
    91→        table.addCell(Cell().add(createStyledParagraph(section, font, 9f, isBold = true)))
    92→        val statusP = createStyledParagraph(status, font, 9f, isBold = true)
    93→        if (status.contains("Safe", true) || status.contains("آمن", true)) {
    94→            statusP.setFontColor(SUCCESS_COLOR)
    95→        } else {
    96→            statusP.setFontColor(ERROR_COLOR)
    97→        }
    98→        table.addCell(Cell().add(statusP))
    99→    }
   100→
   101→    private fun createSummaryCell(label: String, value: String, font: PdfFont?): Cell {
   102→        val cell = Cell().setPadding(10f).setTextAlignment(TextAlignment.CENTER)
   103→        cell.add(createStyledParagraph(label, font, 10f).setFontColor(ColorConstants.GRAY))
   104→        cell.add(createStyledParagraph(value, font, 14f, isBold = true).setFontColor(PRIMARY_COLOR))
   105→        return cell
   106→    }

   108→    fun generateProfessionalReport(
   109→        context: Context,
   110→        title: String,
   111→        designType: String,
   112→        inputs: Map<String, String>,
   113→        results: Map<String, String>,
   114→        safetyChecks: List<CalculatorEngine.DesignSafetyCheck>,
   115→        isSafe: Boolean,
   116→        drawingBitmap: Bitmap? = null
   117→    ): File {
   118→        val fileName = "CivilEngPro_${designType}_${System.currentTimeMillis()}.pdf"
   119→        val file = File(context.getExternalFilesDir(null), fileName)
   120→
   121→        val writer = PdfWriter(file)
   122→        val pdf = PdfDocument(writer)
   123→        val document = Document(pdf)
   124→        val arabicFont = getArabicFont()
   125→
   126→        // --- Header Section ---
   127→        val header = Table(UnitValue.createPercentArray(floatArrayOf(70f, 30f))).useAllAvailableWidth()
   128→        val appNamePara = createStyledParagraph(context.getString(R.string.app_name), arabicFont, 26f, isBold = true)
   129→            .setFontColor(PRIMARY_COLOR)
   130→        header.addCell(Cell().add(appNamePara).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
   131→        
   132→        header.addCell(Cell().add(Paragraph("STRUCTURAL REPORT")
   133→            .setTextAlignment(TextAlignment.RIGHT).setFontSize(10f).setFontColor(ColorConstants.GRAY)).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
   134→        document.add(header)
   135→        
   136→        document.add(LineSeparator(SolidLine(1f)))
   137→        document.add(Paragraph("\n"))
   138→
   139→        // --- Title & Status ---
   140→        document.add(createStyledParagraph(title, arabicFont, 18f, isBold = true))
   141→
   142→        val statusText = if (isSafe) "STATUS: SAFE (مطابق للأكواد)" else "STATUS: UNSAFE (غير آمن - مراجعة التصميم)"
   143→        val statusPara = createStyledParagraph(statusText, arabicFont, 12f, isBold = true)
   144→            .setFontColor(if (isSafe) SUCCESS_COLOR else ERROR_COLOR)
   145→            .setPadding(10f)
   146→            .setBorder(com.itextpdf.layout.borders.SolidBorder(if (isSafe) SUCCESS_COLOR else ERROR_COLOR, 1f))
   147→        document.add(statusPara)
   148→        
   149→        document.add(Paragraph("\n"))
   150→
   151→        // --- Split Layout: Info vs Drawing ---
   152→        val mainTable = Table(UnitValue.createPercentArray(floatArrayOf(50f, 50f))).useAllAvailableWidth()
   153→
   154→        // Column 1: Inputs & Results
   155→        val infoCell = Cell().setBorder(com.itextpdf.layout.borders.Border.NO_BORDER)
   156→        infoCell.add(createStyledParagraph("DESIGN DATA", arabicFont, 12f, isBold = true).setFontColor(PRIMARY_COLOR))
   157→        
   158→        val dataTable = Table(UnitValue.createPercentArray(floatArrayOf(60f, 40f))).useAllAvailableWidth()
   159→        inputs.forEach { (k, v) ->
   160→            addArabicCell(dataTable, k, arabicFont, fontSize = 9f)
   161→            addArabicCell(dataTable, v, arabicFont, isBold = true, fontSize = 9f)
   162→        }
   163→        results.forEach { (k, v) ->
   164→            addArabicCell(dataTable, k, arabicFont, fontSize = 9f)
   165→            addArabicCell(dataTable, v, arabicFont, isBold = true, fontSize = 9f)
   166→        }
   167→        infoCell.add(dataTable)
   168→        mainTable.addCell(infoCell)
   169→
   170→        // Column 2: Drawing
   171→        val drawingCell = Cell().setBorder(com.itextpdf.layout.borders.Border.NO_BORDER).setPaddingLeft(10f)
   172→        drawingCell.add(createStyledParagraph("SECTION DRAWING (رسم تسليح القطاع)", arabicFont, 12f, isBold = true).setFontColor(PRIMARY_COLOR))
   173→        if (drawingBitmap != null) {
   174→            val stream = ByteArrayOutputStream()
   175→            drawingBitmap.compress(Bitmap.CompressFormat.PNG, 100, stream)
   176→            val img = Image(ImageDataFactory.create(stream.toByteArray()))
   177→            img.setAutoScale(true)
   178→            drawingCell.add(img)
   179→        } else {
   180→            drawingCell.add(Paragraph("[Drawing Placeholder]").setItalic().setFontColor(ColorConstants.GRAY))
   181→        }
   182→        mainTable.addCell(drawingCell)
   183→        
   184→        document.add(mainTable)
   185→        document.add(Paragraph("\n"))
   186→
   187→        // --- Safety Checks Table ---
   188→        if (safetyChecks.isNotEmpty()) {
   189→            document.add(createStyledParagraph("SAFETY VERIFICATIONS (تحققات الأمان)", arabicFont, 12f, isBold = true).setFontColor(PRIMARY_COLOR))
   190→            val checkTable = Table(UnitValue.createPercentArray(floatArrayOf(40f, 20f, 20f, 20f))).useAllAvailableWidth()
   191→            
   192→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Check Type", arabicFont, 9f, true)))
   193→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Calculated", arabicFont, 9f, true)))
   194→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Limit", arabicFont, 9f, true)))
   195→            checkTable.addHeaderCell(Cell().add(createStyledParagraph("Result", arabicFont, 9f, true)))
   196→
   197→            safetyChecks.forEach { check ->
   198→                addArabicCell(checkTable, check.name, arabicFont, fontSize = 9f)
   199→                checkTable.addCell(Cell().add(Paragraph(String.format(Locale.US, "%.2f %s", check.value, check.unit)).setFontSize(9f)))
   200→                checkTable.addCell(Cell().add(Paragraph(String.format(Locale.US, "%.2f %s", check.limit, check.unit)).setFontSize(9f)))
   201→                val resText = if (check.isSafe) "PASS" else "FAIL"
   202→                val resPara = createStyledParagraph(resText, arabicFont, 9f, isBold = true)
   203→                    .setFontColor(if (check.isSafe) SUCCESS_COLOR else ERROR_COLOR)
   204→                checkTable.addCell(Cell().add(resPara))
   205→            }
   206→            document.add(checkTable)
   207→        }
   208→
   209→        // --- Footer ---
   210→        document.add(Paragraph("\n\n"))
   211→        document.add(Paragraph("This report is generated based on Structural Design Codes (ECP/ACI/SBC). Professional review is mandatory before execution.")
   212→            .setFontSize(8f).setItalic().setFontColor(ColorConstants.GRAY).setTextAlignment(TextAlignment.CENTER))
   213→        
   214→        document.add(Paragraph(java.util.Date().toString())
   215→            .setFontSize(8f).setTextAlignment(TextAlignment.RIGHT))
   216→
   217→        document.close()
   218→        return file
   219→    }

   402→        context: Context,
   403→        result: SteelMemberResult,
   404→        drawingBitmap: Bitmap? = null
   405→    ): File {
   406→        val fileName = "SteelSection_Analysis_${System.currentTimeMillis()}.pdf"
   407→        val file = File(context.getExternalFilesDir(null), fileName)
   408→
   409→        val writer = PdfWriter(file)
   410→        val pdf = PdfDocument(writer)
   411→        val document = Document(pdf)
   412→        val arabicFont = getArabicFont()
   413→
   414→        // --- Header ---
   415→        val header = Table(UnitValue.createPercentArray(floatArrayOf(70f, 30f))).useAllAvailableWidth()
   416→        val appNamePara = createStyledParagraph(context.getString(R.string.app_name), arabicFont, 24f, true)
   417→            .setFontColor(PRIMARY_COLOR)
   418→        header.addCell(Cell().add(appNamePara).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
   419→        header.addCell(Cell().add(Paragraph("STEEL SECTION ANALYSIS").setTextAlignment(TextAlignment.RIGHT).setFontSize(10f).setFontColor(ColorConstants.GRAY)).setBorder(com.itextpdf.layout.borders.Border.NO_BORDER))
   420→        document.add(header)
   421→        document.add(LineSeparator(SolidLine(1f)))
   422→
   423→        document.add(Paragraph("\n"))
   424→        document.add(Paragraph("Technical Datasheet: ${result.sectionType.displayName}").setFontSize(18f).setBold().setTextAlignment(TextAlignment.CENTER))
   425→        
   426→        // Status Card
   427→        val statusText = if (result.isSafe) "SECTION IS SAFE ✅" else "SECTION IS UNSAFE ❌"
   428→        document.add(Paragraph(statusText).setFontColor(if (result.isSafe) SUCCESS_COLOR else ERROR_COLOR).setBold().setPadding(10f).setBorder(com.itextpdf.layout.borders.SolidBorder(if (result.isSafe) SUCCESS_COLOR else ERROR_COLOR, 1f)).setTextAlignment(TextAlignment.CENTER))
   429→
   430→        document.add(Paragraph("\n"))
   431→
   432→        // --- Key Properties Table ---
   433→        document.add(createStyledParagraph("Geometric Properties (الخصائص الهندسية)", arabicFont, 12f, true).setFontColor(PRIMARY_COLOR))
   434→        
   435→        val propTable = Table(UnitValue.createPercentArray(floatArrayOf(50f, 50f))).useAllAvailableWidth()
   436→        propTable.addCell(createDataCell("Area (A)", "%.2f cm²".format(result.sectionType.getArea()/100.0), arabicFont))
   437→        propTable.addCell(createDataCell("Weight (W)", "%.2f kg/m".format(result.weight), arabicFont))
   438→        
   439→        val section = result.sectionType
   440→        if (section is SteelSectionType.ISection) {
   441→            propTable.addCell(createDataCell("Depth (h)", "${section.h} mm", arabicFont))
   442→            propTable.addCell(createDataCell("Flange Width (bf)", "${section.bf} mm", arabicFont))
   443→            propTable.addCell(createDataCell("Web Thickness (tw)", "${section.tw} mm", arabicFont))
   444→            propTable.addCell(createDataCell("Flange Thickness (tf)", "${section.tf} mm", arabicFont))
   445→        }
   446→        document.add(propTable)
   447→
   448→        document.add(Paragraph("\n"))
   449→
   450→        // --- Design Capacities ---
   451→        document.add(createStyledParagraph("Design Capacities (سعة التحمل القصوى)", arabicFont, 12f, true).setFontColor(PRIMARY_COLOR))
   452→        
   453→        val capacityTable = Table(UnitValue.createPercentArray(floatArrayOf(50f, 50f))).useAllAvailableWidth()
   454→        capacityTable.addCell(createDataCell("Axial Capacity (Pn)", "%.1f kN".format(result.axialCapacity), arabicFont))
   455→        capacityTable.addCell(createDataCell("Flexural Capacity (Mn)", "%.1f kN.m".format(result.flexuralCapacity), arabicFont))
   456→        capacityTable.addCell(createDataCell("Shear Capacity (Vn)", "%.1f kN".format(result.shearCapacity), arabicFont))
   457→        capacityTable.addCell(createDataCell("Utilization Ratio", "%.2f%%".format(result.utilizationRatio * 100), arabicFont))
   458→        document.add(capacityTable)
   459→
   460→        // --- Drawing Section ---
   461→        if (drawingBitmap != null) {
   462→            document.add(Paragraph("\n"))
   463→            document.add(createStyledParagraph("Section Visualization (رسم تخطيطي)", arabicFont, 12f, true).setFontColor(PRIMARY_COLOR))
   464→            val stream = ByteArrayOutputStream()
   465→            drawingBitmap.compress(Bitmap.CompressFormat.PNG, 100, stream)
   466→            val img = Image(ImageDataFactory.create(stream.toByteArray()))
   467→            img.setMaxWidth(200f).setHorizontalAlignment(HorizontalAlignment.CENTER)
   468→            document.add(img)
   469→        }
   470→
   471→        // --- Buckling & Stability ---
   472→        result.bucklingCheck?.let { check ->
   473→            document.add(createStyledParagraph("\nBuckling & Stability Analysis (تحليل الانبعاج)", arabicFont, 12f, true).setFontColor(PRIMARY_COLOR))
   474→            
   475→            val buckTable = Table(UnitValue.createPercentArray(floatArrayOf(50f, 50f))).useAllAvailableWidth()
   476→            buckTable.addCell(createDataCell("Slenderness Ratio (λ)", "%.2f".format(check.slendernessRatio), arabicFont))
   477→            buckTable.addCell(createDataCell("Critical Stress (Fcr)", "%.1f MPa".format(check.criticalStress), arabicFont))
   478→            buckTable.addCell(createDataCell("Buckling Mode", check.bucklingMode.name, arabicFont))
   479→            buckTable.addCell(createDataCell("Stability Status", if (check.isSafe) "Stable" else "Unstable", arabicFont))
   480→            document.add(buckTable)
   481→        }
   482→
   483→        // --- Warnings & Notes ---
   484→        if (result.warnings.isNotEmpty()) {
   485→            document.add(createStyledParagraph("\nWarnings (تحذيرات فنية)", arabicFont, 12f, true).setFontColor(ERROR_COLOR))
   486→            val warningList = com.itextpdf.layout.element.List()
   487→            result.warnings.forEach { warning -> 
   488→                val li = ListItem(warning)
   489→                if (containsArabic(warning) && arabicFont != null) {
   490→                    li.setFont(arabicFont).setBaseDirection(BaseDirection.RIGHT_TO_LEFT)
   491→                }
   492→                warningList.add(li) 
   493→            }
   494→            document.add(warningList)
   495→        }
   496→
   497→        // --- Footer ---
   498→        document.add(Paragraph("\n\n"))
   499→        document.add(Paragraph("Detailed technical analysis provided by CivilEG Pro Engine. This report is for engineering guidance only.")
   500→            .setFontSize(8f).setItalic().setFontColor(ColorConstants.GRAY).setTextAlignment(TextAlignment.CENTER))
   501→        document.add(Paragraph("Generated: ${java.util.Date()}").setFontSize(8f).setTextAlignment(TextAlignment.RIGHT))
   502→
   503→        document.close()
   504→        return file
   505→    }
