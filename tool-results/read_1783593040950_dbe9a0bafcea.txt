   281→   281→@Composable
   282→   282→fun StructuralAnalysisVisualizer(inputs: SteelWarehouseInputs, result: SteelWarehouseAnalysisResult) {
   283→   283→    Card(
   284→   284→        modifier = Modifier.fillMaxWidth().height(220.dp).padding(vertical = 8.dp),
   285→   285→        border = BorderStroke(1.dp, Color.LightGray)
   286→   286→    ) {
   287→   287→        ComposeCanvas(modifier = Modifier.fillMaxSize().background(Color(0xFFF8F9FA))) {
   288→   288→            val padding = 40.dp.toPx()
   289→   289→            val w = size.width - 2 * padding
   290→   290→            val h = size.height - 2 * padding
   291→   291→            val baseY = size.height - padding
   292→   292→            
   293→   293→            // Scale calculations
   294→   294→            val scale = (w / inputs.span).toFloat()
   295→   295→            val eh = (inputs.eaveHeight * scale).toFloat()
   296→   296→            val rh = (inputs.ridgeHeight * scale).toFloat()
   297→   297→            val midX = padding + w / 2
   298→   298→            
   299→   299→            // Draw Main Frame Outline
   300→   300→            val framePath = Path().apply {
   301→   301→                moveTo(padding, baseY)
   302→   302→                lineTo(padding, baseY - eh)
   303→   303→                lineTo(midX, baseY - rh)
   304→   304→                lineTo(padding + w, baseY - eh)
   305→   305→                lineTo(padding + w, baseY)
   306→   306→            }
   307→   307→            drawPath(framePath, Color.Gray, style = androidx.compose.ui.graphics.drawscope.Stroke(1f))
   308→   308→            
   309→   309→            // Draw Distributed Load Arrows
   310→   310→            val loadColor = Color(0xFFE53935)
   311→   311→            for (i in 0..12) {
   312→   312→                val x = padding + (i * w / 12)
   313→   313→                val y = if (x <= midX) {
   314→   314→                    baseY - eh - (x - padding) / (midX - padding) * (rh - eh)
   315→   315→                } else {
   316→   316→                    baseY - rh + (x - midX) / (padding + w - midX) * (rh - eh)
   317→   317→                }
   318→   318→                drawLine(loadColor, Offset(x, y - 25f), Offset(x, y), strokeWidth = 2f)
   319→   319→                // Arrow heads
   320→   320→                drawLine(loadColor, Offset(x - 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   321→   321→                drawLine(loadColor, Offset(x + 4f, y - 6f), Offset(x, y), strokeWidth = 2f)
   322→   322→            }
   323→   323→            
   324→   324→            // Draw Moment Diagram (Simplified)
   325→   325→            val momentColor = Color(0xFF1E88E5)
   326→   326→            val momentPath = Path().apply {
   327→   327→                moveTo(padding, baseY - eh)
   328→   328→                // Simulated parabolic curve for moment
   329→   329→                cubicTo(
   330→   330→                    padding + w * 0.25f, baseY - eh + 50f,
   331→   331→                    padding + w * 0.75f, baseY - eh + 50f,
   332→   332→                    padding + w, baseY - eh
   333→   333→                )
   334→   334→            }
   335→   335→            drawPath(momentPath, momentColor.copy(alpha = 0.15f))
   336→   336→            drawPath(momentPath, momentColor, style = androidx.compose.ui.graphics.drawscope.Stroke(2f))
   337→   337→            
   338→   338→            // Labels
   339→   339→            // Note: NativeCanvas is used for text if needed, but we'll keep it simple
   340→   340→        }
   341→   341→    }

   391→   391→                )

   393→   393→                val baseY = size.height - padding
   394→   394→

   397→   397→                        drawFrontElevation(inputs, scale, startX, baseY)
   398→   398→                    }
   399→   399→                    1 -> { // Plan View
   400→   400→                        drawPlanView(inputs, scale, startX, baseY)
   401→   401→                    }
   402→   402→                    2 -> { // Side View
   403→   403→                        drawSideElevation(inputs, scale, startX, baseY)
   404→   404→                    }
   405→   405→                    3 -> { // 3D Perspective
   406→   406→                        draw3DView(inputs, scale, startX, baseY)
   407→   407→                    }
   408→   408→                }
   409→   409→            }

   433→   433→private fun androidx.compose.ui.graphics.drawscope.DrawScope.drawPlanView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {

   470→   470→private fun androidx.compose.ui.graphics.drawscope.DrawScope.draw3DView(inputs: SteelWarehouseInputs, scale: Double, startX: Float, baseY: Float) {
   471→   471→    val s = scale.toFloat() * 0.65f
   472→   472→    val depthOff = 35f
   473→   473→    
   474→   474→    // Draw from back to front for better perspective look
   475→   475→    for (i in 4 downTo 0) {
   476→   476→        val off = i * depthOff
   477→   477→        val x0 = startX + off
   478→   478→        val y0 = baseY - off
   479→   479→        val spanW = (inputs.span * s).toFloat()
   480→   480→        val eaveH = (inputs.eaveHeight * s).toFloat()
   481→   481→        val ridgeH = (inputs.ridgeHeight * s).toFloat()
   482→   482→        
   483→   483→        val color = if (i == 0) Color.Black else Color.Gray.copy(alpha = 0.4f)
   484→   484→        val stroke = if (i == 0) 3f else 1.5f
   485→   485→        
   486→   486→        // Sections of the frame
   487→   487→        drawLine(color, Offset(x0, y0), Offset(x0, y0 - eaveH), strokeWidth = stroke)
   488→   488→        drawLine(color, Offset(x0 + spanW, y0), Offset(x0 + spanW, y0 - eaveH), strokeWidth = stroke)
   489→   489→        drawLine(color, Offset(x0, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   490→   490→        drawLine(color, Offset(x0 + spanW, y0 - eaveH), Offset(x0 + spanW / 2, y0 - ridgeH), strokeWidth = stroke)
   491→   491→        
   492→   492→        // Purlins/Girts connections (longitudinal)
   493→   493→        if (i > 0) {
   494→   494→            val px = x0 - depthOff
   495→   495→            val py = y0 + depthOff
   496→   496→            val lineCol = Color.LightGray.copy(alpha = 0.5f)
   497→   497→            drawLine(lineCol, Offset(x0, y0 - eaveH), Offset(px, py - eaveH), strokeWidth = 1f)
   498→   498→            drawLine(lineCol, Offset(x0 + spanW, y0 - eaveH), Offset(px + spanW, py - eaveH), strokeWidth = 1f)
   499→   499→            drawLine(lineCol, Offset(x0 + spanW / 2, y0 - ridgeH), Offset(px + spanW / 2, py - ridgeH), strokeWidth = 1f)
   500→   500→        }
   501→   501→    }
   502→   502→}

   536→   536→private fun drawCanvasElevation(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float, result: SteelWarehouseAnalysisResult? = null) {
   537→   537→    val p = Paint().apply { 
   538→   538→        color = android.graphics.Color.BLACK
   539→   539→        strokeWidth = 6f
   540→   540→        isAntiAlias = true
   541→   541→        style = Paint.Style.STROKE
   542→   542→    }
   543→   543→    val dimPaint = Paint().apply {
   544→   544→        color = android.graphics.Color.DKGRAY
   545→   545→        strokeWidth = 2f
   546→   546→        textSize = 24f
   547→   547→        isAntiAlias = true
   548→   548→    }
   549→   549→    val textP = Paint().apply {
   550→   550→        color = android.graphics.Color.BLACK
   551→   551→        textSize = 22f
   552→   552→        isFakeBoldText = true
   553→   553→        textAlign = Paint.Align.CENTER
   554→   554→    }
   555→   555→    
   556→   556→    val s = scale * 0.7f
   557→   557→    val w = inputs.span.toFloat() * s
   558→   558→    val eh = inputs.eaveHeight.toFloat() * s
   559→   559→    val rh = inputs.ridgeHeight.toFloat() * s
   560→   560→    val midX = x + w/2
   561→   561→    
   562→   562→    // Draw Main Frame (Columns & Rafters)
   563→   563→    canvas.drawLine(x, y, x, y - eh, p)
   564→   564→    canvas.drawLine(x + w, y, x + w, y - eh, p)
   565→   565→    canvas.drawLine(x, y - eh, midX, y - rh, p)
   566→   566→    canvas.drawLine(x + w, y - eh, midX, y - rh, p)
   567→   567→    
   568→   568→    // Foundations
   569→   569→    val footP = Paint().apply { color = android.graphics.Color.LTGRAY; style = Paint.Style.FILL }
   570→   570→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, footP)
   571→   571→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, footP)
   572→   572→    canvas.drawRect(x - 20f, y, x + 20f, y + 10f, p.apply { strokeWidth = 2f })
   573→   573→    canvas.drawRect(x + w - 20f, y, x + w + 20f, y + 10f, p)
   574→   574→    p.strokeWidth = 6f
   575→   575→    
   576→   576→    // --- Dimension Lines ---
   577→   577→    // Span Dimension
   578→   578→    drawDimensionLine(canvas, x, y + 60f, x + w, y + 60f, "${inputs.span} m", dimPaint)
   579→   579→    
   580→   580→    // Eave Height Dimension
   581→   581→    drawDimensionLine(canvas, x - 60f, y, x - 60f, y - eh, "${inputs.eaveHeight} m", dimPaint)
   582→   582→    
   583→   583→    // Ridge Height Dimension
   584→   584→    drawDimensionLine(canvas, x + w + 60f, y, x + w + 60f, y - rh, "${inputs.ridgeHeight} m", dimPaint)
   585→   585→
   586→   586→    // --- Section Labels ---
   587→   587→    result?.let { res ->
   588→   588→        val sectionP = Paint().apply { color = android.graphics.Color.BLUE; textSize = 20f; isFakeBoldText = true }
   589→   589→        
   590→   590→        // Column Label
   591→   591→        canvas.save()
   592→   592→        canvas.rotate(-90f, x + 25f, y - eh/2)
   593→   593→        canvas.drawText(res.mainFrame.columnSection.displayName, x + 25f, y - eh/2, sectionP.apply { textAlign = Paint.Align.CENTER })
   594→   594→        canvas.restore()
   595→   595→        
   596→   596→        // Rafter Label
   597→   597→        canvas.save()
   598→   598→        val angle = Math.toDegrees(Math.atan2((rh - eh).toDouble(), (w/2).toDouble())).toFloat()
   599→   599→        canvas.rotate(-angle, x + w/4, y - eh - (rh-eh)/4 - 15f)
   600→   600→        canvas.drawText(res.mainFrame.rafterSection.displayName, x + w/4, y - eh - (rh-eh)/4 - 15f, sectionP.apply { textAlign = Paint.Align.CENTER })
   601→   601→        canvas.restore()
   602→   602→    }
   603→   603→
   604→   604→    // Connection Markers (Circles at joints)
   605→   605→    val connPaint = Paint().apply { color = android.graphics.Color.RED; style = Paint.Style.STROKE; strokeWidth = 2f }
   606→   606→    canvas.drawCircle(x, y - eh, 15f, connPaint) // Eave joint left
   607→   607→    canvas.drawCircle(x + w, y - eh, 15f, connPaint) // Eave joint right
   608→   608→    canvas.drawCircle(midX, y - rh, 15f, connPaint) // Apex joint
   609→   609→}

   611→   611→private fun drawDimensionLine(canvas: Canvas, x1: Float, y1: Float, x2: Float, y2: Float, text: String, paint: Paint) {
   612→   612→    val tickSize = 15f
   613→   613→    canvas.drawLine(x1, y1, x2, y2, paint)
   614→   614→    
   615→   615→    // Draw Ticks (45 degree lines)
   616→   616→    val angle = Math.atan2((y2 - y1).toDouble(), (x2 - x1).toDouble())
   617→   617→    val tickAngle = angle + Math.PI / 4
   618→   618→    
   619→   619→    canvas.drawLine(x1 - (Math.cos(tickAngle) * tickSize).toFloat(), y1 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   620→   620→                    x1 + (Math.cos(tickAngle) * tickSize).toFloat(), y1 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   621→   621→    canvas.drawLine(x2 - (Math.cos(tickAngle) * tickSize).toFloat(), y2 - (Math.sin(tickAngle) * tickSize).toFloat(), 
   622→   622→                    x2 + (Math.cos(tickAngle) * tickSize).toFloat(), y2 + (Math.sin(tickAngle) * tickSize).toFloat(), paint)
   623→   623→
   624→   624→    // Text label
   625→   625→    val midX = (x1 + x2) / 2
   626→   626→    val midY = (y1 + y2) / 2
   627→   627→    val textPaint = Paint(paint).apply { textAlign = Paint.Align.CENTER }
   628→   628→    
   629→   629→    if (x1 == x2) { // Vertical
   630→   630→        canvas.save()
   631→   631→        canvas.rotate(-90f, midX - 10f, midY)
   632→   632→        canvas.drawText(text, midX - 10f, midY, textPaint)
   633→   633→        canvas.restore()
   634→   634→    } else { // Horizontal
   635→   635→        canvas.drawText(text, midX, midY - 10f, textPaint)
   636→   636→    }
   637→   637→}

   639→   639→private fun drawCanvasPlan(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   640→   640→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 4f; style = Paint.Style.STROKE; isAntiAlias = true }
   641→   641→    val dimPaint = Paint().apply { color = android.graphics.Color.DKGRAY; strokeWidth = 1.5f; textSize = 20f; isAntiAlias = true }
   642→   642→    
   643→   643→    val s = scale * 0.6f
   644→   644→    val w = inputs.span.toFloat() * s
   645→   645→    val l = inputs.length.toFloat() * s
   646→   646→    val planTop = y - l
   647→   647→    
   648→   648→    canvas.drawRect(x, planTop, x + w, y, p)
   649→   649→    
   650→   650→    val baySpacing = inputs.baySpacing.toFloat() * s
   651→   651→    var curY = planTop
   652→   652→    var count = 0
   653→   653→    while (curY <= y + 0.1f) {
   654→   654→        canvas.drawLine(x, curY, x + w, curY, p.apply { strokeWidth = 2f })
   655→   655→        
   656→   656→        // Column markers
   657→   657→        canvas.drawRect(x - 5f, curY - 5f, x + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   658→   658→        canvas.drawRect(x + w - 5f, curY - 5f, x + w + 5f, curY + 5f, Paint().apply { color = android.graphics.Color.BLACK })
   659→   659→        
   660→   660→        if (curY + baySpacing <= y + 0.1f) {
   661→   661→            drawDimensionLine(canvas, x - 40f, curY, x - 40f, curY + baySpacing, "${inputs.baySpacing}m", dimPaint)
   662→   662→        }
   663→   663→        
   664→   664→        curY += baySpacing
   665→   665→        count++
   666→   666→    }
   667→   667→    
   668→   668→    // Span Dimension
   669→   669→    drawDimensionLine(canvas, x, y + 40f, x + w, y + 40f, "البحر: ${inputs.span}m", dimPaint)
   670→   670→    // Total Length Dimension
   671→   671→    drawDimensionLine(canvas, x + w + 40f, planTop, x + w + 40f, y, "الطول: ${inputs.length}m", dimPaint)
   672→   672→}

   674→   674→private fun drawCanvasSide(canvas: Canvas, inputs: SteelWarehouseInputs, x: Float, y: Float, scale: Float) {
   675→   675→    val p = Paint().apply { color = android.graphics.Color.BLACK; strokeWidth = 5f; isAntiAlias = true }
   676→   676→    val textP = Paint().apply { color = android.graphics.Color.DKGRAY; textSize = 22f; textAlign = Paint.Align.CENTER }
   677→   677→    
   678→   678→    val s = scale * 0.6f
   679→   679→    val l = inputs.length.toFloat() * s
   680→   680→    val eh = inputs.eaveHeight.toFloat() * s
   681→   681→    
   682→   682→    canvas.drawLine(x, y, x + l, y, p) // Ground
   683→   683→    val bay = inputs.baySpacing.toFloat() * s
   684→   684→    var cur = x
   685→   685→    while (cur <= x + l + 0.1f) {
   686→   686→        canvas.drawLine(cur, y, cur, y - eh, p)
   687→   687→        cur += bay
   688→   688→    }
   689→   689→    canvas.drawLine(x, y - eh, x + l, y - eh, p)
   690→   690→    
   691→   691→    canvas.drawText("الطول الكلي: ${inputs.length}m", x + l/2, y + 35f, textP)
   692→   692→}

  1345→  1345→fun SteelSectionDrawing(section: SteelSectionType) {
  1346→  1346→    Box(modifier = Modifier.fillMaxWidth().height(250.dp).padding(16.dp), contentAlignment = Alignment.Center) {
  1347→  1347→        Column(horizontalAlignment = Alignment.CenterHorizontally) {
  1348→  1348→            ComposeCanvas(modifier = Modifier.size(150.dp)) {
  1349→  1349→                val w = size.width
  1350→  1350→                val h = size.height
  1351→  1351→                
  1352→  1352→                when (section) {
  1353→  1353→                    is SteelSectionType.ISection -> {
  1354→  1354→                        val tf = (section.flangeThickness / section.depth * h).toFloat().coerceIn(5f, 25f)
  1355→  1355→                        val tw = (section.webThickness / section.width * w).toFloat().coerceIn(5f, 20f)
  1356→  1356→                        val bf = (section.width / section.depth * w).toFloat().coerceAtMost(w)
  1357→  1357→                        val offsetBF = (w - bf) / 2
  1358→  1358→                        
  1359→  1359→                        // Top Flange
  1360→  1360→                        drawRect(Color.DarkGray, Offset(offsetBF, 0f), Size(bf, tf))
  1361→  1361→                        // Bottom Flange
  1362→  1362→                        drawRect(Color.DarkGray, Offset(offsetBF, h - tf), Size(bf, tf))
  1363→  1363→                        // Web
  1364→  1364→                        drawRect(Color.DarkGray, Offset((w - tw) / 2, tf), Size(tw, h - 2 * tf))
  1365→  1365→                        
  1366→  1366→                        // Stress Distribution (Schematic)
  1367→  1367→                        drawLine(Color.Red.copy(alpha = 0.3f), Offset(w + 20f, 0f), Offset(w + 20f, h), strokeWidth = 2f)
  1368→  1368→                        drawLine(Color.Red, Offset(w + 20f, 0f), Offset(w + 50f, 0f), strokeWidth = 3f)
  1369→  1369→                        drawLine(Color.Red, Offset(w + 20f, h), Offset(w - 10f, h), strokeWidth = 3f)
  1370→  1370→                        drawLine(Color.Red, Offset(w + 50f, 0f), Offset(w - 10f, h), strokeWidth = 2f)
  1371→  1371→                    }
  1372→  1372→                    is SteelSectionType.LSection -> {
  1373→  1373→                        val t = (section.thickness / section.legA * h).toFloat().coerceIn(5f, 20f)
  1374→  1374→                        drawRect(Color.DarkGray, Offset(0f, 0f), Size(t, h))
  1375→  1375→                        drawRect(Color.DarkGray, Offset(t, h - t), Size(w - t, t))
  1376→  1376→                    }
  1377→  1377→                    is SteelSectionType.RHS -> {
  1378→  1378→                        val t = (section.thickness / section.height * h).toFloat().coerceIn(5f, 20f)
  1379→  1379→                        drawRect(Color.DarkGray, Offset(0f, 0f), size, style = androidx.compose.ui.graphics.drawscope.Stroke(t))
  1380→  1380→                    }
  1381→  1381→                    else -> {
  1382→  1382→                        drawRect(Color.DarkGray, Offset(0f, 0f), size)
  1383→  1383→                    }
  1384→  1384→                }
  1385→  1385→            }
  1386→  1386→            Spacer(Modifier.height(12.dp))
  1387→  1387→            Text("توزيع الإجهادات المرن", style = MaterialTheme.typography.labelSmall, color = Color.Gray)
  1388→  1388→        }
  1389→  1389→    }
